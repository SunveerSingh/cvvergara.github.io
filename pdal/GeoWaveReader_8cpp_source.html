219  List attribs;
220  if (m_useFeatCollDataAdapter)
221  attribs = java_cast<FeatureCollectionDataAdapter>(accumuloAdapterStore.getAdapter(java_new<ByteArrayId>(std::to_string(m_pointsPerEntry) + m_featureTypeName))).getType().getAttributeDescriptors();
222  else
223  attribs = java_cast<FeatureDataAdapter>(accumuloAdapterStore.getAdapter(java_new<ByteArrayId>(m_featureTypeName))).getType().getAttributeDescriptors();
224 
225  for (int i = 0; i < attribs.size(); ++i){
226  std::string name = java_cast<AttributeDescriptor>(attribs.get(i)).getLocalName();
227  if (name.compare("location") != 0 && name.compare("X") != 0 && name.compare("Y") != 0)
228  layout->registerDim(Dimension::id(name));

284  point_count_t GeoWaveReader::read(PointViewPtr view, point_count_t count)
285  {
286  using namespace Dimension;
287 
288  String location = java_new<String>("location");
289  point_count_t numRead = 0;
290 
291  if (m_useFeatCollDataAdapter)
292  {
293  while (m_iterator.hasNext() && count > 0)
294  {
295  SimpleFeatureCollection featureCollection = java_cast<SimpleFeatureCollection>(m_iterator.next());
296  SimpleFeatureIterator featItr = featureCollection.features();
297 
298  while (featItr.hasNext() && count-- > 0)
299  {
300  SimpleFeature simpleFeature = java_cast<SimpleFeature>(featItr.next());
301  List attribs = simpleFeature.getType().getAttributeDescriptors();
302 
303  for (int i = 0; i < attribs.size(); ++i){
304  String name = java_cast<AttributeDescriptor>(attribs.get(i)).getLocalName();
305 
306  if (!name.equals(location))
307  view->setField(id(name), numRead, java_cast<Double>(simpleFeature.getAttribute(name)).doubleValue());
308  }
309 
310  if (m_cb)
311  m_cb(*view, numRead);
312 
313  ++numRead;
314  }
315  featItr.close();
316  }
317  }
318  else
319  {
320  while (m_iterator.hasNext() && count-- > 0){
321  SimpleFeature simpleFeature = java_cast<SimpleFeature>(m_iterator.next());
322  List attribs = simpleFeature.getType().getAttributeDescriptors();
323 
324  for (int i = 0; i < attribs.size(); ++i){
325  String name = java_cast<AttributeDescriptor>(attribs.get(i)).getLocalName();
326 
327  if (!name.equals(location))
328  view->setField(id(name), numRead, java_cast<Double>(simpleFeature.getAttribute(name)).doubleValue());
329  }
330 
331  if (m_cb)
332  m_cb(*view, numRead);
333 
334  ++numRead;
335  }
336  }
337 
338  return numRead;
339  }
