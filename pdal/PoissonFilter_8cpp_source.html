46 class PointViewSource : public PointSource
47 {
48 public:
49  PointViewSource(PointView& view) : m_view(view), m_current(0)
50  {}
51 
52  virtual void reset()
53  { m_current = 0; }
54  virtual bool nextPoint(Point& point)
55  {
56  using namespace Dimension;
57 
58  if (m_current >= m_view.size())
59  return false;
60  point.p.coords[0] = m_view.getFieldAs<double>(Id::X, m_current);
61  point.p.coords[1] = m_view.getFieldAs<double>(Id::Y, m_current);
62  point.p.coords[2] = m_view.getFieldAs<double>(Id::Z, m_current);
63  point.n.coords[0] = m_view.getFieldAs<double>(Id::NormalX, m_current);
64  point.n.coords[1] = m_view.getFieldAs<double>(Id::NormalY, m_current);
65  point.n.coords[2] = m_view.getFieldAs<double>(Id::NormalZ, m_current);
66  m_current++;
67  return true;
68  }
69 
70 private:
71  PointView& m_view;
72  PointId m_current;
73 };

218 void PoissonFilter::addDimensions(PointLayoutPtr layout)
219 {
220  if (layout->hasDim(Dimension::Id::Red) &&
221  layout->hasDim(Dimension::Id::Green) &&
222  layout->hasDim(Dimension::Id::Blue))
223  m_doColor = true;
224 
225  if (layout->hasDim(Dimension::Id::NormalX))
226  {
227  if ((!layout->hasDim(Dimension::Id::NormalY)) ||
228  (!layout->hasDim(Dimension::Id::NormalZ)))
229  throwError("If normals are provided as part of the input "
230  "dataset, all of X, Y and Z must be provided.");
231  m_normalsProvided = true;
232  }
233  else
234  layout->registerDims( {Dimension::Id::NormalX, Dimension::Id::NormalY,
235  Dimension::Id::NormalZ} );
236 }

247 PointViewSet PoissonFilter::run(PointViewPtr view)
248 {
249  if (!m_normalsProvided)
250  NormalFilter().doFilter(*view);
251 
252  std::unique_ptr<PointSource> source;
253  if (m_doColor)
254  source.reset(new ColorPointViewSource(*view));
255  else
256  source.reset(new PointViewSource(*view));
257 
258  PoissonOpts<double> opts;
259 
260  opts.m_depth = m_depth;
261  opts.m_density = m_density;
262  opts.m_solveDepth = m_depth;
263  opts.m_kernelDepth = m_depth - 2;
264  if (m_doColor)
265  {
266  opts.m_color = 16;
267  opts.m_hasColor = true;
268  }
269 
270  PoissonRecon<double> recon(opts, *source);
271  recon.execute();
272  recon.evaluate();
273 
274  PointViewSet s;
275  PointViewPtr outView = view->makeNew();
276  s.insert(outView);
277  PointViewMesh mesh(*outView, m_doColor);
278  recon.extractMesh(mesh);
279  return s;
280 }
