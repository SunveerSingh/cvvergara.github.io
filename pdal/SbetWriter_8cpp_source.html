65 void SbetWriter::write(const PointViewPtr view)
66 {
67  Dimension::IdList dims = fileDimensions();
68  for (PointId idx = 0; idx < view->size(); ++idx)
69  {
70  for (auto di = dims.begin(); di != dims.end(); ++di)
71  {
72  // If a dimension doesn't exist, write 0.
73  Dimension::Id dim = *di;
74  *m_stream << (view->hasDim(dim) ?
75  view->getFieldAs<double>(dim, idx) : 0.0);
76  }
77  }
78 }
