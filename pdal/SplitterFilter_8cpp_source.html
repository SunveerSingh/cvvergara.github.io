79 PointViewSet SplitterFilter::run(PointViewPtr inView)
80 {
81  PointViewSet viewSet;
82  if (!inView->size())
83  return viewSet;
84 
85  auto addPoint = [this, &inView](PointId idx, int xpos, int ypos) {
86  Coord loc(xpos, ypos);
87  PointViewPtr& outView = m_viewMap[loc];
88  if (!outView)
89  outView = inView->makeNew();
90  outView->appendPoint(*inView.get(), idx);
91  };
92 
93  // Use the location of the first point as the origin, unless specified.
94  // (!= test == isnan(), which doesn't exist on windows)
95  if (m_xOrigin != m_xOrigin)
96  m_xOrigin = inView->getFieldAs<double>(Dimension::Id::X, 0);
97  if (m_yOrigin != m_yOrigin)
98  m_yOrigin = inView->getFieldAs<double>(Dimension::Id::Y, 0);
99  // Overlay a grid of squares on the points (m_length sides). Each square
100  // corresponds to a new point buffer. Place the points falling in the
101  // each square in the corresponding point buffer.
102  for (PointId idx = 0; idx < inView->size(); idx++)
103  {
104  double x = inView->getFieldAs<double>(Dimension::Id::X, idx);
105  double dx = x - m_xOrigin;
106  int xpos = dx / m_length;
107  if (dx < 0)
108  xpos--;
109 
110  double y = inView->getFieldAs<double>(Dimension::Id::Y, idx);
111  double dy = y - m_yOrigin;
112  int ypos = dy / m_length;
113  if (dy < 0)
114  ypos--;
115 
116  addPoint(idx, xpos, ypos);
117 
118  if (m_buffer > 0.0) {
119  if (squareContains(xpos - 1, ypos, x, y)) {
120  addPoint(idx, xpos - 1, ypos);
121  } else if (squareContains(xpos + 1, ypos, x, y)) {
122  addPoint(idx, xpos + 1, ypos);
123  }
124  if (squareContains(xpos, ypos - 1, x, y)) {
125  addPoint(idx, xpos, ypos - 1);
126  } else if (squareContains(xpos, ypos + 1, x, y)) {
127  addPoint(idx, xpos, ypos + 1);
128  }
129  }
130  }
131 
132  // Pull the buffers out of the map and stick them in the standard
133  // output set.
134  for (auto bi = m_viewMap.begin(); bi != m_viewMap.end(); ++bi)
135  viewSet.insert(bi->second);
136  return viewSet;
137 }

139 bool SplitterFilter::squareContains(int xpos, int ypos, double x, double y) const {
140  double minx = m_xOrigin + xpos * m_length - m_buffer;
141  double maxx = minx + m_length + 2 * m_buffer;
142  double miny = m_yOrigin + ypos * m_length - m_buffer;
143  double maxy = miny + m_length + 2 * m_buffer;
144  return minx < x && x < maxx && miny < y && y < maxy;
145 }
