32 static PyObject* Stdout_write(PyObject* self, PyObject* args)
33 {
34  std::size_t written(0);
35  Stdout* selfimpl = reinterpret_cast<Stdout*>(self);
36  if (selfimpl->write)
37  {
38  char* data;
39  if (!PyArg_ParseTuple(args, "s", &data))
40  return 0;
41 
42  std::string str(data);
43  selfimpl->write(str);
44  written = str.size();
45  }
46  return PyLong_FromSize_t(written);
47 }

194 void Redirector::set_stdout(stdout_write_type write, stdout_flush_type flush)
195 {
196  if (!m_stdout)
197  {
198  m_stdout_saved =
199  PySys_GetObject(const_cast<char*>("stdout")); // borrowed
200  m_stdout = StdoutType.tp_new(&StdoutType, 0, 0);
201  }
202 
203  Stdout* impl = reinterpret_cast<Stdout*>(m_stdout);
204  impl->write = write;
205  impl->flush = flush;
206  PySys_SetObject(const_cast<char*>("stdout"), m_stdout);
207 }
