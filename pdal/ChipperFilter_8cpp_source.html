39 /**
40 The objective is to split the region into non-overlapping blocks, each
41 containing approximately the same number of points, as specified by the
42 user. We'd also like the blocks closer to square than not.
43 
44 First, the points are read into arrays - one for the x direction, and one for
45 the y direction. The arrays are sorted and are initialized with indices into
46 the other array of the location of the other coordinate of the same point.
47 
48 Partitions are created that place the maximum number of points in a
49 block, subject to the user-defined threshold, using a cumulate and round
50 procedure.
51 
52 The distance of the point-space is checked in each direction and the
53 wider dimension is chosen for splitting at an appropriate partition point.
54 The points in the narrower direction are copied to locations in the spare
55 array at one side or the other of the chosen partition, and that portion
56 of the spare array then becomes the active array for the narrow direction.
57 This avoids resorting of the arrays, which are already sorted.
58 
59 This procedure is then recursively applied to the created blocks until
60 they contains only one or two partitions. In the case of one or two
61 partitions we are done, and we simply store away the contents of the
62 blocks.
63 **/

100 void ChipperFilter::load(PointView& view, ChipRefList& xvec, ChipRefList& yvec,
101  ChipRefList& spare)
102 {
103  point_count_t idx;
104  std::vector<ChipPtRef>::iterator it;
105 
106  xvec.reserve(view.size());
107  yvec.reserve(view.size());
108  spare.resize(view.size());
109 
110  for (PointId i = 0; i < view.size(); ++i)
111  {
112  ChipPtRef xref;
113 
114  xref.m_pos = view.getFieldAs<double>(Dimension::Id::X, i);
115  xref.m_ptindex = i;
116  xvec.push_back(xref);
117 
118  ChipPtRef yref;
119 
120  yref.m_pos = view.getFieldAs<double>(Dimension::Id::Y, i);
121  yref.m_ptindex = i;
122  yvec.push_back(yref);
123  }
124 
125  // Sort xvec and assign other index in yvec to sorted indices in xvec.
126  std::stable_sort(xvec.begin(), xvec.end());
127  for (size_t i = 0; i < xvec.size(); ++i)
128  {
129  idx = xvec[i].m_ptindex;
130  yvec[idx].m_oindex = i;
131  }
132 
133  // Sort yvec.
134  std::stable_sort(yvec.begin(), yvec.end());
135 
136  // Iterate through the yvector, setting the xvector appropriately.
137  for (size_t i = 0; i < yvec.size(); ++i)
138  xvec[yvec[i].m_oindex].m_oindex = i;
139 }

158 // Build a list of partitions. The partition is the size of each block in
159 // the x and y directions in number of points.
160 void ChipperFilter::partition(point_count_t size)
161 {
162  size_t num_partitions;
163 
164  num_partitions = size / m_threshold;
165  if (size % m_threshold)
166  num_partitions++;
167 
168  // This is a standard statistics cumulate and round. It distributes
169  // the points into partitions such the "extra" points are reasonably
170  // distributed among the partitions.
171  double total(0.0);
172  double partition_size = static_cast<double>(size) / num_partitions;
173  m_partitions.push_back(0);
174  for (size_t i = 0; i < num_partitions; ++i)
175  {
176  total += partition_size;
177  size_t itotal = lround(total);
178  m_partitions.push_back(itotal);
179  }
180 }

183 void ChipperFilter::decideSplit(ChipRefList& v1, ChipRefList& v2, ChipRefList& spare,
184  PointId pleft, PointId pright)
185 {
186  double v1range;
187  double v2range;
188  uint32_t left = m_partitions[pleft];
189  uint32_t right = m_partitions[pright] - 1;
190 
191  // Decide the wider direction of the block, and split in that direction
192  // to maintain squareness.
193  v1range = v1[right].m_pos - v1[left].m_pos;
194  v2range = v2[right].m_pos - v2[left].m_pos;
195  if (v1range > v2range)
196  split(v1, v2, spare, pleft, pright);
197  else
198  split(v2, v1, spare, pleft, pright);
199 }

201 void ChipperFilter::split(ChipRefList& wide, ChipRefList& narrow, ChipRefList& spare,
202  PointId pleft, PointId pright)
203 {
204  PointId lstart;
205  PointId rstart;
206  PointId pcenter;
207  PointId left;
208  PointId right;
209  PointId center;
210 
211  left = m_partitions[pleft];
212  right = m_partitions[pright] - 1;
213 
214  // There are two cases in which we are done.
215  // 1) We have a distance of two between left and right.
216  // 2) We have a distance of three between left and right.
217 
218  if (pright - pleft == 1)
219  emit(wide, left, right);
220  else if (pright - pleft == 2) {
221  center = m_partitions[pright - 1];
222  emit(wide,
223  left,
224  center - 1);
225  emit(wide,
226  center,
227  right);
228  } else {
229  pcenter = (pleft + pright) / 2;
230  center = m_partitions[pcenter];
231 
232  // We are splitting in the wide direction - split elements in the
233  // narrow array by copying them to the spare array in the correct
234  // partition. The spare array then becomes the active narrow array
235  // for the [left,right] partition.
236  lstart = left;
237  rstart = center;
238  for (PointId i = left; i <= right; ++i)
239  {
240  if (narrow[i].m_oindex < center)
241  {
242  spare[lstart] = narrow[i];
243  wide[narrow[i].m_oindex].m_oindex = lstart;
244  lstart++;
245  }
246  else
247  {
248  spare[rstart] = narrow[i];
249  wide[narrow[i].m_oindex].m_oindex = rstart;
250  rstart++;
251  }
252  }
253 
254  decideSplit(wide, spare, narrow, pleft, pcenter);
255  decideSplit(wide, spare, narrow, pcenter, pright);
256  }
257 }
