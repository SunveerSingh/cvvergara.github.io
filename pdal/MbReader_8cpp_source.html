129 bool MbReader::loadData()
130 {
131  int verbose = 0;
132  int kind;
133  int pings;
134  int pingTime[7];
135  double pingTimeT;
136  double lon;
137  double lat;
138  double speed;
139  double heading;
140  double distance;
141  double altitude;
142  double sonarDepth;
143  int numBath;
144  int numAmp;
145  int numSs;
146  char comment[MB_COMMENT_MAXLINE];
147  int error;
148 
149  while (true)
150  {
151  int status = mb_read(verbose, m_ctx, &kind, &pings, pingTime,
152  &pingTimeT, &lon, &lat, &speed, &heading, &distance, &altitude,
153  &sonarDepth, &numBath, &numAmp, &numSs, m_bathflag, m_bath,
154  m_amp, m_bathlon, m_bathlat, m_ss, m_sslon, m_sslat, comment,
155  &error);
156 
157  if (status == 0)
158  {
159  if (error > 0 && error != MB_ERROR_EOF)
160  throwError("Error reading data: " + MbError::text(error));
161  return false;
162  }
163 
164  if (kind == 1)
165  {
166  for (size_t i = 0; i < (size_t)numBath; ++i)
167  {
168  if (m_bathflag[i] & 1)
169  continue;
170  m_bathQueue.emplace(m_bathlon[i], m_bathlat[i], -m_bath[i],
171  m_amp[i]);
172  }
173  if (numBath != numAmp)
174  log()->get(LogLevel::Warning) << getName() << ": Number of "
175  "bathymetry values doesn't match number of amplitude "
176  "values." << std::endl;
177  if (m_bathQueue.size())
178  break;
179  }
180  }
181  return true;
182 }

185 bool MbReader::processOne(PointRef& point)
186 {
187  if (m_bathQueue.empty())
188  if (!loadData())
189  return false;
190 
191  BathData& bd = m_bathQueue.front();
192 
193  point.setField(Dimension::Id::X, bd.m_bathlon);
194  point.setField(Dimension::Id::Y, bd.m_bathlat);
195  point.setField(Dimension::Id::Z, bd.m_bath);
196  point.setField(Dimension::Id::Amplitude, bd.m_amp);
197  m_bathQueue.pop();
198  return true;
199 }
