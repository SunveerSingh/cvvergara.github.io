189  Options filterOps;
190  filterOps.add("dimensions", " , X, Z ");
191  StatsFilter filter;
192  filter.setInput(*reader);
193  filter.setOptions(filterOps);
194 
195  PointTable table;
196  filter.prepare(table);
197  filter.execute(table);
198  MetadataNode m = filter.getMetadata();
199  std::vector<MetadataNode> children = m.children("statistic");
200 
201  auto findNode = [](MetadataNode m,
202  const std::string name, const std::string val)
203  {
204  auto findNameVal = [name, val](MetadataNode m)
205  { return (m.name() == name && m.value() == val); };
206 
207  return m.find(findNameVal);
208  };
209 
210  for (auto mi = children.begin(); mi != children.end(); ++mi)
211  {
212  if (findNode(*mi, "name", "X").valid())
213  {
214  EXPECT_DOUBLE_EQ(mi->findChild("average").value<double>(), 1.0);
215  EXPECT_DOUBLE_EQ(mi->findChild("minimum").value<double>(), 1.0);
216  EXPECT_DOUBLE_EQ(mi->findChild("maximum").value<double>(), 1.0);
217  EXPECT_DOUBLE_EQ(mi->findChild("count").value<double>(), 1000.0);
218  }
219  if (findNode(*mi, "name", "Z").valid())
220  {
221  EXPECT_DOUBLE_EQ(mi->findChild("average").value<double>(), 3.0);
222  EXPECT_DOUBLE_EQ(mi->findChild("minimum").value<double>(), 3.0);
223  EXPECT_DOUBLE_EQ(mi->findChild("maximum").value<double>(), 3.0);
224  EXPECT_DOUBLE_EQ(mi->findChild("count").value<double>(), 1000.0);
225  }
226  }
