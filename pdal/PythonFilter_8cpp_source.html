87 PointViewSet PythonFilter::run(PointViewPtr view)
88 {
89  log()->get(LogLevel::Debug5) << "filters.python " << *m_script <<
90  " processing " << view->size() << " points." << std::endl;
91  m_pythonMethod->resetArguments();
92  m_pythonMethod->begin(*view, m_totalMetadata);
93 
94  if (!m_pdalargs.empty())
95  {
96  std::ostringstream args;
97  args << m_pdalargs;
98  m_pythonMethod->setKWargs(args.str());
99  }
100  m_pythonMethod->execute();
101 
102  PointViewSet viewSet;
103 
104  if (m_pythonMethod->hasOutputVariable("Mask"))
105  {
106  PointViewPtr outview = view->makeNew();
107 
108  void *pydata =
109  m_pythonMethod->extractResult("Mask", Dimension::Type::Unsigned8);
110  char *ok = (char *)pydata;
111  for (PointId idx = 0; idx < view->size(); ++idx)
112  if (*ok++)
113  outview->appendPoint(*view, idx);
114 
115  viewSet.insert(outview);
116  }
117  else
118  {
119  m_pythonMethod->end(*view, getMetadata());
120  viewSet.insert(view);
121  }
122 
123  return viewSet;
124 
125 }
