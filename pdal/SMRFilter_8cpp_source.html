126 PointViewSet SMRFilter::run(PointViewPtr view)
127 {
128  PointViewSet viewSet;
129  if (!view->size())
130  return viewSet;
131 
132  // Segment input view into ignored/kept views.
133  PointViewPtr ignoredView = view->makeNew();
134  PointViewPtr keptView = view->makeNew();
135  if (m_ignored.m_id == Dimension::Id::Unknown)
136  keptView->append(*view);
137  else
138  Segmentation::ignoreDimRange(m_ignored, view, keptView, ignoredView);
139 
140  // Segment kept view into last/other-than-last return views.
141  PointViewPtr lastView = keptView->makeNew();
142  PointViewPtr nonlastView = keptView->makeNew();
143  if (m_lastOnly)
144  Segmentation::segmentLastReturns(keptView, lastView, nonlastView);
145  else
146  lastView->append(*keptView);
147 
148  for (PointId i = 0; i < nonlastView->size(); ++i)
149  nonlastView->setField(Dimension::Id::Classification, i, 1);
150 
151  m_srs = lastView->spatialReference();
152 
153  lastView->calculateBounds(m_bounds);
154  m_cols = ((m_bounds.maxx - m_bounds.minx) / m_cell) + 1;
155  m_rows = ((m_bounds.maxy - m_bounds.miny) / m_cell) + 1;
156 
157  // Create raster of minimum Z values per element.
158  std::vector<double> ZImin = createZImin(lastView);
159 
160  // Create raster mask of pixels containing low outlier points.
161  std::vector<int> Low = createLowMask(ZImin);
162 
163  // Create raster mask of net cuts. Net cutting is used to when a scene
164  // contains large buildings in highly differentiated terrain.
165  std::vector<int> isNetCell = createNetMask();
166 
167  // Apply net cutting to minimum Z raster.
168  std::vector<double> ZInet = createZInet(ZImin, isNetCell);
169 
170  // Create raster mask of pixels containing object points. Note that we use
171  // ZInet, the result of net cutting, to identify object pixels.
172  std::vector<int> Obj = createObjMask(ZInet);
173 
174  // Create raster representing the provisional DEM. Note that we use the
175  // original ZImin (not ZInet), however the net cut mask will still force
176  // interpolation at these pixels.
177  std::vector<double> ZIpro =
178  createZIpro(lastView, ZImin, Low, isNetCell, Obj);
179 
180  // Classify ground returns by comparing elevation values to the provisional
181  // DEM.
182  classifyGround(lastView, ZIpro);
183 
184  PointViewPtr outView = view->makeNew();
185  outView->append(*ignoredView);
186  outView->append(*nonlastView);
187  outView->append(*lastView);
188  viewSet.insert(outView);
189 
190  return viewSet;
191 }

193 void SMRFilter::classifyGround(PointViewPtr view, std::vector<double>& ZIpro)
194 {
195  // "While many authors use a single value for the elevation threshold, we
196  // suggest that a second parameter be used to increase the threshold on
197  // steep slopes, transforming the threshold to a slope-dependent value. The
198  // total permissible distance is then equal to a fixed elevation threshold
199  // plus the scaling value multiplied by the slope of the DEM at each LIDAR
200  // point. The rationale behind this approach is that small horizontal and
201  // vertical displacements yield larger errors on steep slopes, and as a
202  // result the BE/OBJ threshold distance should be more permissive at these
203  // points."
204  MatrixXd gsurfs(m_rows, m_cols);
205  MatrixXd thresh(m_rows, m_cols);
206  {
207  MatrixXd ZIproM = Map<MatrixXd>(ZIpro.data(), m_rows, m_cols);
208  MatrixXd scaled = ZIproM / m_cell;
209 
210  MatrixXd gx = gradX(scaled);
211  MatrixXd gy = gradY(scaled);
212  gsurfs = (gx.cwiseProduct(gx) + gy.cwiseProduct(gy)).cwiseSqrt();
213  std::vector<double> gsurfsV(gsurfs.data(),
214  gsurfs.data() + gsurfs.size());
215  std::vector<double> gsurfs_fillV = knnfill(view, gsurfsV);
216  gsurfs = Map<MatrixXd>(gsurfs_fillV.data(), m_rows, m_cols);
217  thresh = (m_threshold + m_scalar * gsurfs.array()).matrix();
218 
219  if (!m_dir.empty())
220  {
221  std::string fname = FileUtils::toAbsolutePath("gx.tif", m_dir);
222  writeMatrix(gx, fname, "GTiff", m_cell, m_bounds, m_srs);
223 
224  fname = FileUtils::toAbsolutePath("gy.tif", m_dir);
225  writeMatrix(gy, fname, "GTiff", m_cell, m_bounds, m_srs);
226 
227  fname = FileUtils::toAbsolutePath("gsurfs.tif", m_dir);
228  writeMatrix(gsurfs, fname, "GTiff", m_cell, m_bounds, m_srs);
229 
230  fname = FileUtils::toAbsolutePath("gsurfs_fill.tif", m_dir);
231  MatrixXd gsurfs_fill =
232  Map<MatrixXd>(gsurfs_fillV.data(), m_rows, m_cols);
233  writeMatrix(gsurfs_fill, fname, "GTiff", m_cell, m_bounds, m_srs);
234 
235  fname = FileUtils::toAbsolutePath("thresh.tif", m_dir);
236  writeMatrix(thresh, fname, "GTiff", m_cell, m_bounds, m_srs);
237  }
238  }
239 
240  for (PointId i = 0; i < view->size(); ++i)
241  {
242  double x = view->getFieldAs<double>(Id::X, i);
243  double y = view->getFieldAs<double>(Id::Y, i);
244  double z = view->getFieldAs<double>(Id::Z, i);
245 
246  size_t c = static_cast<size_t>(std::floor(x - m_bounds.minx) / m_cell);
247  size_t r = static_cast<size_t>(std::floor(y - m_bounds.miny) / m_cell);
248 
249  // TODO(chambbj): We don't quite do this by the book and yet it seems to
250  // work reasonably well:
251  // "The calculation requires that both elevation and slope are
252  // interpolated from the provisional DEM. There are any number of
253  // interpolation techniques that might be used, and even nearest
254  // neighbor approaches work quite well, so long as the cell size of the
255  // DEM nearly corresponds to the resolution of the LIDAR data. Based on
256  // these results, we find that a splined cubic interpolation provides
257  // the best results."
258  if (std::isnan(ZIpro[c * m_rows + r]))
259  continue;
260 
261  if (std::isnan(gsurfs(r, c)))
262  continue;
263 
264  // "The final step of the algorithm is the identification of
265  // ground/object LIDAR points. This is accomplished by measuring the
266  // vertical distance between each LIDAR point and the provisional
267  // DEM, and applying a threshold calculation."
268  if (std::fabs(ZIpro[c * m_rows + r] - z) > thresh(r, c))
269  view->setField(Id::Classification, i, 1);
270  else
271  view->setField(Id::Classification, i, 2);
272  }
273 }

275 std::vector<int> SMRFilter::createLowMask(std::vector<double> const& ZImin)
276 {
277  // "[The] minimum surface is checked for low outliers by inverting the point
278  // cloud in the z-axis and applying the filter with parameters (slope =
279  // 500%, maxWindowSize = 1). The resulting mask is used to flag low outlier
280  // cells as OBJ before the inpainting of the provisional DEM."
281 
282  // Need to add a step to negate ZImin
283  std::vector<double> negZImin;
284  std::transform(ZImin.begin(), ZImin.end(), std::back_inserter(negZImin),
285  [](double v) { return -v; });
286  std::vector<int> LowV = progressiveFilter(negZImin, 5.0, 1.0);
287 
288  if (!m_dir.empty())
289  {
290  std::string fname = FileUtils::toAbsolutePath("zilow.tif", m_dir);
291  MatrixXi Low = Map<MatrixXi>(LowV.data(), m_rows, m_cols);
292  writeMatrix(Low.cast<double>(), fname, "GTiff", m_cell, m_bounds,
293  m_srs);
294  }
295 
296  return LowV;
297 }

299 std::vector<int> SMRFilter::createNetMask()
300 {
301  // "To accommodate the removal of [very large buildings on highly
302  // differentiated terrain], we implemented a feature in the published SMRF
303  // algorithm which is helpful in removing such features. We accomplish this
304  // by introducing into the initial minimum surface a "net" of minimum values
305  // at a spacing equal to the maximum window diameter, where these minimum
306  // values are found by applying a morphological open operation with a disk
307  // shaped structuring element of radius (2*wkmax)."
308  std::vector<int> isNetCell(m_rows * m_cols, 0);
309  if (m_cut > 0.0)
310  {
311  int v = std::ceil(m_cut / m_cell);
312 
313  for (auto c = 0; c < m_cols; c += v)
314  {
315  for (auto r = 0; r < m_rows; ++r)
316  {
317  isNetCell[c * m_rows + r] = 1;
318  }
319  }
320  for (auto c = 0; c < m_cols; ++c)
321  {
322  for (auto r = 0; r < m_rows; r += v)
323  {
324  isNetCell[c * m_rows + r] = 1;
325  }
326  }
327  }
328 
329  return isNetCell;
330 }

350 std::vector<double> SMRFilter::createZImin(PointViewPtr view)
351 {
352  using namespace Dimension;
353 
354  // "As with many other ground filtering algorithms, the first step is
355  // generation of ZImin from the cell size parameter and the extent of the
356  // data."
357  std::vector<double> ZIminV(m_rows * m_cols,
358  std::numeric_limits<double>::quiet_NaN());
359 
360  for (PointId i = 0; i < view->size(); ++i)
361  {
362  double x = view->getFieldAs<double>(Id::X, i);
363  double y = view->getFieldAs<double>(Id::Y, i);
364  double z = view->getFieldAs<double>(Id::Z, i);
365 
366  int c = static_cast<int>(floor(x - m_bounds.minx) / m_cell);
367  int r = static_cast<int>(floor(y - m_bounds.miny) / m_cell);
368 
369  if (z < ZIminV[c * m_rows + r] || std::isnan(ZIminV[c * m_rows + r]))
370  ZIminV[c * m_rows + r] = z;
371  }
372 
373  // "...some grid points of ZImin will go unfilled. To fill these values, we
374  // rely on computationally inexpensive image inpainting techniques. Image
375  // inpainting involves the replacement of the empty cells in an image (or
376  // matrix) with values calculated from other nearby values."
377  std::vector<double> ZImin_fillV = knnfill(view, ZIminV);
378 
379  if (!m_dir.empty())
380  {
381  std::string fname = FileUtils::toAbsolutePath("zimin.tif", m_dir);
382  MatrixXd ZImin = Map<MatrixXd>(ZIminV.data(), m_rows, m_cols);
383  writeMatrix(ZImin, fname, "GTiff", m_cell, m_bounds, m_srs);
384 
385  fname = FileUtils::toAbsolutePath("zimin_fill.tif", m_dir);
386  MatrixXd ZImin_fill = Map<MatrixXd>(ZImin_fillV.data(), m_rows, m_cols);
387  writeMatrix(ZImin_fill, fname, "GTiff", m_cell, m_bounds, m_srs);
388  }
389 
390  return ZImin_fillV;
391 }

393 std::vector<double> SMRFilter::createZInet(std::vector<double> const& ZImin,
394  std::vector<int> const& isNetCell)
395 {
396  // "To accommodate the removal of [very large buildings on highly
397  // differentiated terrain], we implemented a feature in the published SMRF
398  // algorithm which is helpful in removing such features. We accomplish this
399  // by introducing into the initial minimum surface a "net" of minimum values
400  // at a spacing equal to the maximum window diameter, where these minimum
401  // values are found by applying a morphological open operation with a disk
402  // shaped structuring element of radius (2*wkmax)."
403  std::vector<double> ZInetV = ZImin;
404  if (m_cut > 0.0)
405  {
406  int v = std::ceil(m_cut / m_cell);
407  std::vector<double> bigErode =
408  erodeDiamond(ZImin, m_rows, m_cols, 2 * v);
409  std::vector<double> bigOpen =
410  dilateDiamond(bigErode, m_rows, m_cols, 2 * v);
411  for (auto c = 0; c < m_cols; ++c)
412  {
413  for (auto r = 0; r < m_rows; ++r)
414  {
415  if (isNetCell[c * m_rows + r] == 1)
416  {
417  ZInetV[c * m_rows + r] = bigOpen[c * m_rows + r];
418  }
419  }
420  }
421  }
422 
423  if (!m_dir.empty())
424  {
425  std::string fname = FileUtils::toAbsolutePath("zinet.tif", m_dir);
426  MatrixXd ZInet = Map<MatrixXd>(ZInetV.data(), m_rows, m_cols);
427  writeMatrix(ZInet, fname, "GTiff", m_cell, m_bounds, m_srs);
428  }
429 
430  return ZInetV;
431 }

433 std::vector<double> SMRFilter::createZIpro(PointViewPtr view,
434  std::vector<double> const& ZImin,
435  std::vector<int> const& Low,
436  std::vector<int> const& isNetCell,
437  std::vector<int> const& Obj)
438 {

468 // Fill voids with the average of eight nearest neighbors.
469 std::vector<double> SMRFilter::knnfill(PointViewPtr view,
470  std::vector<double> const& cz)
471 {
472  // Create a temporary PointView that encodes our raster values so that we
473  // can construct a 2D KDIndex and perform nearest neighbor searches.
474  PointViewPtr temp = view->makeNew();
475  PointId i(0);
476  for (int c = 0; c < m_cols; ++c)
477  {
478  for (int r = 0; r < m_rows; ++r)
479  {
480  if (std::isnan(cz[c * m_rows + r]))
481  continue;
482 
483  temp->setField(Id::X, i, m_bounds.minx + (c + 0.5) * m_cell);
484  temp->setField(Id::Y, i, m_bounds.miny + (r + 0.5) * m_cell);
485  temp->setField(Id::Z, i, cz[c * m_rows + r]);
486  i++;
487  }
488  }
489 
490  KD2Index kdi(*temp);
491  kdi.build();
492 
493  // Where the raster has voids (i.e., NaN), we search for that cell's eight
494  // nearest neighbors, and fill the void with the average value of the
495  // neighbors.
496  std::vector<double> out = cz;
497  for (int c = 0; c < m_cols; ++c)
498  {
499  for (int r = 0; r < m_rows; ++r)
500  {
501  if (!std::isnan(out[c * m_rows + r]))
502  continue;
503 
504  double x = m_bounds.minx + (c + 0.5) * m_cell;
505  double y = m_bounds.miny + (r + 0.5) * m_cell;
506  int k = 8;
507  std::vector<PointId> neighbors(k);
508  std::vector<double> sqr_dists(k);
509  kdi.knnSearch(x, y, k, &neighbors, &sqr_dists);
510 
511  double M1(0.0);
512  size_t j(0);
513  for (auto const& n : neighbors)
514  {
515  j++;
516  double delta = temp->getFieldAs<double>(Id::Z, n) - M1;
517  M1 += (delta / j);
518  }
519 
520  out[c * m_rows + r] = M1;
521  }
522  }
523 
524  return out;
525 };

527 // Iteratively open the estimated surface. progressiveFilter can be used to
528 // identify both low points and object (i.e., non-ground) points, depending on
529 // the inputs.
530 std::vector<int> SMRFilter::progressiveFilter(std::vector<double> const& ZImin,
531  double slope, double max_window)
532 {
533  // "The maximum window radius is supplied as a distance metric (e.g., 21 m),
534  // but is internally converted to a pixel equivalent by dividing it by the
535  // cell size and rounding the result toward positive infinity (i.e., taking
536  // the ceiling value)."
537  int max_radius = std::ceil(max_window / m_cell);
538  std::vector<double> prevSurface = ZImin;
539  std::vector<double> prevErosion = ZImin;
540 
541  // "...the radius of the element at each step [is] increased by one pixel
542  // from a starting value of one pixel to the pixel equivalent of the maximum
543  // value."
544  std::vector<int> Obj(m_rows * m_cols, 0);
545  for (int radius = 1; radius <= max_radius; ++radius)
546  {
547  // "On the first iteration, the minimum surface (ZImin) is opened using
548  // a disk-shaped structuring element with a radius of one pixel."
549  std::vector<double> curErosion =
550  erodeDiamond(prevErosion, m_rows, m_cols, 1);
551  std::vector<double> curOpening =
552  dilateDiamond(curErosion, m_rows, m_cols, radius);
553  prevErosion = curErosion;
554 
555  // "An elevation threshold is then calculated, where the value is equal
556  // to the supplied slope tolerance parameter multiplied by the product
557  // of the window radius and the cell size."
558  double threshold = slope * m_cell * radius;
559 
560  // "This elevation threshold is applied to the difference of the minimum
561  // and the opened surfaces."
562 
563  // Need to provide means of diffing two vectors.
564  std::vector<double> diff;
565  std::transform(prevSurface.begin(), prevSurface.end(),
566  curOpening.begin(), std::back_inserter(diff),
567  [&](double l, double r) { return std::fabs(l - r); });
568 
569  // "Any grid cell with a difference value exceeding the calculated
570  // elevation threshold for the iteration is then flagged as an OBJ
571  // cell."
572  std::vector<int> foo;
573  std::transform(diff.begin(), diff.end(), std::back_inserter(foo),
574  [threshold](double x) {
575  return (x > threshold) ? int(1) : int(0);
576  });
577  std::transform(Obj.begin(), Obj.end(), foo.begin(), Obj.begin(),
