114 void OGRWriter::readyFile(const std::string& filename,
115  const SpatialReference& srs)
116 {
117  m_outputFilename = filename;
118  m_ds = m_driver->CreateDataSource(filename.data(), nullptr);
119  if (!m_ds)
120  throwError("Unable to open OGR datasource '" + filename + "'.\n");
121  OGRSpatialReference ogrSrs(srs.getWKT().data());
122  m_layer = m_ds->CreateLayer("points", &ogrSrs, m_geomType, nullptr);
123  if (!m_layer)
124  throwError("Can't create OGR layer for points.\n");
125  m_feature = OGRFeature::CreateFeature(m_layer->GetLayerDefn());
126 }

140 bool OGRWriter::processOne(PointRef& point)
141 {
142  double x = point.getFieldAs<double>(Dimension::Id::X);
143  double y = point.getFieldAs<double>(Dimension::Id::Y);
144  double z = point.getFieldAs<double>(Dimension::Id::Z);
145  double m = point.getFieldAs<double>(m_measureDim);
146 
147  OGRPoint pt(x, y, z);
148  m_curCount++;
149 
150  if (m_multiCount > 1)
151  m_multiPoint.addGeometry(&pt);
152  if (m_curCount == m_multiCount)
153  {
154  if (m_multiCount > 1)
155  {
156  m_feature->SetGeometry(&m_multiPoint);
157  m_multiPoint.empty();
158  }
159  else
160  m_feature->SetGeometry(&pt);
161  if (m_layer->CreateFeature(m_feature))
162  throwError("Couldn't create feature.");
163  m_curCount = 0;
164  }
165  return true;
166 }
