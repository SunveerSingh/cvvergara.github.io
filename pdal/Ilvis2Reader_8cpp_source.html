154  point.setField(pdal::Dimension::Id::LatitudeCentroid,

156  point.setField(pdal::Dimension::Id::ElevationCentroid,
157  convert<double>(s, "ELEVATION_CENTROID", 5));
158  point.setField(pdal::Dimension::Id::LongitudeLow,
159  Utils::normalizeLongitude(convert<double>(s, "LONGITUDE_LOW", 6)));
160  point.setField(pdal::Dimension::Id::LatitudeLow,
161  convert<double>(s, "LATITUDE_LOW", 7));
162  point.setField(pdal::Dimension::Id::ElevationLow,
163  convert<double>(s, "ELEVATION_LOW", 8));
164  point.setField(pdal::Dimension::Id::LongitudeHigh,
165  Utils::normalizeLongitude(convert<double>(s, "LONGITUDE_HIGH", 9)));
166  point.setField(pdal::Dimension::Id::LatitudeHigh,
167  convert<double>(s, "LATITUDE_HIGH", 10));
168  point.setField(pdal::Dimension::Id::ElevationHigh,
169  convert<double>(s, "ELEVATION_HIGH", 11));
170 
171  double x, y, z;
172  pdal::Dimension::Id xd, yd, zd;
173 
174  xd = m_layout->findDim("LONGITUDE_" + pointMap);
175  yd = m_layout->findDim("LATITUDE_" + pointMap);
176  zd = m_layout->findDim("ELEVATION_" + pointMap);
177 
178  x = point.getFieldAs<double>(xd);
179  y = point.getFieldAs<double>(yd);
180  z = point.getFieldAs<double>(zd);
181 
182  point.setField(pdal::Dimension::Id::X, x);
183  point.setField(pdal::Dimension::Id::Y, y);
184  point.setField(pdal::Dimension::Id::Z, z);
185 }

217 bool Ilvis2Reader::processOne(PointRef& point)
218 {
219  std::string line;
220 
221 // Format:
222 // LVIS_LFID SHOTNUMBER TIME LONGITUDE_CENTROID LATITUDE_CENTROID ELEVATION_CENTROID LONGITUDE_LOW LATITUDE_LOW ELEVATION_LOW LONGITUDE_HIGH LATITUDE_HIGH ELEVATION_HIGH
223 
224  try
225  {
226  // This handles the second time through for this data line when we have
227  // an "ALL" mapping and the high and low elevations are different.
228  if (m_resample)
229  {
230  readPoint(point, m_fields, "HIGH");
231  m_resample = false;
232  return true;
233  }
234 
235  if (!std::getline(m_stream, line))
236  return false;
237  m_fields = Utils::split2(line, ' ');
238  if (m_fields.size() != 12)
239  throwError("Invalid format for line " +
240  Utils::toString(m_lineNum) + ". Expected 12 fields, got " +
241  Utils::toString(m_fields.size()) + ".");
242 
243  double low_elev = convert<double>(m_fields, "ELEVATION_LOW", 8);
244  double high_elev = convert<double>(m_fields, "ELEVATION_HIGH", 11);
245 
246  // write LOW point if specified, or for ALL
247  if (m_mapping == IlvisMapping::LOW || m_mapping == IlvisMapping::ALL)
248  {
249  readPoint(point, m_fields, "LOW");
250  // If we have ALL mapping and the high elevation is different
251  // from that of the low elevation, we'll a second point with the
252  // high elevation.
253  if (m_mapping == IlvisMapping::ALL && (low_elev != high_elev))
254  m_resample = true;
255  }
256  else if (m_mapping == IlvisMapping::HIGH)
257  readPoint(point, m_fields, "HIGH");
258  }
259  catch (const error& err)
260  {
261  throwError(err.what());
262  }
263  return true;
264 }

267 point_count_t Ilvis2Reader::read(PointViewPtr view, point_count_t count)
268 {
269  PointId idx = view->size();
270  point_count_t numRead = 0;
271 
272  PointRef point = PointRef(*view, 0);
273  while (numRead < count)
274  {
275  point.setPointId(idx++);
276  if (!processOne(point))
277  break;
278  if (m_cb)
279  m_cb(*view, idx);
280  numRead++;
281  }
282 
283  return numRead;
284 }
