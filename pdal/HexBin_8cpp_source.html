96 void HexBin::done(PointTableRef table)
97 {
98  m_grid->processSample();
99 
100  try
101  {
102  m_grid->findShapes();
103  m_grid->findParentPaths();
104  }
105  catch (hexer::hexer_error& e)
106  {
107  m_metadata.add("error", e.what(), "Hexer threw an error and was unable to compute a boundary");
108  m_metadata.add("boundary", "MULTIPOLYGON EMPTY", "Empty polygon -- unable to compute boundary");
109 
110  return;
111 
112  }
113 
114 
115  std::ostringstream offsets;
116  offsets << "MULTIPOINT (";
117  for (int i = 0; i < 6; ++i)
118  {
119  hexer::Point p = m_grid->offset(i);
120  offsets << p.m_x << " " << p.m_y;
121  if (i != 5)
122  offsets << ", ";
123  }
124  offsets << ")";
125 
126  m_metadata.add("edge_length", m_edgeLength, "The edge length of the "
127  "hexagon to use in situations where you do not want to estimate "
128  "based on a sample");
129  m_metadata.add("estimated_edge", m_grid->height(),
130  "Estimated computed edge distance");
131  m_metadata.add("threshold", m_grid->denseLimit(),
132  "Minimum number of points inside a hexagon to be considered full");
133  m_metadata.add("sample_size", m_sampleSize, "Number of samples to use "
134  "when estimating hexagon edge size. Specify 0.0 or omit options "
135  "for edge_size if you want to compute one.");
136  m_metadata.add("hex_offsets", offsets.str(), "Offset of hex corners from "
137  "hex centers.");
138 
139  std::ostringstream polygon;
140  polygon.setf(std::ios_base::fixed, std::ios_base::floatfield);
141  polygon.precision(m_precision);
142  m_grid->toWKT(polygon);
143 
144  if (m_outputTesselation)
145  {
146  MetadataNode hexes = m_metadata.add("hexagons");
147  for (HexIter hi = m_grid->hexBegin(); hi != m_grid->hexEnd(); ++hi)
148  {
149  HexInfo h = *hi;
150 
151  MetadataNode hex = hexes.addList("hexagon");
152  hex.add("density", h.density());
153 
154  hex.add("gridpos", Utils::toString(h.xgrid()) + " " +
155  Utils::toString((h.ygrid())));
156  std::ostringstream oss;
157  // Using stream limits precision (default 6)
158  oss << "POINT (" << h.x() << " " << h.y() << ")";
159  hex.add("center", oss.str());
160  }
161  m_metadata.add("hex_boundary", polygon.str(),
162  "Boundary MULTIPOLYGON of domain");
163  }
164 
165 
166  /***
167  We want to make these bumps on edges go away, which means that
168  we want to elimnate both B and C. If we take a line from A -> C,
169  we need the tolerance to eliminate B. After that we're left with
170  the triangle ACD and we want to eliminate C. The perpendicular
171  distance from AD to C is the hexagon height / 2, so we set the
172  tolerance a little larger than that. This is larger than the
173  perpendicular distance needed to eliminate B in ABC, so should
174  serve for both cases.
175 
176  B ______ C
177  / \
178  A / \ D
179 
180  ***/
181  double tolerance = 1.1 * m_grid->height() / 2;
182 
183  double cull = m_cullArg->set() ? m_cullArea : (6 * tolerance * tolerance);
184 
185  SpatialReference srs(table.anySpatialReference());
186  pdal::Polygon p(polygon.str(), srs);
187  pdal::Polygon density_p(polygon.str(), srs);
188 
189  // If the SRS was geographic, use relevant
190  // UTM for area and density computation
191  if (srs.isGeographic())
192  {
193  // Compute a UTM polygon
194  BOX3D box = p.bounds();
195  int zone = SpatialReference::calculateZone(box.minx, box.miny);
196 
197  auto makezone = [] (int zone) -> std::string
198  {
199 
200  std::ostringstream z;
201 
202  // Use WGS84 UTM zones
203  z << "EPSG:327" << abs(zone);
204  return z.str();
205  };
206 
207  SpatialReference utm(makezone(zone));
208  density_p = p.transform(utm);
209  }
210 
211  if (m_doSmooth)
212  p = p.simplify(tolerance, cull);
213 
214  std::string boundary_text = p.wkt(m_precision);
215 
216  m_metadata.add("boundary", boundary_text, "Approximated MULTIPOLYGON of domain");
217  m_metadata.addWithType("boundary_json", p.json(), "json", "Approximated MULTIPOLYGON of domain");
218  double area = density_p.area();
219 
220 // double density = (double) m_grid->densePointCount() / area ;
221  double density = (double) m_count/ area ;
222  m_metadata.add("density", density, "Number of points per square unit (total area)");
223  m_metadata.add("area", area, "Area in square units of tessellated polygon");
224 
225  double moving_avg(0.0);
226  double avg_count(0.0);
227 
228  double hex_area(((3 * SQRT_3)/2.0) * (m_grid->height() * m_grid->height()));
229  int n(0);
230  point_count_t totalCount(0);
231  double totalArea(0.0);
232  for (HexIter hi = m_grid->hexBegin(); hi != m_grid->hexEnd(); ++hi)
233  {
234  HexInfo h = *hi;
235  totalCount += h.density();
236  totalArea += hex_area;
237  ++n;
238  }
239 
240  double avg_density = totalArea /(double) totalCount;
241  m_metadata.add("avg_pt_per_sq_unit", avg_density, "Number of points per square unit (tessellated area within inclusions)");
242 
243  double avg_spacing = std::sqrt(1/density);
244  m_metadata.add("avg_pt_spacing", avg_spacing, "Avg point spacing (x/y units)");
245 
246 }
