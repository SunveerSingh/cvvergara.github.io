65 void Summary::extractMetadata(MetadataNode &m)
66 {
67  uint32_t cnt = static_cast<uint32_t>(count());
68  m.add("count", cnt, "count");
69  m.add("minimum", minimum(), "minimum");
70  m.add("maximum", maximum(), "maximum");
71  m.add("average", average(), "average");
72 
73  double std = stddev();
74  if (!std::isinf(std) && !std::isnan(std))
75  m.add("stddev", std, "standard deviation");
76 
77  double k = kurtosis();
78  if (!std::isinf(k) && !std::isnan(k))
79  m.add("kurtosis", k, "kurtosis");
80 
81  double sk = skewness();
82  if (!std::isinf(sk) && !std::isnan(sk))
83  m.add("skewness", skewness(), "skewness");
84 
85  double v = variance();
86  if (!std::isinf(v) && !std::isnan(v))
87  m.add("variance", v, "variance");
88  m.add("name", m_name, "name");
89  if (m_enumerate == Enumerate)
90  {
91  for (auto& v : m_values)
92  m.addList("values", v.first);
93  }
94  else if (m_enumerate == Global)
95  {
96  computeGlobalStats();
97  m.add("median", m_median);
98  m.add("mad", m_mad);
99  }
100  else if (m_enumerate == Count)
101  {
102  for (auto& v : m_values)
103  {
104  std::string val =
105  std::to_string(v.first) + "/" + std::to_string(v.second);
106  m.addList("counts", val);
107  }
108  }

111 void Summary::computeGlobalStats()
112 {
113  auto compute_median = [](std::vector<double> vals)
114  {
115  std::nth_element(vals.begin(), vals.begin()+vals.size()/2, vals.end());
116 
117  return *(vals.begin()+vals.size()/2);
118  };
119 
120  // TODO add quantiles
121  m_median = compute_median(m_data);
122  std::transform(m_data.begin(), m_data.end(), m_data.begin(),
123  [this](double v) { return std::fabs(v - this->m_median); });
124  m_mad = compute_median(m_data);

175 void StatsFilter::prepared(PointTableRef table)
176 {
177  PointLayoutPtr layout(table.layout());
178  std::unordered_map<std::string, Summary::EnumType> dims;
179 
180  auto getWarn([this]()->std::ostream&
181  {
182  return log()->get(LogLevel::Warning);
183  });
184 
185  // Add dimensions to the list.
186  if (m_dimNames.empty())
187  {
188  for (auto id : layout->dims())
189  dims[layout->dimName(id)] = Summary::NoEnum;
190  }
191  else
192  {
193  for (auto& s : m_dimNames)
194  {
195  if (layout->findDim(s) == Dimension::Id::Unknown)
196  getWarn() << "Dimension '" << s << "' listed in --dimensions "
197  "option does not exist. Ignoring." << std::endl;
198  else
199  dims[s] = Summary::NoEnum;
200  }
201  }
202 
203  // Set the enumeration flag for those dimensions specified.
204  for (auto& s : m_enums)
205  {
206  if (dims.find(s) == dims.end())
207  getWarn() << "Dimension '" << s << "' listed in --enumerate option "
208  "does not exist. Ignoring." << std::endl;
209  else
210  dims[s] = Summary::Enumerate;
211  }
212 
213  // Set the count flag for those dimensions specified.
214  for (auto& s : m_counts)
215  {
216  if (dims.find(s) == dims.end())
217  getWarn() << "Dimension '" << s << "' listed in --count option "
218  "does not exist. Ignoring." << std::endl;
219  else
220  dims[s] = Summary::Count;
221  }
222 
223  // Set the global flag for those dimensions specified.
224  for (auto& s : m_global)
225  {
226  if (dims.find(s) == dims.end())
227  getWarn() << "Dimension '" << s << "' listed in --global option "
228  "does not exist. Ignoring." << std::endl;
229  else
230  dims[s] = Summary::Global;
231  }
232  // Create the summary objects.
233  for (auto& dv : dims)
234  m_stats.insert(std::make_pair(layout->findDim(dv.first),
235  Summary(dv.first, dv.second)));
236 }

239 void StatsFilter::extractMetadata(PointTableRef table)
240 {
241  uint32_t position(0);
242 
243  bool bNoPoints(true);
244  for (auto di = m_stats.begin(); di != m_stats.end(); ++di)
245  {
246  Summary& s = di->second;
247 
248  bNoPoints = (bool)s.count();
249 
250  MetadataNode t = m_metadata.addList("statistic");
251  t.add("position", position++);
252  s.extractMetadata(t);
253  }
254 
255  // If we have X, Y, & Z dims, output bboxes
256  auto xs = m_stats.find(Dimension::Id::X);
257  auto ys = m_stats.find(Dimension::Id::Y);
258  auto zs = m_stats.find(Dimension::Id::Z);
259  if (xs != m_stats.end() &&
260  ys != m_stats.end() &&
261  zs != m_stats.end() &&
262  bNoPoints)
263  {
264  BOX3D box(xs->second.minimum(), ys->second.minimum(),
265  zs->second.minimum(), xs->second.maximum(), ys->second.maximum(),
266  zs->second.maximum());
267  pdal::Polygon p(box);
268 
269  MetadataNode mbox = Utils::toMetadata(box);
270  MetadataNode box_metadata = m_metadata.add("bbox");
271  MetadataNode metadata = box_metadata.add("native");
272 
273  Json::Reader jsonReader;
274  Json::Value json;
275  jsonReader.parse(p.json(), json);
276 
277  MetadataNode boundary = metadata.addWithType("boundary",
278  json.toStyledString(), "json", "GeoJSON boundary");
279  MetadataNode bbox = metadata.add(mbox);
280  SpatialReference ref = table.anySpatialReference();
281  // if we don't get an SRS from the PointTableRef,
282  // we won't add another metadata node
283  if (!ref.empty())
284  {
285  p.setSpatialReference(ref);
286  SpatialReference epsg4326("EPSG:4326");
287  pdal::Polygon pdd = p.transform(epsg4326);
288  BOX3D ddbox = pdd.bounds();
289  MetadataNode epsg_4326_box = Utils::toMetadata(ddbox);
290  MetadataNode dddbox = box_metadata.add("EPSG:4326");
291  dddbox.add(epsg_4326_box);
292 
293  Json::Reader jsonReader;
294  Json::Value json;
295  jsonReader.parse(pdd.json(), json);
296 
297  MetadataNode ddboundary = dddbox.addWithType("boundary",
298  json.toStyledString(), "json", "GeoJSON boundary");
299  }
300  }
