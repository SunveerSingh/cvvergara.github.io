72 PointViewSet MADFilter::run(PointViewPtr view)
73 {
74  using namespace Dimension;
75 
76  PointViewPtr output = view->makeNew();
77 
78  auto estimate_median = [](std::vector<double> vals)
79  {
80  std::nth_element(vals.begin(), vals.begin()+vals.size()/2, vals.end());
81  return *(vals.begin()+vals.size()/2);
82  };
83 
84  std::vector<double> z(view->size());
85  for (PointId j = 0; j < view->size(); ++j)
86  z[j] = view->getFieldAs<double>(m_dimId, j);
87 
88  double median = estimate_median(z);
89  log()->get(LogLevel::Debug) << getName() <<
90  " estimated median value: " << median << std::endl;
91 
92  std::transform(z.begin(), z.end(), z.begin(),
93  [median](double v) { return std::fabs(v - median); });
94  double mad = estimate_median(z)*m_madMultiplier;
95  log()->get(LogLevel::Debug) << getName() << " mad " << mad << std::endl;
96 
97  for (PointId j = 0; j < view->size(); ++j)
98  {
99  if (z[j]/mad < m_multiplier)
100  output->appendPoint(*view, j);
101  }
102 
103  double low_fence = median - m_multiplier * mad;
104  double hi_fence = median + m_multiplier * mad;
105 
106  log()->get(LogLevel::Debug) << getName() << " cropping " << m_dimName
107  << " in the range (" << low_fence
108  << "," << hi_fence << ")" << std::endl;
109 
110  PointViewSet viewSet;
111  viewSet.insert(output);
112  return viewSet;
113 }
