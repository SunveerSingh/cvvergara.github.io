106  void insert(double value)
107  {
108  double delta, delta_n, delta_n2, term1;
109 
110  point_count_t n1(m_cnt);
111 
112  m_cnt++;
113  point_count_t n(m_cnt);
114  m_min = (std::min)(m_min, value);
115  m_max = (std::max)(m_max, value);
116  m_avg += (value - m_avg) / m_cnt;
117  if (m_enumerate != NoEnum)
118  m_values[value]++;
119  if (m_enumerate == Global)
120  {
121  if (m_data.capacity() - m_data.size() < 10000)
122  m_data.reserve(m_data.capacity() + m_cnt);
123  m_data.push_back(value);
124  }
125 
126  // stolen from http://www.johndcook.com/blog/skewness_kurtosis/
127 
128  n1 = n;
129  delta = value - M1;
130  delta_n = delta / n;
131  delta_n2 = delta_n * delta_n;
132  term1 = delta * delta_n * n1;
133  M1 += delta_n;
134  M4 += term1 * delta_n2 * (n*n - 3*n + 3) +
135  (6 * delta_n2 * M2) - (4 * delta_n * M3);
136  M3 += term1 * delta_n * (n - 2) - 3 * delta_n * M2;
137  M2 += term1;
138  }
