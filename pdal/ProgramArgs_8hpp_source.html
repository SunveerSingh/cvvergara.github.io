57 class ArgValList
58 {
59  struct ArgVal
60  {
61  std::string m_val;
62  bool m_consumed;
63 
64  ArgVal(const std::string& s) :
65  m_val(s), m_consumed(false)
66  {}
67  };
68 
69 public:
70  ArgValList(const std::vector<std::string>& slist) : m_unconsumedStart(0)
71  {
72  for (const std::string& s : slist)
73  add(s);
74  }
75 
76  void add(const std::string& s)
77  {
78  if (s.empty())
79  return;
80 
81  // Turn a short arg list into a set of short args: -afv -> -a -f -v
82  // so that each argval represents a single arg.
83  if (s.size() > 1 && s[0] == '-' && s[1] != '-')
84  for (size_t i = 1; i < s.size(); i++)
85  m_vals.push_back({std::string("-") + s[i]});
86  else
87  m_vals.push_back({s});
88  }
89 
90  void consume(size_t i)
91  {
92  m_vals[i].m_consumed = true;
93  if (i == m_unconsumedStart)
94  while (i < m_vals.size() - 1 && consumed(++i))
95  m_unconsumedStart++;
96  }
97 
98  std::vector<std::string> unconsumedArgs() const
99  {
100  std::vector<std::string> remainingVals;
101 
102  for (size_t i = firstUnconsumed(); i < size(); ++i)
103  if (!consumed(i))
104  remainingVals.push_back(m_vals[i].m_val);
105  return remainingVals;
106  }
107 
108  size_t size() const
109  { return m_vals.size(); }
110  const std::string& operator[](size_t i) const
111  { return m_vals[i].m_val; }
112  bool consumed(size_t i) const
113  { return m_vals[i].m_consumed; }
114  size_t firstUnconsumed() const
115  { return m_unconsumedStart; }
116 private:
117  std::vector<ArgVal> m_vals;
118  size_t m_unconsumedStart;
119 };

383  /**
384  Set a an argument's value from a string.
385 
386  Throws an arg_error exception if \a s can't be converted to
387  the argument's type. Values must be provided for with the
388  option name.
389  \note Not intended to be called from user code.
390 
391  \param s Value to set.
392  */
393  virtual void setValue(const std::string& s)
394  {
395  if (m_set)
396  {

410 
411  if (error.empty())

413  m_longname + "'.";
414  throw arg_val_error(error);
415  }
416  m_set = true;
417  }

419  /**
420  Reset the argument's state.
421 
422  Set the interval state of the argument and it's referenced variable
423  as if no command-line parsing had occurred.
424  \note For testing. Not intended to be called from user code.

427  {

429  m_set = false;
430  m_hidden = false;

435 
436  If no value is provided for a required positional option, an arg_error

438  \note Not intended to be called from user code.
439 
440  \param vals The list of command-line args.
441  */
442  virtual void assignPositional(ArgValList& vals)

453  return;

458  }
459 
460  /**

463  \return Whether a default was provided.

472  */
473  virtual std::string defaultVal() const
474  { return Utils::toString(m_defaultVal); }
475 
476 private:

478  T m_defaultVal;
479  bool m_defaultProvided;

1438  Long arguments with values can be specified as
1439  "--name=value" or "--name value".
1440 
1441  \param name Name of argument specified on command line.
1442  \param value Potential value assigned to argument.
1443  \return Number of strings consumed (1 for positional arguments or
1444  arguments that don't take values or 2 otherwise).
1445  */
1446  int parseLongArg(const std::string& inName, const std::string& inValue)
1447  {
1448  bool attachedValue = false;
1449 
1450  if (inName.size() == 2)
1451  throw arg_error("No argument found following '--'.");
1452 
1453  std::string name = inName.substr(2);
1454  std::string value = inValue;
1455 
1456  std::size_t pos = name.find_first_of("=");
1457  if (pos != std::string::npos)
1458  {
1459  if (pos < name.size() + 1)
1460  {
1461  value = name.substr(pos + 1);
1462  name = name.substr(0, pos);
1463  attachedValue = true;
1464  }
1465  }
1466  else if (value.size() && value[0] == '-')
1467  {
1468  // If a value starts with a '-' and isn't attached to a name,
1469  // we assume it's an option and not a value.
1470  value.clear();
1471  }
1472 
1473  Arg *arg = findLongArg(name);
1474  if (!arg)
1475  throw arg_error("Unexpected argument '" + name + "'.");
1476 
1477  if (!arg->needsValue())
1478  {
1479  if (attachedValue)
1480  {
1481  if (value != "true" && value != "false")
1482  {
1483  throw arg_error("Value '" + value +
1484  "' provided for argument '" + name +
1485  "' when none is expected.");
1486  }
1487  }
1488  else
1489  value = "invert";
1490  arg->setValue(value);
1491  return 1;
1492  }
1493 
1494  arg->setValue(value);
1495  return (attachedValue ? 1 : 2);
1496  }

1500  Short arguments with values are specified as "-name value".
1501 
1502  \param name Name of argument specified on command line.
1503  \param value Potential value assigned to argument.
1504  \return Number of strings consumed (1 for positional arguments or
1505  arguments that don't take values or 2 otherwise).
1506  */
1507  int parseShortArg(const std::string& name, const std::string& value)
1508  {
1509  if (name.size() == 1)
1510  throw arg_error("No argument found following '-'.");
1511  assert(name.size() == 2);
1512 
1513  Arg *arg = findShortArg(name[1]);
1514  if (!arg)
1515  throw arg_error("Unexpected argument '-" + std::string(1, name[1]) +
1516  "'.");
1517 
1518  int cnt;
1519  if (arg->needsValue())
1520  {
1521  // If the value starts with a '-', assume it's an option
1522  // rather than a value.
1523  if (value.empty() || value[0] == '-')
1524  {
1525  throw arg_error("Short option '" + name + "' expects value "
1526  "but none directly follows.");
