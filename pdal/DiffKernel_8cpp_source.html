64 void DiffKernel::checkPoints(const PointView& source_data,
65  const PointView& candidate_data, MetadataNode errors)
66 {
67  uint32_t MAX_BADBYTES(20);
68  uint32_t badbytes(0);
69 
70  // Both schemas have already been determined to be equal, so are the
71  // same size and in the same order.
72  Dimension::IdList const& sourceDims = source_data.dims();
73  Dimension::IdList const& candidateDims = candidate_data.dims();
74 
75  char sbuf[8];
76  char cbuf[8];
77  for (PointId idx = 0; idx < source_data.size(); ++idx)
78  {
79  for (size_t d = 0; d < sourceDims.size(); ++d)
80  {
81  Dimension::Id sd = sourceDims[d];
82  Dimension::Id cd = candidateDims[d];
83 
84  source_data.getRawField(sd, idx, (void *)sbuf);
85  candidate_data.getRawField(cd, idx, (void *)cbuf);
86  Dimension::Type t = Dimension::defaultType(cd);
87  size_t size = Dimension::size(t);
88  if (memcmp(sbuf, cbuf, size))
89  {
90  std::ostringstream oss;
91 
92  oss << "Point " << idx << " differs for dimension \"" <<
93  Dimension::name(sd) << "\" for source and candidate";
94  errors.add("data.error", oss.str());
95  badbytes++;
96  }
97  }
98  if (badbytes > MAX_BADBYTES )
99  break;
100  }
101 }

104 int DiffKernel::execute()
105 {
106  PointTable sourceTable;
107 
108  Stage& source = makeReader(m_sourceFile, m_driverOverride);
109  source.prepare(sourceTable);
110  PointViewSet sourceSet = source.execute(sourceTable);
111 
112  MetadataNode errors;
113 
114  PointTable candidateTable;
115 
116  Stage& candidate = makeReader(m_candidateFile, m_driverOverride);
117  candidate.prepare(candidateTable);
118  PointViewSet candidateSet = candidate.execute(candidateTable);
119 
120  assert(sourceSet.size() == 1);
121  assert(candidateSet.size() == 1);
122  PointViewPtr sourceView = *sourceSet.begin();
123  PointViewPtr candidateView = *candidateSet.begin();
124  if (candidateView->size() != sourceView->size())
125  {
126  std::ostringstream oss;
127 
128  oss << "Source and candidate files do not have the same point count";
129  errors.add("count.error", oss.str());
130  errors.add("count.candidate", candidateView->size());
131  errors.add("count.source", sourceView->size());
132  }
133 
134  MetadataNode source_metadata = sourceTable.metadata();
135  MetadataNode candidate_metadata = candidateTable.metadata();
136  if (source_metadata != candidate_metadata)
137  {
138  std::ostringstream oss;
139 
140  oss << "Source and candidate files do not have the same metadata count";
141  errors.add("metadata.error", oss.str());
142  errors.add(source_metadata);
143  errors.add(candidate_metadata);
144  }
145 
146  if (candidateTable.layout()->dims().size() !=
147  sourceTable.layout()->dims().size())
148  {
149  std::ostringstream oss;
150 
151  oss << "Source and candidate files do not have the same "
152  "number of dimensions";
153  }
154 
155  return 0;
156 }
