93 /*
94  Build a pipeline from a JSON filter specification.
95 */
96 void TranslateKernel::makeJSONPipeline()
97 {
98  std::string json;
99 
100  if (pdal::FileUtils::fileExists(m_filterJSON))
101  json = pdal::FileUtils::readFileIntoString(m_filterJSON);
102 
103  if (json.empty())
104  json = m_filterJSON;
105  std::stringstream in(json);
106  m_manager.readPipeline(in);
107 
108  std::vector<Stage *> roots = m_manager.roots();
109  if (roots.size() > 1)
110  throw pdal_error("Can't process pipeline with more than one root.");
111 
112  Stage *r(nullptr);
113  if (roots.size())
114  r = dynamic_cast<Reader *>(roots[0]);
115  if (r)
116  {
117  StageCreationOptions ops { m_inputFile, m_readerType, nullptr,
118  Options(), r->tag() };
119  m_manager.replace(r, &m_manager.makeReader(ops));
120  }
121  else
122  {
123  r = &m_manager.makeReader(m_inputFile, m_readerType);
124  if (roots.size())
125  roots[0]->setInput(*r);
126  }
127 
128  std::vector<Stage *> leaves = m_manager.leaves();
129  if (leaves.size() != 1)
130  throw pdal_error("Can't process pipeline with more than one "
131  "terminal stage.");
132 
133  Stage *w = dynamic_cast<Writer *>(leaves[0]);
134  if (w)
135  m_manager.replace(w, &m_manager.makeWriter(m_outputFile, m_writerType));
136  else
137  {
138  // We know we have a leaf because we added a reader.
139  StageCreationOptions ops { m_outputFile, m_writerType, leaves[0],
140  Options(), "" }; // These last two args just keep compiler quiet.
141  m_manager.makeWriter(ops);
142  }
143 }

146 /*
147  Build a pipeline from filters specified as command-line arguments.
148 */
149 void TranslateKernel::makeArgPipeline()
150 {
151  std::string readerType(m_readerType);
152  if (!readerType.empty() && !Utils::startsWith(readerType, "readers."))
153  readerType.insert(0, "readers.");
154  Stage& reader = m_manager.makeReader(m_inputFile, readerType);
155  Stage* stage = &reader;
156 
157  // add each filter provided on the command-line,
158  // updating the stage pointer
159  for (auto const f : m_filterType)
160  {
161  std::string filter_name(f);
162 
163  if (!Utils::startsWith(f, "filters."))
164  filter_name.insert(0, "filters.");
165 
166  Stage& filter = m_manager.makeFilter(filter_name, *stage);
167  stage = &filter;
168  }
169  std::string writerType(m_writerType);
170  if (!writerType.empty() && !Utils::startsWith(writerType, "writers."))
171  writerType.insert(0, "writers.");
172  m_manager.makeWriter(m_outputFile, writerType, *stage);
173 }

176 int TranslateKernel::execute()
177 {
178  std::ostream *metaOut(nullptr);
179 
180  if (m_filterJSON.size() && m_filterType.size())
181  throw pdal_error("Cannot set both --filter options and --json options");
182 
183  if (m_metadataFile.size())
184  {
185  if (m_pipelineOutputFile.size())
186  m_log->get(LogLevel::Info) << "Metadata will not be written. "
187  "'pipeline' option prevents execution.";
188  else
189  {
190  metaOut = FileUtils::createFile(m_metadataFile);
191  if (! metaOut)
192  throw pdal_error("Couldn't output metadata output file '" +
193  m_metadataFile + "'.");
194  }
195  }
196 
197  if (!m_filterJSON.empty())
198  makeJSONPipeline();
199  else
200  makeArgPipeline();
201 
202  // If we write pipeline output, we don't run, and therefore don't write
203  if (m_pipelineOutputFile.size() > 0)
204  {
205  PipelineWriter::writePipeline(m_manager.getStage(),
206  m_pipelineOutputFile);
207  return 0;
208  }
209  m_manager.execute();
210  if (metaOut)
211  {
212  MetadataNode m = m_manager.getMetadata();
213  *metaOut << Utils::toJSON(m);
214  FileUtils::closeFile(metaOut);
215  }
216 
217  return 0;
218 }
