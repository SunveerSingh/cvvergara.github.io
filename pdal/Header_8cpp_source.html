108 ILeStream& operator>>(ILeStream& in, Header& h)
109 {
110  uint8_t versionMajor;
111  uint32_t legacyPointCount;
112  uint32_t legacyReturnCount;
113 
114  in.get(h.m_fileSig, 4);
115  if (!h.signatureValid())
116  throw Exception("Not a LAS/LAZ file. Invalid file signature.");
117  in >> h.m_sourceId >> h.m_globalEncoding;
118  get(in, h.m_projectGuid);
119  in >> versionMajor >> h.m_versionMinor;
120  in.get(h.m_systemId, 32);
121 
122  in.get(h.m_softwareId, 32);
123  in >> h.m_createDOY >> h.m_createYear >> h.m_vlrOffset >>
124  h.m_pointOffset >> h.m_vlrCount >> h.m_pointFormat >>
125  h.m_pointLen >> legacyPointCount;
126  h.m_pointCount = legacyPointCount;
127 
128  // Although it isn't part of the LAS spec, the two high bits have been used
129  // to indicate compression, though only the high bit is currently used.
130  if (h.m_pointFormat & 0x80)
131  h.setCompressed(true);
132  h.m_pointFormat &= ~0xC0;
133 
134  for (size_t i = 0; i < Header::LEGACY_RETURN_COUNT; ++i)
135  {
136  in >> legacyReturnCount;
137  h.m_pointCountByReturn[i] = legacyReturnCount;
138  }
139 
140  in >> h.m_scales[0] >> h.m_scales[1] >> h.m_scales[2];
141  in >> h.m_offsets[0] >> h.m_offsets[1] >> h.m_offsets[2];
142 
143  double maxX, minX;
144  double maxY, minY;
145  double maxZ, minZ;
146  in >> maxX >> minX >> maxY >> minY >> maxZ >> minZ;
147  h.m_bounds = BOX3D(minX, minY, minZ, maxX, maxY, maxZ);
148 
149  if (h.versionAtLeast(1, 3))
150  {
151  uint64_t waveformOffset;
152  in >> waveformOffset;
153  }
154  if (h.versionAtLeast(1, 4))
155  {
156  uint32_t numPoints = (uint32_t)h.m_pointCount;
157  in >> h.m_eVlrOffset >> h.m_eVlrCount >> h.m_pointCount;
158  for (size_t i = 0; i < Header::RETURN_COUNT; ++i)
159  in >> h.m_pointCountByReturn[i];
160  if (numPoints && (numPoints != h.m_pointCount))
161  {
162  std::stringstream ss;
163  ss << "1.4 point count (" << h.m_pointCount << ") "
164  "doesn't match legacy point count (" << numPoints << ").";
165  throw Exception(ss.str());
166  }
167  }
168 
169  return in;
170 }
