59 void AssignRange::parse(const std::string& r)
60 {
61  std::string::size_type pos, count;
62  const char *start;
63  char *end;
64 
65  pos = subParse(r);
66  count = Utils::extractSpaces(r, pos);
67  pos += count;
68 
69  if (r[pos] != '=')
70  throw error("Missing '=' assignment separator.");
71  pos++;
72 
73  count = Utils::extractSpaces(r, pos);
74  pos += count;
75 
76  // Extract value
77  start = r.data() + pos;
78  m_value = std::strtod(start, &end);
79  if (start == end)
80  throw error("Missing value to assign following '='.");
81  pos += (end - start);
82 
83  if (pos != r.size())
84  throw error("Invalid characters following valid range.");
85 }
