93 
94  pdal::gdal::SpatialRef spatialref(m_srs);
95  OGRSpatialReferenceH ref = (OGRSpatialReferenceH)spatialref.get();
96 
97  if (!m_layerName.size())
98  m_layerName = m_filename;

102  throw pdal::pdal_error("Layer creation was null!");
103  }
104 
105  OGRFieldDefnH hFieldDefn;

129 void OGR::writeBoundary(hexer::HexGrid *grid)
130 {
131  OGRGeometryH multi = OGR_G_CreateGeometry(wkbMultiPolygon);
132 
133  const std::vector<hexer::Path *>& paths = grid->rootPaths();
134  for (auto pi = paths.begin(); pi != paths.end(); ++pi)
135  {
136  OGRGeometryH polygon = OGR_G_CreateGeometry(wkbPolygon);
137  collectPath(*pi, polygon);
138 
139  if( OGR_G_AddGeometryDirectly(multi, polygon ) != OGRERR_NONE )
140  {
141  std::ostringstream oss;
142  oss << "Unable to add polygon to multipolygon with error '"
143  << CPLGetLastErrorMsg() << "'";
144  throw pdal::pdal_error(oss.str());
145  }
146  }
147 
148  OGRFeatureH hFeature;
149 
150  hFeature = OGR_F_Create(OGR_L_GetLayerDefn(m_layer));
151  OGR_F_SetFieldInteger( hFeature, OGR_F_GetFieldIndex(hFeature, "ID"), 0);
152 
153  OGR_F_SetGeometry(hFeature, multi);
154  OGR_G_DestroyGeometry(multi);
155 
156  if( OGR_L_CreateFeature( m_layer, hFeature ) != OGRERR_NONE )
157  {
158  std::ostringstream oss;
159  oss << "Unable to create feature for multipolygon with error '"
160  << CPLGetLastErrorMsg() << "'";
161  throw pdal::pdal_error(oss.str());

165 void OGR::collectPath(hexer::Path* path, OGRGeometryH polygon)
166 {
167  OGRGeometryH ring = OGR_G_CreateGeometry(wkbLinearRing);
168 
169  vector<hexer::Point> pts = path->points();
170 
171  vector<hexer::Point>::const_iterator i;
172  for (i = pts.begin(); i != pts.end(); ++i)
173  {
174  OGR_G_AddPoint_2D(ring, i->m_x, i->m_y);
175  }
176 
177  if( OGR_G_AddGeometryDirectly(polygon, ring) != OGRERR_NONE )
178  {
179  std::ostringstream oss;
180  oss << "Unable to add geometry with error '" <<
181  CPLGetLastErrorMsg() << "'";
182  throw pdal::pdal_error(oss.str());
183  }
184 
185  vector<hexer::Path *> paths = path->subPaths();
186  for (vector<hexer::Path*>::size_type pi = 0; pi != paths.size(); ++pi)
187  {
188  hexer::Path* p = paths[pi];
189  collectPath(p, polygon);
190  }
191 }

193 OGRGeometryH OGR::collectHexagon(hexer::HexInfo const& info, hexer::HexGrid const* grid)
194 {
195  OGRGeometryH ring = OGR_G_CreateGeometry(wkbLinearRing);
196 
197  hexer::Point pos = info.m_center;
198  pos += grid->origin();
199 
200 
201  OGR_G_AddPoint_2D(ring, pos.m_x, pos.m_y);
202  for (int i = 1; i <= 5; ++i)
203  {
204  hexer::Point p = pos + grid->offset(i);
205  OGR_G_AddPoint_2D(ring, p.m_x, p.m_y);
206  }
207  OGR_G_AddPoint_2D(ring, pos.m_x, pos.m_y);
208 
209  OGRGeometryH polygon = OGR_G_CreateGeometry(wkbPolygon);
210  if( OGR_G_AddGeometryDirectly(polygon, ring ) != OGRERR_NONE )
211  {
212  std::ostringstream oss;
213  oss << "Unable to add ring to polygon in collectHexagon '"
214  << CPLGetLastErrorMsg() << "'";
215  throw pdal::pdal_error(oss.str());
216  }
217 
218  return polygon;

223 void OGR::writeDensity(hexer::HexGrid *grid)
224 {
225  int counter(0);
226  for (hexer::HexIter iter = grid->hexBegin(); iter != grid->hexEnd(); ++iter)
227  {
228 
229  hexer::HexInfo hi = *iter;
230  OGRGeometryH polygon = collectHexagon(hi, grid);
231  OGRFeatureH hFeature;
232 
233  hFeature = OGR_F_Create(OGR_L_GetLayerDefn(m_layer));
234  OGR_F_SetFieldInteger( hFeature, OGR_F_GetFieldIndex(hFeature, "ID"),
235  counter);
236  OGR_F_SetFieldInteger( hFeature, OGR_F_GetFieldIndex(hFeature, "COUNT"),
237  hi.m_density);
238 
239  OGR_F_SetGeometry(hFeature, polygon);
240  OGR_G_DestroyGeometry(polygon);
241 
242  if( OGR_L_CreateFeature( m_layer, hFeature ) != OGRERR_NONE )
243  {
244  std::ostringstream oss;
245  oss << "Unable to create feature for multipolygon with error '"
246  << CPLGetLastErrorMsg() << "'";
247  throw pdal::pdal_error(oss.str());
248  }
249  OGR_F_Destroy( hFeature );
250  counter++;
251  }
252 }
