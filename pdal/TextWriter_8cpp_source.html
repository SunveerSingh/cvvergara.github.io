100 void TextWriter::ready(PointTableRef table)
101 {
102  m_stream->precision(m_precision);
103  *m_stream << std::fixed;
104 
105  // Find the dimensions listed and put them on the id list.
106  StringList dimNames = Utils::split2(m_dimOrder, ',');
107  for (std::string dim : dimNames)
108  {
109  Utils::trim(dim);
110  Dimension::Id d = table.layout()->findDim(dim);
111  if (d == Dimension::Id::Unknown)
112  throwError("Dimension not found with name '" + dim + "'.");
113  m_dims.push_back(d);
114  }
115 
116  // Add the rest of the dimensions to the list if we're doing that.
117  // Yes, this isn't efficient when, but it's simple.
118  if (m_dimOrder.empty() || m_writeAllDims)
119  {
120  Dimension::IdList all = table.layout()->dims();
121  for (auto di = all.begin(); di != all.end(); ++di)
122  if (!Utils::contains(m_dims, *di))
123  m_dims.push_back(*di);
124  }
125 
126  if (!m_writeHeader)
127  log()->get(LogLevel::Debug) << "Not writing header" << std::endl;
128  else
129  writeHeader(table);
130 }

164 void TextWriter::writeCSVHeader(PointTableRef table)
165 {
166  const PointLayoutPtr layout(table.layout());

168  {
169  if (di != m_dims.begin())
170  *m_stream << m_delimiter;
171 
172  if (m_quoteHeader)
173  *m_stream << "\"" << layout->dimName(*di) << "\"";
174  else

176  }
177  *m_stream << m_newline;
178 }

180 void TextWriter::writeCSVBuffer(const PointViewPtr view)

194 void TextWriter::writeGeoJSONBuffer(const PointViewPtr view)
195 {
196  using namespace Dimension;
197 
198  for (PointId idx = 0; idx < view->size(); ++idx)
199  {
200  if (idx)
201  *m_stream << ",";
202 
203  *m_stream << "{ \"type\":\"Feature\",\"geometry\": "
204  "{ \"type\": \"Point\", \"coordinates\": [";
205  *m_stream << view->getFieldAs<double>(Id::X, idx) << ",";
206  *m_stream << view->getFieldAs<double>(Id::Y, idx) << ",";
207  *m_stream << view->getFieldAs<double>(Id::Z, idx) << "]},";
208 
209  *m_stream << "\"properties\": {";
210 
211  for (auto di = m_dims.begin(); di != m_dims.end(); ++di)
212  {
213  if (di != m_dims.begin())
214  *m_stream << ",";
215 
216  *m_stream << "\"" << view->dimName(*di) << "\":";
217  *m_stream << "\"";
218  *m_stream << view->getFieldAs<double>(*di, idx);
219  *m_stream <<"\"";
220  }
221  *m_stream << "}"; // end properties
222  *m_stream << "}"; // end feature
223  }
