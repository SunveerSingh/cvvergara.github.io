54 void MatlabReader::initialize(PointTableRef table)
55 {
56  m_matfile = matOpen(m_filename.c_str(), "r");
57  if (!m_matfile)
58  throwError("Could not open file '" + m_filename + "' for reading.");
59 
60  m_structArray = matGetVariable(m_matfile, m_structName.c_str());
61  if (!m_structArray)
62  {
63  std::ostringstream oss;
64  oss << "Array struct with name '" << m_structName << "' not found ";

69  m_pointIndex = 0;

89 void MatlabReader::addDimensions(PointLayoutPtr layout)
90 {
91  log()->get(LogLevel::Debug) << "Opening file" <<
92  " '" << m_filename << "'." << std::endl;
93 
94 
95 
96  // For now we read all of the fields in the given
97  // array structure
98  m_numFields = mxGetNumberOfFields(m_structArray);
99  if (!m_numFields)
100  throw pdal::pdal_error("Selected struct array must have fields!");
101 
102  // Fetch the first array and determine number of elements
103  // it has. We're only going to read that many elements no
104  // matter what
105  mxArray* f = mxGetFieldByNumber(m_structArray, 0, 0);
106  if (!f)
107  {
108  throwError("Unable to fetch first array in array struct to determine number of elements!");
109  }
110  m_numElements = mxGetNumberOfElements(f);
111 
112  // Get the dimensions and their types from the array struct
113  PointLayoutPtr matlabLayout = mlang::Script::getStructLayout(m_structArray, log());
114  const Dimension::IdList& dims = matlabLayout->dims();
115 
116  int nDimensionNumber(0);
117  for(auto d: dims)
118  {
119  std::string dimName = matlabLayout->dimName(d);
120  const Dimension::Detail* detail = matlabLayout->dimDetail(d);
121  Dimension::Id id = detail->id();
122  Dimension::Type t = detail->type();
123  layout->registerDim(id, t);
124 
125  // Keep a map of the Matlab dimension number to
126  // both the PDAL type and PDAL id
127  std::pair<int, int> pd = std::make_pair(nDimensionNumber, (int)id);
128  m_dimensionIdMap.insert(pd);
129  std::pair<int, int> pt = std::make_pair(nDimensionNumber, (int)t);
130  m_dimensionTypeMap.insert(pt);
131  nDimensionNumber++;
132  }

134 }

137 point_count_t MatlabReader::read(PointViewPtr view, point_count_t numPts)
138 {
139  PointId idx = view->size();
140  point_count_t cnt = 0;
141  PointRef point(*view, idx);
142  while (cnt < numPts)
143  {
144  point.setPointId(idx);
145  if (!processOne(point))
146  break;
147  cnt++;
148  idx++;
149  }
150  return cnt;
151 }

154 bool MatlabReader::processOne(PointRef& point)
155 {
156  // We read them all
157  if (m_pointIndex == m_numElements)
158  return false;
159 
160  for (int i=0; i < m_numFields; ++i)
161  {
162  Dimension::Id d = (Dimension::Id) m_dimensionIdMap[i];
163  Dimension::Type t = (Dimension::Type) m_dimensionTypeMap[i];
164 
165  mxArray* f = mxGetFieldByNumber(m_structArray, 0, i);
166  if (!f)
167  {
168  std::ostringstream oss;
169  oss << "Unable to fetch array for point " << m_pointIndex;
170  throwError(oss.str());
171  }
172  PointId numElements = (PointId) mxGetNumberOfElements(f);
173  if (numElements >= m_pointIndex )
174  {
175  size_t size = mxGetElementSize(f);
176  char* p = (char*)mxGetData(f) + (m_pointIndex*size);
177  point.setField(d, t, (void*)p);
178  }
179  }
180 
181  m_pointIndex++;
182 
183  return true;
184 }
