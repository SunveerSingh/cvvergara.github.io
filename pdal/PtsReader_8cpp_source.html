101 point_count_t PtsReader::read(PointViewPtr view, point_count_t numPts)
102 {
103  PointId idx = view->size();
104 
105  point_count_t cnt = 0;
106  size_t line = 1;
107 
108  while (m_istream->good() && cnt < numPts)
109  {
110  std::string buf;
111  StringList fields;
112 
113  std::getline(*m_istream, buf);
114  line++;
115  if (buf.empty())
116  continue;
117 
118  fields = Utils::split2(buf, m_separator);
119  if (fields.size() != m_dims.size())
120  {
121  log()->get(LogLevel::Error) << "Line " << line <<
122  " in '" << m_filename << "' contains " << fields.size() <<
123  " fields when " << m_dims.size() << " were expected. "
124  "Ignoring." << std::endl;
125  continue;
126  }
127 
128  double d;
129  for (size_t i = 0; i < fields.size(); ++i)
130  {
131  if (!Utils::fromString(fields[i], d))
132  {
133  log()->get(LogLevel::Error) << "Can't convert "
134  "field '" << fields[i] << "' to numeric value on line " <<
135  line << " in '" << m_filename << "'. Setting to 0." <<
136  std::endl;
137  d = 0;
138  }
139  if (i == 3) // Intensity field in PTS is -2048 to 2047, we map to 0 4095
140  {
141  d += 2048;
142  }
143  view->setField(m_dims[i], idx, d);
144  }
145  cnt++;
146  idx++;
147  }
148  return cnt;
149 }
