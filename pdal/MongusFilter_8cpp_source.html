139

144  cz.setConstant(std::numeric_limits<double>::max());
145 
146  // find initial set of Z minimums at native resolution
147  for (point_count_t i = 0; i < np; ++i)
148  {
149  using namespace Dimension;
150  double x = view->getFieldAs<double>(Id::X, i);
151  double y = view->getFieldAs<double>(Id::Y, i);
152  double z = view->getFieldAs<double>(Id::Z, i);
153 
154  int c = Utils::clamp(getColIndex(x, m_cellSize), 0, m_numCols-1);
155  int r = Utils::clamp(getRowIndex(y, m_cellSize), 0, m_numRows-1);

162  }

168  // sufficient accuracy is achieved by using a larger window size for opening
169  // (W11) than for closing (W9).
170  MatrixXd mo = eigen::matrixOpen(cz, 11);
171  writeControl(cx, cy, mo, "grid_open.laz");
172  MatrixXd mc = eigen::matrixClose(mo, 9);
173  writeControl(cx, cy, mc, "grid_close.laz");
174 
175  // ...in order to minimize the distortions caused by such filtering, the
176  // output points ... are compared to C and only ci with significantly lower
177  // elevation [are] replaced... In our case, d = 1.0 m was used.
178  for (auto i = 0; i < cz.size(); ++i)
179  {
180  if ((mc(i) - cz(i)) >= 1.0)
181  cz(i) = mc(i);
182  }
183  // cz is still at native resolution, with low points replaced by
184  // morphological operators
185  writeControl(cx, cy, cz, "grid_mins_adjusted.laz");
186 
187  // downsample control at max_level
188  int level = m_l;
189  double cur_cell_size = m_cellSize * std::pow(2, level);
190  // for max level = 8 and cell size 1, this is 256
191 
192  MatrixXd x_prev, y_prev, z_prev;
193 
194  // Top-level control samples are assumed to be ground points, no filtering
195  // is applied.
196  downsampleMin(&cx, &cy, &cz, &x_prev, &y_prev, &z_prev, cur_cell_size);
197  // x|y|z_prev are control points downsampled to coarsest resolution for
198  // the hierarchy, e.g., for 512x512, this would be 2x2
199  writeControl(x_prev, y_prev, z_prev, "control_init.laz");
200 
201  // Point-filtering is performed iteratively at each level of the
202  // control-points hierarchy in a top-down fashion
203  for (auto l = level-1; l > 0; --l)
204  {
205  std::cerr << "Level " << l << std::endl;
206  cur_cell_size /= 2;
207  // 128, 64, 32, 16, 8, 4, 1
208 
209  // compute TPS with update control at level
210 
211  // The interpolated surface is estimated based on the filtered set of
212  // TPS control-points at the previous level of hierarchy
213  // MatrixXd surface = TPS(x_prev, y_prev, z_prev, cur_cell_size);
214  // 4x4, 8x8, 16x16, 32x32, 64x64, 128x128, 256x256

268  // median takes an unsorted vector, possibly containing NANs, and

270  auto median = [&](std::vector<double> vals)
271  {
272  // Begin by partitioning the vector by isnan.
273  auto ptr = std::partition(vals.begin(), vals.end(), [](double p)
274  {
275  return std::isnan(p);
276  });
277 
278  // Copy the actual values, thus eliminating NANs, and sort it.
279  std::vector<double> cp(ptr, vals.end());
280  std::sort(cp.begin(), cp.end());
281 
282  std::cerr << "median troubleshooting\n";

461 void MongusFilter::downsampleMin(Eigen::MatrixXd *cx, Eigen::MatrixXd *cy,
462  Eigen::MatrixXd* cz, Eigen::MatrixXd *dcx,
463  Eigen::MatrixXd *dcy, Eigen::MatrixXd* dcz,
464  double cell_size)
465 {
466  int nr = ceil(cz->rows() / cell_size);
467  int nc = ceil(cz->cols() / cell_size);
468 
469  // std::cerr << nr << "\t" << nc << "\t" << cell_size << std::endl;
470 
471  dcx->resize(nr, nc);
472  dcx->setConstant(std::numeric_limits<double>::quiet_NaN());
473 
474  dcy->resize(nr, nc);
475  dcy->setConstant(std::numeric_limits<double>::quiet_NaN());
476 
477  dcz->resize(nr, nc);
478  dcz->setConstant(std::numeric_limits<double>::max());
479 
480  for (auto c = 0; c < cz->cols(); ++c)
481  {
482  for (auto r = 0; r < cz->rows(); ++r)
483  {
484  if ((*cz)(r, c) == std::numeric_limits<double>::max())
485  continue;
486 
487  int rr = std::floor(r/cell_size);
488  int cc = std::floor(c/cell_size);
489 
490  if ((*cz)(r, c) < (*dcz)(rr, cc))
491  {
492  (*dcx)(rr, cc) = (*cx)(r, c);
493  (*dcy)(rr, cc) = (*cy)(r, c);
494  (*dcz)(rr, cc) = (*cz)(r, c);
495  }
496  }
497  }
498 }

501 {

533  " ground returns!\n";
534 
535  // create new PointView containing only ground returns
536  PointViewPtr output = view->makeNew();

538  {
539  output->appendPoint(*view, i);
540  }

542  viewSet.erase(view);
543  viewSet.insert(output);
544  }
545  }
546  else
547  {
548  if (idx.empty())
549  log()->get(LogLevel::Debug2) << "Filtered cloud has no "

554  "--extract\n";
555 
556  // return the view buffer unchanged
557  viewSet.insert(view);
558  }
559
