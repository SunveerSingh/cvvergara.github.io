46 void movePoints(PointViewPtr moving, const cpd::Matrix& result)
47 {
48  assert(moving->size() == (point_count_t)result.rows());
49  for (PointId i = 0; i < moving->size(); ++i)

87 PointViewSet CpdFilter::run(PointViewPtr view)
88 {
89  PointViewSet viewSet;
90  if (this->m_complete)
91  {
92  throw pdal_error(
93  "filters.cpd must have two point view inputs, no more, no less");
94  }
95  else if (this->m_fixed)
96  {
97  log()->get(LogLevel::Debug2) << "Adding moving points\n";
98  PointViewPtr result = this->change(this->m_fixed, view);
99  viewSet.insert(result);
100  this->m_complete = true;
101  log()->get(LogLevel::Debug2) << "filters.cpd complete\n";
102  }
103  else
104  {
105  log()->get(LogLevel::Debug2) << "Adding fixed points\n";
106  this->m_fixed = view;
107  }
108  return viewSet;
109 }

120 PointViewPtr CpdFilter::change(PointViewPtr fixed, PointViewPtr moving)
121 {
122  MetadataNode root = this->getMetadata();
123  root.add("method", this->m_method);
124  log()->get(LogLevel::Debug2)
125  << "filters.cpd running method:" << this->m_method << "\n";
126  if (this->m_method == "rigid")
127  {
128  this->cpd_rigid(fixed, moving);
129  }
130  else if (this->m_method == "affine")
131  {
132  this->cpd_affine(fixed, moving);
133  }
134  else if (this->m_method == "nonrigid")
135  {
136  this->cpd_nonrigid(fixed, moving);
137  }
138  else
139  {
140  std::stringstream ss;
141  ss << "Invalid cpd detection method: " << this->m_method;
142  throw pdal_error(ss.str());
143  }
144  return moving;
145 }

147 void CpdFilter::cpd_rigid(PointViewPtr fixed, PointViewPtr moving)
148 {
149  cpd::Matrix fixedMatrix = eigen::pointViewToEigen(*fixed);
150  cpd::Matrix movingMatrix = eigen::pointViewToEigen(*moving);
151  cpd::RigidResult result = cpd::rigid(fixedMatrix, movingMatrix);
152  movePoints(moving, result.points);
153  addMetadata(this, static_cast<cpd::Result>(result));
154  MetadataNode root = getMetadata();
155  root.add("transform", result.matrix());
156 }
