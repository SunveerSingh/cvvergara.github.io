70 TEST(Compression, simple)
71 {
72  const std::string file(Support::datapath("las/1.2-with-color.las"));
73 
74  const pdal::Option opt_filename("filename", file);
75  pdal::Options opts;
76  opts.add(opt_filename);
77 
78  LasReader reader;
79  reader.setOptions(opts);
80 
81  PointTable table;
82  PointLayoutPtr layout(table.layout());
83 
84  reader.prepare(table);
85  PointViewSet viewSet = reader.execute(table);
86  PointViewPtr view = *viewSet.begin();
87 
88  EXPECT_EQ(layout->pointSize(), 52U);
89 
90  std::vector<unsigned char> rawBuf;
91  //LazPerfBuf b(rawBuf);
92 
93  DimTypeList dimTypes = layout->dimTypes();
94  auto cb = [&rawBuf](char *buf, size_t bufsize)
95  {
96  unsigned char *ubuf = reinterpret_cast<unsigned char *>(buf);
97  rawBuf.insert(rawBuf.end(), ubuf, ubuf + bufsize);
98  };
99 
100  LazPerfCompressor compressor(cb, dimTypes);
101 
102  std::vector<char> tmpbuf(layout->pointSize());
103  for (PointId idx = 0; idx < view->size(); ++idx)
104  {
105  view->getPackedPoint(dimTypes, idx, tmpbuf.data());
106  compressor.compress(tmpbuf.data(), layout->pointSize());
107  }
108  compressor.done();
109 
110  EXPECT_EQ(view->size() * layout->pointSize(), (size_t)55380);
111  EXPECT_EQ(rawBuf.size(), (size_t)30945);
112 
113  LazPerfBuf b2(rawBuf);
114  LazPerfDecompressor<LazPerfBuf> decompressor(b2, dimTypes);
115 
116  size_t outbufSize = decompressor.pointSize() * view->size();
117  std::vector<char> outbuf(outbufSize);
118  decompressor.decompress(outbuf.data(), outbufSize);
119 
120  PointViewPtr otherView(new PointView(table));
121 
122  char *pos = outbuf.data();
123  for (PointId nextId = 0; nextId < 11; nextId++)
124  {
125  otherView->setPackedPoint(dimTypes, nextId, pos);
126  pos += decompressor.pointSize();
127  }
128  EXPECT_EQ(otherView->size(), 11U);
129  EXPECT_EQ(getBytes(otherView).size(), (size_t)(52 * 11));
130 
131  uint16_t r = otherView->getFieldAs<uint16_t>(Dimension::Id::Red, 10);
132  EXPECT_EQ(r, 64U);
133  int32_t x = otherView->getFieldAs<int32_t>(Dimension::Id::X, 10);
134  EXPECT_EQ(x, 636038);
135  double xd = otherView->getFieldAs<double>(Dimension::Id::X, 10);
136  EXPECT_FLOAT_EQ(xd, 636037.53);
137  int32_t y = otherView->getFieldAs<int32_t>(Dimension::Id::Y, 10);
138  EXPECT_EQ(y, 849338);
139 }

142 TEST(Compression, types)
143 {
144  using namespace Dimension;
145  Type types[] = {
146  Type::Unsigned8, Type::Unsigned16, Type::Unsigned32, Type::Unsigned64,
147  Type::Signed8, Type::Signed16, Type::Signed32, Type::Signed64,
148  Type::Float, Type::Double
149  };
150  // Size is 42.
151 
152  std::default_random_engine generator;
153  std::uniform_int_distribution<int> dist(std::numeric_limits<int>::min());
154  char pts[3][42];
155 
156  // Fill three "points" with some random data.
157  char *c = &pts[0][0];
158  for (size_t i = 0; i < 3 * 42; ++i)
159  {
160  int v = dist(generator);
161  memcpy(c++, &v, sizeof(char));
162  }
163 
164  DimTypeList dimTypes;
165  for (auto ti = std::begin(types); ti != std::end(types); ++ti)
166  dimTypes.push_back(DimType(Dimension::Id::Unknown, *ti));
167 
168  std::vector<unsigned char> rawBuf;
169  auto cb = [&rawBuf](char *buf, size_t bufsize)
170  {
171  unsigned char *ubuf = reinterpret_cast<unsigned char *>(buf);
172  rawBuf.insert(rawBuf.begin(), ubuf, ubuf + bufsize);
173  };
174 
175  LazPerfCompressor compressor(cb, dimTypes);
176  for (size_t i = 0; i < 50; i++)
177  {
178  compressor.compress(pts[0], 42);
179  compressor.compress(pts[1], 42);
180  compressor.compress(pts[2], 42);
181  }
182  compressor.done();
183 
184  LazPerfBuf b2(rawBuf);
185 
186  LazPerfDecompressor<LazPerfBuf> decompressor(b2, dimTypes);
187  char oPts[3][42];
188  for (size_t i = 0; i < 50; ++i)
189  {
190  decompressor.decompress(oPts[0], 42);
191  decompressor.decompress(oPts[1], 42);
192  decompressor.decompress(oPts[2], 42);
193  EXPECT_EQ(memcmp(pts[0], oPts[0], 42), 0);
194  EXPECT_EQ(memcmp(pts[1], oPts[1], 42), 0);
195  EXPECT_EQ(memcmp(pts[2], oPts[2], 42), 0);
196  memset(oPts[0], 0, 42);
197  memset(oPts[1], 0, 42);
198  memset(oPts[2], 0, 42);
199  }
200 }
