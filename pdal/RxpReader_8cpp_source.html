59 Dimension::IdList getRxpDimensions(bool syncToPps, bool minimal, bool reflectanceAsIntensity)
60 {
61  using namespace Dimension;
62  Dimension::IdList ids;
63 
64  ids.push_back(Id::X);
65  ids.push_back(Id::Y);
66  ids.push_back(Id::Z);
67  ids.push_back(getTimeDimensionId(syncToPps));
68  if (!minimal)
69  {
70  ids.push_back(Id::ReturnNumber);
71  ids.push_back(Id::NumberOfReturns);
72  ids.push_back(Id::Amplitude);
73  ids.push_back(Id::Reflectance);
74  ids.push_back(Id::EchoRange);
75  ids.push_back(Id::Deviation);
76  ids.push_back(Id::BackgroundRadiation);
77  ids.push_back(Id::IsPpsLocked);
78  }
79  if (reflectanceAsIntensity) {
80  ids.push_back(Id::Intensity);
81  }
82 
83  return ids;
84 }
