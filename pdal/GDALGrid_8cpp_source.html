75 void GDALGrid::expand(size_t width, size_t height, size_t xshift, size_t yshift)
76 {
77  if (width < m_width)
78  throw error("Expanded grid must have width at least as large "
79  "as existing grid.");
80  if (height < m_height)
81  throw error("Expanded grid must have height at least as large "
82  "as existing grid.");
83  if (m_width + xshift > width || m_height + yshift > height)
84  throw error("Can't shift existing grid outside of new grid "
85  "during expansion.");
86  if (width == m_width && height == m_height)
87  return;
88 
89  // Grid (raster) works upside down from standard X/Y.
90  yshift = height - (m_height + yshift);
91  auto moveVec = [=](DataPtr& src, double initializer = 0)
92  {
93  // Compute an index in the destination given source index coords.
94  auto dstIndex = [width, xshift, yshift](size_t i, size_t j)
95  {
96  return ((yshift + j) * width) + i + xshift;
97  };
98 
99  size_t size(width * height);
100  DataPtr dst(new DataVec(size, initializer));
101  for (size_t j = 0; j < m_height; ++j)
102  {
103  size_t srcPos = index(0, j);
104  size_t dstPos = dstIndex(0, j);
105  std::copy(src->begin() + srcPos, src->begin() + srcPos + m_width,
106  dst->begin() + dstPos);
107  }
108  src = std::move(dst);
109  };
110 
111  moveVec(m_count);
112  if (m_outputTypes & statMin)
113  moveVec(m_min, std::numeric_limits<double>::max());
114  if (m_outputTypes & statMax)
115  moveVec(m_max, std::numeric_limits<double>::lowest());
116  if (m_outputTypes & statIdw)
117  {
118  moveVec(m_idw);
119  moveVec(m_idwDist);
120  }
121  if ((m_outputTypes & statMean) || (m_outputTypes & statStdDev))
122  moveVec(m_mean);
123  if (m_outputTypes & statStdDev)
124  moveVec(m_stdDev);
125  m_width = width;
126  m_height = height;
127 }

168 void GDALGrid::addPoint(double x, double y, double z)
169 {
170  int iOrigin = horizontalIndex(x);
171  int jOrigin = verticalIndex(y);
172 
173  // Here's the logic... we divide the cells around the subject cell
174  // (at iOrigin, jOrigin) into four quadrants. We move outward from the
175  // subject cell, checking distance until we find that we're farther than
176  // permitted by the radius, then we move up (down, over, whatever) a row
177  // or column and do it again until we find a case where there's not a
178  // single cell in the row that meets our criteria.
179  // There are easier ways to figure out which cells will be close enough,
180  // be we need the precise distance for all those cells that we already
181  // know will qualify, so I don't think there's much overhead here that
182  // we can avoid.
183 
184  // The four quadrant cases could certainly be merged into one, but I
185  // think it would be harder to follow and there's really not that
186  // much code here.
187 
188  // The quadrants are the standard mathematical ones. Here's a picture
189  // of how things kind of work. At the end we update the central cell.
190 
191  // ^ ->
192  // ^ | -->
193  // ^ | | --->
194  // ^ | | | ---->
195  // <------- X ------>
196  // <------ | | | v
197  // <----- | | v
198  // <--- | v
199  // <- v
200 
201 
202  // First quadrant;
203  int i = iOrigin + 1;
204  int j = jOrigin;
205  while (i < (int)m_width && j >= 0)
206  {
207  double d = distance(i, j, x, y);
208  if (d < m_radius)
209  {
210  update(i, j, z, d);
211  i++;
212  }
213  else
214  {
215  if (i == iOrigin + 1)
216  break;
217  i = iOrigin + 1;
218  j--;
219  }
220  }
221 
222  // Second quadrant;
223  i = iOrigin;
224  j = jOrigin - 1;
225  while (i >= 0 && j >= 0)
226  {
227  double d = distance(i, j, x, y);
228  if (d < m_radius)
229  {
230  update(i, j, z, d);
231  j--;
232  }
233  else
234  {
235  if (j == jOrigin - 1)
236  break;
237  j = jOrigin - 1;
238  i--;
239  }
240  }
241 
242  // Third quadrant;
243  i = iOrigin - 1;
244  j = jOrigin;
245  while (i >= 0 && j < (int)m_height)
246  {
247  double d = distance(i, j, x, y);
248  if (d < m_radius)
249  {
250  update(i, j, z, d);
251  i--;
252  }
253  else
254  {
255  if (i == iOrigin - 1)
256  break;
257  i = iOrigin - 1;
258  j++;
259  }
260  }
261  // Fourth quadrant;
262  i = iOrigin;
263  j = jOrigin + 1;
264  while (i < (int)m_width && j < (int)m_height)
265  {
266  double d = distance(i, j, x, y);
267  if (d < m_radius)
268  {
269  update(i, j, z, d);
270  j++;
271  }
272  else
273  {
274  if (j == jOrigin + 1)
275  break;
276  j = jOrigin + 1;
277  i++;
278  }
279  }
280 
281  // This is a questionable case. If a point is in a cell, shouldn't
282  // it just be counted?
283  double d = distance(iOrigin, jOrigin, x, y);
284  if (d < m_radius &&
285  iOrigin >= 0 && jOrigin >= 0 &&
286  iOrigin < (int)m_width && jOrigin <= (int)m_height)
287  update(iOrigin, jOrigin, z, d);
288 }

290 void GDALGrid::update(int i, int j, double val, double dist)
291 {
292  // Once we determine that a point is close enough to a cell to count it,
293  // this function does the actual math. We use the value of the
294  // point (val) and its distance from the cell center (dist). There's
295  // a little math that needs to be done once all points are added. See
296  // finalize() for that.
297 
298  // See
299  // https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance
300  // https://en.wikipedia.org/wiki/Inverse_distance_weighting
301 
302  size_t offset = index(i, j);
303 
304  double& count = (*m_count)[offset];
305  count++;
306 
307  if (m_min)
308  {
309  double& min = (*m_min)[offset];
310  min = std::min(val, min);
311  }
312 
313  if (m_max)
314  {
315  double& max = (*m_max)[offset];
316  max = std::max(val, max);
317  }
318 
319  if (m_mean)
320  {
321  double& mean = (*m_mean)[offset];
322  double delta = val - mean;
323 
324  mean += delta / count;
325  if (m_stdDev)
326  {
327  double& stdDev = (*m_stdDev)[offset];
328  stdDev += delta * (val - mean);
329  }
330  }
331 
332  if (m_idw)
333  {
334  double& idw = (*m_idw)[offset];
335  double& idwDist = (*m_idwDist)[offset];
336 
337  // If the distance is 0, we set the idwDist to nan to signal that
338  // we should ignore the distance and take the value as is.
339  if (!std::isnan(idwDist))
340  {
341  if (dist == 0)
342  {
343  idw = val;
344  idwDist = std::numeric_limits<double>::quiet_NaN();
345  }
346  else
347  {
348  idw += val / dist;
349  idwDist += 1 / dist;
350  }
351  }
352  }
353 }

355 void GDALGrid::finalize()
356 {
357  // See
358  // https://en.wikipedia.org/wiki/Algorithms_for_calculating_variance
359  // https://en.wikipedia.org/wiki/Inverse_distance_weighting
360  if (m_stdDev)
361  for (size_t i = 0; i < m_count->size(); ++i)
362  if (!empty(i))
363  (*m_stdDev)[i] = sqrt((*m_stdDev)[i] / (*m_count)[i]);
364 
365  if (m_idw)
366  for (size_t i = 0; i < m_count->size(); ++i)
367  if (!empty(i))
368  {
369  double& distSum = (*m_idwDist)[i];
370 
371  if (!std::isnan(distSum))
372  (*m_idw)[i] /= distSum;
373  }
374 
375  if (m_windowSize > 0)
376  windowFill();
377  else
378  for (size_t i = 0; i < m_count->size(); ++i)
379  if (empty(i))
380  fillNodata(i);
381 }

399 void GDALGrid::windowFill(size_t dstI, size_t dstJ)
400 {
401  size_t istart = dstI > m_windowSize ? dstI - m_windowSize : (size_t)0;
402  size_t iend = std::min(width(), dstI + m_windowSize + 1);
403  size_t jstart = dstJ > m_windowSize ? dstJ - m_windowSize : (size_t)0;
404  size_t jend = std::min(height(), dstJ + m_windowSize + 1);
405 
406  double distSum = 0;
407  size_t dstIdx = index(dstI, dstJ);
408 
409  // Initialize to 0 (rather than numeric_limits::max/lowest) since we're
410  // going to accumulate and average.
411  if (m_min)
412  (*m_min)[dstIdx] = 0;
413  if (m_max)
414  (*m_max)[dstIdx] = 0;
415 
416  for (size_t i = istart; i < iend; ++i)
417  for (size_t j = jstart; j < jend; ++j)
418  {
419  size_t srcIdx = index(i, j);
420  if ((srcIdx == dstIdx) || empty(srcIdx))
421  continue;
422  // The ternaries just avoid underflow UB. We're just trying to
423  // find the distance from j to dstJ or i to dstI.
424  double distance = std::max(j > dstJ ? j - dstJ : dstJ - j,
425  i > dstI ? i - dstI : dstI - i);
426  windowFillCell(srcIdx, dstIdx, distance);
427  distSum += (1 / distance);
428  }
429 
430  // Divide summed values by the (inverse) distance sum.
431  if (distSum > 0)
432  {
433  if (m_min)
434  (*m_min)[dstIdx] /= distSum;
435  if (m_max)
436  (*m_max)[dstIdx] /= distSum;
437  if (m_mean)
438  (*m_mean)[dstIdx] /= distSum;
439  if (m_idw)
440  (*m_idw)[dstIdx] /= distSum;
441  if (m_stdDev)
442  (*m_stdDev)[dstIdx] /= distSum;
443  }
444  else
445  fillNodata(dstIdx);
446 }
