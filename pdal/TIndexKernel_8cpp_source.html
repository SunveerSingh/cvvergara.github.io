112 void TIndexKernel::validateSwitches(ProgramArgs& args)
113 {
114  if (m_merge)
115  {
116  if (!m_wkt.empty() && !m_bounds.empty())
117  throw pdal_error("Can't specify both 'polygon' and "
118  "'bounds' options.");
119  if (!m_bounds.empty())
120  m_wkt = m_bounds.toWKT();
121  if (m_filespec.empty())

123  StringList invalidArgs;

125  invalidArgs.push_back("src_srs_name");
126  invalidArgs.push_back("stdin");
127  invalidArgs.push_back("fast_boundary");
128  for (auto arg : invalidArgs)
129  if (args.set(arg))
130  {
131  std::ostringstream out;
132 
133  out << "option '" << arg << "' not supported during merge.";
134  throw pdal_error(out.str());
135  }
136  }
137  else
138  {
139  if (m_filespec.empty() && !m_usestdin)
140  throw pdal_error("No input pattern specified");
141  if (args.set("polygon"))
142  throw pdal_error("'polygon' option not supported when building "
143  "index.");
144  if (args.set("bounds"))
145  throw pdal_error("'bounds' option not supported when building "
146  "index.");
147  if (args.set("a_srs"))
148  m_overrideASrs = true;
149  }

215 void TIndexKernel::createFile()
216 {
217  if (!m_usestdin)
218  m_files = FileUtils::glob(m_filespec);
219  else
220  m_files = readSTDIN();
221 
222  if (m_absPath)
223  for (auto& s : m_files)
224  s = FileUtils::toAbsolutePath(s);
225 
226  if (m_files.empty())
227  {
228  std::ostringstream out;
229  out << "Couldn't find files to index: " << m_filespec << ".";
230  throw pdal_error(out.str());
231  }
232 
233 //ABELL - Remove CPLGetBasename use.
234  const std::string filename = m_files.front();
235  if (m_layerName.empty())
236  m_layerName = CPLGetBasename(filename.c_str());
237 
238  // Open or create the dataset.
239  if (!openDataset(m_idxFilename))
240  if (!createDataset(m_idxFilename))
241  {
242  std::ostringstream out;
243  out << "Couldn't open or create index dataset file '" <<
244  m_idxFilename << "'.";
245  throw pdal_error(out.str());
246  }
247 
248  // Open or create a layer
249  if (!openLayer(m_layerName))
250  if (!createLayer(m_layerName))
251  {
252  std::ostringstream out;
253  out << "Couldn't open or create layer '" << m_layerName <<
254  "' in output file '" << m_idxFilename << "'.";
255  throw pdal_error(out.str());
256  }
257 
258  FieldIndexes indexes = getFields();
259 
260  StageFactory factory(false);
261  for (auto f : m_files)
262  {
263  //ABELL - Not sure why we need to get absolute path here.
264  f = FileUtils::toAbsolutePath(f);
265  FileInfo info = getFileInfo(factory, f);
266  if (!isFileIndexed(indexes, info))
267  {
268  if (createFeature(indexes, info))
269  m_log->get(LogLevel::Info) << "Indexed file " << f << std::endl;
270  else
271  m_log->get(LogLevel::Error) << "Failed to create feature for "
272  "file '" << f << "'" << std::endl;
273 
274  }
275  }
276  OGR_DS_Destroy(m_dataset);
277 }

280 void TIndexKernel::mergeFile()
281 {
282  using namespace gdal;
283 
284  std::ostringstream out;
285 
286  if (!openDataset(m_idxFilename))
287  {
288  std::ostringstream out;
289  out << "Couldn't open index dataset file '" << m_idxFilename << "'.";
290  throw pdal_error(out.str());
291  }
292  if (!openLayer(m_layerName))
293  {
294  std::ostringstream out;
295  out << "Couldn't open layer '" << m_layerName <<
296  "' in output file '" << m_idxFilename << "'.";
297  throw pdal_error(out.str());
298  }
299 
300  FieldIndexes indexes = getFields();
301 
302  SpatialRef outSrs(m_tgtSrsString);
303  if (!outSrs)
304  throw pdal_error("Couldn't interpret target SRS string.");
305 
306  if (!m_wkt.empty())
307  {
308  Geometry g(m_wkt, outSrs);
309 
310  if (!g)
311  throw pdal_error("Couldn't interpret geometry filter string.");
312  OGR_L_SetSpatialFilter(m_layer, g.get());
313  }
314 
315  std::vector<FileInfo> files;
316 
317  // Docs are bad here. You need this call even if you haven't read anything
318  // or nothing happens.
319  OGR_L_ResetReading(m_layer);
320  while (true)
321  {
322  OGRFeatureH feature = OGR_L_GetNextFeature(m_layer);
323  if (!feature)
324  break;
325 
326  FileInfo fileInfo;
327  fileInfo.m_filename =
328  OGR_F_GetFieldAsString(feature, indexes.m_filename);
329  fileInfo.m_srs =
330  OGR_F_GetFieldAsString(feature, indexes.m_srs);
331  files.push_back(fileInfo);
332 
333  OGR_F_Destroy(feature);
334  }
335 
336  Options cropOptions;
337  if (!m_bounds.empty())
338  cropOptions.add("bounds", m_bounds);
339  else
340  cropOptions.add("polygon", m_wkt);
341 
342  Stage& merge = makeFilter("filters.merge");
343  for (auto f : files)
344  {
345  Stage& reader = makeReader(f.m_filename, m_driverOverride);
346  Stage *premerge = &reader;
347 
348  if (m_tgtSrsString != f.m_srs)
349  {
350  Options reproOptions;
351  reproOptions.add("out_srs", m_tgtSrsString);
352  reproOptions.add("in_srs", f.m_srs);
353  Stage& repro = makeFilter("filters.reprojection", reader,
354  reproOptions);
355  premerge = &repro;
356  }
357 
358  // WKT is set, even if we're using a bounding box for fitering, so
359  // can be used as a test here.
360  if (!m_wkt.empty())
361  {
362  Stage& crop = makeFilter("filters.crop", *premerge, cropOptions);
363  premerge = &crop;
364  }
365  merge.setInput(*premerge);
366  }
367 
368  Options writerOptions;
369  writerOptions.add("offset_x", "auto");
370  writerOptions.add("offset_y", "auto");
371  writerOptions.add("offset_z", "auto");
372  Stage& writer = makeWriter(m_filespec, merge, "", writerOptions);
373 
374  PointTable table;
375  writer.prepare(table);
376  writer.execute(table);
377 }

380 bool TIndexKernel::createFeature(const FieldIndexes& indexes,
381  FileInfo& fileInfo)
382 {
383  using namespace gdal;
384 
385  OGRFeatureH hFeature = OGR_F_Create(OGR_L_GetLayerDefn(m_layer));
386 
387  // Set the creation time into the feature.
388  setDate(hFeature, fileInfo.m_ctime, indexes.m_ctime);
389 
390  // Set the file mod time into the feature.
391  setDate(hFeature, fileInfo.m_mtime, indexes.m_mtime);
392 
393  // Set the filename into the feature.
394  OGR_F_SetFieldString(hFeature, indexes.m_filename,
395  fileInfo.m_filename.c_str());
396 
397  // Set the SRS into the feature.
398  // We override if m_assignSrsString is set
399  if (fileInfo.m_srs.empty() || m_overrideASrs)
400  fileInfo.m_srs = m_assignSrsString;
401 
402  SpatialRef srcSrs(fileInfo.m_srs);
403  if (srcSrs.empty())
404  {
405  std::ostringstream oss;
406 
407  oss << "Unable to import source spatial reference '" <<
408  fileInfo.m_srs << "' for file '" <<
409  fileInfo.m_filename << "'.";
410  throw pdal_error(oss.str());
411  }
412 
413  // We have a limit of like 254 characters in some formats (notably
414  // shapefile), so try to get the condensed version of the SRS.
415 
416  // Failing that, get the proj.4 version. Not sure what's supposed to
417  // happen if we overflow 254 with proj.4.
418 
419  const char* pszAuthorityCode = OSRGetAuthorityCode(srcSrs.get(), NULL);
420  const char* pszAuthorityName = OSRGetAuthorityName(srcSrs.get(), NULL);
421  if (pszAuthorityName && pszAuthorityCode)
422  {
423  std::string auth = std::string(pszAuthorityName) + ":" +
424  pszAuthorityCode;
425  OGR_F_SetFieldString(hFeature, indexes.m_srs, auth.data());
426  }
427  else
428  {
429  char* pszProj4 = NULL;
430  int err = -1;
431  try
432  {
433  err = OSRExportToProj4(srcSrs.get(), &pszProj4);
434  }
435  catch (pdal_error)
436  {}
437  if (err != OGRERR_NONE)
438  {
439  m_log->get(LogLevel::Warning) << "Unable to convert SRS to "
440  "proj.4 format for file '" << fileInfo.m_filename << "'" <<
441  std::endl;
442  return false;
443  }
444  std::string srs = std::string(pszProj4);
445  OGR_F_SetFieldString(hFeature, indexes.m_srs, srs.c_str());
446  CPLFree(pszProj4);
447  }
448 
449  // Set the geometry in the feature
450  Geometry g = prepareGeometry(fileInfo);
451  char *pgeom;
452  OGR_G_ExportToWkt(g.get(), &pgeom);
453  OGR_F_SetGeometry(hFeature, g.get());
454 
455  return (OGR_L_CreateFeature(m_layer, hFeature) == OGRERR_NONE);
456 }

459 TIndexKernel::FileInfo TIndexKernel::getFileInfo(StageFactory& factory,
460  const std::string& filename)
461 {
462  FileInfo fileInfo;
463 
464  PipelineManager manager;
465  manager.commonOptions() = m_manager.commonOptions();
466  manager.stageOptions() = m_manager.stageOptions();
467 
468  // Need to make sure options get set.
469  Stage& reader = manager.makeReader(filename, "");
470 
471  // If we aren't able to make a hexbin filter, we
472  // will just do a simple fast_boundary.
473  Stage* hexer = &manager.makeFilter("filters.hexbin", reader);
474  if (m_fastBoundary || !hexer)
475  {
476  QuickInfo qi = reader.preview();
477 
478  std::stringstream polygon;
479  polygon << "POLYGON ((";
480 
481  polygon << qi.m_bounds.minx << " " << qi.m_bounds.miny;
482  polygon << ", " << qi.m_bounds.maxx << " " << qi.m_bounds.miny;
483  polygon << ", " << qi.m_bounds.maxx << " " << qi.m_bounds.maxy;
484  polygon << ", " << qi.m_bounds.minx << " " << qi.m_bounds.maxy;
485  polygon << ", " << qi.m_bounds.minx << " " << qi.m_bounds.miny;
486  polygon << "))";
487  fileInfo.m_boundary = polygon.str();
488  if (!qi.m_srs.empty())
489  fileInfo.m_srs = qi.m_srs.getWKT();
490  }
491  else
492  {
493 
494  PointTable table;
495  hexer->prepare(table);
496  PointViewSet set = hexer->execute(table);
497 
498  MetadataNode m = table.metadata();
499  m = m.findChild("filters.hexbin:boundary");
500  fileInfo.m_boundary = m.value();
501 
502  PointViewPtr v = *set.begin();
503  if (!v->spatialReference().empty())
504  fileInfo.m_srs = v->spatialReference().getWKT();
505  }
506 
507  FileUtils::fileTimes(filename, &fileInfo.m_ctime, &fileInfo.m_mtime);
508  fileInfo.m_filename = filename;
509 
510  return fileInfo;
511 }
