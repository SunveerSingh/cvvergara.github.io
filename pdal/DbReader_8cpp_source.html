46 void DbReader::loadSchema(PointLayoutPtr layout, const XMLSchema& schema)
47 {
48  m_layout = layout;
49  m_dims = schema.xmlDims();
50 
51  // Override XYZ to doubles and use those going forward
52  // we will apply any scaling set before handing it off
53  // to PDAL.
54  layout->registerDim(Dimension::Id::X);
55  layout->registerDim(Dimension::Id::Y);
56  layout->registerDim(Dimension::Id::Z);
57 
58  m_orientation = schema.orientation();
59  m_packedPointSize = 0;
60  for (auto di = m_dims.begin(); di != m_dims.end(); ++di)
61  {
62  di->m_dimType.m_id =
63  layout->registerOrAssignDim(di->m_name, di->m_dimType.m_type);
64  m_packedPointSize += Dimension::size(di->m_dimType.m_type);
65  }
66 }

107 void DbReader::writeField(PointView& view, const char *pos, const DimType& dim,
108  PointId idx)
109 {
110  using namespace Dimension;
111 
112  if (dim.m_id == Id::X || dim.m_id == Id::Y || dim.m_id == Id::Z)
113  {
114  Everything e;
115 
116  memcpy(&e, pos, Dimension::size(dim.m_type));
117  double d = Utils::toDouble(e, dim.m_type);
118  d = (d * dim.m_xform.m_scale.m_val) + dim.m_xform.m_offset.m_val;
119  view.setField(dim.m_id, idx, d);
120  }
121  else
122  view.setField(dim.m_id, dim.m_type, idx, pos);
123 }
