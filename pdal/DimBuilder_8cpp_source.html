124 bool DimBuilder::execute()
125 {
126  Json::Reader reader;
127 
128  std::ifstream in(m_input);
129 
130  if (!in)
131  return false;
132 
133  Json::Value root;
134  if (!reader.parse(in, root))
135  {
136  std::cerr << reader.getFormattedErrorMessages();
137  return false;
138  }
139  Json::Value dims = root.get("dimensions", Json::Value());
140  if (root.size() != 1 || !dims.isArray())
141  {
142  std::ostringstream oss;
143 
144  oss << "Root node must contain a single 'dimensions' array.";
145  throw dimbuilder_error(oss.str());
146  }
147  for (size_t i = 0; i < dims.size(); ++i)
148  {
149  Json::Value& dim = dims[(int)i];
150  if (!dim.isObject())
151  {
152  std::ostringstream oss;
153 
154  oss << "Found a dimension that is not an object: " <<
155  dim.asString();
156  throw dimbuilder_error(oss.str());
157  }
158  extractDim(dim);
159  }
160 
161  std::ofstream out(m_output);
162  if (!out)
163  {
164  std::ostringstream oss;
165 
166  oss << "Unable to open output file '" << m_output << "'.";
167  throw dimbuilder_error(oss.str());
168  }
169  writeOutput(out);
170  return true;
171 }

174 void DimBuilder::extractDim(Json::Value& dim)
175 {
176  DimSpec d;
177 
178  // Get dimension name.
179  Json::Value name = dim.removeMember("name");
180  if (name.isNull())
181  throw dimbuilder_error("Dimension missing name.");
182  if (!name.isString())
183  throw dimbuilder_error("Dimension name must be a string.");
184  d.m_name = name.asString();
185  validateDimension(d.m_name);
186 
187  // Get dimension description.
188  Json::Value description = dim.removeMember("description");
189  if (description.isNull())
190  {
191  std::ostringstream oss;
192 
193  oss << "Dimension '" << d.m_name << "' must have a description.";
194  throw dimbuilder_error(oss.str());
195  }
196  if (!description.isString())
197  {
198  std::ostringstream oss;
199 
200  oss << "Description of dimension '" << d.m_name << "' must be a "
201  "string.";
202  throw dimbuilder_error(oss.str());
203  }
204  d.m_description = description.asString();
205 
206  // Get dimension type
207  Json::Value type = dim.removeMember("type");
208  if (type.isNull())
209  {
210  std::ostringstream oss;
211 
212  oss << "Dimension '" << d.m_name << "' must have a type.";
213  throw dimbuilder_error(oss.str());
214  }
215  d.m_type = Dimension::type(type.asString());
216  if (d.m_type == Dimension::Type::None)
217  {
218  std::ostringstream oss;
219 
220  oss << "Invalid type '" << type.asString() << "' specified for "
221  "dimension '" << d.m_name << "'.";
222  throw dimbuilder_error(oss.str());
223  }
224 
225  Json::Value altNames = dim.removeMember("alt_names");
226  if (!altNames.isNull())
227  {
228  if (!altNames.isString())
229  {
230  std::ostringstream oss;
231 
232  oss << "Alternate names for dimension '" << d.m_name << "' must "
233  "be a string.";
234  throw dimbuilder_error(oss.str());
235  }
236  d.m_altNames = Utils::split2(altNames.asString(), ',');
237  for (auto& s : d.m_altNames)
238  {
239  Utils::trim(s);
240  validateDimension(s);
241  }
242  }
243  if (dim.size() > 0)
244  {
245  std::ostringstream oss;
246 
247  oss << "Unexpected member '" << dim.getMemberNames()[0] << "' when "
248  "reading dimension '" << d.m_name << "'.";
249  throw dimbuilder_error(oss.str());
250  }
251  m_dims.push_back(d);
252 }

254 void DimBuilder::validateDimension(const std::string& dimName)
255 {
256  if (Dimension::extractName(dimName, 0) != dimName.size())
257  {
258  std::ostringstream oss;
259 
260  oss << "Invalid dimension name '" << dimName << "'. Dimension "
261  "names must start with a letter and be followed by letters, "
262  "digits or underscores.";
263  throw dimbuilder_error(oss.str());
264  }
265  for (DimSpec& d : m_dims)
266  {
267  if (d.m_name == dimName)
268  {
269  std::ostringstream oss;
270 
271  oss << "Duplicate dimension name '" << dimName << "' found. "
272  "Dimension names must be unique.";
273  throw dimbuilder_error(oss.str());
274  }
275  }

345 void DimBuilder::writeDescriptions(std::ostream& out)
346 {
347  out << "/// Get a description of a predefined dimension.\n";
348  out << "/// \\param[in] id Dimension ID.\n";
349  out << "/// \\return Dimension description.\n";
350  out << "inline std::string description(Id id)\n";
351  out << "{\n";
352  out << " switch (id)\n";
353  out << " {\n";
354  for (auto& d : m_dims)
355  {
356  std::vector<std::string> pieces = Utils::wordWrap2(d.m_description,
357  63, 60);
358  out << " case Id::" << d.m_name << ":\n";
359  out << " return \"" << pieces[0] << "\"";
360  if (pieces.size() == 1)
361  out << ";";
362  out << "\n";
363  auto pi = pieces.begin();
364  pi++;
365  for (; pi != pieces.end(); ++pi)
366  {
367  std::string piece = *pi;
368  out << " \"" << piece << "\"";
369  if (pi + 1 == pieces.end())
370  out << ";";
371  out << "\n";
372  }
373  }
374  out << " case Id::Unknown:\n";
375  out << " return \"\";\n";
376  out << " }\n";
377  out << " return \"\";\n";
378  out << "}\n";

382 void DimBuilder::writeNameToId(std::ostream& out)
383 {
384  out << "/// Get a predefined dimension ID given a dimension name. "
385  "Multiple names\n";
386  out << "/// may map to the same dimension for convenience. Names "
387  "are case-insensitive.\n";
388  out << "/// \\param[in] s Name of dimension.\n";
389  out << "/// \\return Dimension ID associated with the name. "
390  "Id::Unknown is returned\n";
391  out << "/// if the name doesn't map to a predefined dimension.\n";
392  out << "inline Id id(std::string s)\n";
393  out << "{\n";
394  out << " s = Utils::toupper(s);\n";
395  out << "\n";
396  for (auto& d : m_dims)
397  {
398  std::vector<std::string> names;
399  names.push_back(d.m_name);
400  names.insert(names.end(), d.m_altNames.begin(), d.m_altNames.end());
401  for (std::string& s : names)
402  {
403  out << " if (s == \"" << Utils::toupper(s) << "\")\n";
404  out << " return Id::" << d.m_name << ";\n";
405  }
406  }
407  out << " return Id::Unknown;\n";
408  out << "}\n";
