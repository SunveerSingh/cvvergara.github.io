72 void Geometry::update(const std::string& wkt_or_json)
73 {
74  bool isJson = wkt_or_json.find("{") != wkt_or_json.npos ||
75  wkt_or_json.find("}") != wkt_or_json.npos;
76 
77  GEOSWKTReader* geosreader = GEOSWKTReader_create_r(m_geoserr.ctx());
78 
79  if (!isJson)
80  {
81  geos::GeometryDeleter geom_del(m_geoserr);
82  GEOSGeomPtr p(GEOSWKTReader_read_r(m_geoserr.ctx(), geosreader,
83  wkt_or_json.c_str()), geom_del);
84  m_geom.swap(p);
85  }
86  else
87  {
88  // Assume it is GeoJSON and try constructing from that
89  OGRGeometryH json = OGR_G_CreateGeometryFromJson(wkt_or_json.c_str());
90 
91  if (!json)
92  throw pdal_error("Unable to create geometry from "
93  "input GeoJSON");
94 
95  char* gdal_wkt(0);
96  OGRErr err = OGR_G_ExportToWkt(json, &gdal_wkt);
97 
98  geos::GeometryDeleter geom_del(m_geoserr);
99  GEOSGeomPtr p(GEOSWKTReader_read_r(m_geoserr.ctx(), geosreader,
100  gdal_wkt), geom_del);
101  m_geom.swap(p);
102 
103  OGRFree(gdal_wkt);
104  OGR_G_DestroyGeometry(json);
105  }
106  prepare();
107 
108  GEOSWKTReader_destroy_r(m_geoserr.ctx(), geosreader);
109 }

165 Geometry::Geometry(OGRGeometryH g, const SpatialReference& srs)
166  : m_srs(srs)
167  , m_geoserr(geos::ErrorHandler::get())
168 {
169 
170  OGRGeometry *ogr_g = (OGRGeometry*)g;
171 
172  // Convert the the GDAL geom to WKB in order to avoid the version
173  // context and DLL boundary issues with exporting directoly to GEOS
174  // from GDAL
175  OGRwkbByteOrder bo =
176  GEOS_getWKBByteOrder() == GEOS_WKB_XDR ? wkbXDR : wkbNDR;
177  int wkbSize = ogr_g->WkbSize();
178  std::vector<unsigned char> wkb(wkbSize);
179 
180  ogr_g->exportToWkb(bo, wkb.data());
181 
182  GEOSWKBReader* reader = GEOSWKBReader_create_r(m_geoserr.ctx());
183 
184  geos::GeometryDeleter geom_del(m_geoserr);
185  GEOSGeomPtr p(GEOSWKBReader_read_r(m_geoserr.ctx(), reader, wkb.data(),
186  wkbSize), geom_del);
187  m_geom.swap(p);
188  prepare();
189 
190  GEOSWKBReader_destroy_r(m_geoserr.ctx(), reader);
191 }

194 Geometry Geometry::transform(const SpatialReference& ref) const
195 {
196  if (m_srs.empty())
197  throw pdal_error("Geometry::transform failed. "
198  "Source missing spatial reference.");
199  if (ref.empty())
200  throw pdal_error("Geometry::transform failed. "
201  "Invalid destination spatial reference.");
202  if (ref == m_srs)
203  return *this;
204 
205  gdal::SpatialRef fromRef(m_srs.getWKT());
206  gdal::SpatialRef toRef(ref.getWKT());
207  gdal::Geometry geom(wkt(12, true), fromRef);
208  geom.transform(toRef);
209  return Geometry(geom.wkt(), ref);
210 }

213 BOX3D Geometry::bounds() const
214 {
215  uint32_t numInputDims;
216  BOX3D output;
217 
218  GEOSGeometry* boundary = GEOSGeom_clone_r(m_geoserr.ctx(), m_geom.get());
219 
220  // Smash out multi*
221  if (GEOSGeomTypeId_r(m_geoserr.ctx(), m_geom.get()) > 3)
222  boundary = GEOSEnvelope_r(m_geoserr.ctx(), m_geom.get());
223 
224  GEOSGeometry const* ring = GEOSGetExteriorRing_r(m_geoserr.ctx(), boundary);
225  GEOSCoordSequence const* coords = GEOSGeom_getCoordSeq_r(m_geoserr.ctx(), ring);
226 
227  GEOSCoordSeq_getDimensions_r(m_geoserr.ctx(), coords, &numInputDims);
228 
229  uint32_t count(0);
230  GEOSCoordSeq_getSize_r(m_geoserr.ctx(), coords, &count);
231 
232  double x(0.0);
233  double y(0.0);
234  double z(0.0);
235  for (unsigned i = 0; i < count; ++i)
236  {
237  GEOSCoordSeq_getOrdinate_r(m_geoserr.ctx(), coords, i, 0, &x);
238  GEOSCoordSeq_getOrdinate_r(m_geoserr.ctx(), coords, i, 1, &y);
239  if (numInputDims > 2)
240  GEOSCoordSeq_getOrdinate_r(m_geoserr.ctx(), coords, i, 2, &z);
241  output.grow(x, y, z);
242  }
243  GEOSGeom_destroy_r(m_geoserr.ctx(), boundary);
244 
245  return output;
246 }
