61 // Parse dimension name:band number:scale factor
62 // Unsupplied band numbers start at 1. The default scale factor is 1.0
63 //
64 ColorizationFilter::BandInfo parseDim(const std::string& dim,
65  uint32_t defaultBand)
66 {
67  std::string::size_type pos, count;
68  const char *start;
69  char *end;
70  std::string name;
71  uint32_t band = defaultBand;
72  double scale = 1.0;
73 
74  pos = 0;
75  // Skip leading whitespace.
76  count = Utils::extractSpaces(dim, pos);
77  pos += count;
78 
79  count = Dimension::extractName(dim, pos);
80  if (count == 0)
81  throw std::string("No dimension name provided.");
82  name = dim.substr(pos, count);
83  pos += count;
84 
85  count = Utils::extractSpaces(dim, pos);
86  pos += count;
87 
88  if (pos < dim.size() && dim[pos] == ':')
89  {
90  pos++;
91  start = dim.data() + pos;
92  band = std::strtoul(start, &end, 10);
93  if (start == end)
94  band = defaultBand;
95  pos += (end - start);
96 
97  count = Utils::extractSpaces(dim, pos);
98  pos += count;
99 
100  if (pos < dim.size() && dim[pos] == ':')
101  {
102  pos++;
103  start = dim.data() + pos;
104  scale = std::strtod(start, &end);
105  if (start == end)
106  scale = 1.0;
107  pos += (end - start);
108  }
109  }
110 
111  count = Utils::extractSpaces(dim, pos);
112  pos += count;
113 
114  if (pos != dim.size())
115  {
116  std::ostringstream oss;
117  oss << "Invalid character '" << dim[pos] <<
118  "' following dimension specification.";
119  throw oss.str();
120  }
121  return ColorizationFilter::BandInfo(name, band, scale);
122 }

133 void ColorizationFilter::initialize()
134 {
135  if (m_dimSpec.empty())
136  m_dimSpec = { "Red", "Green", "Blue" };
137 
138  uint32_t defaultBand = 1;
139  for (std::string& dim : m_dimSpec)
140  {
141  try
142  {
143  BandInfo bi = parseDim(dim, defaultBand);
144  defaultBand = bi.m_band + 1;
145  m_bands.push_back(bi);
146  }
147  catch(const std::string& what)
148  {
149  throwError("invalid --dimensions option: '" + dim + "': " + what);
150  }
151  }
152 
153  gdal::registerDrivers();
154 }

188 bool ColorizationFilter::processOne(PointRef& point)
189 {
190  static std::vector<double> data;
191 
192  double x = point.getFieldAs<double>(Dimension::Id::X);
193  double y = point.getFieldAs<double>(Dimension::Id::Y);
194 
195  if (m_raster->read(x, y, data) == gdal::GDALError::None)
196  {
197  int i(0);
198  for (auto bi = m_bands.begin(); bi != m_bands.end(); ++bi)
199  {
200  BandInfo& b = *bi;
201  point.setField(b.m_dim, data[i] * b.m_scale);
202  ++i;
203  }
204  return true;
205  }
206  return false;
207 }
