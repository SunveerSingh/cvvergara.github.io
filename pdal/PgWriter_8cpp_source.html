111 void PgWriter::writeInit()
112 {
113  if (m_schema_is_initialized)
114  return;
115 
116  // Start up the database connection
117  pg_begin(m_session);
118 
119  // Pre-SQL can be *either* a SQL file to execute, *or* a SQL statement
120  // to execute. We find out which one here.
121  if (m_pre_sql.size())
122  {
123  std::string sql = FileUtils::readFileIntoString(m_pre_sql);
124  if (!sql.size())
125  {
126  // if there was no file to read because the data in pre_sql was
127  // actually the sql code the user wanted to run instead of the
128  // filename to open, we'll use that instead.
129  sql = m_pre_sql;
130  }
131  pg_execute(m_session, sql);
132  }
133 
134  bool bHaveTable = CheckTableExists(m_table_name);
135 
136  // Apply the over-write preference if it is set
137  if (m_overwrite && bHaveTable)
138  {
139  DeleteTable(m_schema_name, m_table_name);
140  bHaveTable = false;
141  }
142 
143  // Read or create a PCID for our new table
144  m_pcid = SetupSchema(m_srid);
145 
146  // Create the table!
147  if (! bHaveTable)
148  {
149  CreateTable(m_schema_name, m_table_name, m_column_name, m_pcid);
150  }
151 
152  m_schema_is_initialized = true;
153 }

184 uint32_t PgWriter::SetupSchema(uint32_t srid)
185 {
186 
187  // If the user has specified a PCID they want to use,
188  // does it exist in the database?
189  std::ostringstream oss;
190  long schema_count;
191  if (m_pcid)
192  {
193  oss << "SELECT Count(pcid) FROM pointcloud_formats WHERE pcid = " <<
194  m_pcid;
195  std::string count_str = pg_query_once(m_session, oss.str());
196  if (count_str.empty())
197  throwError("Unable to count pcid's in table `pointcloud_formats`");
198  schema_count = atoi(count_str.c_str());
199  if (schema_count == 0)
200  throwError("Requested PCID '" + Utils::toString(m_pcid) +
201  "' does not exist in POINTCLOUD_FORMATS");
202  return m_pcid;
203  }
204 
205  // Do we have any existing schemas in the POINTCLOUD_FORMATS table?
206  uint32_t pcid = 0;
207  oss.clear();
208  oss << "SELECT Count(pcid) FROM pointcloud_formats";
209  std::string schema_count_str = pg_query_once(m_session, oss.str());
210  if (schema_count_str.empty())
211  throwError("Unable to count pcid's in table 'pointcloud_formats'.");
212  schema_count = atoi(schema_count_str.c_str());
213  oss.str("");
214 
215  // Create an XML output schema.
216  std::string compression;
217  /* If the writer specifies a compression, we should set that */
218  if (m_patch_compression_type == CompressionType::Dimensional)
219  compression = "dimensional";
220  else if (m_patch_compression_type == CompressionType::Ght)
221  compression = "ght";
222 
223  Metadata metadata;
224  MetadataNode m = metadata.getNode();
225  m.add("compression", compression);
226 
227  XMLSchema schema(dbDimTypes(), m);
228  std::string xml = schema.xml();
229 
230  // Do any of the existing schemas match the one we want to use?
231  if (schema_count > 0)
232  {
233  PGresult *result = pg_query_result(m_session,
234  "SELECT pcid, schema FROM pointcloud_formats");
235  for (int i = 0; i < PQntuples(result); ++i)
236  {
237  char *pcid_str = PQgetvalue(result, i, 0);
238  char *schema_str = PQgetvalue(result, i, 1);
239 
240  if (xml == schema_str)
241  {
242  m_pcid = atoi(pcid_str);
243  PQclear(result);
244  return m_pcid;
245  }
246  }
247  PQclear(result);
248  }
249 
250  if (schema_count == 0)
251  {
252  // Try using the sequence
253  // pgpointcloud with the sequence is not yet
254  // released. See https://github.com/PDAL/PDAL/issues/1101 for
255  // SQL to create this sequence on the pointcloud_formats
256  // table.
257  std::string have_seq = pg_query_once(m_session,
258  "select count(*) from pg_class where relname = 'pointcloud_formats_pcid_sq'");
259  int seq_count = atoi(have_seq.c_str());
260  if (seq_count)
261  {
262  // We have the sequence, use its nextval
263  std::string pcid_str = pg_query_once(m_session,
264  "SELECT nextval('pointcloud_formats_pcid_sq')");
265  if (pcid_str.empty())
266  throwError("Unable to select nextval from "
267  "'pointcloud_formats_pcid_seq'.");
268  pcid = atoi(pcid_str.c_str());
269  }
270  else
271  {
272  // We don't have the sequence installed, all we can do is
273  // set to 1 and increment
274  pcid = 1;
275  }
276  }
277  else
278  {
279  std::string pcid_str = pg_query_once(m_session,
280  "SELECT Max(pcid)+1 AS pcid FROM pointcloud_formats");
281  if (pcid_str.empty())
282  throw("Unable to get the max pcid from 'pointcloud_formats'.");
283  pcid = atoi(pcid_str.c_str());
284  }
285 
286  const char* paramValues = xml.c_str();
287  oss << "INSERT INTO pointcloud_formats (pcid, srid, schema) "
288  "VALUES (" << pcid << "," << srid << ",$1)";
289  PGresult *result = PQexecParams(m_session, oss.str().c_str(), 1,
290  NULL, &paramValues, NULL, NULL, 0);
291  if (PQresultStatus(result) != PGRES_COMMAND_OK)
292  throwError(PQresultErrorMessage(result));
293  PQclear(result);
294  m_pcid = pcid;
295  return m_pcid;
296 }

354 bool PgWriter::CheckTableExists(std::string const& name)
355 {
356  std::ostringstream oss;
357  oss << "SELECT count(*) FROM pg_tables WHERE tablename ILIKE '" <<
358  name << "'";
359 
360  log()->get(LogLevel::Debug) << "checking for table '" << name <<
361  "' existence ... " << std::endl;
362 
363  std::string count_str = pg_query_once(m_session, oss.str());
364  if (count_str.empty())
365  throwError("Unable to check for the existence of 'pg_table'.");
366  int count = atoi(count_str.c_str());
367 
368  if (count == 1)
369  return true;
370  else if (count > 1)
371  log()->get(LogLevel::Debug) << "found more than 1 table named '" <<
372  name << "'" << std::endl;
373  return false;

411 void PgWriter::writeTile(const PointViewPtr view)
412 {
413  std::vector<char> storage(packedPointSize());
414  std::string hexrep;
415  size_t maxHexrepSize = packedPointSize() * view->size() * 2;
416  hexrep.reserve(maxHexrepSize);
417 
418  m_insert.clear();
419  m_insert.reserve(maxHexrepSize + 3000);
420 
421  for (PointId idx = 0; idx < view->size(); ++idx)
422  {
423  size_t size = readPoint(*view.get(), idx, storage.data());
424 
425  /* We are always getting uncompressed bytes off the block_data */
426  /* so we always used compression type 0 (uncompressed) in writing */
427  /* our WKB */
428  static char syms[] = "0123456789ABCDEF";
429  for (size_t i = 0; i != size; i++)
430  {
431  hexrep.push_back(syms[((storage[i] >> 4) & 0xf)]);
432  hexrep.push_back(syms[storage[i] & 0xf]);
433  }
434  }
435 
436  std::string insert_into("INSERT INTO ");
437  std::string values(" (" + pg_quote_identifier(m_column_name) +
438  ") VALUES ('");
439 
440  m_insert.append(insert_into);
441 
442  if (m_schema_name.size())
443  {
444  m_insert.append(pg_quote_identifier(m_schema_name));
445  m_insert.append(".");
446  }
447 
448  m_insert.append(pg_quote_identifier(m_table_name));
449  m_insert.append(values);
450 
451  std::ostringstream options;
452 
453  uint32_t num_points = htobe32(view->size());
454  int32_t pcid = htobe32(m_pcid);
455  CompressionType compression_v = CompressionType::None;
456  uint32_t compression = htobe32(static_cast<uint32_t>(compression_v));
457 
458 #if BYTE_ORDER == LITTLE_ENDIAN
459  options << "01";
460 #elif BYTE_ORDER == BIG_ENDIAN
461  options << "00";
462 #endif
463 
464  // needs to be 4 bytes
465  options << std::hex << std::setfill('0') << std::setw(8) << pcid;
466  // needs to be 4 bytes
467  options << std::hex << std::setfill('0') << std::setw(8) << compression;
468  // needs to be 4 bytes
469  options << std::hex << std::setfill('0') << std::setw(8) << num_points;
470 
471  m_insert.append(options.str());
472  m_insert.append(hexrep);
473  m_insert.append("')");
474 
475  pg_execute(m_session, m_insert);
476 }
