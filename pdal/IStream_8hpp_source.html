181  /**
182  Pop the current stream and return it. The last stream on the stack
183  cannot be popped.
184 
185  \return Pointer to the popped stream.
186  */
187  PDAL_DLL std::istream *popStream()
188  {
189  // Can't pop the last stream for now.
190  if (m_streams.empty())
191  return nullptr;
192  std::istream *strm = m_stream;
193  m_stream = m_streams.top();
194  m_streams.pop();
195  return strm;
196  }

198  /**
199  Fetch data from the stream into a string.
200 
201  \param s String to fill.
202  \param size Number of bytes to extract.
203  */
204  PDAL_DLL void get(std::string& s, size_t size)
205  {
206  // Could do this by appending to a string with a stream, but this
207  // is probably fast enough for now (there's only a simple increment
208  // to advance an istream iterator, which you'd have to call in a loop).
209  std::unique_ptr<char[]> buf(new char[size+1]);
210  m_stream->read(buf.get(), size);
211  buf[size] = '\0';
212  s = buf.get();
213  }
