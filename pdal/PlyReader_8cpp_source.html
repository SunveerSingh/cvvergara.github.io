59 std::string PlyReader::readLine()
60 {
61  m_line.clear();
62  if (m_lines.size())
63  {
64  m_line = m_lines.top();
65  m_lines.pop();
66  }
67  else
68  {
69  do
70  {
71  std::getline(*m_stream, m_line);
72  } while (m_line.empty() && m_stream->good());
73  }
74  Utils::trimTrailing(m_line);
75  m_linePos = Utils::extract(m_line, 0,
76  [](char c){ return !std::isspace(c); });
77  return std::string(m_line, 0, m_linePos);
78 }

88 {
89  std::string s;

91  m_linePos += cnt;

94

96  [](char c){ return !std::isspace(c); });
97  s = std::string(m_line, m_linePos, cnt);

99  return s;

124 void PlyReader::extractFormat()
125 {
126  std::string word = readLine();
127  if (word != "format")
128  throwError("Expected format line not found in PLY file.");
129 
130  word = nextWord();
131  if (word == "ascii")
132  m_format = Format::Ascii;
133  else if (word == "binary_big_endian")
134  m_format = Format::BinaryBe;
135  else if (word == "binary_little_endian")
136  m_format = Format::BinaryLe;
137  else
138  throwError("Unrecognized PLY format: '" + word + "'.");
139 
140  word = nextWord();
141  if (word != "1.0")
142  throwError("Unsupported PLY version: '" + word + "'.");
143 }

179 void PlyReader::extractProperty(Element& element)
180 {
181  std::string word = nextWord();
182  Dimension::Type type = getType(word);
183 
184  if (type != Dimension::Type::None)
185  {
186  std::string name = nextWord();
187  if (name.empty())
188  throwError("No name for property of element '" +
189  element.m_name + "'.");
190  element.m_properties.push_back(
191  std::unique_ptr<Property>(new SimpleProperty(name, type)));
192  }
193  else if (word == "list")
194  {
195  if (element.m_name == "vertex")
196  throwError("List properties are not supported for the 'vertex' "
197  "element.");
198 
199  word = nextWord();
200  Dimension::Type countType = getType(word);
201  if (countType == Dimension::Type::None)
202  throwError("No valid count type for list property of element '" +
203  element.m_name + "'.");
204  word = nextWord();
205  Dimension::Type listType = getType(word);
206  if (listType == Dimension::Type::None)
207  throwError("No valid list type for list property of element '" +
208  element.m_name + "'.");
209  std::string name = nextWord();
210  if (name.empty())
211  throwError("No name for property of element '" +
212  element.m_name + "'.");
213  element.m_properties.push_back(
214  std::unique_ptr<Property>(new ListProperty(name, countType,
215  listType)));
216  }
217  else
218  throwError("Invalid property type '" + word + "'.");
219 }

240 bool PlyReader::extractElement()
241 {
242  std::string word = readLine();
243  if (word == "comment" || word == "obj_info")
244  return true;
245  else if (word == "end_header")
246  {
247  pushLine();
248  return false;
249  }
250  else if (word == "element")
251  {
252  std::string name = nextWord();
253  if (name.empty())
254  throwError("Missing element name.");
255  long count = std::stol(nextWord());
256  if (count < 0)
257  throwError("Invalid count for element '" + name + "'.");
258  m_elements.emplace_back(name, count);
259  extractProperties(m_elements.back());
260  return true;
261  }
262  throwError("Invalid keyword '" + word + "' when expecting an element.");
263  return false; // quiet compiler
264 }

336 void PlyReader::SimpleProperty::read(std::istream *stream,
337  PlyReader::Format format, PointRef& point)
338 {
339  if (format == Format::Ascii)
340  {
341  double d;
342  *stream >> d;
343  point.setField(m_dim, d);
344  }
345  else if (format == Format::BinaryLe)
346  {
347  ILeStream in(stream);

349  point.setField(m_dim, m_type, &e);

362 void PlyReader::ListProperty::read(std::istream *stream,
363  PlyReader::Format format, PointRef& point)
364 {
365  if (format == Format::Ascii)
366  {
367  size_t cnt;
368  *stream >> cnt;
369 
370  double d;
371  while (cnt--)
372  *stream >> d;
373  }
374  else if (format == Format::BinaryLe)
375  {
376  ILeStream istream(stream);
377  Everything e = Utils::extractDim(istream, m_countType);
378  size_t cnt = (size_t)Utils::toDouble(e, m_countType);
379  cnt *= Dimension::size(m_listType);
380  istream.seek(cnt, std::ios_base::cur);

382  else if (format == Format::BinaryBe)

384  IBeStream istream(stream);

386  size_t cnt = (size_t)Utils::toDouble(e, m_countType);
387  cnt *= Dimension::size(m_listType);
388  istream.seek(cnt, std::ios_base::cur);

402 void PlyReader::ready(PointTableRef table)
403 {
404  m_stream = Utils::openFile(m_filename, true);
405  if (m_stream)
406  m_stream->seekg(m_dataPos);
407  for (Element& elt : m_elements)
408  {
409  if (&elt == m_vertexElt)
410  break;
411 
412  // We read an element into point 0. Since the element's properties
413  // weren't registered as dimensions, we'll try to write the data
414  // to a NULL dimension, which is a noop.
415  // This essentially just gets us to the vertex element.
416  // In binary mode, this is all silliness, since we should be able
417  // to seek right where we want to go, but in text mode, you've got
418  // to go through the data.
419  PointRef point(table, 0);
420  for (PointId idx = 0; idx < elt.m_count; ++idx)
421  readElement(elt, point);
422  }
423  m_index = 0;
424 }
