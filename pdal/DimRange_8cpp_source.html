42 std::string::size_type DimRange::subParse(const std::string& r)
43 {
44  bool& ilb(m_inclusive_lower_bound);
45  bool& iub(m_inclusive_upper_bound);
46  bool& negate(m_negate);
47  double& ub(m_upper_bound);
48  double& lb(m_lower_bound);
49  std::string& name(m_name);
50 
51  std::string::size_type pos, count;
52  const char *start;
53  char *end;
54 
55  ilb = true;
56  iub = true;
57  negate = false;
58  pos = 0;
59  // Skip leading whitespace.
60  count = Utils::extractSpaces(r, pos);
61  pos += count;
62 
63  count = Dimension::extractName(r, pos);
64  if (count == 0)
65  throw error("No dimension name.");
66  name = r.substr(pos, count);
67  pos += count;
68 
69  if (r[pos] == '!')
70  {
71  negate = true;
72  pos++;
73  }
74 
75  if (r[pos] == '(')
76  ilb = false;
77  else if (r[pos] != '[')
78  throw error("Missing '(' or '['.");
79  pos++;
80 
81  // Extract lower bound.
82  start = r.data() + pos;
83  lb = std::strtod(start, &end);
84  if (start == end)
85  lb = std::numeric_limits<double>::lowest();
86  pos += (end - start);
87 
88  count = Utils::extractSpaces(r, pos);
89  pos += count;
90 
91  if (r[pos] != ':')
92  throw error("Missing ':' limit separator.");
93  pos++;
94 
95  start = r.data() + pos;
96  ub = std::strtod(start, &end);
97  if (start == end)
98  ub = std::numeric_limits<double>::max();
99  pos += (end - start);
100 
101  count = Utils::extractSpaces(r, pos);
102  pos += count;
103 
104  if (r[pos] == ')')
105  iub = false;
106  else if (r[pos] != ']')
107  throw error("Missing ')' or ']'.");
108  pos++;
109 
110  count = Utils::extractSpaces(r, pos);
111  pos += count;
112  return pos;
113 }

116 bool DimRange::valuePasses(double v) const
117 {
118  // Determine if a point passes a range.
119  bool fail = std::isnan(v) ||
120  ((m_inclusive_lower_bound && v < m_lower_bound) ||
121  (!m_inclusive_lower_bound && v <= m_lower_bound) ||
122  (m_inclusive_upper_bound && v > m_upper_bound) ||
123  (!m_inclusive_upper_bound && v >= m_upper_bound));
124  if (m_negate)
125  fail = !fail;
126  return !fail;
127 }
