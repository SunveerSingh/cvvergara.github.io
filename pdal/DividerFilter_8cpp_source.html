109 PointViewSet DividerFilter::run(PointViewPtr inView)
110 {
111  if (m_sizeMode == SizeMode::Capacity)
112  m_size = ((inView->size() - 1) / m_size) + 1;
113 
114  PointViewSet result;
115  std::vector<PointViewPtr> views;
116  for (point_count_t i = 0; i < m_size; ++i)
117  {
118  PointViewPtr v(inView->makeNew());
119  views.push_back(v);
120  result.insert(v);
121  }
122 
123  if (m_mode == Mode::Partition)
124  {
125  point_count_t limit = ((inView->size() - 1) / m_size) + 1;
126  unsigned viewNum = 0;
127  for (PointId i = 0; i < inView->size();)
128  {
129  views[viewNum]->appendPoint(*inView, i++);
130  if (i % limit == 0)
131  viewNum++;
132  }
133  }
134  else // RoundRobin
135  {
136  unsigned viewNum = 0;
137  for (PointId i = 0; i < inView->size(); ++i)
138  {
139  views[viewNum]->appendPoint(*inView, i);
140  viewNum++;
141  if (viewNum == m_size)
142  viewNum = 0;
143  }
144  }
145  return result;
146 }
