62 inline int32_t cksum(const void *c, size_t size)
63 {
64  int32_t sum = 0;
65 
66  const int32_t *p = static_cast<const int32_t *>(c);
67  while (size)
68  {
69  int32_t val = 0;
70  memcpy(&val, p, std::min(sizeof(int32_t), size));
71  sum += val;
72  p++;
73  if (size <= sizeof(int32_t))
74  break;
75  size -= sizeof(int32_t);
76  }
77  return -sum;
78 }
