95 void NormalFilter::filter(PointView& view)
96 {
97  KD3Index& kdi = view.build3dIndex();
98 
99  for (PointId i = 0; i < view.size(); ++i)
100  {
101  // find the k-nearest neighbors
102  auto ids = kdi.neighbors(i, m_knn);
103 
104  // compute covariance of the neighborhood
105  auto B = eigen::computeCovariance(view, ids);
106 
107  // perform the eigen decomposition
108  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3f> solver(B);
109  if (solver.info() != Eigen::Success)
110  throwError("Cannot perform eigen decomposition.");
111  auto eval = solver.eigenvalues();
112  Eigen::Vector3f normal = solver.eigenvectors().col(0);
113 
114  if (m_viewpointArg->set())
115  {
116  PointRef p = view.point(i);
117  Eigen::Vector3f vp(
118  m_viewpoint.x - p.getFieldAs<double>(Dimension::Id::X),
119  m_viewpoint.y - p.getFieldAs<double>(Dimension::Id::Y),
120  m_viewpoint.z - p.getFieldAs<double>(Dimension::Id::Z));
121  if (vp.dot(normal) < 0)
122  normal *= -1.0;
123  }
124  else if (m_up)
125  {
126  if (normal[2] < 0)
127  normal *= -1.0;
128  }
129 
130  view.setField(Dimension::Id::NormalX, i, normal[0]);
131  view.setField(Dimension::Id::NormalY, i, normal[1]);
132  view.setField(Dimension::Id::NormalZ, i, normal[2]);
133 
134  double sum = eval[0] + eval[1] + eval[2];
135  view.setField(Dimension::Id::Curvature, i,
136  sum ? std::fabs(eval[0] / sum) : 0);
137  }
138 }
