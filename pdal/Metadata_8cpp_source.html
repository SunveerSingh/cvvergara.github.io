49 std::string Metadata::inferType(const std::string& val)
50 {
51  size_t pos;
52 
53  long l = 0;
54  try
55  {
56  pos = 0;
57  l = std::stol(val, &pos);
58  }
59  catch (std::invalid_argument)
60  {}
61  if (pos == val.length())
62  return (l < 0 ? "nonNegativeInteger" : "integer");
63 
64  try
65  {
66  pos = 0;
67  std::stod(val, &pos);
68  }
69  catch(std::invalid_argument)
70  {}
71 
72  if (pos == val.length())
73  return "double";
74 
75  BOX2D b2d;
76  std::istringstream iss1(val);
77  iss1 >> b2d;
78  if (iss1.good())
79  return "bounds";
80 
81  BOX3D b3d;

83  iss2 >> b3d;
84  if (iss2.good())
85  return "bounds";
86 
87  if (val == "true" || val == "false")
88  return "boolean";
89 
90  try
91  {
92  SpatialReference s(val);
93  return "spatialreference";
94  }
95  catch (pdal_error&)
96  {
97  }
98 
99  Uuid uuid(val);
100  if (!uuid.isNull())
101  return "uuid";
102 
103  return "string";
104 }
