285 // Orders Bounds by their midpoint. This is really only useful if the bounds
286 // are arranged in a grid and are of equal size (like during a MetaQuery).
287 PDAL_DLL inline bool operator<(const Bounds& lhs, const Bounds& rhs)
288 {
289  const auto& lhsMid(lhs.mid());
290  const auto& rhsMid(rhs.mid());
291 
292  return
293  (lhsMid.x < rhsMid.x) ||
294  (lhsMid.x == rhsMid.x && lhsMid.y < rhsMid.y) ||
295  (lhsMid.x == rhsMid.x && lhsMid.y == rhsMid.y && lhsMid.z < rhsMid.z);
296 }
