81 void checkPointsEqualReader(const PointViewSet& pointViewSet, double tolerance)
82 {
83  auto reader = newReader();
84  PointTable table;
85  reader->prepare(table);
86  PointViewSet readerPointViewSet = reader->execute(table);
87  ASSERT_EQ(1u, readerPointViewSet.size());
88  PointViewPtr expected = *readerPointViewSet.begin();
89  ASSERT_EQ(1u, pointViewSet.size());
90  PointViewPtr actual = *pointViewSet.begin();
91 
92  ASSERT_EQ(expected->size(), actual->size());
93 
94  for (PointId i = 0; i < expected->size(); ++i)
95  {
96  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::X, i),
97  actual->getFieldAs<double>(Dimension::Id::X, i), tolerance);
98  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::Y, i),
99  actual->getFieldAs<double>(Dimension::Id::Y, i), tolerance);
100  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::Z, i),
101  actual->getFieldAs<double>(Dimension::Id::Z, i), tolerance);
102  }
103 }
104 }
