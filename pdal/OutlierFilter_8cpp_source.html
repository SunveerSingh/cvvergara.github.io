75 Indices OutlierFilter::processRadius(PointViewPtr inView)
76 {
77  KD3Index index(*inView);
78  index.build();
79 
80  point_count_t np = inView->size();
81 
82  std::vector<PointId> inliers, outliers;
83 
84  for (PointId i = 0; i < np; ++i)
85  {
86  auto ids = index.radius(i, m_radius);
87  if (ids.size() > size_t(m_minK))
88  inliers.push_back(i);
89  else
90  outliers.push_back(i);
91  }
92 
93  return Indices{inliers, outliers};

96 Indices OutlierFilter::processStatistical(PointViewPtr inView)
97 {
98  KD3Index index(*inView);
99  index.build();
100 
101  point_count_t np = inView->size();
102 
103  std::vector<PointId> inliers, outliers;
104 
105  std::vector<double> distances(np, 0.0);
106  for (PointId i = 0; i < np; ++i)
107  {
108  // we increase the count by one because the query point itself will
109  // be included with a distance of 0
110  point_count_t count = m_meanK + 1;
111 
112  std::vector<PointId> indices(count);
113  std::vector<double> sqr_dists(count);
114  index.knnSearch(i, count, &indices, &sqr_dists);
115 
116  for (size_t j = 1; j < count; ++j)
117  {
118  double delta = std::sqrt(sqr_dists[j]) - distances[i];
119  distances[i] += (delta / j);
120  }
121  }
122 
123  size_t n(0);
124  double M1(0.0);
125  double M2(0.0);
126  for (auto const& d : distances)
127  {
128  size_t n1(n);
129  n++;
130  double delta = d - M1;
131  double delta_n = delta / n;
132  M1 += delta_n;
133  M2 += delta * delta_n * n1;
134  }
135  double mean = M1;
136  double variance = M2 / (n - 1.0);
137  double stdev = std::sqrt(variance);
138 
139  double threshold = mean + m_multiplier * stdev;
140 
141  for (PointId i = 0; i < np; ++i)
142  {
143  if (distances[i] < threshold)
144  inliers.push_back(i);
145  else
146  outliers.push_back(i);
147  }
148 
149  return Indices{inliers, outliers};
150 }

152 PointViewSet OutlierFilter::run(PointViewPtr inView)
153 {
154  PointViewSet viewSet;
155  if (!inView->size())
156  return viewSet;
157 
158  Indices indices;
159  if (Utils::iequals(m_method, "statistical"))
160  {
161  indices = processStatistical(inView);
162  }
163  else if (Utils::iequals(m_method, "radius"))
164  {
165  indices = processRadius(inView);
166  }
167  else
168  {
169  log()->get(LogLevel::Warning) << "Requested method is unrecognized. "
170  "Please choose from \"statistical\" "
171  "or \"radius\".\n";
172  viewSet.insert(inView);
173  return viewSet;
174  }
175 
176  if (indices.inliers.empty())
177  {
178  log()->get(LogLevel::Warning) << "Requested filter would remove all "
179  "points. Try a larger radius/smaller "
180  "minimum neighbors.\n";
181  viewSet.insert(inView);
182  return viewSet;
183  }
184 
185  if (!indices.outliers.empty())
186  {
187  log()->get(LogLevel::Debug2)
188  << "Labeled " << indices.outliers.size() << " outliers as noise!\n";
189 
190  // set the classification label of outlier returns
191  for (const auto& i : indices.outliers)
192  inView->setField(Dimension::Id::Classification, i, m_class);
193 
194  viewSet.insert(inView);
195  }
196  else
197  {
198  if (indices.outliers.empty())
199  log()->get(LogLevel::Warning)
200  << "Filtered cloud has no outliers!\n";
201 
202  // return the input buffer unchanged
203  viewSet.insert(inView);
204  }
205 
206  return viewSet;
207 }
