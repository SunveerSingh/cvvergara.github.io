63 template <typename T, T MIN = std::numeric_limits<T>::lowest(),
64  T MAX = std::numeric_limits<T>::max()>

76  bool setVal(T val)
77  {
78 #pragma GCC diagnostic push
79 #pragma GCC diagnostic ignored "-Wtype-limits"
80  if (val >= MIN && val <= MAX)
81  {
82  this->m_val = val;

84  return true;
85  }
86  return false;
87 #pragma GCC diagnostic pop
88  }
