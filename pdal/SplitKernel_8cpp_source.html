99 int SplitKernel::execute()
100 {
101  PointTable table;
102 
103  Stage& reader = makeReader(m_inputFile, m_driverOverride);
104 
105  Options filterOpts;
106  std::string driver = (m_length ? "filters.splitter" : "filters.chipper");
107  if (m_length)
108  {
109  filterOpts.add("length", m_length);
110  filterOpts.add("origin_x", m_xOrigin);
111  filterOpts.add("origin_y", m_yOrigin);
112  }
113  else
114  {
115  filterOpts.add("capacity", m_capacity);
116  }
117  Stage& f = makeFilter(driver, reader, filterOpts);
118  f.prepare(table);
119  PointViewSet pvSet = f.execute(table);
120 
121  int filenum = 1;
122  for (auto& pvp : pvSet)
123  {
124  BufferReader reader;
125  reader.addView(pvp);
126 
127  std::string filename = makeFilename(m_outputFile, filenum++);
128  Stage& writer = makeWriter(filename, reader, "");
129 
130  writer.prepare(table);
131  writer.execute(table);
132  }
133  return 0;
134 }
