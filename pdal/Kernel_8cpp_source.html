65 bool Kernel::parseStageOption(std::string o, std::string& stage,
66  std::string& option, std::string& value)
67 {
68  value.clear();
69  if (o.size() < 2)
70  return false;
71  if (o[0] != '-' || o[1] != '-')
72  return false;
73 
74  o = o.substr(2);
75 
76  // Options are stage_type.stage_name.option_name
77  // stage_type is always lowercase stage_names start with lowercase and
78  // then are lowercase or digits. Option names start with lowercase and
79  // then contain lowercase, digits or underscore.
80 
81  // This awfulness is to work around the multiply-defined islower. Seems
82  // a bit better than the cast solution.
83  auto islc = [](char c)
84  { return std::islower(c); };
85 
86  std::string::size_type pos = 0;
87  std::string::size_type count = 0;
88 
89  // Get stage_type.
90  count = Utils::extract(o, pos, islc);
91  pos += count;
92  std::string stageType = o.substr(0, pos);
93  if (!isStagePrefix(stageType))
94  return false;
95  if (pos >= o.length() || o[pos++] != '.')
96  return false;
97 
98  // Get stage_name.
99  bool ok;
100  if (stageType == "stage")
101  ok = Stage::parseTagName(o, pos);
102  else
103  ok = Stage::parseName(o, pos);
104  if (!ok)
105  return false;
106  stage = o.substr(0, pos);
107  if (pos >= o.length() || o[pos++] != '.')
108  return false;
109 
110  // Get option name.
111  std::string::size_type optionStart = pos;
112  count = Option::parse(o, pos);
113  pos += count;
114  option = o.substr(optionStart, count);
115 
116  // We've gotten a good option name, so return true, even if the value
117  // is missing. The caller can handle the missing value if desired.
118  if (pos >= o.length() || o[pos++] != '=')
119  return true;
120 
121  // The command-line parser takes care of quotes around an argument
122  // value and such. May want to do something to handle escaped characters?
123  value = o.substr(pos);
124  return true;
125 }

135 void Kernel::doSwitches(const StringList& cmdArgs, ProgramArgs& args)
136 {
137  OptionsMap& stageOptions = m_manager.stageOptions();
138  StringList stringArgs;
139 
140  // Scan the argument vector for extra stage options. Pull them out and
141  // stick them in the list. Let the ProgramArgs handle everything else.
142  // NOTE: This depends on the format being "option=value" rather than
143  // "option value". This is what we've always expected, so no problem,
144  // but it would be better to be more flexible.
145  for (size_t i = 0; i < cmdArgs.size(); ++i)
146  {
147  std::string stageName, opName, value;
148 
149  if (parseStageOption(cmdArgs[i], stageName, opName, value))
150  {
151  if (value.empty())
152  {
153  std::ostringstream oss;
154  oss << "Stage option '" << stageName << "." << opName <<
155  "' must be specified " << " as --" << stageName << "." <<
156  opName << "=<value>" << ".";
157  throw pdal_error(oss.str());
158  }
159  Option op(opName, value);
160  stageOptions[stageName].add(op);
161  }
162  else
163  stringArgs.push_back(cmdArgs[i]);
164  }
165 
166  try
167  {
168  // parseSimple allows us to scan for the help option without
169  // raising exception about missing arguments and so on.
170  // It also removes consumed args from the arg list, so for now,
171  // parse a copy that will be ignored by parse().
172  ProgramArgs hargs;
173  hargs.add("help,h", "Print help message", m_showHelp);
174  hargs.parseSimple(stringArgs);
175 
176  addBasicSwitches(args);
177  addSwitches(args);
178  if (!m_showHelp)
179  args.parse(stringArgs);
180  }
181  catch (arg_error& e)
182  {
183  throw pdal_error(getName() + ": " + e.m_error);
184  }
185 }
