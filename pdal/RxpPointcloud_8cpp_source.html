82 point_count_t RxpPointcloud::read(PointViewPtr view, point_count_t count)
83 {
84  point_count_t numRead = 0;
85 
86  if (m_idx < m_view->size())
87  {
88  numRead += writeSavedPoints(view, count);
89  if (numRead == count)
90  return numRead;
91  }
92 
93  for (m_dec.get(m_rxpbuf); !m_dec.eoi(); m_dec.get(m_rxpbuf))
94  {
95  dispatch(m_rxpbuf.begin(), m_rxpbuf.end());
96  if (m_view->size() - m_idx + numRead >= count) break;
97  }
98 
99  numRead += writeSavedPoints(view, count - numRead);
100 
101  return numRead;
102 }

105 point_count_t RxpPointcloud::writeSavedPoints(PointViewPtr view, point_count_t count)
106 {
107  point_count_t numRead = 0;
108  while (m_idx < m_view->size() && numRead < count)
109  {
110  view->appendPoint(*m_view, m_idx);
111  ++m_idx, ++numRead;
112  }
113  return numRead;

117 void RxpPointcloud::on_echo_transformed(echo_type echo)
118 {
119  if (!(scanlib::pointcloud::single == echo || scanlib::pointcloud::last == echo))
120  {
121  // Come back later, when we've got all the echos
122  return;
123  }
124 
125  using namespace Dimension;
126 
127  point_count_t idx = m_view->size();
128  unsigned int returnNumber = 1;
129  Id timeId = getTimeDimensionId(m_syncToPps);
130  for (const auto& t : targets)
131  {
132  m_view->setField(Id::X, idx, t.vertex[0]);
133  m_view->setField(Id::Y, idx, t.vertex[1]);
134  m_view->setField(Id::Z, idx, t.vertex[2]);
135  m_view->setField(timeId, idx, t.time);
136  if (!m_minimal)
137  {
138  m_view->setField(Id::Amplitude, idx, t.amplitude);
139  m_view->setField(Id::Reflectance, idx, t.reflectance);
140  m_view->setField(Id::ReturnNumber, idx, returnNumber);
141  m_view->setField(Id::NumberOfReturns, idx, targets.size());
142  m_view->setField(Id::EchoRange, idx, t.echo_range);
143  m_view->setField(Id::Deviation, idx, t.deviation);
144  m_view->setField(Id::BackgroundRadiation, idx, t.background_radiation);
145  m_view->setField(Id::IsPpsLocked, idx, t.is_pps_locked);
146  }
147  if (m_reflectanceAsIntensity) {
148  uint16_t intensity;
149  if (t.reflectance > m_maxReflectance) {
150  intensity = std::numeric_limits<uint16_t>::max();
151  } else if (t.reflectance < m_minReflectance) {
152  intensity = 0;
153  } else {
154  intensity = uint16_t(std::roundf(double(std::numeric_limits<uint16_t>::max()) *
155  (t.reflectance - m_minReflectance) / (m_maxReflectance - m_minReflectance)));
156  }
157  m_view->setField(Id::Intensity, idx, intensity);
158  }
159  ++idx, ++returnNumber;
160  }
161 }
