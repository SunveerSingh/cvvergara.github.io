78  double d;

87 point_count_t SbetReader::read(PointViewPtr view, point_count_t count)
88 {
89  PointId nextId = view->size();
90  PointId idx = m_index;
91  point_count_t numRead = 0;
92  seek(idx);
93  while (numRead < count && idx < m_numPts)
94  {
95  PointRef point = view->point(nextId);
96  processOne(point);
97  if (m_cb)
98  m_cb(*view, nextId);
99 
100  idx++;
101  nextId++;
102  numRead++;
103  }
104  m_index = idx;
105  return numRead;
106 }
