146 std::istream& operator>>(std::istream& in, MbFormat& f)
147 {
148  std::string s;
149 
150  f.m_value = 0;
151  in >> s;
152  try
153  {
154  int val = stoi(s);
155  for (MbFormatInfo& fi : formats)
156  if (val == fi.second)
157  {
158  f.m_value = val;
159  break;
160  }
161  }
162  catch (std::exception)
163  {
164  s = Utils::toupper(s);
165  for (MbFormatInfo& fi : formats)
166  if (s == fi.first)
167  {
168  f.m_value = fi.second;
169  break;
170  }
171  }
172  if (f.m_value == 0)
173  in.setstate(std::ios_base::failbit);
174  return in;
175 }
