50 void BasePointTable::addSpatialReference(const SpatialReference& spatialRef)
51 {
52  auto it = std::find(m_spatialRefs.begin(), m_spatialRefs.end(), spatialRef);
53 
54  // If not found, add to the beginning.
55  if (it == m_spatialRefs.end())
56  m_spatialRefs.push_front(spatialRef);
57  // If not the first element, move the found element to the front.
58  else if (it != m_spatialRefs.begin())
59  m_spatialRefs.splice(m_spatialRefs.begin(), m_spatialRefs, it);
60 }
