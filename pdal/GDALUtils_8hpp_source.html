314 /*
315  Slight abstraction of a GDAL raster band.
316 */
317 template<typename T>
318 class Band
319 {
320 friend class Raster;
321 
322 private:
323  GDALDataset *m_ds; /// Dataset handle
324  int m_bandNum; /// Band number. Band numbers start at 1.
325  double m_dstNoData; /// Output no data value.
326  GDALRasterBand *m_band; /// Band handle
327  int m_xTotalSize, m_yTotalSize; /// Total size (x and y) of the raster
328  int m_xBlockSize, m_yBlockSize; /// Size (x and y) of blocks
329  int m_xBlockCnt, m_yBlockCnt; /// Number of blocks in each direction
330  std::vector<T> m_buf; /// Block read buffer.
331  std::string m_name; /// Band name.
332 
333  /**
334  Create an object for reading a band of a GDAL dataset.
335 
336  \param ds GDAL dataset handle.
337  \param dstNoData The no data value to be used when writing the band.
338  \param bandNum Band number (1-indexed).
339  \param name Name of the raster band.
340  */
341  Band(GDALDataset *ds, int bandNum, double dstNoData = -9999.0,
342  const std::string& name = "") :
343  m_ds(ds), m_bandNum(bandNum), m_dstNoData(dstNoData),
344  m_xBlockSize(0), m_yBlockSize(0)
345  {
346  m_band = m_ds->GetRasterBand(m_bandNum);
347  if (!m_band)
348  throw InvalidBand();
349 
350  if (name.size())
351  {
352  m_band->SetDescription(name.data());
353  // We don't care about offset, but this sets the flag to indicate
354  // that the metadata has changed.
355  m_band->SetOffset(m_band->GetOffset(NULL) + .00001);
356  m_band->SetOffset(m_band->GetOffset(NULL) - .00001);
357  }
358 
359  m_xTotalSize = m_band->GetXSize();
360  m_yTotalSize = m_band->GetYSize();
361 
362  m_band->GetBlockSize(&m_xBlockSize, &m_yBlockSize);
363  m_buf.resize(m_xBlockSize * m_yBlockSize);
364 
365  m_xBlockCnt = ((m_xTotalSize - 1) / m_xBlockSize) + 1;
366  m_yBlockCnt = ((m_yTotalSize - 1) / m_yBlockSize) + 1;
367  }
368 
369  /*
370  Read the band into the vector. Reads a block at a time. Each
371  block is either fully populated with data or a partial block.
372  Partial blocks appear at the X and Y margins when the total size in
373  the doesn't divide evenly by the block size for both the X and Y
374  dimensions.
375 
376  \param Data Vector into which the data should be read. The vector is
377  resized as necessary.
378  */
379  void read(std::vector<T>& data)
380  {
381  data.resize(m_xTotalSize * m_yTotalSize);
382 
383  for (int y = 0; y < m_yBlockCnt; ++y)
384  for (int x = 0; x < m_xBlockCnt; ++x)
385  readBlock(x, y, data);
386  }
387 
388  /*
389  Read a block's worth of data.
390 
391  Read data into a block-sized buffer. Then copy data from the
392  block buffer into the destination array at the proper location to
393  build a complete raster.
394 
395  \param x X coordinate of the block to read.
396  \param y Y coordinate of the block to read.
397  \param data Pointer to the data vector that contains the
398  raster information.
399  */
400  void readBlock(int x, int y, std::vector<T>& data)
401  {
402  uint8_t *buf = reinterpret_cast<uint8_t *>(m_buf.data());
403  if (m_band->ReadBlock(x, y, buf) != CPLE_None)
404  throw CantReadBlock();
405 
406  int xWidth = 0;
407  if (x == m_xBlockCnt - 1)
408  xWidth = m_xTotalSize % m_xBlockSize;
409  if (xWidth == 0)
410  xWidth = m_xBlockSize;
411 
412  int yHeight = 0;
413  if (y == m_yBlockCnt - 1)
414  yHeight = m_yTotalSize % m_yBlockSize;
415  if (yHeight == 0)
416  yHeight = m_yBlockSize;
417 
418  auto bi = m_buf.begin();
419  // Go through rows copying data. Increment the buffer pointer by the
420  // width of the row.
421  for (int row = 0; row < yHeight; ++row)
422  {
423  int wholeRows = m_xTotalSize * ((y * m_yBlockSize) + row);
424  int partialRows = m_xBlockSize * x;
425  auto di = data.begin() + (wholeRows + partialRows);
426  std::copy(bi, bi + xWidth, di);
427 
428  // Blocks are always full-sized, even if only some of the data
429  // is valid, so we use m_xBlockSize instead of xWidth.
430  bi += m_xBlockSize;
431  }
432  }
433 
434  /*
435  Write linearized data pointed to by \c data into the band.
436 
437  \param data Pointer to beginning of band
438  */
439  template <typename SOURCE_ITER>
440  void write(SOURCE_ITER si, ITER_VAL<SOURCE_ITER> srcNoData)
441  {
442  for (int y = 0; y < m_yBlockCnt; ++y)
443  for (int x = 0; x < m_xBlockCnt; ++x)
444  writeBlock(x, y, si, srcNoData);
445  }
446 
447  T getNoData() const
448  {
449  // The destination nodata value was set when the raster was opened.
450  // Make sure it's valid for the band type and convert.
451  T t;
452  if (!Utils::numericCast(m_dstNoData, t))
453  {
454  throw CantWriteBlock("Invalid nodata value " +
455  Utils::toString(m_dstNoData) + " for output data_type '" +
456  Utils::typeidName<T>() + "'.");
457  }
458  return t;
459  }
460 
461  template <typename SOURCE_ITER>
462  void writeBlock(int x, int y, SOURCE_ITER sourceBegin,
463  ITER_VAL<SOURCE_ITER> srcNoData)
464  {
465  int xWidth = 0;
466  if (x == m_xBlockCnt - 1)
467  xWidth = m_xTotalSize % m_xBlockSize;
468  if (xWidth == 0)
469  xWidth = m_xBlockSize;
470 
471  int yHeight = 0;
472  if (y == m_yBlockCnt - 1)
473  yHeight = m_yTotalSize % m_yBlockSize;
474  if (yHeight == 0)
475  yHeight = m_yBlockSize;
476 
477  T dstNoData = getNoData();
478  auto di = m_buf.begin();
479  // Go through rows copying data. Increment the destination iterator
480  // by the width of the row.
481  for (int row = 0; row < yHeight; ++row)
482  {
483  // Find the offset location in the source container.
484  int wholeRowElts = m_xTotalSize * ((y * m_yBlockSize) + row);
485  int partialRowElts = m_xBlockSize * x;
486 
487  auto si = sourceBegin + (wholeRowElts + partialRowElts);
488  std::transform(si, si + m_xBlockSize, di,
489  [srcNoData, dstNoData](ITER_VAL<SOURCE_ITER> s){
490  T t;
491 
492  if (srcNoData == s ||
493  (std::isnan(srcNoData) && std::isnan(s)))
494  t = dstNoData;
495  else
496  {
497  if (!Utils::numericCast(s, t))
498  {
499  throw CantWriteBlock("Unable to convert data for "
500  "raster type as requested: " + Utils::toString(s) +
501  " -> " + Utils::typeidName<T>());
502  }
503  }
504  return t;
505  });
506 
507  // Blocks are always full-sized, even if only some of the data
508  // is valid, so we use m_xBlockSize instead of xWidth.
509  di += m_xBlockSize;
510  }
511  if (m_band->WriteBlock(x, y, m_buf.data()) != CPLE_None)
512  throw CantWriteBlock();
513  }
514 };

568  /**

602  /**
603  Write an entire raster band (layer) into raster to be written with GDAL.
604 
605  \param data Linearized raster data to be written.
606  \param noData No-data value in the source data.
607  \param nBand Band number to write.
608  \param name Name of the raster band.
609  */
610  template<typename SOURCE_ITER>
611  GDALError writeBand(SOURCE_ITER si, ITER_VAL<SOURCE_ITER> srcNoData,
612  int nBand, const std::string& name = "")
613  {
614  try
615  {
616  switch(m_bandType)
617  {
618  case Dimension::Type::Unsigned8:
619  Band<uint8_t>(m_ds, nBand, m_dstNoData, name).
620  write(si, srcNoData);
621  break;
622  case Dimension::Type::Signed8:
623  Band<int8_t>(m_ds, nBand, m_dstNoData, name).
624  write(si, srcNoData);
625  break;
626  case Dimension::Type::Unsigned16:
627  Band<uint16_t>(m_ds, nBand, m_dstNoData, name).
628  write(si, srcNoData);
629  break;
630  case Dimension::Type::Signed16:
631  Band<int16_t>(m_ds, nBand, m_dstNoData, name).
632  write(si, srcNoData);
633  break;
634  case Dimension::Type::Unsigned32:
635  Band<uint32_t>(m_ds, nBand, m_dstNoData, name).
636  write(si, srcNoData);
637  break;
638  case Dimension::Type::Signed32:
639  Band<int32_t>(m_ds, nBand, m_dstNoData, name).
640  write(si, srcNoData);
641  break;
642  case Dimension::Type::Unsigned64:
643  Band<uint64_t>(m_ds, nBand, m_dstNoData, name).
644  write(si, srcNoData);
645  break;
646  case Dimension::Type::Signed64:
647  Band<int64_t>(m_ds, nBand, m_dstNoData, name).
648  write(si, srcNoData);
649  break;
650  case Dimension::Type::Float:
651  Band<float>(m_ds, nBand, m_dstNoData, name).
652  write(si, srcNoData);
653  break;
654  case Dimension::Type::Double:
655  Band<double>(m_ds, nBand, m_dstNoData, name).
656  write(si, srcNoData);
657  break;
658  case Dimension::Type::None:
659  throw CantWriteBlock();
660  }
661  }
662  catch (CantWriteBlock err)
663  {
664  std::ostringstream oss;
665  oss << "Unable to write block for for raster '" << m_filename <<
666  "'.";
667  if (err.what.size())
668  oss << "\n" << err.what;
669  m_errorMsg = oss.str();
670  return GDALError::CantWriteBlock;
671  }
672  return GDALError::None;
673  }
