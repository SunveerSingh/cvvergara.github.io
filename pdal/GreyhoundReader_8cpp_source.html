100 std::string GreyhoundReader::Args::base() const
101 {
102  // The url may be of the form:
103  // server.com
104  // server.com/resource/asdf/read?<query>
105  //
106  // We want to extract, from either form (resource may be specified
107  // separately):
108  // server.com/resource/asdf/
109 
110  // Strip off query string.
111  std::string s(url.substr(0, url.find('?')));
112 
113  if (std::regex_match(s, std::regex(".+/resource/.+/read$")))
114  {
115  // If the URL is a fully-specified READ query, strip off the "read".
116  if (resource.size())
117  throw pdal_error("Cannot specify resource twice");
118 
119  s = s.substr(0, s.size() - 4);
120  }
121 
122  if (s.empty())
123  throw pdal_error("No server specified");
124 
125  if (resource.size())
126  {
127  // Otherwise, if resource is specified separately, add it now.
128  s = s + (s.back() == '/' ? "" : "/") + "resource/" + resource;
129  }
130 
131  // Make sure we end up with a trailing slash.
132  if (s.back() != '/')
133  s = s + '/';
134 
135  // Finally, ensure an HTTP prefix so arbiter interprets it correctly.
136  if (s.find("http://") == 0 || s.find("https://") == 0)
137  return s;
138  else return "http://" + s;
139 }

141 std::string GreyhoundReader::Args::query() const
142 {
143  const std::size_t q(url.find('?'));
144  std::string s(url.substr(q != std::string::npos ? q : url.size()));
145 
146  auto add([&s](std::string k, std::string v)
147  {
148  s += (s.size() ? '&' : '?') + k + '=' + v;
149  });
150 
151  if (sbounds.size())
152  {
153  greyhound::Bounds gbounds;
154  if (sbounds.find('(') != std::string::npos)
155  {
156  // This is a PDAL-specified bounds.
157  Bounds pdalBounds;
158  std::istringstream iss(sbounds);
159  iss >> pdalBounds;
160  if (pdalBounds.is3d())
161  {
162  const auto box(pdalBounds.to3d());
163  gbounds = greyhound::Bounds(box.minx, box.miny, box.minz,
164  box.maxx, box.maxy, box.maxz);
165  }
166  else
167  {
168  const auto box(pdalBounds.to2d());
169  gbounds = greyhound::Bounds(box.minx, box.miny,
170  box.maxx, box.maxy);
171  }
172  }
173  else
174  {
175  gbounds = greyhound::Bounds(parse(sbounds));
176  }
177 
178  add("bounds", dense(gbounds.toJson()));
179  }
180 
181  if (depthBegin)
182  add("depthBegin", std::to_string(depthBegin));
183  if (depthEnd)
184  add("depthEnd", std::to_string(depthEnd));
185 
186  Json::Value f(filter);
187 
188  if (f.isString())
189  f = parse(f.asString());
190 
191  if (tilePath.size())
192  f["Path"] = tilePath;
193 
194  if (!f.isNull())
195  add("filter", dense(f));
196 
197  if (!schema.isNull())
198  add("schema", dense(schema));
199 
200 #ifdef PDAL_HAVE_LAZPERF
201  add("compress", "true");
202 #endif
203 
204  return s;
205 }

238 void GreyhoundReader::addDimensions(PointLayoutPtr layout)
239 {
240  auto isXyz([](const std::string& name)
241  {
242  return name == "X" || name == "Y" || name == "Z";
243  });
244 
245  for (const auto& d : m_info["schema"])
246  {
247  const auto name(d["name"].asString());
248 
249  const int baseType(
250  Utils::toNative(Dimension::fromName(d["type"].asString())));

252  const int size(d["size"].asInt());
253 
254  const Dimension::Type type(
255  isXyz(name) ?

263 void GreyhoundReader::prepared(PointTableRef table)
264 {
265  auto& layout(*table.layout());
266 
267  // Note that we must construct the schema (which drives the formatting of
268  // Greyhound's responses) here, rather than just from the 'info' that comes
269  // back from Greyhound. This is because other Reader instances may add
270  // dimensions that don't exist in this resource. Also, the dimensions may
271  // be re-ordered in the layout from what we obtained in 'info'. Greyhound
272  // will zero-fill non-existent dimentions in the responses, which will
273  // compress to pretty much nothing.
274 
275  std::map<std::size_t, const Dimension::Detail> details;
276 
277  for (const Dimension::Id id : layout.dims())
278  {
279  const auto& detail(*layout.dimDetail(id));
280  details.emplace(detail.offset(), detail);
281  }
282 
283  m_args.schema = Json::Value();
284  for (const auto& p : details)
285  {
286  const Dimension::Detail& d(p.second);
287  const std::string name(layout.dimName(d.id()));
288 
289  Json::Value j;
290  j["name"] = name;
291  j["type"] = Dimension::toName(base(d.type()));
292  j["size"] = static_cast<int>(Dimension::size(d.type()));
293  m_args.schema.append(j);
294 
295  m_dims.push_back(layout.findDimType(name));
296  if (m_dims.back().m_id == Dimension::Id::Unknown)
297  throw pdal_error("Could not find dimension " + name);
298  }
299 
300  log()->get(LogLevel::Debug) << "Schema: " << m_args.schema << std::endl;
301 }

303 point_count_t GreyhoundReader::read(PointViewPtr view, point_count_t count)
304 {
305  const std::string url(m_args.base() + "read" + m_args.query());
306  log()->get(LogLevel::Debug) << "Reading: " << url << std::endl;
307 
308  auto response(m_arbiter->getBinary(url));
309  const std::size_t pointSize(view->layout()->pointSize());
310 
311  uint32_t numPoints(0);
312  std::copy(
313  response.data() + response.size() - sizeof(uint32_t),
314  response.data() + response.size(),
315  reinterpret_cast<char*>(&numPoints));
316 
317  log()->get(LogLevel::Debug) <<
318  "Fetched " << numPoints << " points" << std::endl;
319  log()->get(LogLevel::Debug) <<
320  "Fetched " << response.size() << " bytes" << std::endl;
321 
322  response.resize(response.size() - sizeof(uint32_t));
323 
324  std::vector<char*> points;
325  points.reserve(numPoints);
326 
327  const PointId startId(view->size());
328  PointId id(startId);
329  size_t pointNum(0);
330 
331  for (std::size_t i(0); i < numPoints; ++i)
332  {
333  points.push_back(view->getOrAddPoint(id));
334  ++id;
335  }
336 
337  // Because we are requesting the data in the exact PointLayout of our
338  // PointView, we can just decompress directly into that memory. Any
339  // non-existent dimensions (for example those added by other readers in the
340  // pipeline) will be zero-filled.
341 
342 #ifdef PDAL_HAVE_LAZPERF
343  auto cb = [&points, &pointNum](char *buf, size_t bufsize)
344  {
345  std::copy(buf, buf + bufsize, points[pointNum++]);
346  };
347  LazPerfDecompressor(cb, m_dims, numPoints).
348  decompress(response.data(), response.size());
349 #else
350  const char* pos(response.data());
351  const char* end(pos + numPoints * pointSize);
352  std::size_t i(0);
353 
354  while (pos < end)
355  {
356  std::copy(pos, pos + pointSize, points[i]);
357  ++i;
358  pos += pointSize;
359  }
360 #endif
361 
362  if (m_cb)
363  {
364  for (std::size_t i(0); i < numPoints; ++i)
365  {
366  m_cb(*view, startId + i);
367  }
368  }
369 
370  return numPoints;
371 }
