105 void ReprojectionFilter::createTransform(const SpatialReference& srsSRS)
106 {
107  if (m_inferInputSRS)
108  {
109  m_inSRS = srsSRS;
110  if (m_inSRS.empty())
111  throwError("source data has no spatial reference and "
112  "none is specified with the 'in_srs' option.");
113  }
114 
115  if (m_in_ref_ptr)
116  OSRDestroySpatialReference(m_in_ref_ptr);
117  m_in_ref_ptr = OSRNewSpatialReference(m_inSRS.getWKT().c_str());
118  if (!m_in_ref_ptr)
119  throwError("Invalid input spatial reference '" + m_inSRS.getWKT() +
120  "'. This is usually caused by a bad value for the 'in_srs' "
121  "option or an invalid spatial reference in the source file.");
122  if (m_transform_ptr)
123  OCTDestroyCoordinateTransformation(m_transform_ptr);
124  m_transform_ptr = OCTNewCoordinateTransformation(m_in_ref_ptr,
125  m_out_ref_ptr);
126  if (!m_transform_ptr)
127  throwError("Could not construct coordinate transformation object "
128  "in createTransform");
129 }

132 PointViewSet ReprojectionFilter::run(PointViewPtr view)
133 {
134  PointViewSet viewSet;
135  PointViewPtr outView = view->makeNew();
136 
137  createTransform(view->spatialReference());
138 
139  PointRef point(*view, 0);
140  for (PointId id = 0; id < view->size(); ++id)
141  {
142  point.setPointId(id);
143  if (processOne(point))
144  outView->appendPoint(*view, id);
145  }
146 
147  viewSet.insert(outView);
148  view->setSpatialReference(m_outSRS);
149  outView->setSpatialReference(m_outSRS);
150 
151  return viewSet;
152 }

155 bool ReprojectionFilter::processOne(PointRef& point)
156 {
157  double x(point.getFieldAs<double>(Dimension::Id::X));
158  double y(point.getFieldAs<double>(Dimension::Id::Y));
159  double z(point.getFieldAs<double>(Dimension::Id::Z));
160 
161  if (OCTTransform(m_transform_ptr, 1, &x, &y, &z))
162  {
163  point.setField(Dimension::Id::X, x);
164  point.setField(Dimension::Id::Y, y);
165  point.setField(Dimension::Id::Z, z);
166 
167  return true;
168  }
169  else
170  {
171  return false;
172  }
173 }
