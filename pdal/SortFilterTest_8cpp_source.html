51 void doSort(point_count_t count, const std::string & dim="X", const std::string & order="")
52 {
53  Options opts;
54 
55  opts.add("dimension", dim);
56  if (!order.empty())
57  opts.add("order", order);
58 
59  SortFilter filter;
60  filter.setOptions(opts);
61 
62  PointTable table;
63  PointViewPtr view(new PointView(table));
64 
65  table.layout()->registerDim(Dimension::Id::X);
66 
67  std::default_random_engine generator;
68  std::uniform_real_distribution<double> dist(0.0, (double)count);
69 
70  for (PointId i = 0; i < count; ++i)
71  view->setField(Dimension::Id::X, i, dist(generator));
72 
73  filter.prepare(table);
74  FilterWrapper::ready(filter, table);
75  FilterWrapper::filter(filter, *view.get());
76  FilterWrapper::done(filter, table);
77 
78  EXPECT_EQ(count, view->size());
79  for (PointId i = 1; i < count; ++i)
80  {
81  double d1 = view->getFieldAs<double>(Dimension::Id::X, i - 1);
82  double d2 = view->getFieldAs<double>(Dimension::Id::X, i);
83  if(order.empty() || order == "ASC")
84  EXPECT_TRUE(d1 <= d2);
85  else // DES(cending)
86  EXPECT_TRUE(d1 >= d2);
87  }
88 }
