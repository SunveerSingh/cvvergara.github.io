263 bool PluginManager::shutdown()
264 {
265  std::lock_guard<std::mutex> lock(m_mutex);
266 
267  bool success(true);
268 
269  for (auto const& func : m_exitFuncVec)
270  {
271  try
272  {
273  // Exit functions return 0 if successful.
274  if ((*func)() != 0)
275  {
276  success = false;
277  }
278  }
279  catch (...)
280  {
281  success = false;
282  }
283  }
284 
285  // This clears the handles on the dynamic libraries so that they won't
286  // be closed with dlclose(). Depending on the order of dll unloading,
287  // it's possible for a plugin library to be unloaded before this function
288  // (which is usually in a dll) is called, which can raise an error on
289  // some systems when dlclose() is called on a library for which the
290  // reference count is already 0. Since we're exiting anyway and all the
291  // dlls will be closed, there is no need to call dlclose() on them
292  // explicitly.
293  for (auto l : m_dynamicLibraryMap)
294  l.second->clear();
295 
296  m_dynamicLibraryMap.clear();
297  m_plugins.clear();
298  m_exitFuncVec.clear();
299 
300  return success;
301 }

310 bool PluginManager::l_loadPlugin(const std::string& driverFileName)
311 {
312  std::vector<std::string> driverPathVec;
313  driverPathVec = Utils::split2(driverFileName, '.');
314 
315  if ((FileUtils::extension(driverFileName) != dynamicLibraryExtension) ||
316  FileUtils::isDirectory(driverFileName))
317  return false;
318 
319  std::vector<std::string> driverNameVec;
320  driverNameVec = Utils::split2(driverPathVec[0], '_');
321 
322  std::string ptype;
323  if (driverNameVec.size() >= 3)
324  ptype = driverNameVec[2];
325 
326  PF_PluginType type;
327  if (Utils::iequals(ptype, "reader"))
328  type = PF_PluginType_Reader;
329  else if (Utils::iequals(ptype,"kernel"))
330  type = PF_PluginType_Kernel;
331  else if (Utils::iequals(ptype, "filter"))
332  type = PF_PluginType_Filter;
333  else if (Utils::iequals(ptype, "writer"))
334  type = PF_PluginType_Writer;
335  else
336  throw pdal_error("Unknown plugin type '" + ptype + "'");
337 
338  return loadByPath(driverFileName, type);
339 }

342 bool PluginManager::guessLoadByPath(const std::string& driverName)
343 {
344  // parse the driver name into an expected pluginName, e.g.,
345  // writers.las => libpdal_plugin_writer_las
346 
347  std::vector<std::string> driverNameVec;
348  driverNameVec = Utils::split2(driverName, '.');
349 
350  for (const auto& pluginPath : pluginSearchPaths())
351  {
352  m_log->get(LogLevel::Debug) <<
353  "Plugin search path: '" << pluginPath << "'" << std::endl;
354  if (!FileUtils::fileExists(pluginPath) ||
355  !FileUtils::isDirectory(pluginPath))
356  continue;
357 
358  StringList files = FileUtils::directoryList(pluginPath);
359  for (auto file : files)
360  {
361  if ((FileUtils::extension(file) != dynamicLibraryExtension) ||
362  FileUtils::isDirectory(file))
363  continue;
364 
365  std::string stem = FileUtils::stem(file);
366  std::string::size_type pos = stem.find_last_of('_');
367  if (pos == std::string::npos || pos == stem.size() - 1 ||
368  stem.substr(pos + 1) != driverNameVec[1])
369  continue;
370 
371  PF_PluginType type;
372  if (driverNameVec[0] == "readers")
373  type = PF_PluginType_Reader;
374  else if (driverNameVec[0] == "kernels")
375  type = PF_PluginType_Kernel;
376  else if (driverNameVec[0] == "filters")
377  type = PF_PluginType_Filter;
378  else if (driverNameVec[0] == "writers")
379  type = PF_PluginType_Writer;
380  else
381  type = PF_PluginType_Reader;
382 
383  if (loadByPath(file, type))
384  return true;
385  }
386  }
387 
388  return false;
389 }

392 bool PluginManager::loadByPath(const std::string& pluginPath, int type)
393 {
394  // Only filenames that start with libpdal_plugin are candidates to be loaded
395  // at runtime. PDAL plugins are to be named in a specified form:
396 
397  // libpdal_plugin_{stagetype}_{name}
398 
399  // For example, libpdal_plugin_writer_text or libpdal_plugin_filter_color
400 
401  bool loaded(false);
402 
403  std::string filename = Utils::tolower(FileUtils::getFilename(pluginPath));
404 
405  // If we are a valid type, and we're not yet already
406  // loaded in the LibraryMap, load it.
407 
408  if (pluginTypeValid(filename, type) && !libraryLoaded(pluginPath))
409  {
410  std::string errorString;
411  auto completePath(FileUtils::toAbsolutePath(pluginPath));
412 
413  m_log->get(LogLevel::Debug) << "Attempting to load plugin '" <<
414  completePath << "'." << std::endl;
415 
416  if (DynamicLibrary *d = loadLibrary(completePath, errorString))
417  {
418  m_log->get(LogLevel::Debug) << "Loaded plugin '" << completePath <<
419  "'." << std::endl;
420  if (PF_InitFunc initFunc =
421  (PF_InitFunc)(d->getSymbol("PF_initPlugin")))
422  {
423  loaded = initializePlugin(initFunc);
424  m_log->get(LogLevel::Debug) << "Initialized plugin '" <<
425  completePath << "'." << std::endl;
426  }
427  else
428  m_log->get(LogLevel::Error) <<
429  "Failed to initialize plugin function for plugin '" <<
430  completePath << "'." << std::endl;
431  }
432  else
433  m_log->get(LogLevel::Error) << "Plugin '" << completePath <<
434  "' found but failed to load: " << errorString << std::endl;
435  }
436 
437  return loaded;
438 }
