431 template <class T>
432 inline T PointView::getFieldAs(Dimension::Id dim,
433  PointId pointIndex) const
434 {
435  assert(pointIndex < m_size);
436  T retval;
437  const Dimension::Detail *dd = layout()->dimDetail(dim);
438  double val;
439 
440  switch (dd->type())
441  {
442  case Dimension::Type::Float:
443  val = getFieldInternal<float>(dim, pointIndex);
444  break;
445  case Dimension::Type::Double:
446  val = getFieldInternal<double>(dim, pointIndex);
447  break;
448  case Dimension::Type::Signed8:
449  val = getFieldInternal<int8_t>(dim, pointIndex);
450  break;
451  case Dimension::Type::Signed16:
452  val = getFieldInternal<int16_t>(dim, pointIndex);
453  break;
454  case Dimension::Type::Signed32:
455  val = getFieldInternal<int32_t>(dim, pointIndex);
456  break;
457  case Dimension::Type::Signed64:
458  val = static_cast<double>(getFieldInternal<int64_t>(dim, pointIndex));
459  break;
460  case Dimension::Type::Unsigned8:
461  val = getFieldInternal<uint8_t>(dim, pointIndex);
462  break;
463  case Dimension::Type::Unsigned16:
464  val = getFieldInternal<uint16_t>(dim, pointIndex);
465  break;
466  case Dimension::Type::Unsigned32:
467  val = getFieldInternal<uint32_t>(dim, pointIndex);
468  break;
469  case Dimension::Type::Unsigned64:
470  val = static_cast<double>(getFieldInternal<uint64_t>(dim, pointIndex));
471  break;
472  case Dimension::Type::None:
473  default:
474  val = 0;
475  break;
476  }
477 
478  if (!Utils::numericCast(val, retval))
479  {
480  std::ostringstream oss;
481  oss << "Unable to fetch data and convert as requested: ";
482  oss << Dimension::name(dim) << ":" <<
483  Dimension::interpretationName(dd->type()) <<
484  "(" << (double)val << ") -> " << Utils::typeidName<T>();
485  throw pdal_error(oss.str());
486  }
487 
488  return retval;
489 }
