86 inline bool operator < (const uuid& u1, const uuid& u2)
87 {
88  if (u1.time_low != u2.time_low)
89  return u1.time_low < u2.time_low;
90  if (u1.time_mid != u2.time_mid)
91  return u1.time_mid < u2.time_mid;
92  if (u1.time_hi_and_version != u2.time_hi_and_version)
93  return u1.time_hi_and_version < u2.time_hi_and_version;
94  for (size_t i = 0; i < sizeof(u1.node); ++i)
95  if (u1.node[i] != u2.node[i])
96  return u1.node[i] < u2.node[i];
97  return false;

134  bool parse(const std::string& s)
135  {
136  if (s.length() != 36)
137  return false;
138 
139  // Format validation.
140  const char *cp = s.data();
141  for (size_t i = 0; i < 36; i++) {
142  if ((i == 8) || (i == 13) || (i == 18) || (i == 23))
143  {
144  if (*cp != '-')
145  return false;
146  }
147  else if (!isxdigit(*cp))
148  return false;
149  ++cp;
150  }
151 
152  cp = s.data();
153  m_data.time_low = strtoul(cp, NULL, 16);
154  m_data.time_mid = (uint16_t)strtoul(cp + 9, NULL, 16);
155  m_data.time_hi_and_version = (uint16_t)strtoul(cp + 14, NULL, 16);
156  m_data.clock_seq = (uint16_t)strtoul(cp + 19, NULL, 16);
157 
158  // Extract bytes as pairs of hex digits.
159  cp = s.data() + 24;
160  char buf[3];
161  buf[2] = 0;
162  for (size_t i = 0; i < 6; i++) {
163  buf[0] = *cp++;
164  buf[1] = *cp++;
165  m_data.node[i] = (uint8_t)strtoul(buf, NULL, 16);
166  }
167  return true;
168  }
