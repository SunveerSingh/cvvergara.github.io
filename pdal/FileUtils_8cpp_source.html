364 std::vector<std::string> glob(std::string path)
365 {
366  std::vector<std::string> filenames;
367 #ifdef WIN32
368  WIN32_FIND_DATA ffd;
369  HANDLE handle = FindFirstFile(path.c_str(), &ffd);
370 
371  if (INVALID_HANDLE_VALUE == handle)
372  return filenames;
373 
374  size_t found = path.find_last_of("/\\");
375  do
376  {
377  if (found == std::string::npos)
378  filenames.push_back(ffd.cFileName);
379  else
380  filenames.push_back(path.substr(0, found) + "\\" + ffd.cFileName);
381 
382  } while (FindNextFile(handle, &ffd) != 0);
383  FindClose(handle);
384 #else
385  glob_t glob_result;
386 
387  ::glob(path.c_str(), GLOB_NOSORT, NULL, &glob_result);
388  for (unsigned int i = 0; i < glob_result.gl_pathc; ++i)
389  {
390  std::string filename = glob_result.gl_pathv[i];
391  filenames.push_back(filename);
392  }
393  globfree(&glob_result);
394 #endif
395  return filenames;
396 }
