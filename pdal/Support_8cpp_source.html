126 uint32_t Support::diff_text_files(std::istream& str1, std::istream& str2,
127  int32_t ignoreLine1)
128 {
129  uint32_t numdiffs = 0;
130  int32_t currLine = 1;
131 
132  while (!str1.eof() && !str2.eof())
133  {
134  std::string buf1;
135  std::string buf2;
136  std::getline(str1, buf1);
137  std::getline(str2, buf2);
138 
139  if (currLine == ignoreLine1)
140  {
141  ++currLine;
142  continue;
143  }
144 
145  if (str1.eof() && str2.eof())
146  {
147  // hit end on both together
148  break;
149  }
150  else if (str1.eof() && !str2.eof())
151  {
152  // str1 ended, but str2 still going
153  while (!str2.eof())
154  {
155  std::getline(str2, buf2);
156  ++numdiffs;
157  }
158  break;
159  }
160  else if (!str1.eof() && str2.eof())
161  {
162  // str2 ended, but str1 still going
163  while (!str1.eof())
164  {
165  std::getline(str1, buf1);
166  ++numdiffs;
167  }
168  break;
169  }
170 
171  if (buf1 != buf2)
172  ++numdiffs;
173 
174  ++currLine;
175  }
176 
177  assert(str1.eof());
178  assert(str2.eof());
179 
180  return numdiffs;
181 }

214 uint32_t Support::diff_files(std::istream& str1, std::istream& str2,
215  uint32_t* ignorable_start, uint32_t* ignorable_length,
216  uint32_t num_ignorables)
217 {
218  uint32_t numdiffs = 0;
219  char p, q;
220 
221  for (uint32_t i = 0; ; ++i)
222  {
223  str1.get(p);
224  str2.get(q);
225  if (!str1 || !str2)
226  {
227  if (!str1 != !str2)
228  numdiffs++;
229  break;
230  }
231  if (p == q)
232  continue;
233 
234  if (num_ignorables == 0)
235  ++numdiffs;
236  else
237  {
238  // only count the difference if we are NOT in an ignorable
239  // region
240  bool is_ignorable = false;
241  for (uint32_t region = 0; region < num_ignorables; region++)
242  {
243  uint32_t start = ignorable_start[region];
244  uint32_t end = start + ignorable_length[region];
245  if (i >= start && i < end)
246  {
247  // we are in an ignorable region!
248  is_ignorable = true;
249  break;
250  }
251  }
252  if (!is_ignorable)
253  ++numdiffs;
254  }
255  }
256  return numdiffs;
257 }

288 void Support::checkXYZ(const std::string& file1, const std::string& file2)
289 {
290  StageFactory f;
291 
292  const std::string driver1 = f.inferReaderDriver(file1);
293  EXPECT_NE(driver1, "") << "Can't find driver";
294 
295  const std::string driver2 = f.inferReaderDriver(file2);
296  EXPECT_NE(driver2, "") << "Can't find driver";
297 
298  Stage *reader1 = f.createStage(driver1);
299  EXPECT_NE(reader1, (Stage *)NULL) << "Couldn't create stage";
300 
301  Stage *reader2 = f.createStage(driver2);
302  EXPECT_NE(reader1, (Stage *)NULL) << "Couldn't create stage";
303 
304  Options o1;
305  o1.add("filename", file1);
306  reader1->setOptions(o1);
307  PointTable t1;
308  reader1->prepare(t1);
309  PointViewSet s1 = reader1->execute(t1);
310  EXPECT_EQ(s1.size(), 1u);
311  PointViewPtr v1 = *s1.begin();
312 
313  Options o2;
314  o2.add("filename", file2);
315  reader2->setOptions(o2);
316  PointTable t2;
317  reader2->prepare(t2);
318  PointViewSet s2 = reader2->execute(t2);
319  EXPECT_EQ(s2.size(), 1u);
320  PointViewPtr v2 = *s2.begin();
321 
322  EXPECT_EQ(v1->size(), v2->size()) << "Files " << file1 << " and " <<
323  file2 << " have different point counts.";
324  for (PointId idx = 0; idx < v1->size(); ++idx)
325  {
326  using namespace Dimension;
327  ASSERT_DOUBLE_EQ(v1->getFieldAs<double>(Id::X, idx),
328  v2->getFieldAs<double>(Id::X, idx)) << "Index = " << idx;
329  ASSERT_DOUBLE_EQ(v1->getFieldAs<double>(Id::Y, idx),
330  v2->getFieldAs<double>(Id::Y, idx)) << "Index = " << idx;
331  ASSERT_DOUBLE_EQ(v1->getFieldAs<double>(Id::Z, idx),
332  v2->getFieldAs<double>(Id::Z, idx)) << "Index = " << idx;
333  }
334 }
