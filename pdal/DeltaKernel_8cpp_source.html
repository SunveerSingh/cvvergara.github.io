78 int DeltaKernel::execute()
79 {
80  PointTable srcTable;
81  PointTable candTable;
82  DimIndexMap dims;
83 
84  PointViewPtr srcView = loadSet(m_sourceFile, srcTable);
85  PointViewPtr candView = loadSet(m_candidateFile, candTable);
86 
87  PointLayoutPtr srcLayout = srcTable.layout();
88  PointLayoutPtr candLayout = candTable.layout();
89 
90  Dimension::IdList ids = srcLayout->dims();
91  for (Dimension::Id dim : ids)
92  {
93  std::string name = srcLayout->dimName(dim);
94  if (!m_allDims)
95  if (name != "X" && name != "Y" && name != "Z")
96  continue;
97  DimIndex d;
98  d.m_name = name;
99  d.m_srcId = dim;
100  dims[name] = d;
101  }
102  ids = candLayout->dims();
103  for (Dimension::Id dim : ids)
104  {
105  std::string name = candLayout->dimName(dim);
106  auto di = dims.find(name);
107  if (di == dims.end())
108  continue;
109  DimIndex& d = di->second;
110  d.m_candId = dim;
111  }
112 
113  // Remove dimensions that aren't in both the source and candidate lists.
114  for (auto di = dims.begin(); di != dims.end();)
115  {
116  DimIndex& d = di->second;
117  if (d.m_candId == Dimension::Id::Unknown)
118  dims.erase(di++);
119  else
120  ++di;
121  }
122 
123  // Index the candidate data.
124  KD3Index index(*candView);
125  index.build();
126 
127  MetadataNode root;
128 
129  if (m_detail)
130  root = dumpDetail(srcView, candView, index, dims);
131  else
132  root = dump(srcView, candView, index, dims);
133  Utils::toJSON(root, std::cout);
134 
135  return 0;
136 }

139 MetadataNode DeltaKernel::dump(PointViewPtr& srcView, PointViewPtr& candView,
140  KD3Index& index, DimIndexMap& dims)
141 {
142  MetadataNode root;
143 
144  for (PointId id = 0; id < srcView->size(); ++id)
145  {
146  PointRef point = srcView->point(id);
147  PointId candId = index.neighbor(point);
148 
149  // It may be faster to put in a special case to avoid having to
150  // fetch X, Y and Z, more than once but this is simpler and
151  // I'm thinking in most cases it will make no practical difference.
152  for (auto di = dims.begin(); di != dims.end(); ++di)
153  {
154  DimIndex& d = di->second;
155  double sv = srcView->getFieldAs<double>(d.m_srcId, id);
156  double cv = candView->getFieldAs<double>(d.m_candId, candId);
157  accumulate(d, sv - cv);
158  }
159  }
160 
161  root.add("source", m_sourceFile);
162  root.add("candidate", m_candidateFile);
163  for (auto dpair : dims)
164  {
165  DimIndex& d = dpair.second;
166 
167  MetadataNode dimNode = root.add(d.m_name);
168  dimNode.add("min", d.m_min);
169  dimNode.add("max", d.m_max);
170  dimNode.add("mean", d.m_avg);
171  }
172  return root;
173 }

176 void DeltaKernel::accumulate(DimIndex& d, double v)
177 {
178  d.m_cnt++;
179  d.m_min = std::min(v, d.m_min);
180  d.m_max = std::max(v, d.m_max);
181  d.m_avg += (v - d.m_avg) / d.m_cnt;
182 }

185 MetadataNode DeltaKernel::dumpDetail(PointViewPtr& srcView,
186  PointViewPtr& candView, KD3Index& index, DimIndexMap& dims)
187 {
188  MetadataNode root;
189 
190  for (PointId id = 0; id < srcView->size(); ++id)
191  {
192  PointRef point = srcView->point(id);
193  PointId candId = index.neighbor(point);
194 
195  MetadataNode delta = root.add("delta");
196  delta.add("i", id);
197  for (auto di = dims.begin(); di != dims.end(); ++di)
198  {
199  DimIndex& d = di->second;
200  double sv = srcView->getFieldAs<double>(d.m_srcId, id);
201  double cv = candView->getFieldAs<double>(d.m_candId, candId);
202 
203  delta.add(d.m_name, sv - cv);
204  }
205  }
206  return root;
207 }
