97 void RangeFilter::prepared(PointTableRef table)
98 {
99  const PointLayoutPtr layout(table.layout());
100 
101  for (auto& r : m_range_list)
102  {
103  r.m_id = layout->findDim(r.m_name);
104  if (r.m_id == Dimension::Id::Unknown)
105  throwError("Invalid dimension name in 'limits' option: '" +
106  r.m_name + "'.");
107  }
108  std::sort(m_range_list.begin(), m_range_list.end());
109 }
110 
111 
112 // The range list is sorted by dimension, so the logic here should work
113 // as ORs between ranges of the same dimension and ANDs between ranges
114 // of different dimensions. This is simple logic, but is probably the most
115 // common case.
116 bool RangeFilter::processOne(PointRef& point)
117 {
118  Dimension::Id lastId = m_range_list.front().m_id;
119  bool passes = false;
120  for (auto const& r : m_range_list)
121  {
122  // If we're at a new dimension, return false if we haven't passed
123  // the dimension, otherwise reset passes to false for the next
124  // dimension and keep checking.
125  if (r.m_id != lastId)
126  {
127  if (!passes)
128  return false;
129  lastId = r.m_id;
130  passes = false;
131  }
132  // If we've already passed this dimension, continue until we find
133  // a new dimension.
134  else if (passes)
135  continue;
136  passes = r.valuePasses(point.getFieldAs<double>(r.m_id));
137  }
138  return passes;
139 }

142 PointViewSet RangeFilter::run(PointViewPtr inView)
143 {
144  PointViewSet viewSet;
145  if (!inView->size())
146  return viewSet;
147 
148  PointViewPtr outView = inView->makeNew();
149 
150  for (PointId i = 0; i < inView->size(); ++i)
151  {
152  PointRef point = inView->point(i);
153  if (processOne(point))
154  outView->appendPoint(*inView, i);
155  }
156 
157  viewSet.insert(outView);
158  return viewSet;
159 }
