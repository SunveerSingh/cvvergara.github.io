55 Eigen::Vector3f computeCentroid(PointView& view, std::vector<PointId> ids)
56 {
57  using namespace Eigen;
58 
59  auto n = ids.size();
60 
61  double mx, my, mz;

67  mz += view.getFieldAs<double>(Dimension::Id::Z, j);

76 Eigen::Matrix3f computeCovariance(PointView& view, std::vector<PointId> ids)
77 {
78  using namespace Eigen;
79 
80  auto n = ids.size();
81 
82  Vector3f centroid = computeCentroid(view, ids);
83 
84  // demean the neighborhood
85  MatrixXf A(3, n);
86  size_t k = 0;
87  for (auto const& j : ids)
88  {
89  A(0, k) = view.getFieldAs<double>(Dimension::Id::X, j) - centroid[0];
90  A(1, k) = view.getFieldAs<double>(Dimension::Id::Y, j) - centroid[1];
91  A(2, k) = view.getFieldAs<double>(Dimension::Id::Z, j) - centroid[2];
92  k++;
93  }
94 
95  return A * A.transpose() / (ids.size()-1);
96 }

110 Eigen::MatrixXd computeSpline(Eigen::MatrixXd x, Eigen::MatrixXd y,
111  Eigen::MatrixXd z, Eigen::MatrixXd xx,
112  Eigen::MatrixXd yy)
113 {
114  using namespace Eigen;
115 
116  int num_rows = xx.rows();
117  int num_cols = xx.cols();
118 
119  MatrixXd S = MatrixXd::Zero(num_rows, num_cols);
120 
121  for (auto col = 0; col < num_cols; ++col)
122  {
123  for (auto row = 0; row < num_rows; ++row)
124  {
125  // Further optimizations are achieved by estimating only the
126  // interpolated surface within a local neighbourhood (e.g. a 7 x 7
127  // neighbourhood is used in our case) of the cell being filtered.
128  int radius = 3;
129 
130  int c = std::floor(col/2);
131  int r = std::floor(row/2);
132 
133  int cs = Utils::clamp(c-radius, 0, static_cast<int>(z.cols()-1));
134  int ce = Utils::clamp(c+radius, 0, static_cast<int>(z.cols()-1));
135  int col_size = ce - cs + 1;
136  int rs = Utils::clamp(r-radius, 0, static_cast<int>(z.rows()-1));
137  int re = Utils::clamp(r+radius, 0, static_cast<int>(z.rows()-1));
138  int row_size = re - rs + 1;
139 
140  MatrixXd Xn = x.block(rs, cs, row_size, col_size);
141  MatrixXd Yn = y.block(rs, cs, row_size, col_size);
142  MatrixXd Hn = z.block(rs, cs, row_size, col_size);
143 
144  int nsize = row_size * col_size;
145  VectorXd T = VectorXd::Zero(nsize);
146  MatrixXd P = MatrixXd::Zero(nsize, 3);
147  MatrixXd K = MatrixXd::Zero(nsize, nsize);
148 
149  for (auto id = 0; id < nsize; ++id)
150  {
151  double xj = Xn(id);
152  double yj = Yn(id);
153  double zj = Hn(id);
154  if (std::isnan(xj) || std::isnan(yj) || std::isnan(zj))
155  continue;
156  T(id) = zj;
157  P.row(id) << 1, xj, yj;
158  for (auto id2 = 0; id2 < nsize; ++id2)
159  {
160  if (id == id2)
161  continue;
162  double xk = Xn(id2);
163  double yk = Yn(id2);
164  double zk = Hn(id2);
165  if (std::isnan(xk) || std::isnan(yk) || std::isnan(zk))
166  continue;
167  double rsqr = (xj - xk) * (xj - xk) + (yj - yk) * (yj - yk);
168  if (rsqr == 0.0)
169  continue;
170  K(id, id2) = rsqr * std::log10(std::sqrt(rsqr));
171  }
172  }
173 
174  MatrixXd A = MatrixXd::Zero(nsize+3, nsize+3);
175  A.block(0,0,nsize,nsize) = K;
176  A.block(0,nsize,nsize,3) = P;
177  A.block(nsize,0,3,nsize) = P.transpose();
178 
179  VectorXd b = VectorXd::Zero(nsize+3);
180  b.head(nsize) = T;
181 
182  VectorXd x = A.fullPivHouseholderQr().solve(b);
183 
184  Vector3d a = x.tail(3);
185  VectorXd w = x.head(nsize);
186 
187  double sum = 0.0;
188  double xi2 = xx(row, col);
189  double yi2 = yy(row, col);
190  for (auto j = 0; j < nsize; ++j)
191  {
192  double xj = Xn(j);
193  double yj = Yn(j);
194  double zj = Hn(j);
195  if (std::isnan(xj) || std::isnan(yj) || std::isnan(zj))
196  continue;
197  double rsqr = (xj - xi2) * (xj - xi2) + (yj - yi2) * (yj - yi2);
198  if (rsqr == 0.0)
199  continue;
200  sum += w(j) * rsqr * std::log10(std::sqrt(rsqr));
201  }
202 
203  S(row, col) = a(0) + a(1)*xi2 + a(2)*yi2 + sum;
204  }
205  }
206 
207  return S;
208 }

267  kdi.build();
268 
269  MatrixXd ZImax(rows, cols);
270  ZImax.setConstant(std::numeric_limits<double>::quiet_NaN());
271 
272  // for each grid center, search PointView for neighbors, and find max of those
273  for (int c = 0; c < cols; ++c)

276

288  val = z;

292  }
293  }

298 Eigen::MatrixXd extendedLocalMinimum(PointView& view, int rows, int cols,
299  double cell_size, BOX2D bounds)
300 {
301  using namespace Dimension;
302  using namespace Eigen;
303 
304  // Index elevation values by row and column.
305  std::map<uint32_t, std::vector<double>> hash;
306  for (PointId i = 0; i < view.size(); ++i)
307  {
308  double x = view.getFieldAs<double>(Id::X, i);
309  double y = view.getFieldAs<double>(Id::Y, i);
310  double z = view.getFieldAs<double>(Id::Z, i);
311 
312  int c = Utils::clamp(static_cast<int>(floor(x-bounds.minx)/cell_size), 0, cols-1);
313  int r = Utils::clamp(static_cast<int>(floor(y-bounds.miny)/cell_size), 0, rows-1);
314 
315  hash[r*cols+c].push_back(z);
316  }
317 
318  // For each grid cell, sort elevations and detect local minimum, rejecting
319  // low outliers.
320  MatrixXd ZImin(rows, cols);
321  ZImin.setConstant(std::numeric_limits<double>::quiet_NaN());
322  for (int c = 0; c < cols; ++c)
323  {
324  for (int r = 0; r < rows; ++r)
325  {
326  std::vector<double> cp(hash[r*cols+c]);
327  if (cp.empty())
328  continue;
329  std::sort(cp.begin(), cp.end());
330  if (cp.size() == 1)
331  {
332  ZImin(r, c) = cp[0];
333  continue;
334  }
335  for (size_t i = 0; i < cp.size()-1; ++i)
336  {
337  if (std::fabs(cp[i] - cp[i+1]) < 1.0)
338  {
339  ZImin(r, c) = cp[i];
340  break;
341  }
342  }
343  }
344  }
345 
346  return ZImin;
347 }

349 Eigen::MatrixXd matrixClose(Eigen::MatrixXd data, int radius)
350 {
351  using namespace Eigen;
352 
353  MatrixXd data2 = padMatrix(data, radius);
354 
355  int nrows = data2.rows();
356  int ncols = data2.cols();
357 
358  MatrixXd minZ(nrows, ncols);
359  minZ.setConstant(std::numeric_limits<double>::max());
360  MatrixXd maxZ(nrows, ncols);
361  maxZ.setConstant(std::numeric_limits<double>::lowest());
362  for (auto c = 0; c < ncols; ++c)
363  {
364  int cs = Utils::clamp(c-radius, 0, ncols-1);
365  int ce = Utils::clamp(c+radius, 0, ncols-1);
366 
367  for (auto r = 0; r < nrows; ++r)
368  {
369  int rs = Utils::clamp(r-radius, 0, nrows-1);
370  int re = Utils::clamp(r+radius, 0, nrows-1);
371 
372  for (auto col = cs; col <= ce; ++col)
373  {
374  for (auto row = rs; row <= re; ++row)
375  {
376  if ((row-r)*(row-r)+(col-c)*(col-c) > radius*radius)
377  continue;
378  if (data2(row, col) > maxZ(r, c))
379  maxZ(r, c) = data2(row, col);
380  }
381  }
382  }
383  }
384  for (auto c = 0; c < ncols; ++c)
385  {
386  int cs = Utils::clamp(c-radius, 0, ncols-1);
387  int ce = Utils::clamp(c+radius, 0, ncols-1);
388 
389  for (auto r = 0; r < nrows; ++r)
390  {
391  int rs = Utils::clamp(r-radius, 0, nrows-1);
392  int re = Utils::clamp(r+radius, 0, nrows-1);
393 
394  for (auto col = cs; col <= ce; ++col)
395  {
396  for (auto row = rs; row <= re; ++row)
397  {
398  if ((row-r)*(row-r)+(col-c)*(col-c) > radius*radius)
399  continue;
400  if (maxZ(row, col) < minZ(r, c))
401  minZ(r, c) = maxZ(row, col);
402  }
403  }
404  }
405  }
406 
407  return minZ.block(radius, radius, data.rows(), data.cols());
408 }

472 {

474  std::vector<size_t> idx(5);
475 
476  for (int iter = 0; iter < iterations; ++iter)
477  {
478  for (size_t col = 0; col < cols; ++col)
479  {
480  size_t index = col*rows;

482  {
483  size_t j = 0;
484  idx[j++] = index+row;
485  if (row > 0)
486  idx[j++] = idx[0]-1;
487  if (row < rows-1)
488  idx[j++] = idx[0]+1;
489  if (col > 0)
490  idx[j++] = idx[0]-rows;
491  if (col < cols-1)
492  idx[j++] = idx[0]+rows;
493  for (size_t i = 0; i < j; ++i)
494  {
495  if (data[idx[i]] > out[index+row])
496  out[index+row] = data[idx[i]];
497  }
498  }
499  }
500  data.swap(out);
501  }
502  return data;
