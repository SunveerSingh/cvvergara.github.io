76 int RandomKernel::execute()
77 {
78  Options readerOptions;
79 
80  if (!m_bounds.empty())
81  readerOptions.add("bounds", m_bounds);
82 
83  std::string distribution(Utils::tolower(m_distribution));
84  if (distribution == "uniform")
85  readerOptions.add("mode", "uniform");
86  else if (distribution == "normal")
87  readerOptions.add("mode", "normal");
88  else if (distribution == "random")
89  readerOptions.add("mode", "random");
90  else
91  throw pdal_error("invalid distribution: " + m_distribution);
92  readerOptions.add("count", m_numPointsToWrite);
93  Stage& reader = makeReader("", "readers.faux", readerOptions);
94 
95  Options writerOptions;
96  if (m_bCompress)
97  writerOptions.add("compression", true);
98  Stage& writer = makeWriter(m_outputFile, reader, "", writerOptions);
99 
100  PointTable table;
101  writer.prepare(table);
102  writer.execute(table);
103 
104  return 0;
105 }
