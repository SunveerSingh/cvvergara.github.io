67 // The VSIFILE* that VSIFileFromMemBuffer creates in this
68 // macro is never cleaned up. We're opening seven PNGs in the
69 // ColorInterpRamps-ramps.hpp header. We always open them so they're available.
70 //
71 // GDAL forces to keep track of the return value, and its being ignored here,
72 // To avoid the warning message:
73 // warning: ignoring return value of 'VSILFILE* VSIFileFromMemBuffer(....)'
74 // declared with attribute warn_unused_result [-Wunused-result]
75 // Using a tmp variable
76 #define GETRAMP(name) \
77  if (pdal::Utils::iequals(#name, rampFilename)) \
78  { \
79  GByte* location(0); \
80  int size (0); \
81  location = name; \
82  size = sizeof(name); \
83  rampFilename = "/vsimem/" + std::string(#name) + ".png"; \
84  auto tmp(VSIFileFromMemBuffer(rampFilename.c_str(), location, size, FALSE)); \
85  }

87 std::shared_ptr<gdal::Raster> openRamp(std::string& rampFilename)
88 {
89  // If the user selected a default ramp name, it will be opened by
90  // one of these macros if it matches. Otherwise, we just open with the
91  // GDALOpen'able the user gave us
92 
93  // GETRAMP will set rampFilename to the /vsimem filename it
94  // used to actually open the file, and from then on it can be treated
95  // like any other GDAL datasource.
96 
97  GETRAMP(awesome_green);
98  GETRAMP(black_orange);
99  GETRAMP(blue_hue);
100  GETRAMP(blue_red);
101  GETRAMP(heat_map);
102  GETRAMP(pestel_shades);
103  GETRAMP(blue_orange);
104 
105  std::shared_ptr<gdal::Raster>
106  output(new gdal::Raster(rampFilename.c_str()));
107  return output;
108 }

170 void ColorinterpFilter::filter(PointView& view)
171 {
172  double median(0.0);
173  double mad(0.0);
174 
175  // If the user set a 'k' value, we use that
176  // defaulting to using a computed stddev if we
177  // were not told to use MAD.
178  if (m_stdDevThreshold != 0.0)
179  {
180  std::vector<double> values(view.size());
181 
182  stats::Summary summary(Dimension::name(m_interpDim),
183  stats::Summary::NoEnum);
184  for (PointId idx = 0; idx < view.size(); ++idx)
185  {
186  double v = view.getFieldAs<double>(m_interpDim, idx);
187  summary.insert(v);
188  values[idx] = v;
189  }
190 
191  auto compute_median = [](std::vector<double> vals)
192  {
193  std::nth_element(vals.begin(), vals.begin() + vals.size() / 2,
194  vals.end());
195 
196  return *(vals.begin() + vals.size() / 2);
197  };
198 
199  median = compute_median(values);
200  if (m_useMAD)
201  {
202  std::transform(values.begin(), values.end(), values.begin(),
203  [median](double v) { return std::fabs(v - median); });
204  mad = compute_median(values);
205 
206  double threshold = mad * m_madMultiplier * m_stdDevThreshold;
207  m_min = median - threshold;
208  m_max = median + threshold;
209 
210  log()->get(LogLevel::Debug) << getName() << " mad " <<
211  mad << std::endl;
212  log()->get(LogLevel::Debug) << getName() << " median " <<
213  median << std::endl;
214  log()->get(LogLevel::Debug) << getName() << " minimum " <<
215  m_min << std::endl;
216  log()->get(LogLevel::Debug) << getName() << " maximum " <<
217  m_max << std::endl;
218  }
219  else
220  {
221  double threshold = (m_stdDevThreshold * summary.stddev());
222  m_min = median - threshold;
223  m_max = median + threshold;
224 
225  log()->get(LogLevel::Debug) << getName() <<
226  " stddev threshold " << threshold << std::endl;
227  log()->get(LogLevel::Debug) << getName() << " median " <<
228  median << std::endl;
229  log()->get(LogLevel::Debug) << getName() << " minimum " <<
230  m_min << std::endl;
231  log()->get(LogLevel::Debug) << getName() << " maximum " <<
232  m_max << std::endl;
233  }
234  }
235 
236  // If the user didn't set min/max values and hadn't set a stddev, we
237  // compute them.
238  else if (std::isnan(m_min) || std::isnan(m_max))
239  {
240  stats::Summary summary(Dimension::name(m_interpDim),
241  stats::Summary::NoEnum);
242  for (PointId idx = 0; idx < view.size(); ++idx)
243  {
244  double v = view.getFieldAs<double>(m_interpDim, idx);
245  summary.insert(v);
246  }
247 
248  if (std::isnan(m_min))
249  m_min = summary.minimum();
250  if (std::isnan(m_max))
251  m_max = summary.maximum();
252  }
253 
254  PointRef point(view, 0);
255  for (PointId idx = 0; idx < view.size(); ++idx)
256  {
257  point.setPointId(idx);
258  processOne(point);
259  }
260 }

263 bool ColorinterpFilter::processOne(PointRef& point)
264 {
265  // This will throw if min or max aren't defined.
266  if (std::isnan(m_min) || std::isnan(m_max))
267  throwStreamingError();
268 
269  double v = point.getFieldAs<double>(m_interpDim);
270 
271  // Don't color points that aren't in the min/max range.
272  if (v < m_min || v >= m_max)
273  return true;
274 
275  double factor = (v - m_min) / (m_max - m_min);
276  size_t img_width = m_redBand.size();
277  size_t position = std::floor(factor * img_width);
278 
279  if (m_invertRamp)
280  position = (img_width - 1) - position;
281 
282  point.setField(Dimension::Id::Red, m_redBand[position]);
283  point.setField(Dimension::Id::Blue, m_blueBand[position]);
284  point.setField(Dimension::Id::Green, m_greenBand[position]);
285 
286  return true;
287 }
