69 void HAGFilter::filter(PointView& view)
70 {
71  PointViewPtr gView = view.makeNew();
72  PointViewPtr ngView = view.makeNew();
73  std::vector<PointId> gIdx, ngIdx;
74 
75  // First pass: Separate into ground and non-ground views.
76  for (PointId i = 0; i < view.size(); ++i)
77  {
78  double c = view.getFieldAs<double>(Dimension::Id::Classification, i);
79  if (c == 2)
80  {
81  gView->appendPoint(view, i);
82  gIdx.push_back(i);
83  }
84  else
85  {
86  ngView->appendPoint(view, i);
87  ngIdx.push_back(i);
88  }
89  }
90 
91  // Bail if there weren't any points classified as ground.
92  if (gView->size() == 0)
93  throwError("Input PointView does not have any points classified "
94  "as ground");
95 
96  // Build the 2D KD-tree.
97  KD2Index kdi(*gView);
98  kdi.build();
99 
100  // Second pass: Find Z difference between non-ground points and the nearest
101  // neighbor (2D) in the ground view.
102  for (PointId i = 0; i < ngView->size(); ++i)
103  {
104  PointRef point = ngView->point(i);
105  double z0 = point.getFieldAs<double>(Dimension::Id::Z);
106  auto ids = kdi.neighbors(point, 1);
107  double z1 = gView->getFieldAs<double>(Dimension::Id::Z, ids[0]);
108  view.setField(Dimension::Id::HeightAboveGround, ngIdx[i], z0 - z1);
109  }
110 
111  // Final pass: Ensure that all ground points have height value pegged at 0.
112  for (auto const& i : gIdx)
113  view.setField(Dimension::Id::HeightAboveGround, i, 0.0);
114 }
