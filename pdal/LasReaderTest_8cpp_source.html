47 ::testing::AssertionResult CheckEqualCollections(LeftIter left_begin,
48  LeftIter left_end, RightIter right_begin)
49 {
50  bool equal(true);
51 
52  std::string message;
53  size_t index(0);
54  while (left_begin != left_end)
55  {
56  if (*left_begin++ != *right_begin++)
57  {
58  equal = false;
59  message += "\n\tMismatch at index " + std::to_string(index);
60  }
61  ++index;
62  }
63  if (message.size())
64  message += "\n\t";
