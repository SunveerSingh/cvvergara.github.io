81 point_count_t PgReader::getNumPoints() const
82 {
83  if (m_cached_point_count)
84  return m_cached_point_count;
85 
86  std::ostringstream oss;
87  oss << "SELECT Sum(PC_NumPoints(" << pg_quote_identifier(m_column_name) << ")) AS numpoints, ";
88  oss << "Max(PC_NumPoints(" << pg_quote_identifier(m_column_name) <<
89  ")) AS maxpoints FROM ";
90  if (m_schema_name.size())
91  oss << pg_quote_identifier(m_schema_name) << ".";
92  oss << pg_quote_identifier(m_table_name);
93  if (m_where.size())
94  oss << " WHERE " << m_where;
95 
96  PGresult *result = pg_query_result(m_session, oss.str());
97 
98  if (PQresultStatus(result) != PGRES_TUPLES_OK)
99  throwError("Unable to get point count.");
100 
101  m_cached_point_count = atoi(PQgetvalue(result, 0, 0));
102  m_cached_max_points = atoi(PQgetvalue(result, 0, 1));
103  PQclear(result);
104 
105  return m_cached_point_count;
106 }

136 uint32_t PgReader::fetchPcid() const
137 {
138  if (m_pcid)
139  return m_pcid;
140 
141  log()->get(LogLevel::Debug) << "Fetching pcid ..." << std::endl;
142 
143  std::ostringstream oss;
144  oss << "SELECT PC_Typmod_Pcid(a.atttypmod) AS pcid ";
145  oss << "FROM pg_class c, pg_attribute a";
146  if (!m_schema_name.empty())
147  {
148  oss << ", pg_namespace n";
149  }
150  oss << " WHERE c.relname = " << pg_quote_literal(m_table_name);
151  oss << " AND a.attname = " << pg_quote_literal(m_column_name);
152  oss << " AND a.attrelid = c.oid " ;
153  if (!m_schema_name.empty())
154  {
155  oss << " AND c.relnamespace = n.oid AND n.nspname = " <<
156  pg_quote_literal(m_schema_name);
157  }
158 
159  std::string pcid_str = pg_query_once(m_session, oss.str());
160 
161  uint32_t pcid = 0;
162  if (pcid_str.size())
163  {
164  pcid = atoi(pcid_str.c_str());
165  }
166 
167  if (!pcid) // Are pcid == 0 valid?
168  {
169  std::ostringstream oss;
170  oss << "Unable to fetch pcid with column '"
171  << m_column_name <<"' and table ";
172  if (!m_schema_name.empty())
173  oss << "'" << m_schema_name << "'.";
174  oss << "'" << m_table_name << "'";
175  throwError(oss.str());
176  }
177 
178  log()->get(LogLevel::Debug) << " got pcid = " << pcid << std::endl;
179  m_pcid = pcid;
180  return pcid;
181 }

218  oss.str("");

220

280 point_count_t PgReader::readPgPatch(PointViewPtr view, point_count_t numPts)
281 {
282  point_count_t numRemaining = m_patch.remaining;
283  PointId nextId = view->size();
284  point_count_t numRead = 0;
285 
286  size_t offset = (m_patch.count - m_patch.remaining) * packedPointSize();
287  char *pos = (char *)(m_patch.binary.data() + offset);
288 
289  while (numRead < numPts && numRemaining > 0)
290  {
291  writePoint(*view.get(), nextId, pos);
292  pos += packedPointSize();
293  numRemaining--;
294  nextId++;
295  numRead++;
296  }
297  m_patch.remaining = numRemaining;
298  return numRead;
299 }

302 bool PgReader::NextBuffer()
303 {
304  if (m_cur_row >= m_cur_nrows || !m_cur_result)
305  {
306  static std::string fetch = "FETCH 2 FROM cur";
307  if (m_cur_result)
308  PQclear(m_cur_result);
309  m_cur_result = pg_query_result(m_session, fetch);
310  bool logOutput = (log()->getLevel() > LogLevel::Debug3);
311  if (logOutput)
312  log()->get(LogLevel::Debug3) << "SQL: " << fetch << std::endl;
313  if ((PQresultStatus(m_cur_result) != PGRES_TUPLES_OK) ||
314  (PQntuples(m_cur_result) == 0))
315  {
316  PQclear(m_cur_result);
317  m_cur_result = NULL;
318  m_atEnd = true;
319  return false;
320  }
321 
322  m_cur_row = 0;
323  m_cur_nrows = PQntuples(m_cur_result);
324  }
325  m_patch.hex = PQgetvalue(m_cur_result, m_cur_row, 0);
326  m_patch.count = atoi(PQgetvalue(m_cur_result, m_cur_row, 1));
327  m_patch.remaining = m_patch.count;
328  m_patch.update_binary();
329 
330  m_cur_row++;
331  return true;
332 }

335 point_count_t PgReader::read(PointViewPtr view, point_count_t count)
336 {
337  if (eof())
338  return 0;
339 
340  log()->get(LogLevel::Debug) << "readBufferImpl called with "
341  "PointView filled to " << view->size() << " points" <<
342  std::endl;
343 
344  point_count_t totalNumRead = 0;
345  while (totalNumRead < count)
346  {
347  if (m_patch.remaining == 0)
348  if (!NextBuffer())
349  return totalNumRead;
350  PointId bufBegin = view->size();
351  point_count_t numRead = readPgPatch(view, count - totalNumRead);
352  PointId bufEnd = bufBegin + numRead;
353  totalNumRead += numRead;
354  }
355  return totalNumRead;
356 }
