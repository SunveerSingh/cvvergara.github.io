84 PointViewSet MatlabFilter::run(PointViewPtr view)
85 {
86  log()->get(LogLevel::Debug) << "filters.matlab " << m_script <<
87  " processing " << view->size() << " points." << std::endl;
88 
89  int logBufferSize(4096);
90  std::unique_ptr<char[]> buf(new char[logBufferSize]);
91  m_MatlabOutputBuffer.swap(buf);
92 
93  Engine* engine = mlang::Environment::get()->m_engine;
94  engOutputBuffer(engine, m_MatlabOutputBuffer.get(), logBufferSize);
95 
96  Dimension::IdList dims;
97 
98  mxArray* matlabData = mlang::Script::setMatlabStruct(view, dims, m_pdalargs, m_tableMetadata, log());
99  if (engPutVariable(engine, m_structName.c_str(), matlabData))
100  {
101  std::ostringstream oss;
102  oss << "Could not push '" << m_structName << "' struct to Matlab";
103  throwError(oss.str());
104  }
105 
106  engEvalString(engine, m_script.m_source.c_str());
107 
108  std::string noise(m_MatlabOutputBuffer.get(), strlen(m_MatlabOutputBuffer.get()));
109  log()->get(LogLevel::Debug) << "filters.matlab " << noise << std::endl;
110 
111  matlabData = engGetVariable(engine, m_structName.c_str());
112  if (!matlabData)
113  throwError("No 'PDAL' variable is available in Matlab scope!");
114 
115  PointViewSet viewSet;
116 
117  std::string logicalDimensionName = m_script.getLogicalMask(matlabData, log());
118  if (logicalDimensionName.size())
119  {
120  PointViewPtr outview = view->makeNew();
121 
122  mxArray* f = mxGetField(matlabData, 0, logicalDimensionName.c_str());
123  if (!f)
124  {
125  std::ostringstream oss;
126  oss << "Unable to fetch mask dimension '" << logicalDimensionName << "'";
127  throwError(oss.str());
128  }
129 
130  mxLogical* logical = mxGetLogicals(f);
131  if (!logical)
132  {
133  std::ostringstream oss;
134  oss << "Unable to fetch logical mask for dimension '" << logicalDimensionName << "'";
135  throwError(oss.str());
136  }
137 
138  char *ok = (char *)logical;
139  for (PointId idx = 0; idx < view->size(); ++idx)
140  if (*ok++)
141  outview->appendPoint(*view, idx);
142  viewSet.insert(outview);
143  }
144  else
145  {
146  mlang::Script::getMatlabStruct(matlabData, view, dims, m_pdalargs, m_tableMetadata, log());
147  viewSet.insert(view);
148  }
149  return viewSet;
150 
151 }
