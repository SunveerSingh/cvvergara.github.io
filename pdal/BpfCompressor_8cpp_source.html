44 void BpfCompressor::startBlock()
45 {
46  // Initialize the stream.
47  m_strm.zalloc = Z_NULL;
48  m_strm.zfree = Z_NULL;
49  m_strm.opaque = Z_NULL;
50  if (deflateInit(&m_strm, Z_DEFAULT_COMPRESSION) != Z_OK)
51  throw error("Could not initialize BPF compressor.");
52 
53  m_rawSize = 0;
54  m_compressedSize = 0;
55  uint32_t rawSize = 0;
56  uint32_t compressedSize = 0;
57 
58  // Initiailize the streambuf with the backing buffer.
59  m_charbuf.initialize(m_inbuf.data(), m_inbuf.size());
60 
61  // Note our position so that we can rewind here.
62  m_blockStart.mark();
63  m_out << rawSize << compressedSize;
64 
65  // Make a new stream from our charbuf and push it so that future writes
66  // to our stream go to the backing vector.
67  m_out.pushStream(new std::ostream(&m_charbuf));
68 }

71 void BpfCompressor::compress()
72 {
73  // Note our position so that we know how much we've written.
74  uint32_t rawWritten = (uint32_t)m_out.position();
75 
76  // Pop our temp stream so that we can write the real output file.
77  m_out.popStream();
78 
79  m_rawSize += rawWritten;
80 
81  // Deflate the data in the buffer and write it to the output stream.
82  m_strm.avail_in = rawWritten;
83  m_strm.next_in = (unsigned char *)m_inbuf.data();
84  m_strm.avail_out = CHUNKSIZE;
85  m_strm.next_out = m_tmpbuf;
86  while (m_strm.avail_in)
87  {
88  int ret = ::deflate(&m_strm, Z_NO_FLUSH);
89  size_t written = CHUNKSIZE - m_strm.avail_out;
90  m_compressedSize += written;
91  m_out.put(m_tmpbuf, written);
92  m_strm.avail_out = CHUNKSIZE;
93  m_strm.next_out = m_tmpbuf;
94  }
95 
96  // All data has been written. Reinitialize input buffer's streambuf and
97  // push it.
98  m_charbuf.initialize(m_inbuf.data(), m_inbuf.size());
99  m_out.pushStream(new std::ostream(&m_charbuf));
100 }

103 void BpfCompressor::finish()
104 {
105  // Pop our special stream so that we can write the the file.
106  m_out.popStream();
107 
108  // Deflate and write the result to the output file.
109  int ret = Z_OK;
110  while (ret == Z_OK)
111  {
112  ret = ::deflate(&m_strm, Z_FINISH);
113  size_t written = CHUNKSIZE - m_strm.avail_out;
114  m_compressedSize += written;
115  m_out.put(m_tmpbuf, written);
116  m_strm.avail_out = CHUNKSIZE;
117  m_strm.next_out = m_tmpbuf;
118  }
119  if (ret != Z_STREAM_END)
120  throw error("Couldn't close BPF compression stream.");
121  deflateEnd(&m_strm);
122 
123  // Mark our position so that we can get back here.
124  OStreamMarker blockEnd(m_out);
125 
126  // Rewind to the start of the block and write the size bytes.
127  m_blockStart.rewind();
128  m_out << (uint32_t)m_rawSize << (uint32_t)m_compressedSize;
129 
130  // Set the position back to the end of the block.
131  blockEnd.rewind();
132 }
