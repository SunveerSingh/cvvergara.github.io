53 void LasSummaryData::addPoint(double x, double y, double z, int returnNumber)
54 {
55  ++m_totalNumPoints;
56  m_minX = (std::min)(m_minX, x);
57  m_minY = (std::min)(m_minY, y);
58  m_minZ = (std::min)(m_minZ, z);
59  m_maxX = (std::max)(m_maxX, x);
60  m_maxY = (std::max)(m_maxY, y);
61  m_maxZ = (std::max)(m_maxZ, z);
62 
63  // Returns numbers are indexed from one, but the array indexes from 0.
64  returnNumber--;
65  if (returnNumber >= 0 && (size_t)returnNumber < m_returnCounts.size())
66  m_returnCounts[returnNumber]++;
67 }
