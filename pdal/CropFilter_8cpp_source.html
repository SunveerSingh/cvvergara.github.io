147 void CropFilter::transform(const SpatialReference& srs)
148 {
149  // If we don't have any SRS, do nothing.
150  for (auto& geom : m_geoms)
151  {
152  try
153  {
154  geom = geom.transform(srs);
155  }
156  catch (pdal_error& err)
157  {
158  throwError(err.what());
159  }
160  }
161 
162  if (srs.empty() && m_assignedSrs.empty())
163  return;
164  if (srs.empty() || m_assignedSrs.empty())
165  throwError("Unable to transform crop geometry to point "
166  "coordinate system.");
167 
168  for (auto& box : m_boxes)
169  {
170  if (!gdal::reprojectBounds(box, m_assignedSrs.getWKT(), srs.getWKT()))
171  throwError("Unable to reproject bounds.");
172  }
173  for (auto& point : m_centers)
174  {
175  if (!gdal::reprojectPoint(point.x, point.y, point.z,
176  m_assignedSrs.getWKT(), srs.getWKT()))
177  throwError("Unable to reproject point center.");
178  }
179  // Set the assigned SRS for the points/bounds to the one we've
180  // transformed to.
181  m_assignedSrs = srs;
182 }

255 bool CropFilter::crop(const PointRef& point, const filter::Point& center)
256 {
257  double x = point.getFieldAs<double>(Dimension::Id::X);
258  double y = point.getFieldAs<double>(Dimension::Id::Y);
259  x = std::abs(x - center.x);
260  y = std::abs(y - center.y);
261  if (x > m_distance || y > m_distance)
262  return (m_cropOutside);
263 
264  bool inside;
265  if (center.is3d())
266  {
267  double z = point.getFieldAs<double>(Dimension::Id::Z);
268  z = std::abs(z - center.z);
269  if (z > m_distance)
270  return (m_cropOutside);
271  inside = (x * x + y * y + z * z < m_distance2);
272  }
273  else
274  inside = (x * x + y * y < m_distance2);
275  return (m_cropOutside != inside);
276 }
