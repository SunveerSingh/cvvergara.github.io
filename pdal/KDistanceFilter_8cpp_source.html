68 void KDistanceFilter::filter(PointView& view)
69 {
70  using namespace Dimension;
71 
72  // Build the 3D KD-tree.
73  log()->get(LogLevel::Debug) << "Building 3D KD-tree...\n";
74  KD3Index index(view);
75  index.build();
76 
77  // Increment the minimum number of points, as knnSearch will be returning
78  // the neighbors along with the query point.
79  m_k++;
80 
81  // Compute the k-distance for each point. The k-distance is the Euclidean
82  // distance to k-th nearest neighbor.
83  log()->get(LogLevel::Debug) << "Computing k-distances...\n";
84  for (PointId i = 0; i < view.size(); ++i)
85  {
86  std::vector<PointId> indices(m_k);
87  std::vector<double> sqr_dists(m_k);
88  index.knnSearch(i, m_k, &indices, &sqr_dists);
89  view.setField(m_kdist, i, std::sqrt(sqr_dists[m_k-1]));
90  }
91 }
