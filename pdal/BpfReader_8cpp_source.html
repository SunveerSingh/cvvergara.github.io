91 // When the stage is intialized, the schema needs to be populated with the
92 // dimensions in order to allow subsequent stages to be aware of or append to
93 // the dimensions in the PointView.
94 void BpfReader::initialize()
95 {
96  if (m_filename.empty())
97  throwError("Can't read BPF file without filename.");
98 
99  // Logfile doesn't get set until options are processed.
100  m_header.setLog(log());
101 
102  m_stream.open(m_filename);
103 
104  // Resets the stream position in case it was already open.
105  m_stream.seek(0);
106  // In order to know the dimensions we must read the file header.
107  try
108  {
109  if (!m_header.read(m_stream))
110  return;
111  if (!m_header.readDimensions(m_stream, m_dims))
112  return;
113  }
114  catch (const BpfHeader::error& err)
115  {
116  throwError(err.what());
117  }
118 #ifndef PDAL_HAVE_ZLIB
119  if (m_header.m_compression)
120  throwError("Can't read compressed BPF. PDAL wasn't built with "
121  "Zlib support.");
122 #endif
123 
124  std::string code;
125  if (m_header.m_coordType == static_cast<int>(BpfCoordType::Cartesian))
126  code = std::string("EPSG:4326");
127  else if (m_header.m_coordType == static_cast<int>(BpfCoordType::UTM))
128  {
129  uint32_t zone(abs(m_header.m_coordId));
130 
131  if (m_header.m_coordId > 0 && m_header.m_coordId <= 60)
132  code = std::string("EPSG:326");
133  else if (m_header.m_coordId < 0 && m_header.m_coordId >= -60)
134  code = std::string("EPSG:327");
135  else
136  throwError("BPF file contains an invalid UTM zone " +
137  Utils::toString(zone));
138  code += (zone < 10 ? "0" : "") + Utils::toString(zone);
139  }
140  else
141  {
142  // BPF supports something called Terrestrial Centered Rotational
143  // (BpfCoordType::TCR) and East North Up (BpfCoordType::ENU)
144  // which we can figure out when we run into a file with these
145  // coordinate systems.
146  throwError("BPF file contains unsupported coordinate system");
147  }
148  SpatialReference srs(code);
149  setSpatialReference(srs);
150 
151  try
152  {
153  SpatialReference srs(code);
154  setSpatialReference(srs);
155  }
156  catch (...)
157  {
158  log()->get(LogLevel::Error) << "Could not create an SRS" << std::endl;
159  }
160 
161  if (m_header.m_version >= 3)
162  {
163  readUlemData();
164  if (!m_stream)
165  return;
166  readUlemFiles();
167  if (!m_stream)
168  return;
169  readPolarData();
170  }
171 
172  // Read thing after the standard header as metadata->
173  readHeaderExtraData();
174 
175  // Fast forward file to end of header as reported by base header.
176  std::streampos pos = m_stream.position();
177  if (pos > m_header.m_len)
178  throwError("BPF Header length exceeded that reported by file.");
179  m_stream.close();
180 }

329 void BpfReader::readPointMajor(PointRef& point)
330 {
331  double x(0), y(0), z(0);
332 
333  seekPointMajor(m_index);
334  for (size_t dim = 0; dim < m_dims.size(); ++dim)
335  {
336  float f;
337 
338  m_stream >> f;
339  double d = f + m_dims[dim].m_offset;
340  if (m_dims[dim].m_id == Dimension::Id::X)
341  x = d;
342  else if (m_dims[dim].m_id == Dimension::Id::Y)
343  y = d;
344  else if (m_dims[dim].m_id == Dimension::Id::Z)
345  z = d;
346  else
347  point.setField(m_dims[dim].m_id, d);
348  }
349 
350  m_header.m_xform.apply(x, y, z);
351  point.setField(Dimension::Id::X, x);
352  point.setField(Dimension::Id::Y, y);
353  point.setField(Dimension::Id::Z, z);
354  m_index++;
355 }

358 point_count_t BpfReader::readPointMajor(PointViewPtr view, point_count_t count)
359 {
360  PointId nextId = view->size();
361  PointId idx = m_index;
362  point_count_t numRead = 0;
363  seekPointMajor(idx);
364  while (numRead < count && idx < numPoints())
365  {
366  for (size_t d = 0; d < m_dims.size(); ++d)
367  {
368  float f;
369 
370  m_stream >> f;
371  view->setField(m_dims[d].m_id, nextId, f + m_dims[d].m_offset);
372  }
373 
374  // Transformation only applies to X, Y and Z
375  double x = view->getFieldAs<double>(Dimension::Id::X, nextId);
376  double y = view->getFieldAs<double>(Dimension::Id::Y, nextId);
377  double z = view->getFieldAs<double>(Dimension::Id::Z, nextId);
378  m_header.m_xform.apply(x, y, z);
379  view->setField(Dimension::Id::X, nextId, x);
380  view->setField(Dimension::Id::Y, nextId, y);
381  view->setField(Dimension::Id::Z, nextId, z);
382  if (m_cb)
383  m_cb(*view, nextId);
384 
385  idx++;
386  numRead++;
387  nextId++;
388  }
389  m_index = idx;
390  return numRead;
391 }

488 void BpfReader::readByteMajor(PointRef& point)
489 {
490  // We need a temp buffer for the point data
491  union uu
492  {
493  float f;
494  uint32_t u32;
495  } u;
496  double x(0), y(0), z(0);
497  uint8_t u8;
498 
499  for (size_t dim = 0; dim < m_dims.size(); ++dim)
500  {
501  u.u32 = 0;
502  for (size_t b = 0; b < sizeof(float); ++b)

508  }

528 point_count_t BpfReader::readByteMajor(PointViewPtr data, point_count_t count)
529 {
530  PointId idx(0);
531  PointId startId = data->size();
532  point_count_t numRead = 0;
533 
534  // We need a temp buffer for the point data
535  union uu
536  {
537  float f;
538  uint32_t u32;
539  };
540  std::unique_ptr<union uu> uArr(
541  new uu[std::min(count, numPoints() - m_index)]);
542 
543  for (size_t d = 0; d < m_dims.size(); ++d)
544  {
545  for (size_t b = 0; b < sizeof(float); ++b)
546  {
547  idx = m_index;
548  numRead = 0;
549  PointId nextId = startId;
550  seekByteMajor(d, b, idx);
551 
552  for (;numRead < count && idx < numPoints();
553  idx++, numRead++, nextId++)
554  {
555  union uu& u = *(uArr.get() + numRead);
556 
557  if (b == 0)
558  u.u32 = 0;
559  uint8_t u8;
560  m_stream >> u8;
561  u.u32 |= ((uint32_t)u8 << (b * CHAR_BIT));
562  if (b == 3)
563  {
564  u.f += m_dims[d].m_offset;
565  data->setField(m_dims[d].m_id, nextId, u.f);
566  }
567  }
568  }
569  }
570  m_index = idx;
571 
572  // Transformation only applies to X, Y and Z
573  for (PointId idx = startId; idx < data->size(); idx++)
574  {
575  double x = data->getFieldAs<double>(Dimension::Id::X, idx);
576  double y = data->getFieldAs<double>(Dimension::Id::Y, idx);
577  double z = data->getFieldAs<double>(Dimension::Id::Z, idx);

635 int BpfReader::inflate(char *buf, uint32_t insize,
636  char *outbuf, uint32_t outsize)
637 {
638  if (insize == 0)
639  return 0;
640 
641  int ret;
642  z_stream strm;
643 
644  /* allocate inflate state */
645  strm.zalloc = Z_NULL;
646  strm.zfree = Z_NULL;
647  strm.opaque = Z_NULL;
648  strm.avail_in = 0;
649  strm.next_in = Z_NULL;
650  if (inflateInit(&strm) != Z_OK)
651  return -2;
652 
653  strm.avail_in = insize;
654  strm.next_in = (unsigned char *)buf;
655  strm.avail_out = outsize;
656  strm.next_out = (unsigned char *)outbuf;
657 
658  ret = ::inflate(&strm, Z_NO_FLUSH);
659  (void)inflateEnd(&strm);
660  return ret == Z_STREAM_END ? 0 : -1;
661 }
662 #endif // PDAL_HAVE_ZLIB
