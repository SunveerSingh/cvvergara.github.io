128 point_count_t GDALReader::read(PointViewPtr view, point_count_t num)
129 {
130  point_count_t count = std::min(num, m_count - m_index);
131  PointId nextId = view->size();
132 
133  std::array<double, 2> coords;
134  for (int row = 0; row < m_raster->height(); ++row)
135  {
136  for (int col = 0; col < m_raster->width(); ++col)
137  {
138  m_raster->pixelToCoord(col, row, coords);
139  view->setField(Dimension::Id::X, nextId, coords[0]);
140  view->setField(Dimension::Id::Y, nextId, coords[1]);
141  nextId++;
142  }
143  }
144 
145  std::vector<uint8_t> band;
146  std::vector<Dimension::Type> band_types =
147  m_raster->getPDALDimensionTypes();
148 
149  for (int b = 0; b < m_raster->bandCount(); ++b)
150  {
151  // Bands count from 1
152  switch (band_types[b])
153  {
154  case Dimension::Type::Signed8:
155  readBandData<int8_t>(b + 1, view, count);
156  break;
157  case Dimension::Type::Unsigned8:
158  readBandData<uint8_t>(b + 1, view, count);
159  break;
160  case Dimension::Type::Signed16:
161  readBandData<int16_t>(b + 1, view, count);
162  break;
163  case Dimension::Type::Unsigned16:
164  readBandData<uint16_t>(b + 1, view, count);
165  break;
166  case Dimension::Type::Signed32:
167  readBandData<int32_t>(b + 1, view, count);
168  break;
169  case Dimension::Type::Unsigned32:
170  readBandData<uint32_t>(b + 1, view, count);
171  break;
172  case Dimension::Type::Signed64:
173  readBandData<int64_t>(b + 1, view, count);
174  break;
175  case Dimension::Type::Unsigned64:
176  readBandData<uint64_t>(b + 1, view, count);
177  break;
178  case Dimension::Type::Float:
179  readBandData<float>(b + 1, view, count);
180  break;
181  case Dimension::Type::Double:
182  readBandData<double>(b + 1, view, count);
183  break;
184  case Dimension::Type::None:
185  break;
186  }
187  }
188  return view->size();
189 }

193 void GDALReader::readBandData(int band, PointViewPtr view, point_count_t count)
194 {
195  std::vector<T> buf;
196 
197  m_raster->readBand(buf, band);
198  std::stringstream oss;
199  oss << "band-" << (band);
200  log()->get(LogLevel::Info) << "Read band '" << oss.str() << "'" <<
201  std::endl;
202  Dimension::Id d = view->layout()->findDim(oss.str());
203  for (point_count_t i = 0; i < count; ++i)
204  view->setField(d, i, buf[i]);
205 }
