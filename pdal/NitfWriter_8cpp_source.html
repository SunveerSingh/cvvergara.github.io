80 BOX3D NitfWriter::reprojectBoxToDD(const SpatialReference& reference,
81  const BOX3D& box)
82 {
83  if (reference.empty())
84  return BOX3D();
85 
86  BOX3D output(box);
87  if (!gdal::reprojectBounds(output, reference.getWKT(), "EPSG:4326"))
88  throwError("Couldn't reproject corner points to geographic: " +
89  gdal::lastError());
90  return output;
91 }

133 void NitfWriter::doneFile()
134 {
135  finishOutput();
136 
137  std::streambuf *buf = m_oss.rdbuf();
138  long size = buf->pubseekoff(0, m_oss.end);
139  buf->pubseekoff(0, m_oss.beg);
140 
141  std::vector<char> bytes(size);
142  buf->sgetn(bytes.data(), size);
143  m_oss.clear();
144  m_nitf.wrapData(bytes.data(), size);
145  m_nitf.setBounds(reprojectBoxToDD(m_srs, m_bounds));
146 
147  try
148  {
149  m_nitf.write();
150  }
151  catch (const NitfFileWriter::error& err)
152  {
153  throwError(err.what());
154  }
