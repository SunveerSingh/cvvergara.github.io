104 void OverlayFilter::ready(PointTableRef table)
105 {
106  m_ds = OGRDSPtr(OGROpen(m_datasource.c_str(), 0, 0),
107  OGRDataSourceDeleter());
108  if (!m_ds)
109  throwError("Unable to open data source '" + m_datasource + "'");
110 
111  if (m_layer.size())
112  m_lyr = OGR_DS_GetLayerByName(m_ds.get(), m_layer.c_str());
113  else if (m_query.size())
114  m_lyr = OGR_DS_ExecuteSQL(m_ds.get(), m_query.c_str(), 0, 0);
115  else
116  m_lyr = OGR_DS_GetLayer(m_ds.get(), 0);
117 
118  if (!m_lyr)
119  throwError("Unable to select layer '" + m_layer + "'");
120 
121  OGRFeaturePtr feature = OGRFeaturePtr(OGR_L_GetNextFeature(m_lyr),
122  OGRFeatureDeleter());
123 
124  int field_index(1); // default to first column if nothing was set
125  if (m_column.size())
126  {
127  field_index = OGR_F_GetFieldIndex(feature.get(), m_column.c_str());
128  if (field_index == -1)
129  throwError("No column name '" + m_column + "' was found.");
130  }
131 
132  do
133  {
134  OGRGeometryH geom = OGR_F_GetGeometryRef(feature.get());
135  OGRwkbGeometryType t = OGR_G_GetGeometryType(geom);
136  int32_t fieldVal = OGR_F_GetFieldAsInteger(feature.get(), field_index);
137 
138  if (!(t == wkbPolygon ||
139  t == wkbMultiPolygon ||
140  t == wkbPolygon25D ||
141  t == wkbMultiPolygon25D))
142  {
143  throwError("Geometry is not Polygon or MultiPolygon!");
144  }
145 
146  // Don't think Polygon meets criteria for implicit move ctor.
147  m_polygons.push_back(
148  { Polygon(geom, table.anySpatialReference()), fieldVal} );
149 
150  feature = OGRFeaturePtr(OGR_L_GetNextFeature(m_lyr),
151  OGRFeatureDeleter());
152  }
153  while (feature);
154 }

173 bool OverlayFilter::processOne(PointRef& point)
174 {

177  point.setField(m_dim, poly.val);

182 void OverlayFilter::filter(PointView& view)
183 {
184  QuadIndex idx(view);
185 
186  for (const auto& poly : m_polygons)
187  {
188  std::vector<PointId> ids = idx.getPoints(poly.geom.bounds());
189 
190  PointRef point(view, 0);
191  for (PointId id : ids)
192  {
193  point.setPointId(id);
194  if (poly.geom.covers(point))
195  point.setField(m_dim, poly.val);
196  }
197  }
198 }
