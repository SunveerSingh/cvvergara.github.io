249 bool XMLSchema::validate(xmlDocPtr doc, const std::string& xsd)
250 {
251  xmlParserOption parserOption(XML_PARSE_NONET);
252 
253  xmlDocPtr schemaDoc = xmlReadMemory(xsd.c_str(), xsd.size(),
254  NULL, NULL, parserOption);
255  xmlSchemaParserCtxtPtr parserCtxt = xmlSchemaNewDocParserCtxt(schemaDoc);
256  xmlSchemaSetParserStructuredErrors(parserCtxt,
257  &OCISchemaParserStructuredErrorHandler, m_global_context);
258  xmlSchemaPtr schema = xmlSchemaParse(parserCtxt);
259  xmlSchemaValidCtxtPtr validCtxt = xmlSchemaNewValidCtxt(schema);
260  xmlSchemaSetValidErrors(validCtxt, &OCISchemaValidityError,
261  &OCISchemaValidityDebug, m_global_context);
262  bool valid = (xmlSchemaValidateDoc(validCtxt, doc) == 0);
263 
264  xmlFreeDoc(schemaDoc);
265  xmlSchemaFreeParserCtxt(parserCtxt);
266  xmlSchemaFree(schema);
267  xmlSchemaFreeValidCtxt(validCtxt);
268 
269  return valid;
270 }

287 bool XMLSchema::loadMetadata(xmlNode *startNode, MetadataNode& input)
288 {
289 // Expect metadata in the following form
290 // We are going to skip the root element because we are
291 // expecting to be given one with our input
292 // <pc:metadata>
293 // <Metadata name="root" type="">
294 // <Metadata name="compression" type="string">lazperf</Metadata>
295 // <Metadata name="version" type="string">1.0</Metadata>
296 // </Metadata>
297 // </pc:metadata>
298 
299  xmlNode *node = startNode;
300  for (node = startNode; node; node = node->next)
301  {
302  if (node->type != XML_ELEMENT_NODE)
303  continue;
304  if (std::string((const char*)node->name) == "Metadata")
305  {
306  const char *fieldname =
307  (const char*)xmlGetProp(node, (const xmlChar*)"name");
308  const char *etype =
309  (const char*)xmlGetProp(node, (const xmlChar*)"type");
310  const char *description =
311  (const char*)xmlGetProp(node, (const xmlChar*) "description");
312  const char *text = (const char*)xmlNodeGetContent(node);
313 
314  if (!Utils::iequals(fieldname, "root"))
315  {
316  if (!fieldname)
317  {
318  std::cerr << "Unable to read metadata for node '" <<
319  (const char*)node->name << "' no \"name\" was given";
320  return false;
321  }
322  input.add(fieldname, text ? text : "",
323  description ? description : "");
324  }
325  }
326  loadMetadata(node->children, input);
327  }
328  return true;
329 }

332 bool XMLSchema::load(xmlDocPtr doc)
333 {
334  xmlNode* root = xmlDocGetRootElement(doc);
335  // print_element_names(root);
336 
337  if (!Utils::iequals((const char*)root->name, "PointCloudSchema"))
338  {
339  std::cerr << "First node of document was not named 'PointCloudSchema'";
340  return false;
341  }
342 
343  const unsigned SENTINEL_POS = 100000;
344  unsigned missingPos = SENTINEL_POS + 1;
345 
346  xmlNode* dimension = root->children;
347  pdal::Metadata metadata;
348  for (xmlNode *dimension = root->children; dimension;
349  dimension = dimension->next)
350  {
351  // Read off orientation setting
352  if (std::string((const char*)dimension->name) == "orientation")
353  {
354  xmlChar* n = xmlNodeListGetString(doc, dimension->children, 1);
355  if (!n)
356  {
357  std::cerr << "Unable to fetch orientation.\n";
358  return false;
359  }
360  std::string orientation = std::string((const char*)n);
361  xmlFree(n);
362 
363  if (Utils::iequals(orientation, "dimension"))
364  m_orientation = Orientation::DimensionMajor;
365  else
366  m_orientation = Orientation::PointMajor;
367  continue;
368  }
369 
370  if (std::string((const char*)dimension->name) == "metadata")
371  {
372  m_metadata = MetadataNode("root");
373  if (!loadMetadata(dimension, m_metadata))
374  return false;
375  continue;
376  }
377 
378  if (dimension->type != XML_ELEMENT_NODE ||
379  !Utils::iequals((const char*)dimension->name, "dimension"))
380  continue;
381 
382  XMLDim dim;
383  dim.m_position = SENTINEL_POS;
384  for (xmlNode *properties = dimension->children; properties;
385  properties = properties->next)
386  {
387  if (properties->type != XML_ELEMENT_NODE)
388  continue;
389 
390  std::string propName = (const char *)properties->name;
391  propName = Utils::tolower(propName);
392  if (propName == "name")
393  {
394  xmlChar *n = xmlNodeListGetString(doc, properties->children, 1);
395  if (!n)
396  {
397  std::cerr << "Unable to fetch name from XML node.";
398  return false;
399  }
400  dim.m_name = remapOldNames(std::string((const char*)n));
401  xmlFree(n);
402  }
403  if (propName == "description")
404  {
405  xmlChar* n = xmlNodeListGetString(doc, properties->children, 1);
406  if (!n)
407  {
408  std::cerr << "Unable to fetch description.\n";
409  return false;
410  }
411  dim.m_description = std::string((const char*)n);
412  xmlFree(n);
413  }
414  if (propName == "interpretation")
415  {
416  xmlChar* n = xmlNodeListGetString(doc, properties->children, 1);
417  if (!n)
418  {
419  std::cerr << "Unable to fetch interpretation.\n";
420  return false;
421  }
422  dim.m_dimType.m_type = Dimension::type((const char*)n);
423  xmlFree(n);
424  }
425  if (propName == "minimum")
426  {
427  xmlChar* n = xmlGetProp(properties, (const xmlChar*) "value");
428  if (!n)
429  {
430  return false;
431  std::cerr << "Unable to fetch minimum value.\n";
432  }
433  dim.m_min = std::atof((const char*)n);
434  xmlFree(n);
435  }
436  if (propName == "maximum")
437  {
438  xmlChar* n = xmlGetProp(properties, (const xmlChar*) "value");
439  if (!n)
440  {
441  std::cerr << "Unable to fetch maximum value.\n";
442  return false;
443  }
444  dim.m_max = std::atof((const char*)n);
445  xmlFree(n);
446  }
447  if (propName == "position")
448  {
449  xmlChar* n = xmlNodeListGetString(doc, properties->children, 1);
450  if (!n)
451  {
452  std::cerr << "Unable to fetch position value.\n";
453  return false;
454  }
455  dim.m_position = std::atoi((const char*)n);
456  xmlFree(n);
457  }
458  if (propName == "offset")
459  {
460  xmlChar* n = xmlNodeListGetString(doc, properties->children, 1);
461  if (!n)
462  {
463  std::cerr << "Unable to fetch offset value!";
464  return false;
465  }
466  dim.m_dimType.m_xform.m_offset.set((const char*)n);
467  xmlFree(n);
468  }
469  if (propName == "scale")
470  {
471  xmlChar* n = xmlNodeListGetString(doc, properties->children, 1);
472  if (!n)
473  {
474  std::cerr << "Unable to fetch scale value!";
475  return false;
476  }
477  dim.m_dimType.m_xform.m_scale.set((const char*)n);
478  xmlFree(n);
479  }
480  }
481  // If we don't have a position, set it to some value larger than all
482  // previous values.
483  if (dim.m_position == SENTINEL_POS)
484  dim.m_position = missingPos++;
485  m_dims.push_back(dim);
486  }
487  std::sort(m_dims.begin(), m_dims.end());
488 
489  // Renumber dimension positions to be 1..N
490  for (unsigned pos = 0; pos < m_dims.size(); pos++)
491  m_dims[pos].m_position = pos + 1;
492 
493  return true;
494 }

564 void XMLSchema::writeXml(xmlTextWriterPtr w) const
565 {
566  int pos = 0;
567  for (auto di = m_dims.begin(); di != m_dims.end(); ++di, ++pos)
568  {
569  xmlTextWriterStartElementNS(w, (const xmlChar*)"pc",
570  (const xmlChar*)"dimension", NULL);
571 
572  std::ostringstream position;
573  position << (pos + 1);
574  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
575  (const xmlChar*)"position", NULL,
576  (const xmlChar*)position.str().c_str());
577 
578  std::ostringstream size;
579  size << Dimension::size(di->m_dimType.m_type);
580  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
581  (const xmlChar*)"size", NULL, (const xmlChar*)size.str().c_str());
582 
583  std::string description = Dimension::description(di->m_dimType.m_id);
584  if (description.size())
585  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
586  (const xmlChar*)"description", NULL,
587  (const xmlChar*)description.c_str());
588 
589  XForm xform = di->m_dimType.m_xform;
590  if (xform.nonstandard())
591  {
592  std::ostringstream out;
593  out.precision(15);
594 
595  out << xform.m_scale.m_val;
596  std::string scale = out.str();
597 
598  out.str(std::string());
599  out << xform.m_offset.m_val;
600  std::string offset = out.str();
601 
602  out << xform.m_scale.m_val;
603  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
604  (const xmlChar *)"scale", NULL,
605  (const xmlChar *)scale.data());
606  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
607  (const xmlChar *)"offset", NULL,
608  (const xmlChar *)offset.data());
609  }
610 
611  std::string name = di->m_name;
612  if (name.size())
613  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
614  (const xmlChar*)"name", NULL, (const xmlChar*)name.c_str());
615 
616  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
617  (const xmlChar*)"interpretation", NULL,
618  (const xmlChar*)
619  Dimension::interpretationName(di->m_dimType.m_type).c_str());
620 
621  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
622  (const xmlChar*)"active", NULL, (const xmlChar*)"true");
623 
624  xmlTextWriterEndElement(w);
625  xmlTextWriterFlush(w);
626  }
627  std::ostringstream orientation;
628  if (m_orientation == Orientation::PointMajor)
629  orientation << "point";
630  if (m_orientation == Orientation::DimensionMajor)
631  orientation << "dimension";
632  if (!m_metadata.empty())
633  {
634  addMetadataEntry(w, m_metadata);
635  }
636  xmlTextWriterWriteElementNS(w, (const xmlChar*) "pc",
637  (const xmlChar*)"orientation", NULL,
638  (const xmlChar*)orientation.str().c_str());
639 
640  xmlTextWriterWriteElementNS(w, (const xmlChar*)"pc",
641  (const xmlChar*)"version", NULL,
642  (const xmlChar*)PDAL_XML_SCHEMA_VERSION);
643 
644  xmlTextWriterEndElement(w);
645  xmlTextWriterFlush(w);
646 }
