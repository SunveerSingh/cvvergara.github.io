83 std::vector<PArray> Pipeline::getArrays() const
84 {

86 
87  if (!m_executor.executed())
88  throw python_error("call execute() before fetching arrays");
89 
90  const pdal::PointViewSet& pvset = m_executor.getManagerConst().views();
91 
92  for (auto i: pvset)
93  {
94  PArray array = new pdal::python::Array;
95  array->update(i);

97  }
