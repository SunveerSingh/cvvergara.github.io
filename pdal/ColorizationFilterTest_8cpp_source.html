49 void testFile(const Options& filterOps, StringList dimNames,
50  uint16_t expectedRed, uint16_t expectedGreen, uint16_t expectedBlue)
51 {
52  Options readerOps;
53  readerOps.add("filename",
54  Support::datapath("autzen/autzen-point-format-3.las"));
55 
56  LasReader reader;
57  reader.setOptions(readerOps);
58 
59  ColorizationFilter filter;
60  filter.setOptions(filterOps);
61  filter.setInput(reader);
62 
63  PointTable table;
64 
65  filter.prepare(table);
66  PointViewSet viewSet = filter.execute(table);

68  PointViewPtr view = *viewSet.begin();
69 
70  PointLayoutPtr layout = table.layout();

72  uint16_t g = view->getFieldAs<uint16_t>(layout->findDim(dimNames[1]), 0);
73  uint16_t b = view->getFieldAs<uint16_t>(layout->findDim(dimNames[2]), 0);
74 
75  EXPECT_EQ(r, expectedRed);
76  EXPECT_EQ(g, expectedGreen);
77  // We scaled this up to 16bit by multiplying by 255
78  EXPECT_EQ(b, expectedBlue);

112  if (done)
113  return false;

117  EXPECT_EQ(expectedBlue, point.getFieldAs<uint16_t>(id3));
118  done = true;
119 
120  return true;
121  };
122
