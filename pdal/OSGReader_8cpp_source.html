79  void apply(osg::Transform& transform)
80  {
81  osg::Matrix matrix;
82  if (!_matrixStack.empty()) matrix = _matrixStack.back();
83 
84  transform.computeLocalToWorldMatrix(matrix, this);
85 
86  pushMatrix(matrix);
87 
88  traverse(transform);
89 
90  popMatrix();

93  void apply(osg::Geode& geode)
94  {
95  for (unsigned int i = 0; i<geode.getNumDrawables(); ++i)
96  {
97  osg::TriangleFunctor<CollectTriangles> triangleCollector;
98  geode.getDrawable(i)->accept(triangleCollector);
99  for (unsigned int j = 0; j < triangleCollector.verts->size(); j++)
100  {
101  osg::Matrix matrix;
102  if (!_matrixStack.empty())
103  {
104  matrix = _matrixStack.back();
105  }
106  osg::Vec3d v = (*triangleCollector.verts)[j];
107  _points.push_back(v * matrix);
108  }
109  }
110  }
111 
112  inline void pushMatrix(osg::Matrix& matrix) { _matrixStack.push_back(matrix); }
113 
114  inline void popMatrix() { _matrixStack.pop_back(); }
