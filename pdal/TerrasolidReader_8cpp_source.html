55 void TerrasolidReader::initialize()
56 {
57  ILeStream stream(m_filename);
58 
59  TerraSolidHeaderPtr h(new TerraSolidHeader);
60  m_header.swap(h);
61 
62  stream >> m_header->HdrSize >> m_header->HdrVersion >> m_header->RecogVal;
63  stream.get(m_header->RecogStr, 4);
64  stream >> m_header->PntCnt >> m_header->Units >> m_header->OrgX >>
65  m_header->OrgY >> m_header->OrgZ >> m_header->Time >> m_header->Color;
66 
67  if (m_header->RecogVal != 970401)
68  throwError("Header identifier was not '970401', is this "
69  "a TerraSolid .bin file?");
70 
71  m_haveColor = (m_header->Color != 0);
72  m_haveTime = (m_header->Time != 0);
73  m_format = static_cast<TERRASOLID_Format_Type>(m_header->HdrVersion);
74 
75  if ((m_format != TERRASOLID_Format_1) && (m_format != TERRASOLID_Format_2))
76  throwError("Version was '" + Utils::toString(m_format) + "', not '" +
77  Utils::toString(TERRASOLID_Format_1) + "' or '" +
78  Utils::toString(TERRASOLID_Format_2) + "'");
79 
80  log()->get(LogLevel::Debug) << "TerraSolid Reader::initialize format: " <<
81  m_format << std::endl;
82  log()->get(LogLevel::Debug) << "OrgX: " << m_header->OrgX << std::endl;
83  log()->get(LogLevel::Debug) << "OrgY: " << m_header->OrgY << std::endl;
84  log()->get(LogLevel::Debug) << "OrgZ: " << m_header->OrgZ << std::endl;
85  log()->get(LogLevel::Debug) << "Units: " << m_header->Units << std::endl;
86  log()->get(LogLevel::Debug) << "Time: " << m_header->Time << std::endl;
87  log()->get(LogLevel::Debug) << "Color: " << m_header->Color << std::endl;
88  log()->get(LogLevel::Debug) << "Count: " << m_header->PntCnt << std::endl;
89  log()->get(LogLevel::Debug) << "RecogVal: " << m_header->RecogVal <<
90  std::endl;
91 }

94 void TerrasolidReader::addDimensions(PointLayoutPtr layout)
95 {
96  m_size = 0;
97  layout->registerDim(Dimension::Id::Classification);
98  layout->registerDim(Dimension::Id::PointSourceId);
99  layout->registerDim(Dimension::Id::Intensity);
100  layout->registerDim(Dimension::Id::X);
101  layout->registerDim(Dimension::Id::Y);
102  layout->registerDim(Dimension::Id::Z);
103  layout->registerDim(Dimension::Id::ReturnNumber);
104  layout->registerDim(Dimension::Id::NumberOfReturns);
105  if (m_format == TERRASOLID_Format_2)
106  {
107  layout->registerDim(Dimension::Id::Flag);
108  layout->registerDim(Dimension::Id::Mark);
109  }
110  if (m_format == TERRASOLID_Format_1)
111  m_size = 16;
112  else if (m_format == TERRASOLID_Format_2)
113  m_size = 20;
114 
115  if (m_haveTime)
116  {
117  layout->registerDim(Dimension::Id::OffsetTime);
118  m_size += 4;
119  }
120 
121  if (m_haveColor)
122  {
123  layout->registerDim(Dimension::Id::Red);
124  layout->registerDim(Dimension::Id::Green);
125  layout->registerDim(Dimension::Id::Blue);
126  layout->registerDim(Dimension::Id::Alpha);
127  m_size += 4;
128  }

141 point_count_t TerrasolidReader::read(PointViewPtr view, point_count_t count)
142 {
143  count = std::min(count, getNumPoints() - m_index);
144 
145  std::vector<char> buf(m_size * count);
146  m_istream->get(buf);
147  LeExtractor extractor(buf.data(), buf.size());
148 
149  // See https://www.terrasolid.com/download/tscan.pdf
150  // This spec is awful, but it's something.
151  // The scaling adjustments are different than what we used to do and
152  // seem wrong (scaling the offset is odd), but that's what the document
153  // says.
154  // Also modified the fetch of time/color based on header flag (rather
155  // than just not write the data into the buffer).
156  PointId nextId = view->size();
157  while (!eof())
158  {
159  if (m_format == TERRASOLID_Format_1)
160  {
161  uint8_t classification, flight_line, echo_int, x, y, z;
162 
163  extractor >> classification >> flight_line >> echo_int >> x >> y >>
164  z;
165 
166  view->setField(Dimension::Id::Classification, nextId,
167  classification);
168  view->setField(Dimension::Id::PointSourceId, nextId, flight_line);
169  switch (echo_int)
170  {
171  case 0: // only echo
172  view->setField(Dimension::Id::ReturnNumber, nextId, 1);
173  view->setField(Dimension::Id::NumberOfReturns, nextId, 1);
174  break;
175  case 1: // first of many echos
176  view->setField(Dimension::Id::ReturnNumber, nextId, 1);
177  break;
178  default: // intermediate echo or last of many echos
179  break;
180  }
181  view->setField(Dimension::Id::X, nextId,
182  (x - m_header->OrgX) / m_header->Units);
183  view->setField(Dimension::Id::Y, nextId,
184  (y - m_header->OrgY) / m_header->Units);
185  view->setField(Dimension::Id::Z, nextId,
186  (z - m_header->OrgZ) / m_header->Units);
187  }
188 
189  if (m_format == TERRASOLID_Format_2)
190  {
191  int32_t x, y, z;
192  uint8_t classification, echo_int, flag, mark;
193  uint16_t flight_line, intensity;
194 
195  extractor >> x >> y >> z >> classification >> echo_int >> flag >>
196  mark >> flight_line >> intensity;
197 
198  view->setField(Dimension::Id::X, nextId,
199  (x - m_header->OrgX) / m_header->Units);
200  view->setField(Dimension::Id::Y, nextId,
201  (y - m_header->OrgY) / m_header->Units);
202  view->setField(Dimension::Id::Z, nextId,
203  (z - m_header->OrgZ) / m_header->Units);
204  view->setField(Dimension::Id::Classification, nextId,
205  classification);
206  switch (echo_int)
207  {
208  case 0: // only echo
209  view->setField(Dimension::Id::ReturnNumber, nextId, 1);
210  view->setField(Dimension::Id::NumberOfReturns, nextId, 1);
211  break;
212  case 1: // first of many echos
213  view->setField(Dimension::Id::ReturnNumber, nextId, 1);
214  break;
215  default: // intermediate echo or last of many echos
216  break;
217  }
218  view->setField(Dimension::Id::Flag, nextId, flag);
219  view->setField(Dimension::Id::Mark, nextId, mark);
220  view->setField(Dimension::Id::PointSourceId, nextId, flight_line);
221  view->setField(Dimension::Id::Intensity, nextId, intensity);
222  }
223 
224  if (m_haveTime)
225  {
226  uint32_t t;
227 
228  extractor >> t;
229 
230  if (m_index == 0)
231  m_baseTime = t;
232  t -= m_baseTime; // Offset from the beginning of the read.
233  // instead of GPS week.
234  t /= 5; // 5000ths of a second to milliseconds
235  view->setField(Dimension::Id::OffsetTime, nextId, t);
236  }
237 
238  if (m_haveColor)
239  {
240  uint8_t red, green, blue, alpha;
241 
242  extractor >> red >> green >> blue >> alpha;
243 
244  view->setField(Dimension::Id::Red, nextId, red);
245  view->setField(Dimension::Id::Green, nextId, green);
246  view->setField(Dimension::Id::Blue, nextId, blue);
247  view->setField(Dimension::Id::Alpha, nextId, alpha);
248  }
249 
250  if (m_cb)
251  m_cb(*view, nextId);
252 
253  nextId++;
254  m_index++;
255  }
256 
257  return count;
258 }
