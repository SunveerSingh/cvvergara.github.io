61 void FerryFilter::initialize()
62 {
63  std::vector<std::string> toNames;
64  for (auto& dim : m_dimSpec)
65  {
66  StringList s = Utils::split(dim, '=');
67  if (s.size() != 2)
68  throwError("Invalid dimension specified '" + dim + "'. Need "
69  "<from dimension>=<to dimension>. See documentation for "
70  "details.");
71  Utils::trim(s[0]);
72  Utils::trim(s[1]);
73  if (s[0] == s[1])
74  throwError("Can't ferry dimension '" + s[0] + "' to itself.");
75  if (Utils::contains(toNames, s[1]))
76  throwError("Can't ferry two source dimensions to the same "
77  "destination dimension.");
78  toNames.push_back(s[1]);
79  m_dims.emplace_back( s[0], s[1] );
80  }
81 }

104 bool FerryFilter::processOne(PointRef& point)
105 {
106  for (const auto& info : m_dims)
107  {
108  if (info.m_fromId != Dimension::Id::Unknown)
109  {
110  double v = point.getFieldAs<double>(info.m_fromId);
111  point.setField(info.m_toId, v);
112  }
113  }
114  return true;
115 }
