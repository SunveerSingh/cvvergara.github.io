45 // Build the list of dimensions for the output schema.
46 // Placing this here allows validation of dimensions before execution begins.
47 void DbWriter::prepared(PointTableRef table)
48 {
49  using namespace Dimension;
50 
51  PointLayoutPtr layout = table.layout();
52 
53  if (m_outputDims.empty())
54  {
55  for (auto& dimType : layout->dimTypes())
56  m_dbDims.push_back(XMLDim(dimType, layout->dimName(dimType.m_id)));
57  return;
58  }
59 
60  DimTypeList dims;
61  for (std::string& s : m_outputDims)
62  {
63  DimType dt = layout->findDimType(s);
64  if (dt.m_id == Id::Unknown)
65  {
66  std::ostringstream oss;
67  oss << "Invalid dimension '" << s << "' specified for "
68  "'output_dims' option.";
69  throw pdal_error(oss.str());
70  }
71  m_dbDims.push_back(XMLDim(dt, layout->dimName(dt.m_id)));
72  }

76 void DbWriter::ready(PointTableRef /*table*/)
77 {
78  using namespace Dimension;
79 
80  // Determine if X, Y and Z values should be written as Signed32 along with
81  // a scale factor and offset instead of being written as Double.
82  m_locationScaling = m_scaling.nonstandard();
83 
84  auto cmp = [](const XMLDim& d1, const XMLDim& d2) -> bool
85  {
86  long id1 = Utils::toNative(d1.m_dimType.m_id);
87  long id2 = Utils::toNative(d2.m_dimType.m_id);
88 
89  const auto isXyz([](long native)->bool
90  {
91  const Id e(static_cast<Id>(native));
92  return (e == Id::X || e == Id::Y || e == Id::Z);
93  });
94 
95  // Put X, Y and Z at the end of the list.
96  if (isXyz(id1))
97  id1 += 1000000;
98  if (isXyz(id2))
99  id2 += 1000000;
100  return id1 < id2;
101  };
102 
103  // Sort the dimensions so that X, Y & Z are at the end.
104  std::sort(m_dbDims.begin(), m_dbDims.end(), cmp);
105 
106  // Suck the dimTypes out of the dbDims so that they can be used to
107  // retrieve data from the point table.
108  // Set the packed type into the dbDim if necessary and save off the
109  // index of X, Y and Z for location scaling.
110  m_dimTypes.clear();
111  m_xOffsets = std::make_pair(-1, -1);
112  m_yOffsets = std::make_pair(-1, -1);
113  m_zOffsets = std::make_pair(-1, -1);
114  m_packedPointSize = 0;
115  m_dbPointSize = 0;
116  for (auto& xmlDim : m_dbDims)
117  {
118  m_dimTypes.push_back(xmlDim.m_dimType);
119  DimType& dt = m_dimTypes.back();
120  // Dim types are stored in the map to allow fast access in readField.
121  m_dimMap[Utils::toNative(dt.m_id)] = dt;
122 
123  if (m_locationScaling)
124  {
125  if (xmlDim.m_dimType.m_id == Id::X)
126  {
127  xmlDim.m_dimType.m_xform = m_scaling.m_xXform;
128  xmlDim.m_dimType.m_type = Type::Signed32;
129  m_xOffsets = std::make_pair(m_packedPointSize, m_dbPointSize);
130  }
131  if (xmlDim.m_dimType.m_id == Id::Y)
132  {
133  xmlDim.m_dimType.m_xform = m_scaling.m_yXform;
134  xmlDim.m_dimType.m_type = Type::Signed32;
135  m_yOffsets = std::make_pair(m_packedPointSize, m_dbPointSize);
136  }
137  if (xmlDim.m_dimType.m_id == Id::Z)
138  {
139  xmlDim.m_dimType.m_xform = m_scaling.m_zXform;
140  xmlDim.m_dimType.m_type = Type::Signed32;
141  m_zOffsets = std::make_pair(m_packedPointSize, m_dbPointSize);
142  }
143  }
144  m_packedPointSize += Dimension::size(dt.m_type);
145  m_dbPointSize += Dimension::size(xmlDim.m_dimType.m_type);
146  }
147 }

168 /// Read a field from a PointView and write its value as formatted for output
169 /// to the DB schema to the location as requested.
170 /// \param[in] view PointView to read from.
171 /// \param[in] pos Location in which to store field value.
172 /// \param[in] id ID of the dimension to read.
173 /// \param[in] idx Index of point to read.
174 /// \return Size of field as read.
175 size_t DbWriter::readField(const PointView& view, char *pos,
176  Dimension::Id id, PointId idx)
177 {
178  using namespace Dimension;
179 
180  DimType& dt = m_dimMap[(int)id];
181  size_t size = Dimension::size(dt.m_type);
182 
183  // Using the ID instead of a dimType as the arugment hides the complication
184  // of the "type" of the dimension to retrieve in the case of location
185  // scaling.
186  view.getField(pos, id, dt.m_type, idx);
187 
188  auto iconvert = [pos](const XForm& xform, Dimension::Id dim)
189  {
190  double d;
191  int32_t i;
192 
193  memcpy(&d, pos, sizeof(double));
194 
195  d = xform.toScaled(d);
196  if (!Utils::numericCast(d, i))
197  {
198  std::ostringstream oss;
199  oss << "Unable to convert double to int32 for packed DB output: ";
200  oss << Dimension::name(dim) << ": (" << d << ").";
201  throw pdal_error(oss.str());
202  }
203  memcpy(pos, &i, sizeof(int32_t));
204  };
205 
206  if (m_locationScaling)
207  {
208  // For X, Y or Z.
209  if (id == Id::X)
210  {
211  iconvert(m_scaling.m_xXform, Id::X);
212  size = sizeof(int32_t);
213  }
214  else if (id == Id::Y)
215  {
216  iconvert(m_scaling.m_yXform, Id::Y);
217  size = sizeof(int32_t);
218  }
219  else if (id == Id::Z)
220  {
221  iconvert(m_scaling.m_zXform, Id::Z);
222  size = sizeof(int32_t);
223  }
224  }
225  return size;
226 }

229 /// Read a point's data packed into a buffer.
230 /// \param[in] view PointView to read from.
231 /// \param[in] idx Index of point to read.
232 /// \param[in] outbuf Buffer to write to.
233 /// \return Number of bytes written to buffer.
234 size_t DbWriter::readPoint(const PointView& view, PointId idx, char *outbuf)
235 {
236  using namespace Dimension;
237 
238  // Read the data for the output dimensions from the view into the outbuf.
239  view.getPackedPoint(m_dimTypes, idx, outbuf);
240 
241  auto iconvert = [](const XForm& xform, Id dim,
242  const char *inpos, char *outpos)
243  {
244  double d;
245  int32_t i;
246 
247  memcpy(&d, inpos, sizeof(double));
248  d = xform.toScaled(d);
249  if (!Utils::numericCast(d, i))
250  {
251  std::ostringstream oss;
252  oss << "Unable to convert double to int32 for packed DB output: ";
253  oss << Dimension::name(dim) << ": (" << d << ").";
254  throw pdal_error(oss.str());
255  }
256  memcpy(outpos, &i, sizeof(int32_t));
257  };
258 
259  if (m_xOffsets.first >= 0)
260  iconvert(m_scaling.m_xXform, Id::X, outbuf + (size_t)m_xOffsets.first,
261  outbuf + (size_t)m_xOffsets.second);
262  if (m_yOffsets.first >= 0)
263  iconvert(m_scaling.m_yXform, Id::Y, outbuf + (size_t)m_yOffsets.first,
264  outbuf + (size_t)m_yOffsets.second);
265  if (m_zOffsets.first >= 0)
266  iconvert(m_scaling.m_zXform, Id::Z, outbuf + (size_t)m_zOffsets.first,
267  outbuf + (size_t)m_zOffsets.second);
268  return m_dbPointSize;
269 }
