76 void PointView::setFieldInternal(Dimension::Id dim, PointId idx,
77  const void *buf)
78 {
79  PointId rawId = 0;
80  if (idx == size())
81  {
82  rawId = m_pointTable.addPoint();
83  m_index.push_back(rawId);
84  m_size++;
85  assert(m_temps.empty());

89  std::cerr << "Point index must increment.\n";

91  return;
92  }
93  else
94  {
95  rawId = m_index[idx];
96  }
97  m_pointTable.setFieldInternal(dim, rawId, buf);

217 void PointView::dump(std::ostream& ostr) const
218 {
219  using std::endl;
220  PointLayoutPtr layout = m_pointTable.layout();
221  const Dimension::IdList& dims = layout->dims();
222 
223  point_count_t numPoints = size();
224  ostr << "Contains " << numPoints << " points" << endl;
225  for (PointId idx = 0; idx < numPoints; idx++)
226  {
227  ostr << "Point: " << idx << endl;
228 
229  for (auto di = dims.begin(); di != dims.end(); ++di)
230  {
231  Dimension::Id d = *di;
232  const Dimension::Detail *dd = layout->dimDetail(d);
233  ostr << Dimension::name(d) << " (" <<
234  Dimension::interpretationName(dd->type()) << ") : ";
235 
236  switch (dd->type())
237  {
238  case Dimension::Type::Signed8:
239  {
240  ostr << (int)(getFieldInternal<int8_t>(d, idx));
241  break;
242  }

245  ostr << getFieldInternal<int16_t>(d, idx);
