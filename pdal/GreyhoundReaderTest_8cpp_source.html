60 Options toOptions(const Json::Value& json)
61 {
62  Options options;
63  options.add("url", server);
64  options.add("resource", resource);
65 
66  for (const auto k : json.getMemberNames())
67  {
68  const auto& v(json[k]);
69  if (v.isBool()) options.add(k, v.asBool());
70  else if (v.isIntegral()) options.add(k, v.asInt());
71  else if (v.isDouble()) options.add(k, v.asDouble());
72  else if (v.isString()) options.add(k, v.asString());
73  else if (v.isObject() || v.isArray()) options.add(k, v);
74  }
75 
76  return options;
77 }
