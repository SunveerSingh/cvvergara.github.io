98 void BpfWriter::initialize()
99 {
100  m_header.m_coordId = m_coordId.m_val;
101  m_header.m_coordType = Utils::toNative(m_header.m_coordId ?
102  BpfCoordType::UTM : BpfCoordType::Cartesian);
103 #ifndef PDAL_HAVE_ZLIB
104  if (m_compression)
105  throwError("Can't write compressed BPF. PDAL wasn't built with "
106  "Zlib support.");
107 #endif
108  m_header.m_compression = Utils::toNative(
109  m_compression ? BpfCompression::Zlib : BpfCompression::None);
110  m_extraData = Utils::base64_decode(m_extraDataSpec);
111 
112  for (auto file : m_bundledFilesSpec)
113  {
114  if (!FileUtils::fileExists(file))
115  throwError("Bundledfile '" + file + "' doesn't exist.");
116 
117  size_t size = FileUtils::fileSize(file);
118  if (size > (std::numeric_limits<uint32_t>::max)())
119  throwError("Bundledfile '" + file + "' too large.");
120 
121  BpfUlemFile ulemFile(size, FileUtils::getFilename(file), file);
122  if (ulemFile.m_filename.length() > 32)
123  throwError("Bundledfile '" + file + "' name exceeds "
124  "maximum length of 32.");
125  m_bundledFiles.push_back(ulemFile);
126  }
127 
128  // BPF coordinates are always in UTM meters, which can be quite large.
129  // Allowing the writer to proceed with the default offset of 0 can lead to
130  // unexpected quantization of the coordinates. Instead, we force use of
131  // auto offset to subtract the minimum value in XYZ, unless of course, the
132  // user chooses to override with their own offset.
133  if (!m_scaling.m_xOffArg->set())
134  m_scaling.m_xXform.m_offset.m_auto = true;
135  if (!m_scaling.m_yOffArg->set())
136  m_scaling.m_yXform.m_offset.m_auto = true;
137  if (!m_scaling.m_zOffArg->set())
138  m_scaling.m_zXform.m_offset.m_auto = true;
139 }

148 void BpfWriter::readyFile(const std::string& filename,
149  const SpatialReference& srs)
150 {
151  m_curFilename = filename;
152  m_stream.open(filename);
153  m_header.m_version = 3;
154  m_header.m_numDim = m_dims.size();
155  m_header.m_numPts = 0;
156  m_header.setLog(log());
157 
158  if (m_coordId.m_auto)
159  {
160  m_header.m_coordId = 0;
161  if (m_header.trySetSpatialReference(srs))
162  m_header.m_coordType = Utils::toNative(BpfCoordType::UTM);
163  }
164 
165  // We will re-write the header and dimensions to account for the point
166  // count and dimension min/max.
167  try
168  {
169  m_header.write(m_stream);
170  }
171  catch (const BpfHeader::error& err)
172  {
173  throwError(err.what());
174  }
175  m_header.writeDimensions(m_stream, m_dims);
176  for (auto& file : m_bundledFiles)
177  file.write(m_stream);
178  m_stream.put((const char *)m_extraData.data(), m_extraData.size());
179 
180  m_header.m_len = m_stream.position();
181 
182  m_header.m_xform.m_vals[0] = m_scaling.m_xXform.m_scale.m_val;
183  m_header.m_xform.m_vals[5] = m_scaling.m_yXform.m_scale.m_val;
184  m_header.m_xform.m_vals[10] = m_scaling.m_zXform.m_scale.m_val;
185 }

188 void BpfWriter::loadBpfDimensions(PointLayoutPtr layout)
189 {
190  Dimension::IdList dims;
191 
192  if (m_outputDims.size())
193  {
194  for (std::string& s : m_outputDims)
195  {
196  Dimension::Id id = layout->findDim(s);
197  if (id == Dimension::Id::Unknown)
198  throwError("Invalid dimension '" + s + "' specified for "
199  "'output_dims' option.");
200  dims.push_back(id);
201  }
202  }
203  else
204  dims = layout->dims();
205 
206  // Verify that we have X, Y and Z and that they're the first three
207  // dimensions.
208  std::sort(dims.begin(), dims.end());
209  if (dims.size() < 3 || dims[0] != Dimension::Id::X ||
210  dims[1] != Dimension::Id::Y || dims[2] != Dimension::Id::Z)
211  {
212  throwError("Missing one of dimensions X, Y or Z. Can't write BPF.");
213  }
214 
215  for (auto id : dims)
216  {
217  BpfDimension dim;
218  dim.m_id = id;
219  dim.m_label = layout->dimName(id);
220  m_dims.push_back(dim);
221  }
222 }

225 void BpfWriter::writeView(const PointViewPtr dataShared)
226 {
227  m_scaling.setAutoXForm(dataShared);
228 
229  // Avoid reference count overhead internally.
230  const PointView* data(dataShared.get());
231 
232  // We know that X, Y and Z are dimensions 0, 1 and 2.
233  m_dims[0].m_offset = m_scaling.m_xXform.m_offset.m_val;
234  m_dims[1].m_offset = m_scaling.m_yXform.m_offset.m_val;
235  m_dims[2].m_offset = m_scaling.m_zXform.m_offset.m_val;
236 
237  try
238  {
239  switch (m_header.m_pointFormat)
240  {
241  case BpfFormat::PointMajor:
242  writePointMajor(data);
243  break;
244  case BpfFormat::DimMajor:
245  writeDimMajor(data);
246  break;
247  case BpfFormat::ByteMajor:
248  writeByteMajor(data);
249  break;
250  }
251  }
252  catch (const BpfCompressor::error& err)
253  {
254  throwError(err.what());
255  }
256  m_header.m_numPts += data->size();
257 }

260 void BpfWriter::writePointMajor(const PointView* data)
261 {
262  // Blocks of 10,000 points will ensure that we're under 16MB, even
263  // for 255 dimensions.
264  size_t blockpoints = std::min<point_count_t>(10000UL, data->size());
265 
266  // For compression we're going to write to a buffer so that it can be
267  // compressed before it's written to the file stream.
268  BpfCompressor compressor(m_stream,
269  blockpoints * sizeof(float) * m_dims.size());
270  PointId idx = 0;
271  while (idx < data->size())
272  {
273  if (m_header.m_compression)
274  compressor.startBlock();
275  size_t blockId;
276  for (blockId = 0; idx < data->size() && blockId < blockpoints;
277  ++idx, ++blockId)
278  {
279  for (auto & bpfDim : m_dims)
280  {
281  double d = getAdjustedValue(data, bpfDim, idx);
282  m_stream << (float)d;
283  }
284  }
285  if (m_header.m_compression)
286  {
287  compressor.compress();
288  compressor.finish();
289  }
290  }
291 }

317 void BpfWriter::writeByteMajor(const PointView* data)
318 {
319  union
320  {
321  float f;
322  uint32_t u32;
323  } uu;
324 
325  // We're going to pretend for now that we only ever have one point buffer.
326  BpfCompressor compressor(m_stream,
327  data->size() * sizeof(float) * m_dims.size());
328 
329  if (m_header.m_compression)
330  compressor.startBlock();
331  for (auto & bpfDim : m_dims)
332  {
333  for (size_t b = 0; b < sizeof(float); b++)
334  {
335  for (PointId idx = 0; idx < data->size(); ++idx)
336  {
337  uu.f = (float)getAdjustedValue(data, bpfDim, idx);
338  uint8_t u8 = (uint8_t)(uu.u32 >> (b * CHAR_BIT));
339  m_stream << u8;
340  }
341  }
342  }
343  if (m_header.m_compression)
344  {
345  compressor.compress();
346  compressor.finish();
347  }
348 }

351 double BpfWriter::getAdjustedValue(const PointView* data,
352  BpfDimension& bpfDim, PointId idx)
353 {
354  double d = data->getFieldAs<double>(bpfDim.m_id, idx);
355  bpfDim.m_min = std::min(bpfDim.m_min, d);
356  bpfDim.m_max = std::max(bpfDim.m_max, d);
357 
358  if (bpfDim.m_id == Dimension::Id::X)
359  d /= m_scaling.m_xXform.m_scale.m_val;
360  else if (bpfDim.m_id == Dimension::Id::Y)
361  d /= m_scaling.m_yXform.m_scale.m_val;
362  else if (bpfDim.m_id == Dimension::Id::Z)
363  d /= m_scaling.m_zXform.m_scale.m_val;
364  return (d - bpfDim.m_offset);
365 }
