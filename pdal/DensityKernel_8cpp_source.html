87 int DensityKernel::execute()
88 {
89  gdal::registerDrivers();
90 
91  if (m_inputFile == "STDIN" ||
92  (FileUtils::extension(m_inputFile) == ".xml" ||
93  FileUtils::extension(m_inputFile) == ".json"))
94  {
95  m_manager.readPipeline(m_inputFile);
96  }
97  else
98  {
99  m_manager.makeReader(m_inputFile, "");
100  }
101  Options options;
102  options.add("sample_size", m_sampleSize);
103  options.add("threshold", m_density);
104  options.add("edge_length", m_edgeLength);
105  options.add("hole_cull_area_tolerance", m_cullArea);
106  options.add("smooth", m_doSmooth);
107  m_hexbinStage = &(m_manager.makeFilter("filters.hexbin",
108  *m_manager.getStage(), options));
109  m_manager.execute();
110  outputDensity(m_manager.pointTable().anySpatialReference());
111  return 0;
112 }
