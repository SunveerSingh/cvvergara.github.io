272  {

274  EXPECT_EQ(y, view->getFieldAs<int>(Dimension::Id::Y, index));
275  EXPECT_EQ(z, view->getFieldAs<int>(Dimension::Id::Z, index));
276  bool incNext = true;
277  if (xlimit)
278  {

281  x = 0;
282  else
283  incNext = false;
284  }

288  y++;
289  if (y >= (int)ylimit)
290  y = 0;
291  else
292  incNext = false;
293  }
294

296  z++;
297  }
