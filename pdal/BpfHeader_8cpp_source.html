93 {

111 bool BpfHeader::readV3(ILeStream& stream)
112 {
113  m_log->get(LogLevel::Debug) << "BPF: Reading V3\n";
114 
115  uint8_t dummyChar;
116  uint8_t interleave;
117  std::string magic;
118 
119  stream.get(magic, 4);
120  if (magic != "BPF!")
121  return false;
122 
123  stream.get(m_ver, 4);
124  Utils::fromString(m_ver, m_version);
125 
126  uint8_t numDim;
127  stream >> m_len >> numDim >> interleave >> m_compression >>
128  dummyChar >> m_numPts >> m_coordType >> m_coordId >> m_spacing >>
129  m_xform >> m_startTime >> m_endTime;
130  m_numDim = (int32_t)numDim;
131 
132  switch (interleave)
133  {
134  case 0:
135  m_pointFormat = BpfFormat::DimMajor;
136  break;
137  case 1:
138  m_pointFormat = BpfFormat::PointMajor;
139  break;
140  case 2:
141  m_pointFormat = BpfFormat::ByteMajor;
142  break;
143  default:
144  throw error("Invalid BPF file: unknown interleave type.");
145  }
146  return (bool)stream;
147 }

162 bool BpfHeader::readV1(ILeStream& stream)
163 {

166  stream >> m_len;

170 
171  if (m_version == 1)
172  m_pointFormat = BpfFormat::DimMajor;
173  else if (m_version == 2)

188 
189  stream >> xDim.m_offset >> yDim.m_offset >> zDim.m_offset;
190  stream >> xDim.m_min >> yDim.m_min >> zDim.m_min;
191  stream >> xDim.m_max >> yDim.m_max >> zDim.m_max;
192 
193  m_staticDims.resize(3);
194  m_staticDims[0] = xDim;
195  m_staticDims[1] = yDim;

219 bool BpfHeader::readDimensions(ILeStream& stream, BpfDimensionList& dims)
220 {
221  size_t staticCnt = m_staticDims.size();
222 
223  dims.resize(m_numDim);
224 
225  if (static_cast<std::size_t>(m_numDim) < staticCnt)
226  {
227  m_log->get(LogLevel::Error) << "BPF dimension range looks bad.\n";
228  m_log->get(LogLevel::Error) <<
229  "BPF: num dims: " << m_numDim << "\n" <<
230  "BPF: static count: " << staticCnt << "\n";
231 
232  m_log->get(LogLevel::Error) << "Dims:\n";
233  for (auto d : dims)
234  m_log->get(LogLevel::Error) << "\t" << d.m_label << "\n";
235 
236  m_log->get(LogLevel::Error) << "Static:\n";
237  for (auto d : m_staticDims)
238  m_log->get(LogLevel::Error) << "\t" << d.m_label << "\n";
239  }
240 
241  for (size_t d = 0; d < staticCnt; d++)
242  dims.at(d) = m_staticDims[d];
243  if (!BpfDimension::read(stream, dims, staticCnt))
244  return false;
245 
246  // Verify that we have an X, Y and Z, so that we don't have to worry
247  // about it later.
248  bool x = false;
249  bool y = false;
250  bool z = false;
251  for (auto d : dims)
252  {
253  if (d.m_label == "X")
254  x = true;
255  if (d.m_label == "Y")
256  y = true;
257  if (d.m_label == "Z")
258  z = true;
259  }
260  if (!x || !y || !z)
261  throw error("BPF file missing at least one of X, Y or Z dimensions.");
262  return true;
263 }

363 bool BpfUlemFile::write(OLeStream& stream)
364 {
365  stream.put("FILE", 4);
366  stream << m_len;
367  stream.put(m_filename, 32);
368 
369  std::ifstream in(m_filespec, std::ios::binary);
370  uint32_t len = m_len;
371 
372  const uint32_t MAX_BLOCKSIZE = 1000000;
373  char buf[MAX_BLOCKSIZE];
374  while (len)
375  {
376  uint32_t blocksize = std::min(MAX_BLOCKSIZE, len);
377  in.read(buf, blocksize);
378  stream.put(buf, blocksize);
379  len -= blocksize;
380  }
381  return true;
382 }
