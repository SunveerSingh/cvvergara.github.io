166 void addRange(const string& begin, const string& end, vector<PointId>& points)
167 {
168  PointId low = parseInt(begin);
169  PointId high = parseInt(end);
170  if (low > high)
171  throw pdal_error(string("Range invalid: ") + begin + "-" + end);
172  while (low <= high)
173  points.push_back(low++);
174 }

177 vector<PointId> getListOfPoints(std::string p)
178 {
179  vector<PointId> output;
180 
181  //Remove whitespace from string with awful remove/erase idiom.
182  p.erase(remove_if(p.begin(), p.end(), ::isspace), p.end());
183 
184  vector<string> ranges = Utils::split2(p, ',');
185  for (string s : ranges)
186  {
187  vector<string> limits = Utils::split(s, '-');
188  if (limits.size() == 1)
189  output.push_back(parseInt(limits[0]));
190  else if (limits.size() == 2)
191  addRange(limits[0], limits[1], output);
192  else
193  throw pdal_error(string("Invalid point range: ") + s);
194  }
195  return output;
196 }

200 MetadataNode InfoKernel::dumpPoints(PointViewPtr inView) const
201 {
202  MetadataNode root;
203  PointViewPtr outView = inView->makeNew();
204 
205  // Stick points in a inViewfer.
206  std::vector<PointId> points = getListOfPoints(m_pointIndexes);
207  bool oorMsg = false;
208  for (size_t i = 0; i < points.size(); ++i)
209  {
210  PointId id = (PointId)points[i];
211  if (id < inView->size())
212  outView->appendPoint(*inView.get(), id);
213  else if (!oorMsg)
214  {
215  m_log->get(LogLevel::Warning) << "Attempt to display points with "
216  "IDs not available in input dataset." << std::endl;
217  oorMsg = true;
218  }
219  }
220 
221  MetadataNode tree = outView->toMetadata();
222  std::string prefix("point ");
223  for (size_t i = 0; i < outView->size(); ++i)
224  {
225  MetadataNode n = tree.findChild(std::to_string(i));
226  n.add("PointId", points[i]);
227  root.add(n.clone("point"));
228  }
229  return root;

272 {
273  if (!pdal::Utils::fileExists(filename))
274  throw pdal_error("File not found: " + filename);
275 
276  if (filename == "STDIN")
277  {
278  m_manager.readPipeline(std::cin);
279  m_reader = m_manager.getStage();
280  }
281  else if (FileUtils::extension(filename) == ".xml" ||
282  FileUtils::extension(filename) == ".json")

284  m_manager.readPipeline(filename);

292  Stage& reader = m_manager.makeReader(filename, m_driverOverride, ops);
293  m_reader = &reader;
294  }

355 void InfoKernel::dump(MetadataNode& root)
356 {
357  if (m_showSchema)
358  root.add(m_manager.pointTable().layout()->toMetadata().clone("schema"));
359 
360  if (m_PointCloudSchemaOutput.size() > 0)

362 #ifdef PDAL_HAVE_LIBXML2

364 
365  std::ostream *out = Utils::createFile(m_PointCloudSchemaOutput);
366  std::string xml(schema.xml());

371  std::endl;
372 #endif
373 
374  }
375  if (m_showStats)
376  root.add(m_statsStage->getMetadata().clone("stats"));
377 
378  if (m_pipelineFile.size() > 0)
379  PipelineWriter::writePipeline(m_manager.getStage(), m_pipelineFile);
380 
381  if (m_pointIndexes.size())
382  {
383  PointViewSet viewSet = m_manager.views();
384  assert(viewSet.size() == 1);
385  MetadataNode points = dumpPoints(*viewSet.begin());
386  if (points.valid())
387  root.add(points.clone("points"));
388  }
389 
390  if (m_queryPoint.size())
391  {
392  PointViewSet viewSet = m_manager.views();
393  assert(viewSet.size() == 1);
394  root.add(dumpQuery(*viewSet.begin()));
395  }
396 
397  if (m_showMetadata)
398  {
399  // If we have a reader cached, this means we
400  // weren't reading a pipeline file directly. In that
401  // case, use the metadata from the reader (old behavior).
402  // Otherwise, return the full metadata of the entire pipeline
403  if (m_reader)
404  root.add(m_reader->getMetadata().clone("metadata"));
405  else
406  root.add(m_manager.getMetadata().clone("metadata"));
407  }
408 
409  if (m_boundary)
410  {
411  PointViewSet viewSet = m_manager.views();
412  assert(viewSet.size() == 1);
413  if (m_hexbinStage)
414  root.add(m_hexbinStage->getMetadata().clone("boundary"));
415  else
416  {
417  pdal::BOX2D bounds;
418  for (auto const &v: viewSet)
419  {
420  pdal::BOX2D b;
421  v->calculateBounds(b);
422  bounds.grow(b);
423  }

425  polygon << "POLYGON ((";
426 
427  polygon << bounds.minx << " " << bounds.miny;
428  polygon << ", " << bounds.maxx << " " << bounds.miny;
429  polygon << ", " << bounds.maxx << " " << bounds.maxy;
430  polygon << ", " << bounds.minx << " " << bounds.maxy;

443 MetadataNode InfoKernel::dumpQuery(PointViewPtr inView) const
444 {
445  int count;
446  std::string location;
447 
448  // See if there's a provided point count.
449  StringList parts = Utils::split2(m_queryPoint, '/');
450  if (parts.size() == 2)
451  {
452  location = parts[0];
453  count = atoi(parts[1].c_str());
454  }
455  else if (parts.size() == 1)
456  {
457  location = parts[0];
458  count = inView->size();
459  }
460  else
461  count = 0;
462  if (count == 0)
463  throw pdal_error("Invalid location specification. "
464  "--query=\"X,Y[/count]\"");
465 
466  auto seps = [](char c){ return (c == ',' || c == '|' || c == ' '); };
467 
468  std::vector<std::string> tokens = Utils::split2(location, seps);
469  std::vector<double> values;
470  for (auto ti = tokens.begin(); ti != tokens.end(); ++ti)
471  {
472  double d;
473  if (Utils::fromString(*ti, d))
474  values.push_back(d);
475  }
476 
477  if (values.size() != 2 && values.size() != 3)
478  throw pdal_error("--points must be two or three values");
479 
480  PointViewPtr outView = inView->makeNew();
481 
482  std::vector<PointId> ids;
483  if (values.size() >= 3)
484  {
485  KD3Index kdi(*inView);
486  kdi.build();
487  ids = kdi.neighbors(values[0], values[1], values[2], count);
488  }
489  else
490  {
491  KD2Index kdi(*inView);
492  kdi.build();
493  ids = kdi.neighbors(values[0], values[1], count);
494  }
495 
496  for (auto i = ids.begin(); i != ids.end(); ++i)
497  outView->appendPoint(*inView.get(), *i);
498 
499  return outView->toMetadata();
500 }
