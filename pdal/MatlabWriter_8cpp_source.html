63 void MatlabWriter::prepared(PointTableRef table)
64 {
65  if (m_outputDims.empty())
66  {
67  m_dims = table.layout()->dims();
68  }
69  else
70  {
71  for (std::string& s : m_outputDims)
72  {
73 // DimType dimType = table.layout()->findDimType(s);
74  Dimension::Id id = table.layout()->findDim(s);
75  if (id == Dimension::Id::Unknown)
76  throwError("Invalid dimension '" + s + "' specified for "
77  "'output_dims' option.");
78  m_dims.push_back(id);
79  }
80  }
81 }

98  if (matPutVariable(m_matfile, m_structName.c_str(), data))
