388 {
389  Options ops;
390  ops.add("bounds", BOX3D(1, 101, 201, 10, 110, 210));
391  ops.add("mode", "ramp");
392  ops.add("count", 10);
393 
394  FauxReader reader;

399 
400  RangeFilter range;

406 
407  FixedPointTable table(20);
408  f.prepare(table);
409 
410  auto cb = [](PointRef& point)
411  {
412  static int i = 0;
413  int x = point.getFieldAs<int>(Dimension::Id::X);
414  if (i == 0)

416  else if (i == 1)
417  EXPECT_EQ(x, 4);
418  else if (i == 2)
419  EXPECT_EQ(x, 5);
420  else if (i == 3)
421  EXPECT_EQ(x, 8);
422  else if (i == 4)
423  EXPECT_EQ(x, 9);
424  EXPECT_TRUE(i < 5);
425  ++i;
426  return true;
427  };
428  f.setCallback(cb);
429 
430  f.execute(table);
431 }
