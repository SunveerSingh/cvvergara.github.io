168 /// A StreamPointTable must provide storage for point data up to its capacity.
169 /// It must implement getPoint() which returns a pointer to a buffer of
170 /// sufficient size to contain a point's data. The minimum size required
171 /// is constant and can be determined by calling pointsToBytes(1) in the
172 /// finalize() method.
173 class PDAL_DLL StreamPointTable : public SimplePointTable
174 {
175 protected:
176  StreamPointTable(PointLayout& layout) : SimplePointTable(layout)
177  {}
178 
179 public:
180  /// Called when a new point should be added. Probably a no-op for
181  /// streaming.
182  virtual PointId addPoint()
183  { return 0; }
184  /// Called when execute() is started. Typically used to set buffer size
185  /// when all dimensions are known.
186  virtual void finalize()
187  {}
188  /// Called when the contents of StreamPointTable have been consumed and
189  /// the point data will be potentially overwritten.
190  virtual void reset()
191  {}
192  virtual point_count_t capacity() const = 0;
193 };
