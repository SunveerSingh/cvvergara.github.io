51 //
52 // References:
53 // - NITF 2.1 standard: MIL-STD-2500C (01 May 2006)
54 // - Lidar implementation profile v1.0 (2010-09-07)
55 //
56 // To be a proper lidar NITF file, the file must:
57 // - have at least one Image segment ("IM")
58 // - have at least one DES segment ("DE") named LIDARA
59 //
60 // You could have multiple image segments and LIDARA segments, but the
61 // standard doesn't seem to say anything about how you associate which
62 // image segment(s) with which LIDARA segment(s). We will assume only
63 // one image segment and only one LIDARA segment.
64 //
65 // We don't support LIDARA segments that are split into multiple DES
66 // segments via the DES INDEX mechanism. We also don't support wierd
67 // things that Nitro doesn't support: NITF 1.0, streaming segments,
68 // and so on.
69 //
70 // For the metadata, we store:
71 // - the file header fields --> FH.field_name
72 // - the file header TREs --> FH.TRE.tre_name
73 // - the IM segment fields --> IM:0.field_name
74 // - the IM segment TREs --> IM:0.tre_name.field_name
75 // - the DES fields --> DE:0.field_name
76 // - the DES TREs --> DE:0.field_name
77 //
78 // Note we use a ":N" syntax to indicate which segment is being used,
79 // so there is no ambiuity with multisegment NITFs
80 //
81 // We parse out the TRE fields for those TREs that Nitro recognizes
82 // (see tre_plugins.cpp).
83 //
84 // The dimensions we write out are (precisely) the LAS dimensions; we
85 // use the same names, so as not to require upstream stgaes to
86 // understand both LAS dimension names and NITF dimension names.
87 //
88 
89 //
90 // BUG: we should provide an option to set the SRS of the Stage using
91 // the IGEOLO field.

93 // BUG: need to implement addDefaultDimensions() so it does what LAS
94 // does.
95 //
96 // BUG: findIMSegment() allows "None" as an image type (for now, just
97 // to support the autzen test input)
98 //

101 void NitfReader::initialize(PointTableRef table)
102 {
103  try
104  {
105  NitfFileReader nitf(m_filename);
106  nitf.open();
107  nitf.getLasOffset(m_offset, m_length);
108  nitf.extractMetadata(m_metadata);
109  }
110  catch (const NitfFileReader::error& err)
111  {
112  throwError(err.what());
113  }
114  m_metadata.add("DESDATA_OFFSET", m_offset);
115  m_metadata.add("DESDATA_LENGTH", m_length);
116 
117  // Initialize the LAS stuff with its own metadata node.
118  MetadataNode lasNode = m_metadata.add(LasReader::getName());
119  initializeLocal(table, lasNode);
120 }
