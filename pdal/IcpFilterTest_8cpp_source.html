63 void checkPointsEqualReader(const PointViewSet& pointViewSet, double tolerance)
64 {
65  auto reader = newReader();
66  PointTable table;
67  reader->prepare(table);
68  PointViewSet readerPointViewSet = reader->execute(table);
69  ASSERT_EQ(1u, readerPointViewSet.size());
70  PointViewPtr expected = *readerPointViewSet.begin();
71  ASSERT_EQ(1u, pointViewSet.size());
72  PointViewPtr actual = *pointViewSet.begin();
73 
74  ASSERT_EQ(expected->size(), actual->size());
75 
76  for (PointId i = 0; i < expected->size(); ++i)
77  {
78  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::X, i),
79  actual->getFieldAs<double>(Dimension::Id::X, i), tolerance);
80  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::Y, i),
81  actual->getFieldAs<double>(Dimension::Id::Y, i), tolerance);
82  ASSERT_NEAR(expected->getFieldAs<double>(Dimension::Id::Z, i),
83  actual->getFieldAs<double>(Dimension::Id::Z, i), tolerance);
84  }
85 }
86 }
