63  * \brief Convert PCD point cloud to PDAL.
64  *
65  * Converts PCD data to PDAL format.
66  */
67 template <typename CloudT>
68 void PCDtoPDAL(CloudT &cloud, PointViewPtr view, BOX3D const& bounds)
69 {
70  typedef typename pcl::traits::fieldList<typename CloudT::PointType>::type
71  FieldList;
72 
73  if (pcl::traits::has_xyz<typename CloudT::PointType>::value)
74  {
75  auto getX = [&cloud, &bounds](size_t i)
76  { return cloud.points[i].x + bounds.minx; };
77  auto getY = [&cloud, &bounds](size_t i)
78  { return cloud.points[i].y + bounds.miny; };
79  auto getZ = [&cloud, &bounds](size_t i)
80  { return cloud.points[i].z + bounds.minz; };
81  setValues(view, Dimension::Id::X, cloud.points.size(), getX);
82  setValues(view, Dimension::Id::Y, cloud.points.size(), getY);
83  setValues(view, Dimension::Id::Z, cloud.points.size(), getZ);
84  }
85 
86  if (pcl::traits::has_intensity<typename CloudT::PointType>::value)
87  {
88  for (size_t i = 0; i < cloud.points.size(); ++i)
89  {
90  float f(0.0);
91  bool hasIntensity(true);
92 
93  typename CloudT::PointType p = cloud.points[i];
94  pcl::for_each_type<FieldList>
95  (pcl::CopyIfFieldExists<typename CloudT::PointType, float>
96  (p, "intensity", hasIntensity, f));
97  view->setField(Dimension::Id::Intensity, i, f);
98  }
99  }
100 
101  if (pcl::traits::has_color<typename CloudT::PointType>::value)
102  {
103  for (size_t i = 0; i < cloud.points.size(); ++i)
104  {
105  uint32_t v(0);
106 
107  typename CloudT::PointType p = cloud.points[i];
108  pcl::for_each_type<FieldList>
109  (pcl::CopyIfFieldExists<typename CloudT::PointType, uint32_t>
110  (p, "rgba", v));
111  view->setField<uint8_t>(Dimension::Id::Red, i, (v & 0x00FF0000) >> 16);
112  view->setField<uint8_t>(Dimension::Id::Green, i, (v & 0x0000FF00) >> 8);
113  view->setField<uint8_t>(Dimension::Id::Blue, i, (v & 0x000000FF));
114  }
115  }
116 }

127 /**
128  * \brief Convert PDAL point cloud to PCD.
129  *
130  * Converts PDAL data to PCD format.
131  */
132 template <typename CloudT>
133 void PDALtoPCD(PointViewPtr view, CloudT &cloud, BOX3D const& bounds,
134  double const& scale_x, double const& scale_y, double const& scale_z)
135 {
136  typedef typename pcl::traits::fieldList<typename CloudT::PointType>::type
137  FieldList;
138 
139  cloud.width = view->size();
140  cloud.height = 1; // unorganized point cloud
141  cloud.is_dense = false;
142  cloud.points.resize(cloud.width);
143 
144  if (pcl::traits::has_xyz<typename CloudT::PointType>::value)
145  {
146  for (size_t i = 0; i < cloud.points.size(); ++i)
147  {
148  double xd = view->getFieldAs<double>(Dimension::Id::X, i) - bounds.minx;
149  double yd = view->getFieldAs<double>(Dimension::Id::Y, i) - bounds.miny;
150  double zd = view->getFieldAs<double>(Dimension::Id::Z, i) - bounds.minz;
151 
152  typename CloudT::PointType p = cloud.points[i];
153  p.x = (float) (xd / scale_x);
154  p.y = (float) (yd / scale_y);
155  p.z = (float) (zd / scale_z);
156  cloud.points[i] = p;
157  }
158  }
159 
160  if (pcl::traits::has_intensity<typename CloudT::PointType>::value)
161  {
162  for (size_t i = 0; i < cloud.points.size(); ++i)
163  {
164  typename CloudT::PointType p = cloud.points[i];
165 
166  float f = view->getFieldAs<float>(Dimension::Id::Intensity, i);
167  pcl::for_each_type<FieldList>
168  (pcl::SetIfFieldExists<typename CloudT::PointType, float>
169  (p, "intensity", f));
170  cloud.points[i] = p;
171  }
172  }
173 
174  if (pcl::traits::has_color<typename CloudT::PointType>::value)
175  {
176  for (size_t i = 0; i < cloud.points.size(); ++i)
177  {
178  typename CloudT::PointType p = cloud.points[i];
179 
180  uint8_t r = view->getFieldAs<uint8_t>(Dimension::Id::Red, i);
181  uint8_t g = view->getFieldAs<uint8_t>(Dimension::Id::Green, i);
182  uint8_t b = view->getFieldAs<uint8_t>(Dimension::Id::Blue, i);
183  pcl::for_each_type<FieldList> (
184  pcl::SetIfFieldExists<typename CloudT::PointType, uint32_t> (
185  p, "rgba", (r << 16) | (g << 8) | b
186  )
187  );
188  cloud.points[i] = p;
189  }
190  }
191 }
