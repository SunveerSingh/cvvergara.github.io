198 void LasHeader::setSrs()
199 {
200  bool useWkt = false;
201 
202  if (incompatibleSrs())
203  {
204  m_log->get(LogLevel::Error) << "Invalid SRS specification. "
205  "GeoTiff not allowed with point formats 6 - 10." << std::endl;
206  }
207  else if (findVlr(TRANSFORM_USER_ID, WKT_RECORD_ID) &&
208  findVlr(TRANSFORM_USER_ID, GEOTIFF_DIRECTORY_RECORD_ID))
209  {
210  m_log->get(LogLevel::Debug) << "File contains both "
211  "WKT and GeoTiff VLRs which is disallowed." << std::endl;
212  }
213  else
214  useWkt = (m_versionMinor >= 4);
215 
216  try
217  {
218  if (useWkt)
219  setSrsFromWkt();
220  else
221  setSrsFromGeotiff();
222  }
223  catch (...)
224  {
225  m_log->get(LogLevel::Error) << "Could not create an SRS" << std::endl;
226  }

268 void LasHeader::setSrsFromWkt()
269 {
270  const LasVLR *vlr = findVlr(TRANSFORM_USER_ID, WKT_RECORD_ID);
271  if (!vlr)
272  vlr = findVlr(LIBLAS_USER_ID, WKT_RECORD_ID);
273  if (!vlr || vlr->dataLen() == 0)
274  return;
275 
276  // There is supposed to be a NULL byte at the end of the data,
277  // but sometimes there isn't because some people don't follow the
278  // rules. If there is a NULL byte, don't stick it in the
279  // wkt string.
280  size_t len = vlr->dataLen();
281  const char *c = vlr->data() + len - 1;
282  if (*c == 0)
283  len--;
284  m_srs.set(std::string(vlr->data(), len));
285 }

288 void LasHeader::setSrsFromGeotiff()
289 {
290  const LasVLR *vlr = findVlr(TRANSFORM_USER_ID, GEOTIFF_DIRECTORY_RECORD_ID);
291  // We must have a directory entry.
292  if (!vlr)
293  return;
294  uint8_t *data = (uint8_t *)vlr->data();
295  size_t dataLen = vlr->dataLen();
296 
297  std::vector<uint8_t> directoryRec(data, data + dataLen);
298 
299  vlr = findVlr(TRANSFORM_USER_ID, GEOTIFF_DOUBLES_RECORD_ID);
300  data = NULL;
301  dataLen = 0;
302  if (vlr && !vlr->isEmpty())
303  {
304  data = (uint8_t *)vlr->data();
305  dataLen = vlr->dataLen();
306  }
307  std::vector<uint8_t> doublesRec(data, data + dataLen);
308 
309  vlr = findVlr(TRANSFORM_USER_ID, GEOTIFF_ASCII_RECORD_ID);
310  data = NULL;
311  dataLen = 0;
312  if (vlr && !vlr->isEmpty())
313  {
314  data = (uint8_t *)vlr->data();
315  dataLen = vlr->dataLen();
316  }
317  std::vector<uint8_t> asciiRec(data, data + dataLen);
318 
319  GeotiffSrs geotiff(directoryRec, doublesRec, asciiRec);
320  SpatialReference gtiffSrs = geotiff.srs();
321  if (!gtiffSrs.empty())
322  m_srs = gtiffSrs;
323 }

326 ILeStream& operator>>(ILeStream& in, LasHeader& h)
327 {
328  uint8_t versionMajor;
329  uint32_t legacyPointCount;
330  uint32_t legacyReturnCount;
331 
332  in.get(h.m_fileSig, 4);
333  if (!Utils::iequals(h.m_fileSig, LasHeader::FILE_SIGNATURE))
334  throw LasHeader::error("File signature is not 'LASF', "
335  "is this an LAS/LAZ file?");
336 
337  in >> h.m_sourceId >> h.m_globalEncoding;
338  LasHeader::get(in, h.m_projectUuid);
339  in >> versionMajor >> h.m_versionMinor;
340  in.get(h.m_systemId, 32);
341 
342  in.get(h.m_softwareId, 32);
343  in >> h.m_createDOY >> h.m_createYear >> h.m_vlrOffset >>
344  h.m_pointOffset >> h.m_vlrCount >> h.m_pointFormat >>
345  h.m_pointLen >> legacyPointCount;
346  h.m_pointCount = legacyPointCount;
347 
348  // Although it isn't part of the LAS spec, the two high bits have been used
349  // to indicate compression, though only the high bit is currently used.
350  if (h.m_pointFormat & 0x80)
351  h.setCompressed(true);
352  h.m_pointFormat &= ~0xC0;
353 
354  for (size_t i = 0; i < LasHeader::LEGACY_RETURN_COUNT; ++i)
355  {
356  in >> legacyReturnCount;
357  h.m_pointCountByReturn[i] = legacyReturnCount;
358  }
359 
360  in >> h.m_scales[0] >> h.m_scales[1] >> h.m_scales[2];
361  in >> h.m_offsets[0] >> h.m_offsets[1] >> h.m_offsets[2];
362 
363  double maxX, minX;
364  double maxY, minY;
365  double maxZ, minZ;
366  in >> maxX >> minX >> maxY >> minY >> maxZ >> minZ;
367  h.m_bounds = BOX3D(minX, minY, minZ, maxX, maxY, maxZ);
368 
369  if (h.versionAtLeast(1, 3))
370  {
371  uint64_t waveformOffset;
372  in >> waveformOffset;
373  }
374  if (h.versionAtLeast(1, 4))
375  {
376  in >> h.m_eVlrOffset >> h.m_eVlrCount >> h.m_pointCount;
377  for (size_t i = 0; i < LasHeader::RETURN_COUNT; ++i)
378  in >> h.m_pointCountByReturn[i];
379  }
380 
381  // Read regular VLRs.
382  in.seek(h.m_vlrOffset);
383  for (size_t i = 0; i < h.m_vlrCount; ++i)
384  {
385  LasVLR r;
386  in >> r;
387  h.m_vlrs.push_back(std::move(r));
388  }
389 
390  // Read extended VLRs.
391  if (h.versionAtLeast(1, 4))
392  {
393  in.seek(h.m_eVlrOffset);
394  for (size_t i = 0; i < h.m_eVlrCount; ++i)
395  {
396  ExtLasVLR r;
397  in >> r;
398  h.m_vlrs.push_back(std::move(r));
399  }
400  }
401  h.setSrs();
402 
403  return in;
404 }

407 OLeStream& operator<<(OLeStream& out, const LasHeader& h)
408 {
409  uint32_t legacyPointCount = 0;
410  if (h.m_pointCount <= (std::numeric_limits<uint32_t>::max)())
411  legacyPointCount = (uint32_t)h.m_pointCount;
412 
413  out.put(h.m_fileSig, 4);
414  if (h.versionEquals(1, 0))
415  out << (uint32_t)0;
416  else if (h.versionEquals(1, 1))
417  out << h.m_sourceId << (uint16_t)0;
418  else
419  out << h.m_sourceId << h.m_globalEncoding;
420  LasHeader::put(out, h.m_projectUuid);
421  out << (uint8_t)1 << h.m_versionMinor;
422  out.put(h.m_systemId, 32);
423  out.put(h.m_softwareId, 32);
424 
425  uint8_t pointFormat = h.m_pointFormat;
426  if (h.compressed())
427  pointFormat |= 0x80;
428 
429  out << h.m_createDOY << h.m_createYear << h.m_vlrOffset <<
430  h.m_pointOffset << h.m_vlrCount << pointFormat <<
431  h.m_pointLen << legacyPointCount;
432 
433  for (size_t i = 0; i < LasHeader::LEGACY_RETURN_COUNT; ++i)
434  {
435  uint32_t legacyReturnCount = std::min(h.m_pointCountByReturn[i],
436  (uint64_t)(std::numeric_limits<uint32_t>::max)());
437  out << legacyReturnCount;
438  }
439 
440  out << h.m_scales[0] << h.m_scales[1] << h.m_scales[2];
441  out << h.m_offsets[0] << h.m_offsets[1] << h.m_offsets[2];
442 
443  out << h.maxX() << h.minX() << h.maxY() << h.minY() << h.maxZ() << h.minZ();
444 
445  if (h.versionAtLeast(1, 3))
446  out << (uint64_t)0;
447  if (h.versionAtLeast(1, 4))
448  {
449  out << h.m_eVlrOffset << h.m_eVlrCount << h.m_pointCount;
450  for (size_t i = 0; i < LasHeader::RETURN_COUNT; ++i)
451  out << h.m_pointCountByReturn[i];
452  }
453 
454  return out;
455 }
