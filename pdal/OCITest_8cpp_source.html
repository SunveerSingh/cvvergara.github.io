104 class SplitFilter : public Filter
105 {
106 public:
107  SplitFilter(point_count_t viewSize = 100) : m_viewSize(viewSize)
108  {}

110  virtual std::string getName() const
111  { return "split_filter"; }
112 
113 private:
114  point_count_t m_viewSize;
115 
116  virtual PointViewSet run(PointViewPtr view)
117  {
118  PointViewSet out;
119  PointViewPtr v = view->makeNew();
120  for (PointId i = 0; i < view->size(); ++i)
121  {
122  if (i && (i % m_viewSize == 0))
123  {
124  out.insert(v);
125  v = v->makeNew();
126  }
127  v->appendPoint(*view, i);
128  }
129  out.insert(v);
130  return out;
131  }

240 // Compare the source LAS file with the extracted OCI data.
241 // Candidate is the OCI reader's view.
242 void compare(const PointViewPtr candidate, std::string filename)
243 {
244  Options options;
245  Option fn("filename", filename);
246  options.add(fn);
247 
248  PointTable table;
249 
250  LasReader reader;
251  reader.setOptions(options);
252 
253  reader.prepare(table);
254  PointViewSet viewSet = reader.execute(table);
255 
256  EXPECT_EQ(viewSet.size(), 1u);
257  PointViewPtr source = *viewSet.begin();
258 
259  EXPECT_EQ(source->size(), candidate->size());
260  PointId limit = std::min(source->size(), candidate->size());
261 
262  for (PointId i = 0; i < limit; ++i)
263  {
264  using namespace Dimension;
265 
266  int32_t sx = source->getFieldAs<int32_t>(Id::X, i);
267  int32_t sy = source->getFieldAs<int32_t>(Id::Y, i);
268  int32_t sz = source->getFieldAs<int32_t>(Id::Z, i);
269  uint16_t sintensity = source->getFieldAs<uint16_t>(Id::Intensity, i);
270  uint16_t sred = source->getFieldAs<uint16_t>(Id::Red, i);
271  uint16_t sgreen = source->getFieldAs<uint16_t>(Id::Green, i);
272  uint16_t sblue = source->getFieldAs<uint16_t>(Id::Blue, i);
273 
274  int32_t cx = candidate->getFieldAs<int32_t>(Id::X, i);
275  int32_t cy = candidate->getFieldAs<int32_t>(Id::Y, i);
276  int32_t cz = candidate->getFieldAs<int32_t>(Id::Z, i);
277  uint16_t cintensity = candidate->getFieldAs<uint16_t>(Id::Intensity, i);
278  uint16_t cred = candidate->getFieldAs<uint16_t>(Id::Red, i);
279  uint16_t cgreen = candidate->getFieldAs<uint16_t>(Id::Green, i);
280  uint16_t cblue = candidate->getFieldAs<uint16_t>(Id::Blue, i);
281 
282  EXPECT_EQ(sx, cx);
283  EXPECT_EQ(sy, cy);
284  EXPECT_EQ(sz, cz);
285  EXPECT_EQ(sintensity, cintensity);
286  EXPECT_EQ(sred, cred);
287  EXPECT_EQ(sgreen, cgreen);
288  EXPECT_EQ(sblue, cblue);
289  }
290 }
