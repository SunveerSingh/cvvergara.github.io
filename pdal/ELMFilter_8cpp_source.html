73 void ELMFilter::filter(PointView& view)
74 {
75  BOX2D bounds;
76  view.calculateBounds(bounds);
77 
78  size_t cols = ((bounds.maxx - bounds.minx) / m_cell) + 1;
79  size_t rows = ((bounds.maxy - bounds.miny) / m_cell) + 1;
80 
81  // Make an initial pass through the input PointView to index elevation
82  // values and PointIds by row and column.
83  std::map<uint32_t, std::multimap<double, PointId>> hash;
84  for (PointId id = 0; id < view.size(); ++id)
85  {
86  double x = view.getFieldAs<double>(Dimension::Id::X, id);
87  double y = view.getFieldAs<double>(Dimension::Id::Y, id);
88  double z = view.getFieldAs<double>(Dimension::Id::Z, id);
89 
90  size_t c = static_cast<size_t>(floor(x - bounds.minx) / m_cell);
91  size_t r = static_cast<size_t>(floor(y - bounds.miny) / m_cell);
92 
93  hash[c * rows + r].emplace(z, id);
94  }
95 
96  // Count the number of points we classify as noise.
97  point_count_t num(0);
98 
99  // Make a second pass through the now rasterized PointView to compute the
100  // extended local minimum.
101  for (size_t c = 0; c < cols; ++c)
102  {
103  for (size_t r = 0; r < rows; ++r)
104  {
105  std::multimap<double, PointId> ids(hash[c * rows + r]);
106 
107  if (ids.size() <= 1)
108  continue;
109 
110  for (auto it = ids.begin(); it != std::prev(ids.end()); ++it)
111  {
112  // Where the current value is sufficiently close to the next, we
113  // consider that this is not a low outlier and break the current
114  // loop.
115  if (std::fabs(it->first - std::next(it)->first) < m_threshold)
116  break;
117 
118  // Otherwise this point is classified as noise, and we proceed
119  // to the next lowest value.
120  view.setField(Dimension::Id::Classification, it->second,
121  m_class);
122  ++num;
123  }
124  }
125  }
126 
127  log()->get(LogLevel::Info)
128  << "Classified " << num
129  << " points as noise by Extended Local Minimum (ELM).\n";
130 }
