227 TEST(MetadataTest, test_vlr_metadata)
228 {
229  MetadataNode m;
230 
231  MetadataNode bogusvlr = m.add("vlr1", "VLR1VALUE", "VLR1DESC");
232  MetadataNode vlr = m.add("vlr2", "VLR2VALUE", "VLR2DESC");
233  std::string recordId("MYRECOREDID");
234  std::string userId("MYUSERID");
235  vlr.add("record_id", recordId);
236  vlr.add("user_id", userId);
237  // Find a node whose name starts with vlr and that has child nodes
238  // with the name and recordId we're looking for.
239  auto pred = [recordId,userId](MetadataNode n)
240  {
241  auto recPred = [recordId](MetadataNode n)
242  {
243  return n.name() == "record_id" &&
244  n.value() == recordId;
245  };
246  auto userPred = [userId](MetadataNode n)
247  {
248  return n.name() == "user_id" &&
249  n.value() == userId;
250  };
251  return (Utils::startsWith(n.name(), "vlr") &&
252  !n.findChild(recPred).empty() &&
253  !n.findChild(userPred).empty());
254  };
255 
256  MetadataNode found = m.find(pred);
257  EXPECT_EQ(found.name(), "vlr2");
258  EXPECT_EQ(found.value(), "VLR2VALUE");
259  EXPECT_EQ(found.description(), "VLR2DESC");
260 }
