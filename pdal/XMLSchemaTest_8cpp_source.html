50 std::string ReadXML(std::string filename)
51 {
52  std::istream* infile = FileUtils::openFile(filename);
53  std::ifstream::pos_type size;
54  std::vector<char> data;
55  if (infile->good())
56  {
57  infile->seekg(0, std::ios::end);
58  size = infile->tellg();
59  data.resize(static_cast<std::vector<char>::size_type>(size));
60  infile->seekg(0, std::ios::beg);
61  infile->read(&data.front(), size);
62 
63  delete infile;
64  return std::string(&data[0], data.size());
65  }
66  else
67  throw pdal_error("unable to open file!");
68 }
