99 
100  PointLayoutPtr layout = PointLayoutPtr(new PointLayout());
101  std::vector<mxArray*> arrays;
102 
103  mxClassID ml_id = mxGetClassID(array);
104  if (ml_id != mxSTRUCT_CLASS)
105  throw pdal::pdal_error("Selected array must be a Matlab struct array!");
106 
107 
108  int numFields = mxGetNumberOfFields(array);
109 
110  if (!numFields)
111  throw pdal::pdal_error("Selected struct array must have fields!");
112 
113  // This needs to skip mxClassID types that
114  // don't map to PDAL
115  for (int i=0; i < numFields; ++i)
116  {
117  const char* fieldName = mxGetFieldNameByNumber(array, i);
118  mxArray* f = mxGetFieldByNumber(array, 0, i);
119  mxClassID mt = mxGetClassID(f);
120  Dimension::Type pt = Script::getPDALDataType(mt);
121  if (mt == mxDOUBLE_CLASS ||
122  mt == mxSINGLE_CLASS ||
123  mt == mxINT8_CLASS ||
124  mt == mxUINT8_CLASS ||
125  mt == mxINT16_CLASS ||
126  mt == mxUINT16_CLASS ||
127  mt == mxINT32_CLASS ||
128  mt == mxDOUBLE_CLASS ||
129  mt == mxINT32_CLASS ||
130  mt == mxUINT32_CLASS ||

132  mt == mxUINT64_CLASS )
133  {
134  layout->registerOrAssignDim(fieldName, pt);

143 void Script::getMatlabStruct(mxArray* array, PointViewPtr view, const Dimension::IdList& indims, std::string& pdalargs, MetadataNode pdalmetadata, LogPtr log)
144 {
145  std::vector<mxArray*> arrays;
146  std::vector<std::string> dimNames;
147 
148  mxClassID ml_id = mxGetClassID(array);
149  if (ml_id != mxSTRUCT_CLASS)
150  throw pdal::pdal_error("Selected array must be a Matlab struct array!");
151 
152  mxArray* metadata = mxGetField(array, 0, "metadata");
153  if (metadata)
154  {
155 
156  mxArray* args = mxGetField(metadata, 0, "pdalargs");
157  if (args)
158  {
159  mxClassID mt = mxGetClassID(args);
160  if (mt == mxCHAR_CLASS)
161  {
162  size_t len = mxGetN(args);
163  std::string data;
164  data.resize(len+1);
165  int ret = mxGetString(args, (char*)data.data(), len+1);
166  pdalargs = std::string(data);
167  }
168  }
169  }
170  Dimension::IdList dims;
171  if (!indims.size())
172  dims = view->dims();
173  else
174  dims = indims;
175 
176  for (auto d: dims)
177  {
178  std::string dimName = view->dimName(d);
179 
180  mxArray* f = mxGetField(array, 0, dimName.c_str());
181  if (!f)
182  {
183  std::ostringstream oss;
184  oss << "No dimension named '" << dimName << "' exists on struct array.";
185  throw pdal::pdal_error(oss.str());
186  }
187 
188  mwSize numElements = mxGetNumberOfElements(f);
189  if (numElements != view->size())
190  {
191  std::ostringstream oss;
192  oss << "Array shape is not the same as the PDAL PointView. ";
193  oss << "Matlab array is has '" << numElements << "' elements. ";
194  oss << "PointView has '" << view->size() << "' elements.";
195  throw pdal::pdal_error(oss.str());
196  }
197 
198  mxClassID mt = mxGetClassID(f);
199  Dimension::Type pt = Script::getPDALDataType(mt);
200 
201  char* p = (char*)mxGetData(f);
202  if (!p)
203  {
204  std::ostringstream oss;
205  oss << "Unable to fetch Matlab pointer to array for dimension '" << dimName << "'";
206  throw pdal::pdal_error(oss.str());
207  }
208 
209  for (PointId i = 0; i < view->size(); ++i)
210  {
211  view->setField(d, pt, i, p);
212  p += view->dimSize(d);
213  }
214  }
215

217 }

242  (mxComplexity)0);

254 
255  for (point_count_t i = 0; i < view->size(); ++i)
256  {
257  view->getField(pointsPtr, id, t, i);
258  pointsPtr += view->dimSize(id);
259  }
260  }
261 
262  // Push the dimension names into a char**
263  auto convert = [](std::string& s)
264  {
265  return s.c_str();
266  };
267  //
268  // Going into a 1x1 struct
269  mdims[0] = 1;
270  mdims[1] = 1;
271 
272 
273  std::vector<const char*> fieldNames;
274 
275  // Metadata
276  //
277  std::vector<std::string> metadataDimNames;
278  dimNames.push_back("metadata");
279 
280  metadataDimNames.push_back("json");
281  metadataDimNames.push_back("wkt");
282  metadataDimNames.push_back("horizontal");
283  metadataDimNames.push_back("vertical");
284  metadataDimNames.push_back("pdalargs");
285 
286

293

302

309  mxSetField(metadata, 0, "vertical", vertical);
310 
311  mxArray* pargs = mxCreateString(pdalargs.c_str());
312  mxSetField(metadata, 0, "pdalargs", pargs);
313 
314  fieldNames.clear();
315  std::transform(dimNames.begin(), dimNames.end(), std::back_inserter(fieldNames), convert);
316  mxArray* s = mxCreateStructArray(2, mdims, dimNames.size(), fieldNames.data());
317  for (size_t j = 0; j < dims.size(); ++j)
318  {
319  std::string dimName = dimNames[j];
