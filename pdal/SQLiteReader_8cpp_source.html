196 point_count_t SQLiteReader::readPatch(PointViewPtr view, point_count_t numPts)
197 {
198  const row* r = m_session->get();
199  if (!r)
200  throwError("readPatch with no data in session.");
201  std::map<std::string, int32_t> const& columns = m_session->columns();
202 
203  // Availability of positions already validated
204  int32_t position = columns.find("POINTS")->second;
205 
206  MetadataNode comp = m_patch->m_metadata.findChild("compression");
207  m_patch->m_isCompressed = Utils::iequals(comp.value(), "lazperf");
208  m_patch->m_compVersion = m_patch->m_metadata.findChild("version").value();
209 
210  position = columns.find("NUM_POINTS")->second;
211  size_t count;
212  Utils::fromString((*r)[position].data, count);
213  m_patch->remaining = (int32_t)count;
214  m_patch->count = (int32_t)count;
215 
216  log()->get(LogLevel::Debug3) << "patch compression? "
217  << m_patch->m_isCompressed << std::endl;
218 
219  if (m_patch->m_isCompressed)
220  log()->get(LogLevel::Debug3) << "patch compression version: "
221  << m_patch->m_compVersion << std::endl;
222 
223  position = columns.find("POINTS")->second;
224 
225  point_count_t numRead = 0;
226  PointId nextId = view->size();
227  if (m_patch->m_isCompressed)
228  {
229  size_t bufsize = 0;
230 #ifdef PDAL_HAVE_LAZPERF
231  auto cb = [this, view, &nextId, &numRead, &count]
232  (char *buf, size_t bufsize)
233  {
234  writePoint(*view.get(), nextId, buf);
235  if (m_cb)
236  m_cb(*view, nextId);
237  nextId++;
238  numRead++;
239  count--;
240  };
241 
242  const char *buf = reinterpret_cast<const char *>(
243  (*r)[position].blobBuf.data());
244  bufsize = (*r)[position].blobBuf.size();
245  count = std::min(count, size_t(numPts));
246  LazPerfDecompressor(cb, dbDimTypes(), count).
247  decompress(buf, bufsize);
248 
249  // Set the data into the patch.
250 
251 #else
252  throwError("Can't decompress without LAZperf.");
253 #endif
254 
255  log()->get(LogLevel::Debug3) << "Compressed byte size: " <<
256  bufsize << std::endl;
257  if (!bufsize)
258  throwError("Compressed patch size was 0.");
259  log()->get(LogLevel::Debug3) << "Uncompressed byte size: " <<
260  (m_patch->count * packedPointSize()) << std::endl;
261  }
262  else
263  {
264  const char *pos = (const char *)&((*r)[position].blobBuf[0]);
265  while (numRead < numPts && count > 0)
266  {
267  writePoint(*view.get(), nextId, pos);
268 
269  pos += packedPointSize();
270  if (m_cb)
271  m_cb(*view, nextId);
272  nextId++;
273  numRead++;
274  count--;
275  }
276  }
277  m_patch->remaining -= numRead;
278  return numRead;
279 }

282 point_count_t SQLiteReader::read(PointViewPtr view, point_count_t count)
283 {
284  if (eof())
285  return 0;
286 
287  log()->get(LogLevel::Debug4) << "read called with "
288  "PointView filled to " << view->size() << " points" <<
289  std::endl;
290 
291  point_count_t totalNumRead = 0;
292  if (! b_doneQuery)
293  {
294  // read first patch
295  m_session->query(m_query);
296  validateQuery();
297  b_doneQuery = true;
298  totalNumRead = readPatch(view, count);
299  }
300 
301  int patch_count(0);
302  while (totalNumRead < count)
303  {
304  if (m_patch->remaining == 0)
305  {
306  if (!nextBuffer())
307  {
308  m_at_end = true;
309  return totalNumRead;
310  }
311  }
312  PointId bufBegin = view->size();
313  point_count_t numRead = readPatch(view, count - totalNumRead);
314  PointId bufEnd = bufBegin + numRead;
315  totalNumRead += numRead;
316  patch_count++;
317  }
318  return totalNumRead;
319 }
