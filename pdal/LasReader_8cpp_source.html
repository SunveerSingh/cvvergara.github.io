111 void LasReader::handleCompressionOption()
112 {
113  std::string compression = Utils::toupper(m_compression);
114 #if defined(PDAL_HAVE_LAZPERF) && defined(PDAL_HAVE_LASZIP)
115  if (compression == "EITHER")
116  compression = "LASZIP";
117 #endif
118 #if !defined(PDAL_HAVE_LAZPERF) && defined(PDAL_HAVE_LASZIP)
119  if (compression == "EITHER")
120  compression = "LASZIP";
121  if (compression == "LAZPERF")
122  throwError("Can't decompress with LAZperf. PDAL not built "
123  "with LAZperf.");
124 #endif
125 #if defined(PDAL_HAVE_LAZPERF) && !defined(PDAL_HAVE_LASZIP)
126  if (compression == "EITHER")
127  compression = "LAZPERF";
128  if (compression == "LASZIP")
129  throwError("Can't decompress with LASzip. PDAL not built "
130  "with LASzip.");
131 #endif
132 
133 #if defined(PDAL_HAVE_LAZPERF) || defined(PDAL_HAVE_LASZIP)
134  if (compression != "LAZPERF" && compression != "LASZIP")
135  throwError("Invalid value for option for compression: '" +
136  m_compression + "'. Value values are 'lazperf' and 'laszip'.");
137 #endif
138 
139  // Set case-corrected value.
140  m_compression = compression;
141 }

400 void LasReader::readExtraBytesVlr()
401 {
402  const LasVLR *vlr = m_header.findVlr(SPEC_USER_ID,
403  EXTRA_BYTES_RECORD_ID);
404  if (!vlr)
405  return;
406  const char *pos = vlr->data();
407  size_t size = vlr->dataLen();
408  if (size % sizeof(ExtraBytesSpec) != 0)
409  {
410  log()->get(LogLevel::Warning) << "Bad size for extra bytes VLR. "
411  "Ignoring.";
412  return;
413  }
414  size /= sizeof(ExtraBytesSpec);
415  std::vector<ExtraBytesIf> ebList;
416  while (size--)
417  {
418  ExtraBytesIf eb;
419  eb.readFrom(pos);
420  ebList.push_back(eb);
421  pos += sizeof(ExtraBytesSpec);
422  }
423 
424  std::vector<ExtraDim> extraDims;
425  for (ExtraBytesIf& eb : ebList)
426  {
427  std::vector<ExtraDim> eds = eb.toExtraDims();
428  for (auto& ed : eds)
429  extraDims.push_back(std::move(ed));
430  }
431  if (m_extraDims.size() && m_extraDims != extraDims)
432  log()->get(LogLevel::Warning) << "Extra byte dimensions specified "
433  "in pineline and VLR don't match. Ignoring pipeline-specified "
434  "dimensions";
435  m_extraDims = extraDims;
436 }

532 bool LasReader::processOne(PointRef& point)
533 {
534  if (m_index >= getNumPoints())
535  return false;
536 
537  size_t pointLen = m_header.pointLen();
538 
539  if (m_header.compressed())
540  {
541 #ifdef PDAL_HAVE_LASZIP
542  if (m_compression == "LASZIP")
543  {
544  handleLaszip(laszip_read_point(m_laszip));
545  loadPoint(point, *m_laszipPoint);
546  }
547 #endif
548 
549 #ifdef PDAL_HAVE_LAZPERF
550  if (m_compression == "LAZPERF")
551  {
552  m_decompressor->decompress(m_decompressorBuf.data());
553  loadPoint(point, m_decompressorBuf.data(), pointLen);
554  }
555 #endif
556 #if !defined(PDAL_HAVE_LAZPERF) && !defined(PDAL_HAVE_LASZIP)
557  throwError("Can't read compressed file without LASzip or "
558  "LAZperf decompression library.");
559 #endif
560  } // compression
561  else
562  {
563  std::vector<char> buf(m_header.pointLen());
564 
565  m_streamIf->m_istream->read(buf.data(), pointLen);
566  loadPoint(point, buf.data(), pointLen);
567  }
568  m_index++;
569  return true;
570 }

573 point_count_t LasReader::read(PointViewPtr view, point_count_t count)
574 {
575  size_t pointLen = m_header.pointLen();
576  count = std::min(count, getNumPoints() - m_index);
577 
578  PointId i = 0;
579  if (m_header.compressed())
580  {
581 #if defined(PDAL_HAVE_LAZPERF) || defined(PDAL_HAVE_LASZIP)
582  if (m_compression == "LASZIP" || m_compression == "LAZPERF")
583  {
584  for (i = 0; i < count; i++)
585  {
586  PointRef point = view->point(i);
587  PointId id = view->size();
588  processOne(point);
589  if (m_cb)
590  m_cb(*view, id);
591  }
592  }
593 #else
594  throwError("Can't read compressed file without LASzip or "
595  "LAZperf decompression library.");
596 #endif
597  }
598  else
599  {
600  point_count_t remaining = count;
601 
602  // Make a buffer at most a meg.
603  size_t bufsize = std::min<size_t>((point_count_t)1000000,
604  count * pointLen);
605  std::vector<char> buf(bufsize);
606  try
607  {
608  do
609  {
610  point_count_t blockPoints = readFileBlock(buf, remaining);
611  remaining -= blockPoints;
612  char *pos = buf.data();
613  while (blockPoints--)
614  {
615  PointId id = view->size();
616  PointRef point = view->point(id);
617  loadPoint(point, pos, pointLen);
618  if (m_cb)
619  m_cb(*view, id);
620  pos += pointLen;
621  i++;
622  }
623  } while (remaining);
624  }
625  catch (std::out_of_range&)
626  {}
627  catch (invalid_stream&)
628  {}
629  }
630  m_index += i;
631  return (point_count_t)i;
632 }

635 point_count_t LasReader::readFileBlock(std::vector<char>& buf,
636  point_count_t maxpoints)
637 {
638  std::istream *stream(m_streamIf->m_istream);
639 
640  size_t ptLen = m_header.pointLen();
641  point_count_t blockpoints = buf.size() / ptLen;
642 
643  blockpoints = std::min(maxpoints, blockpoints);
644  if (stream->eof())
645  throw invalid_stream("stream is done");
646 
647  stream->read(buf.data(), blockpoints * ptLen);
648  if (stream->gcount() != (std::streamsize)(blockpoints * ptLen))
649  {
650  // we read fewer bytes than we asked for
651  // because the file was either truncated
652  // or the header is bunk.
653  blockpoints = stream->gcount() / ptLen;
654  }
655  return blockpoints;
656 }

899 {

904  {
905  istream.skip(dim.m_size);

907  }

914  dim.m_dimType.m_xform.m_offset.m_val;
915  point.setField(dim.m_dimType.m_id, d);
916  }
917  else
