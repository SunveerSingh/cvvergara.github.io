92 Dimension::Id PointLayout::assignDim(const std::string& name,
93  Dimension::Type type)
94 {
95  if (m_nextFree == Dimension::COUNT)
96  throw pdal_error("No dimension IDs remaining for assignment.");
97  Dimension::Id id = (Dimension::Id)m_nextFree;
98 
99  auto di = m_propIds.find(name);
100  if (di != m_propIds.end())
101  id = di->second;
102  Dimension::Detail dd = m_detail[Utils::toNative(id)];
103  dd.setType(resolveType(type, dd.type()));
104  if (update(dd, name))
105  {
106  if (di == m_propIds.end())
107  {
108  m_nextFree++;
109  m_propIds[name] = id;
110  }
111  return id;
112  }
113  return Dimension::Id::Unknown;
114 }

221 // Update the point layout given dimension detail and the dimension's name.
222 bool PointLayout::update(Dimension::Detail dd, const std::string& name)
223 {
224  if (m_finalized)
225  {
226  throw pdal_error("Can't update layout after points have been added.");
227  }
228 
229  Dimension::DetailList detail;
230 
231  // Update the list of used IDs.
232  bool used = Utils::contains(m_used, dd.id());
233  for (auto id : m_used)
234  {
235  if (id == dd.id())
236  detail.push_back(dd);
237  else
238  detail.push_back(m_detail[Utils::toNative(id)]);
239  }
240  if (!used)
241  detail.push_back(dd);
242 
243  // Find the dimension in the list that we're referring to with
244  // this update.
245  auto di = std::find_if(detail.begin(), detail.end(),
246  [dd](const Dimension::Detail& td){ return td.id() == dd.id(); });
247  Dimension::Detail *cur = &(*di);
248 
249  {
250  auto sorter = [this](const Dimension::Detail& d1,
251  const Dimension::Detail& d2) -> bool
252  {
253  if (d1.size() > d2.size())
254  return true;
255  if (d1.size() < d2.size())
256  return false;
257  return d1.id() < d2.id();
258  };
259 
260  // Sort dimensions based on size and then on ID.
261  int offset = 0;
262  std::sort(detail.begin(), detail.end(), sorter);
263  for (auto& d : detail)
264  {
265  d.setOffset(offset);
266  offset += (int)d.size();
267  }
268  //NOTE - I tried forcing all points to be aligned on 8-byte boundaries
269  // in case this would matter to the optimized memcpy, but it made
270  // no difference. No sense wasting space for no difference.
271  m_pointSize = (size_t)offset;
272  }
273 
274  if (!used)
275  m_used.push_back(dd.id());
276 
277  // Update the detail for the ID.
278  for (auto& dtemp : detail)
279  m_detail[Utils::toNative(dtemp.id())] = dtemp;
280 
281  return true;
282 }

285 // Given two types, find a type that can represent the values of both.
286 Dimension::Type PointLayout::resolveType(Dimension::Type t1,
287  Dimension::Type t2)
288 {
289  using namespace Dimension;
290  if (t1 == Type::None && t2 != Type::None)
291  return t2;
292  if (t2 == Type::None && t1 != Type::None)
293  return t1;
294  if (t1 == t2)
295  return t1;
296  if (base(t1) == base(t2))
297  return std::max(t1, t2);
298  //Prefer floating to non-floating.
299  if (base(t1) == BaseType::Floating && base(t2) != BaseType::Floating)
300  return t1;
301  if (base(t2) == BaseType::Floating && base(t1) != BaseType::Floating)
302  return t2;
303  // Now we're left with cases of a signed/unsigned mix.
304  // If the unsigned type is smaller, take the signed type.
305  if (base(t1) == BaseType::Unsigned && size(t1) < size(t2))
306  return t2;
307  if (base(t2) == BaseType::Unsigned && size(t2) < size(t1))
308  return t1;
309  // Signed type is smaller or the the sizes are equal.
310  switch (std::max(size(t1), size(t2)))
311  {
312  case 1:
313  return Type::Signed16;
314  case 2:
315  return Type::Signed32;
316  case 4:
317  return Type::Signed64;
318  default:
319  return Type::Double;
320  }
321 }
