71 template <typename PointT> void
72 pcl::Pipeline<PointT>::applyPassThrough(PointCloudConstPtr cloud,
73  PointCloud &output, Json::Value const& vt)
74 {
75  // initial setup
76  pcl::PassThrough<PointT> pass;
77  pass.setInputCloud(cloud);
78 
79  // parse params
80  std::string field = vt["setFilterFieldName"].asString();
81  float m1 = vt["setFilterLimits"]
82  .get("min", -std::numeric_limits<float>::max())
83  .asFloat();
84  float m2 = vt["setFilterLimits"]
85  .get("max", std::numeric_limits<float>::max())
86  .asFloat();
87 
88  // summarize settings
89  PCL_DEBUG("\tField name: %s\n", field.c_str());
90  PCL_DEBUG("\tLimits: %f, %f\n", m1, m2);
91 
92  if (field.compare("x") == 0)
93  {
94  if (m1 != -std::numeric_limits<float>::max())
95  m1 -= x_offset_;
96  if (m2 != std::numeric_limits<float>::max())
97  m2 -= x_offset_;
98  }
99 
100  if (field.compare("y") == 0)
101  {
102  if (m1 != -std::numeric_limits<float>::max())
103  m1 -= y_offset_;
104  if (m2 != std::numeric_limits<float>::max())
105  m2 -= y_offset_;
106  }
107 
108  if (field.compare("z") == 0)
109  {
110  if (m1 != -std::numeric_limits<float>::max())
111  m1 -= z_offset_;
112  if (m2 != std::numeric_limits<float>::max())
113  m2 -= z_offset_;
114  }
115 
116  // set params and apply filter
117  pass.setFilterFieldName(field);
118  pass.setFilterLimits(m1, m2);
119  pass.filter(output);
120 
121  PCL_DEBUG("\t%d filtered to %d in passthrough\n", cloud->points.size(),
122  output.points.size());
123 
124  return;
125 }

322 template <typename PointT> void
323 pcl::Pipeline<PointT>::applyFilter(PointCloud &output)
324 {
325  // Has the input dataset been set already?
326  if (!input_)
327  {
328  PCL_WARN("[pcl::%s::applyFilter] No input dataset given!\n",
329  getClassName().c_str());
330  output.width = output.height = 0;
331  output.points.clear();
332  return;
333  }
334 
335  // Do we have a JSON string to process?
336  if (!json_set_)
337  {
338  PCL_WARN("[pcl::%s::applyFilter] No input JSON given!\n",
339  getClassName().c_str());
340  output = *input_;
341  return;
342  }
343 
344  output.is_dense = true;
345 
346  try
347  {
348  typedef pcl::PointCloud<PointT> Cloud;
349  typename Cloud::Ptr cloud(new Cloud);
350  typename Cloud::Ptr cloud_f(new Cloud);
351 
352  pcl::copyPointCloud<PointT> (*input_, *cloud);
353 
354  for (auto const& vt : pt_)
355  {
356  std::string name(vt.get("name", "").asString());
357 
358  using namespace Utils;
359 
360  if (iequals(name, "PassThrough"))
361  applyPassThrough(cloud, *cloud_f, vt);
362  else if (iequals(name, "StatisticalOutlierRemoval"))
363  applyStatisticalOutlierRemoval(cloud, *cloud_f, vt);
364  else if (iequals(name, "RadiusOutlierRemoval"))
365  applyRadiusOutlierRemoval(cloud, *cloud_f, vt);
366  else if (iequals(name, "VoxelGrid"))
367  applyVoxelGrid(cloud, *cloud_f, vt);
368  else if (iequals(name, "GridMinimum"))
369  applyGridMinimum(cloud, *cloud_f, vt);
370  else if (iequals(name, "ApproximateProgressiveMorphologicalFilter"))
371  applyApproximateProgressiveMorphologicalFilter(cloud, *cloud_f,
372  vt);
373  else if (iequals(name, "ProgressiveMorphologicalFilter"))
374  applyProgressiveMorphologicalFilter(cloud, *cloud_f, vt);
375  else
376  PCL_WARN("Requested filter `%s` not implemented! Skipping...\n",
377  name.c_str());
378 
379  cloud.swap(cloud_f);
380 
381  if (cloud->points.size() == 0)
382  {
383  PCL_WARN("No points in filtered cloud. Aborting!\n");
384  break;
385  }
386  }
387 
388  output += *cloud;
389  }
390  catch (std::exception const& e)
391  {
392  PCL_WARN("[pcl::%s] Error parsing JSON and creating pipeline! %s\n",
393  getClassName().c_str(), e.what());
394  }
395 
396  // Resize the output arrays
397  output.width = static_cast<uint32_t>(output.points.size());
398 }
