73 {
74  double r = (double)rand(); // [0..32767]
75  double v = (maximum - minimum) / (double)RAND_MAX;
76  double s = r * v; // [0..(max-min)]
77  double t = minimum + s; // [min..max]
78 
79  assert(t >= minimum);
80  assert(t <= maximum);
81 
82  return t;

491 StringList Utils::wordWrap2(std::string const& s, size_t lineLength,
492  size_t firstLength)
493 {
494  std::vector<std::string> output;
495  if (s.empty())
496  return output;
497 
498  if (firstLength == 0)
499  firstLength = lineLength;
500 
501  auto pushWord = [&s, &output](size_t start, size_t end)
502  {
503  if (start != end)
504  output.push_back(s.substr(start, end - start + 1));
505  };
506 
507  size_t len = firstLength;
508  size_t startPos = 0;
509  while (true)
510  {
511  size_t endPos = std::min(startPos + len - 1, s.size() - 1);
512  if (endPos + 1 == s.size())
513  {
514  pushWord(startPos, endPos);
515  return output;
516  }
517  size_t pos = endPos;
518  while (pos > startPos)
519  {
520  if (std::isspace(s[pos]) && !std::isspace(s[pos + 1]))
521  {
522  endPos = pos;
523  break;
524  }
525  pos--;
526  }
527  pushWord(startPos, endPos);
528  len = lineLength;
529  startPos = endPos + 1;
530  }
531  return output;
532 }

606 std::vector<std::string> Utils::backtrace()
607 {
608  std::vector<std::string> lines;
609 #ifdef PDAL_HAVE_EXECINFO_H
610  const int MAX_STACK_SIZE(100);
611  void* buffer[MAX_STACK_SIZE];
612  std::vector<std::string> prefixes;
613  size_t maxPrefix(0);
614 
615  std::size_t size(::backtrace(buffer, MAX_STACK_SIZE));
616  char** symbols(backtrace_symbols(buffer, size));
617 
618  // Store strings and free symbols. Start at 1 to remove this function
619  // from the stack.
620  for (std::size_t i(1); i < size; ++i)
621  {
622  lines.push_back(symbols[i]);
623  const std::string& symbol = lines.back();
624  std::string prefix;
625  std::size_t pos = symbol.find("0x");
626  if (pos != std::string::npos)
627  prefix = symbol.substr(0, pos);
628  else
629  prefix = std::to_string(i) + " ???";
630  trimTrailing(prefix);
631  prefixes.push_back(prefix);
632  maxPrefix = std::max(prefix.size(), maxPrefix);
633  }
634  free(symbols);
635 
636  // Replace the simple symbol with a better representation if possible.
637  for (std::size_t i(1); i < size; ++i)
638  {
639  std::string& symbol = lines[i - 1];
640  std::string& prefix = prefixes[i - 1];
641  prefix = prefix + std::string(maxPrefix + 2 - prefix.size(), ' ');
642 
643  Dl_info info;
644  if (dladdr(buffer[i], &info))
645  {
646  const std::size_t offset(static_cast<char*>(buffer[i]) -
647  static_cast<char*>(info.dli_saddr));
648 
649  // Replace the address and mangled name with a human-readable
650  // name.
651  symbol = prefix + demangle(info.dli_sname) + " + " +
652  std::to_string(offset);
653  }
654  else
655  {
656  symbol = symbol.substr(maxPrefix + 2);
657  trimLeading(symbol);
658  symbol = prefix + symbol;
659  }
660  }
661 #endif
662  return lines;
663 }
