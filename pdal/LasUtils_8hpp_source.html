51 inline std::istream& operator>>(std::istream& in, LasCompression& c)
52 {
53  std::string s;
54 
55  in >> s;
56  s = Utils::toupper(s);
57  if (s == "LASZIP" || s == "TRUE")
58  c = LasCompression::LasZip;
59  else if (s == "LAZPERF")
60  c = LasCompression::LazPerf;
61  else
62  c = LasCompression::None;
63  return in;
64 }

122 class ExtraBytesIf

126  {
127  for (size_t i = 0; i < 3; ++i)

133 
134  ExtraBytesIf(const std::string& name, Dimension::Type type,
135  const std::string& description) :
136  m_type(type), m_name(name), m_description(description), m_size(0)
137  {
138  for (size_t i = 0; i < 3; ++i)
139  {
140  // Setting the scale to 0 looks wrong, but it isn't. If the
141  // scale option flag isn't set, the scale is supposed to be 0.
142  // When we write the VLR, we always clear the scale flag.
143  m_scale[i] = 0.0;
144  m_offset[i] = 0.0;
145  }
146  m_fieldCnt = (m_type == Dimension::Type::None ? 0 : 1);
147  }
