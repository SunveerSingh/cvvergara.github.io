182 void optionsWrite(const Options& writerOps)

186 
187  const std::string file(Support::datapath("las/1.2-with-color.las"));
188  Options options;

190  reader->setOptions(options);
191 
192  Stage* writer(f.createStage("writers.pgpointcloud"));
193  writer->setOptions(writerOps);
194  writer->setInput(*reader);
195 
196  PointTable table;
197  writer->prepare(table);
198 
199  PointViewSet written = writer->execute(table);
200 
201  point_count_t count(0);
202  for(auto i = written.begin(); i != written.end(); ++i)
203  count += (*i)->size();
204  EXPECT_EQ(written.size(), 1U);
205  EXPECT_EQ(count, 1065U);
206 }
