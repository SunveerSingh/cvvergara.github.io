146 /**
147  Reproject a bounds box from a source projection to a destination.
148  \param box Bounds box to be reprojected in-place.
149  \param srcSrs String in WKT or other suitable format of box coordinates.
150  \param dstSrs String in WKT or other suitable format to which
151  coordinates should be projected.
152  \return Whether the reprojection was successful or not.
153 */
154 bool reprojectBounds(BOX3D& box, const std::string& srcSrs,
155  const std::string& dstSrs)
156 {
157  OGRSpatialReference src;
158  OGRSpatialReference dst;
159 
160  OGRErr srcOk = OSRSetFromUserInput(&src, srcSrs.c_str());
161  OGRErr dstOk = OSRSetFromUserInput(&dst, dstSrs.c_str());
162  if (srcOk != OGRERR_NONE || dstOk != OGRERR_NONE)
163  return false;
164 
165  OGRCoordinateTransformationH transform =
166  OCTNewCoordinateTransformation(&src, &dst);
167 
168  bool ok = (OCTTransform(transform, 1, &box.minx, &box.miny, &box.minz) &&
169  OCTTransform(transform, 1, &box.maxx, &box.maxy, &box.maxz));
170  OCTDestroyCoordinateTransformation(transform);
171  return ok;
172 }

360 GDALError Raster::open(int width, int height, int numBands,
361  Dimension::Type type, double noData, StringList options)
362 {
363  if (m_drivername.empty())
364  m_drivername = "GTiff";
365 
366  m_width = width;
367  m_height = height;
368  m_numBands = numBands;
369  m_bandType = type;
370  m_dstNoData = noData;
371 
372  if (!GDALInvGeoTransform(m_forwardTransform.data(),
373  m_inverseTransform.data()))
374  {
375  m_errorMsg = "Geotransform for raster '" + m_filename + "' not "
376  "invertible";
377  return GDALError::NotInvertible;
378  }
379 
380  GDALDriver *driver = GetGDALDriverManager()->GetDriverByName(
381  m_drivername.data());
382  if (!driver)
383  {
384  m_errorMsg = "Driver '" + m_drivername + "' not found.";
385  return GDALError::DriverNotFound;
386  }
387 
388  std::string item;
389  const char *itemp = driver->GetMetadataItem(GDAL_DCAP_CREATE);
390  if (itemp)
391  item = itemp;
392  if (item != "YES")
393  {
394  m_errorMsg = "Requested driver '" + m_drivername + "' does not "
395  "support file creation.";
396  return GDALError::InvalidDriver;
397  }
398 
399  GDALError error = validateType(type, driver);
400  if (error != GDALError::None)
401  return error;
402 
403  std::vector<const char *> opts;
404  for (size_t i = 0; i < options.size(); ++i)
405  {
406  if (options[i].find("INTERLEAVE") == 0)
407  {
408  m_errorMsg = "INTERLEAVE GDAL driver option not supported.";
409  return GDALError::InvalidOption;
410  }
411  opts.push_back(options[i].data());
412  }
413  opts.push_back("INTERLEAVE=BAND");
414  opts.push_back(NULL);
415 
416  m_ds = driver->Create(m_filename.data(), m_width, m_height, m_numBands,
417  toGdalType(type), const_cast<char **>(opts.data()));
418  if (m_ds == NULL)
419  {
420  m_errorMsg = "Unable to open GDAL datasource '" + m_filename + "'.";
421  return GDALError::CantCreate;
422  }
423 
424  if (m_srs.valid())
425  m_ds->SetProjection(m_srs.getWKT().data());
426 
427  m_ds->SetGeoTransform(m_forwardTransform.data());
428  // If the nodata value is NaN, set a default based on type.
429  if (std::isnan(m_dstNoData))
430  {
431  switch (type)
432  {
433  case Dimension::Type::Unsigned8:
434  m_dstNoData = 255;
435  break;
436  case Dimension::Type::Signed8:
437  m_dstNoData = -127;
438  break;
439  case Dimension::Type::Unsigned16:
440  case Dimension::Type::Unsigned32:
441  m_dstNoData = 9999;
442  break;
443  default:
444  m_dstNoData = -9999;
445  break;
446  }
447  }
448 
449  for (int i = 0; i < m_numBands; ++i)
450  {
451  GDALRasterBand *band = m_ds->GetRasterBand(i + 1);
452  band->SetNoDataValue(m_dstNoData);
453  }
454 
455  return GDALError::None;
456 }

459 GDALError Raster::open()
460 {
461  GDALError error = GDALError::None;
462  if (m_ds)
463  return error;
464 
465 #if (GDAL_VERSION_MAJOR > 1)
466  const char ** driverP = NULL;
467  const char *drivers[2] = {0};
468  if (!m_drivername.empty())
469  {
470  drivers[0] = m_drivername.c_str();
471  driverP = drivers;
472  }
473 
474  m_ds = (GDALDataset *)GDALOpenEx(m_filename.c_str(), GA_ReadOnly, driverP,
475  nullptr, nullptr);
476 #else
477  m_ds = (GDALDataset *)GDALOpen(m_filename.c_str(), GA_ReadOnly);
478 #endif
479  if (m_ds == NULL)
480  {
481  m_errorMsg = "Unable to open GDAL datasource '" + m_filename + "'.";
482  return GDALError::CantOpen;
483  }
484 
485  // An identity transform is returned on error.
486  if (m_ds->GetGeoTransform(m_forwardTransform.data()) != CE_None)
487  {
488  m_errorMsg = "Unable to get geotransform for raster '" +
489  m_filename + "'.";
490  error = GDALError::NoTransform;
491  }
492 
493  if (!GDALInvGeoTransform(m_forwardTransform.data(),
494  m_inverseTransform.data()))
495  {
496  m_errorMsg = "Geotransform for raster '" + m_filename + "' not "
497  "invertible";
498  error = GDALError::NotInvertible;
499  }
500 
501  m_width = m_ds->GetRasterXSize();
502  m_height = m_ds->GetRasterYSize();
503  m_numBands = m_ds->GetRasterCount();
504 
505  if (computePDALDimensionTypes() == GDALError::InvalidBand)
506  error = GDALError::InvalidBand;
507  return error;
508 }

511 /**
512  \param type Reqested type of the raster.
513  \param driver Pointer to the GDAL driver being used to write the raster.
514  \return Requested type, or if not supported, the preferred type to use
515  for the raster.

517 GDALError Raster::validateType(Dimension::Type& type,
518  GDALDriver *driver)
519 {
520  // Convert the string of supported GDAL types to a vector of PDAL types,
521  // ignoring types that aren't supported by PDAL (mostly complex values).
522  std::vector<Dimension::Type> types;
523  const char *itemp = driver->GetMetadataItem(GDAL_DMD_CREATIONDATATYPES);
524  if (itemp)
525  {
526  StringList items = Utils::split2(std::string(itemp), ' ');
527  for (auto& i : items)
528  {
529  Dimension::Type t = toPdalType(i);
530  if (t != Dimension::Type::None)
531  types.push_back(t);
532  }
533  }
534 
535  // If requested type is not supported, return an error.
536  if (type != Dimension::Type::None && !Utils::contains(types, type))
537  {
538  m_errorMsg = "Requested type '" + Dimension::interpretationName(type) +
539  "' not supported by GDAL driver '" + m_drivername + "'.";
540  return GDALError::InvalidType;
541  }
542 
543  // If no type is requested, take the "largest" one.
544  if (type == Dimension::Type::None)
545  {
546  std::sort(types.begin(), types.end());
547  type = types.back();
548  }
549  return GDALError::None;
550 }

553 void Raster::pixelToCoord(int col, int row, std::array<double, 2>& output) const
554 {
555  /**
556  double *xform = const_cast<double *>(m_forwardTransform.data());
557  GDALApplyGeoTransform(xform, col, row, &output[0], &output[1]);
558  **/
559 
560  // from http://gis.stackexchange.com/questions/53617/how-to-find-lat-lon-values-for-every-pixel-in-a-geotiff-file
561  double c = m_forwardTransform[0];
562  double a = m_forwardTransform[1];
563  double b = m_forwardTransform[2];
564  double f = m_forwardTransform[3];
565  double d = m_forwardTransform[4];
566  double e = m_forwardTransform[5];
567 
568  //ABELL - Not sure why this is right. You can think of this like:
569  // output[0] = a * (col + .5) + b * (row + .5) + c;
570  // output[1] = d * (col + .5) + e * (row + .5) + f;
571  // Is there some reason why you want to "move" the points in the raster
572  // to a location between the rows/columns? Seems that you would just
573  // use 'c' and 'f' to shift everything a half-row and half-column if
574  // that's what you wanted.
575  // Also, this isn't what GDALApplyGeoTransform does. And why aren't
576  // we just calling GDALApplyGeoTransform?
577  output[0] = a*col + b*row + a*0.5 + b*0.5 + c;
578  output[1] = d*col + e*row + d*0.5 + e*0.5 + f;
579 }

582 // Determines the pixel/line position given an x/y.
583 // No reprojection is done at this time.
584 bool Raster::getPixelAndLinePosition(double x, double y,
585  int32_t& pixel, int32_t& line)
586 {
587  pixel = (int32_t)std::floor(m_inverseTransform[0] +
588  (m_inverseTransform[1] * x) + (m_inverseTransform[2] * y));
589  line = (int32_t) std::floor(m_inverseTransform[3] +
590  (m_inverseTransform[4] * x) + (m_inverseTransform[5] * y));
591 
592  // Return false if we're out of bounds.
593  return (pixel >= 0 && pixel < m_width &&
594  line >= 0 && line < m_height);
595 }

602 GDALError Raster::computePDALDimensionTypes()
603 {
604  if (!m_ds)
605  {
606  m_errorMsg = "Raster not open.";
607  return GDALError::NotOpen;
608  }
609 
610  m_types.clear();
611  for (int i = 0; i < m_numBands; ++i)
612  {
613  // Raster bands are numbered from 1.
614  GDALRasterBand *band = m_ds->GetRasterBand(i + 1);
615  if (!band)
616  {
617  std::ostringstream oss;
618 
619  oss << "Unable to get band " << (i + 1) <<
620  " from raster data source '" << m_filename << "'.";
621  m_errorMsg = oss.str();
622  return GDALError::InvalidBand;
623  }
624  m_types.push_back(toPdalType(band->GetRasterDataType()));
625  }
626  return GDALError::None;
627 }

630 GDALError Raster::read(double x, double y, std::vector<double>& data)
631 {
632  if (!m_ds)
633  {
634  m_errorMsg = "Raster not open.";
635  return GDALError::NotOpen;
636  }
637 
638  int32_t pixel(0);
639  int32_t line(0);
640  data.resize(m_numBands);
641 
642  std::array<double, 2> pix = { {0.0, 0.0} };
643 
644  // No data at this x,y if we can't compute a pixel/line location
645  // for it.
646  if (!getPixelAndLinePosition(x, y, pixel, line))
647  {
648  m_errorMsg = "Requested location is not in the raster.";
649  return GDALError::NoData;
650  }
651 
652  for (int i=0; i < m_numBands; ++i)
653  {
654  GDALRasterBandH b = GDALGetRasterBand(m_ds, i + 1);
655  if (GDALRasterIO(b, GF_Read, pixel, line, 1, 1,
656  &pix[0], 1, 1, GDT_Float64, 0, 0) == CE_None)
657  {
658  // we read a pixel put its values in our vector
659  data[i] = pix[0];
660  }
661  }
662 
663  return GDALError::None;
664 }
