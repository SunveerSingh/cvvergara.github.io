88 void OptechReader::initialize()
89 {
90  ILeStream stream(Utils::openFile(m_filename));
91  if (!stream)
92  throwError("Unable to open " + m_filename + " for reading.");
93 
94  stream.get(m_header.signature, 4);
95  if (strcmp(m_header.signature, "CSD") != 0)
96  throwError("Invalid header signature when reading CSD file: '" +
97  std::string(m_header.signature) + "'");
98 
99  stream.get(m_header.vendorId, 64);
100  stream.get(m_header.softwareVersion, 32);
101  stream >> m_header.formatVersion >> m_header.headerSize >>
102  m_header.gpsWeek >> m_header.minTime >> m_header.maxTime >>
103  m_header.numRecords >> m_header.numStrips;
104  for (size_t i = 0; i < 256; ++i)
105  {
106  stream >> m_header.stripPointers[i];
107  }
108  stream >> m_header.misalignmentAngles[0] >>
109  m_header.misalignmentAngles[1] >> m_header.misalignmentAngles[2] >>
110  m_header.imuOffsets[0] >> m_header.imuOffsets[1] >>
111  m_header.imuOffsets[2] >> m_header.temperature >> m_header.pressure;
112  stream.get(m_header.freeSpace, 830);
113 
114  m_boresightMatrix = createOptechRotationMatrix(
115  m_header.misalignmentAngles[0] + m_header.imuOffsets[0],
116  m_header.misalignmentAngles[1] + m_header.imuOffsets[1],
117  m_header.misalignmentAngles[2] + m_header.imuOffsets[2]);
118 }

144 point_count_t OptechReader::read(PointViewPtr data,
145  point_count_t countRequested)
146 {
147  point_count_t numRead = 0;
148  point_count_t dataIndex = data->size();
149 
150  while (numRead < countRequested)
151  {
152  if (m_returnIndex == 0)
153  {
154  if (!m_extractor.good())
155  {
156  if (m_recordIndex >= m_header.numRecords)
157  {
158  break;
159  }
160  m_recordIndex += fillBuffer();
161  }
162 
163  m_extractor >> m_pulse.gpsTime >> m_pulse.returnCount >>
164  m_pulse.range[0] >> m_pulse.range[1] >> m_pulse.range[2] >>
165  m_pulse.range[3] >> m_pulse.intensity[0] >>
166  m_pulse.intensity[1] >> m_pulse.intensity[2] >>
167  m_pulse.intensity[3] >> m_pulse.scanAngle >> m_pulse.roll >>
168  m_pulse.pitch >> m_pulse.heading >> m_pulse.latitude >>
169  m_pulse.longitude >> m_pulse.elevation;
170 
171  if (m_pulse.returnCount == 0)
172  {
173  m_returnIndex = 0;
174  continue;
175  }
176 
177  // In all the csd files that we've tested, the longitude
178  // values have been less than -2pi.
179  if (m_pulse.longitude < -M_PI * 2)
180  {
181  m_pulse.longitude = m_pulse.longitude + M_PI * 2;
182  }
183  else if (m_pulse.longitude > M_PI * 2)
184  {
185  m_pulse.longitude = m_pulse.longitude - M_PI * 2;
186  }
187  }
188 
189  georeference::Xyz gpsPoint = georeference::Xyz(
190  m_pulse.longitude, m_pulse.latitude, m_pulse.elevation);
191  georeference::RotationMatrix rotationMatrix =
192  createOptechRotationMatrix(m_pulse.roll, m_pulse.pitch,
193  m_pulse.heading);
194  georeference::Xyz point = pdal::georeference::georeferenceWgs84(
195  m_pulse.range[m_returnIndex], m_pulse.scanAngle,
196  m_boresightMatrix, rotationMatrix, gpsPoint);
197 
198  data->setField(Dimension::Id::X, dataIndex, point.X * 180 / M_PI);
199  data->setField(Dimension::Id::Y, dataIndex, point.Y * 180 / M_PI);
200  data->setField(Dimension::Id::Z, dataIndex, point.Z);
201  data->setField(Dimension::Id::GpsTime, dataIndex, m_pulse.gpsTime);
202  if (m_returnIndex == MaximumNumberOfReturns - 1)
203  {
204  data->setField(Dimension::Id::ReturnNumber, dataIndex,
205  m_pulse.returnCount);
206  }
207  else
208  {
209  data->setField(Dimension::Id::ReturnNumber, dataIndex,
210  m_returnIndex + 1);
211  }
212  data->setField(Dimension::Id::NumberOfReturns, dataIndex,
213  m_pulse.returnCount);
214  data->setField(Dimension::Id::EchoRange, dataIndex,
215  m_pulse.range[m_returnIndex]);
216  data->setField(Dimension::Id::Intensity, dataIndex,
217  m_pulse.intensity[m_returnIndex]);
218  data->setField(Dimension::Id::ScanAngleRank, dataIndex,
219  m_pulse.scanAngle * 180 / M_PI);
220 
221  if (m_cb)
222  m_cb(*data, dataIndex);
223 
224  ++dataIndex;
225  ++numRead;
226  ++m_returnIndex;
227 
228  if (m_returnIndex >= m_pulse.returnCount ||
229  m_returnIndex >= MaximumNumberOfReturns)
230  {
231  m_returnIndex = 0;
232  }
233  }
234  return numRead;
235 }

238 size_t OptechReader::fillBuffer()
239 {
240  size_t numRecords = std::min<size_t>(m_header.numRecords - m_recordIndex,
241  MaxNumRecordsInBuffer);
242 
243  buffer_size_t bufferSize = NumBytesInRecord * numRecords;
244  m_buffer.resize(bufferSize);
245  m_istream->get(m_buffer);
246  m_extractor = LeExtractor(m_buffer.data(), m_buffer.size());
247  return numRecords;
248 }
