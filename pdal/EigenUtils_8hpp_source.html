106 /**
107  Compute second derivative in X direction using central difference method.
108 
109  \param data the incoming Eigen matrix.
110  \param spacing the grid spacing of the matrix.
111  \return the second derivative in X.
112 */
113 template <typename Derived>
114 PDAL_DLL double centralDiffX2(const Eigen::MatrixBase<Derived>& data,
115  double spacing)
116 {
117  if (data.rows() != 3 || data.cols() != 3)
118  throw pdal_error("Must provide 3x3 matrix to centralDiffX2");
119  if (spacing <= 0.0)

121  return (data(1, 2) - 2.0 * data(1, 1) + data(1, 0)) / (spacing * spacing);

358 /**
359  Pad input matrix symmetrically.
360 
361  Symmetrically pads the input matrix with given radius.
362 
363  \param d the input matrix.
364  \param r the radius of the padding.
365  \return the padded matrix.
366 */
367 template <typename Derived>
368 PDAL_DLL Derived padMatrix(const Eigen::MatrixBase<Derived>& d, int r)
369 {
370  using namespace Eigen;
371 
372  Derived out = Derived::Zero(d.rows()+2*r, d.cols()+2*r);
373  out.block(r, r, d.rows(), d.cols()) = d;
374  out.block(r, 0, d.rows(), r) =
375  d.block(0, 0, d.rows(), r).rowwise().reverse();
376  out.block(r, d.cols()+r, d.rows(), r) =
377  d.block(0, d.cols()-r, d.rows(), r).rowwise().reverse();
378  out.block(0, 0, r, out.cols()) =
379  out.block(r, 0, r, out.cols()).colwise().reverse();
380  out.block(d.rows()+r, 0, r, out.cols()) =
381  out.block(out.rows()-r-1, 0, r, out.cols()).colwise().reverse();
382 
383  return out;
384 }

394 /**
395  Replace NaNs with mean.
396 
397  \param data the incoming Eigen matrix.
398  \return the updated Eigen matrix.
399 */
400 template <typename Derived>
401 PDAL_DLL Eigen::MatrixXd replaceNaNs(const Eigen::MatrixBase<Derived>& data)
402 {
403  Derived out(data);
404 
405  double mean(0.0);
406  int nv(0);
407  for (int i = 0; i < data.size(); ++i)
408  {
409  if (std::isnan(data(i)))
410  continue;
411  mean += data(i);
412  nv++;
413  }
414  mean /= nv;
415 
416  // replace NaNs with mean
417  for (int i = 0; i < data.size(); ++i)
418  {
419  if (std::isnan(data(i)))
420  out(i) = mean;
421  }
422 
423  return out;
424 }

439 /**
440  Compute the numerical gradient in the X direction.
441 
442  This is meant to mimic MATLAB's gradient function. The spacing between points
443  in each direction is assumed to be one.
444 
445  \param data the input matrix.
446  \return the X component of the two-dimensional gradient.
447 */
448 template <typename Derived>
449 PDAL_DLL Derived gradX(const Eigen::MatrixBase<Derived>& A)
450 {
451  Derived out = Derived::Zero(A.rows(), A.cols());
452 
453  // Interior points are obtained by central differences.
454  out.block(0, 1, A.rows(), A.cols()-2) =
455  0.5 * (A.rightCols(A.cols()-2) - A.leftCols(A.cols()-2));
456 
457  // Edge columns are obtained by single-sided differences.
458  out.col(0) = A.col(1) - A.col(0);
459  out.col(out.cols()-1) = A.col(A.cols()-1) - A.col(A.cols()-2);
460 
461  return out;
462 };

464 /**
465  Compute the numerical gradient in the Y direction.
466 
467  This is meant to mimic MATLAB's gradient function. The spacing between points
468  in each direction is assumed to be one.
469 
470  \param data the input matrix.
471  \return the Y component of the two-dimensional gradient.
472 */
473 template <typename Derived>
474 PDAL_DLL Derived gradY(const Eigen::MatrixBase<Derived>& A)
475 {
476  Derived out = Derived::Zero(A.rows(), A.cols());
477 
478  // Interior points are obtained by central differences.
479  out.block(1, 0, A.rows()-2, A.cols()) =
480  0.5 * (A.bottomRows(A.rows()-2) - A.topRows(A.rows()-2));
481 
482  // Edge rows are obtained by single-sided differences.
483  out.row(0) = A.row(1) - A.row(0);
484  out.row(out.rows()-1) = A.row(A.rows()-1) - A.row(A.rows()-2);
485 
486  return out;
487 };

489 /**
490  Compute contour curvature for a single 3x3 matrix.
491 
492  \param data the incoming Eigen matrix.
493  \param spacing the spacing between cells (or edge length of a cell).
494  \return the contour curvature.
495 */
496 template <typename Derived>
497 PDAL_DLL double computeContour(const Eigen::MatrixBase<Derived>& data,
498  double spacing)
499 {
500  if (data.rows() != 3 || data.cols() != 3)
501  throw pdal_error("Must provide 3x3 matrix to computeContour");
502  if (spacing <= 0.0)
503  throw pdal_error("Must provide positive spacing to computeContour");
504  Derived filled = replaceNaNs(data);
505  double zXX = centralDiffX2(filled, spacing);
506  double zYY = centralDiffY2(filled, spacing);
507  double zXY = centralDiffXY(filled, spacing);
508  double zX = centralDiffX(filled, spacing);
509  double zY = centralDiffY(filled, spacing);
510  double p = (zX * zX) + (zY * zY);
511  double q = p + 1;
512  return ((zXX*zX*zX)-(2*zXY*zX*zY)+(zYY*zY*zY))/(p*std::sqrt(q*q*q));
513 }

638  \param illumAltitudeDegree the illumination altitude in degrees.
639  \param illumAzimuthDegree the illumination azimuth in degrees.
640  \return the local slope.
641 */
642 template <typename Derived>
643 PDAL_DLL double computeHillshade(const Eigen::MatrixBase<Derived>& data,
644  double spacing, double illumAltitudeDegree,
645  double illumAzimuthDegree)
646 {
647  if (data.rows() != 3 || data.cols() != 3)
648  throw pdal_error("Must provide 3x3 matrix to computeHillshade");
649  if (spacing <= 0.0)
650  throw pdal_error("Must provide positive spacing to computeHillshade");

652  double zenithRad = (90.0 - illumAltitudeDegree) *
653  (3.14159265358979323846 / 180.0);

655  if (azimuthMath >= 360.0)
656  {
657  azimuthMath = azimuthMath - 360.0;
658  }
659  double azimuthRad = azimuthMath * (3.14159265358979323846 / 180.0);
660 
661  double dZdX = computeDZDX(data, spacing);
662  double dZdY = computeDZDY(data, spacing);
663  double slopeRad = computeSlopeRad(dZdX, dZdY);
664 
665  double aspectRad(0.0);
666  if (dZdX == 0.0)
667  {
668  if (dZdY > 0.0)
669  {
670  aspectRad = 3.14159265358979323846 / 2.0;
671  }
672  else if (dZdY < 0.0)
673  {
674  aspectRad = (2 * 3.14159265358979323846) -
675  (3.14159265358979323846 / 2.0);
676  }
677  }
678  else
679  {
680  aspectRad = std::atan2(dZdY, -dZdX);
681  if (aspectRad < 0.0)
682  {
683  aspectRad = 2.0 * 3.14159265358979323846 + aspectRad;
684  }
685  }
686 
687  return std::cos(zenithRad) * std::cos(slopeRad) +
688  std::sin(zenithRad) * std::sin(slopeRad) * std::cos(azimuthRad - aspectRad);
689 }

692  Compute aspect using finite difference method.

694  \param data the incoming Eigen matrix.
695  \param spacing the grid spacing.
696  \return the local aspect.

714  Compute aspect using D8 method.
715 
716  \param data the incoming Eigen matrix.
717  \param spacing the grid spacing.
718  \return the local aspect.
719 */

721 PDAL_DLL double computeAspectD8(const Eigen::MatrixBase<Derived>& data,
722  double spacing)
723 {
724  if (data.rows() != 3 || data.cols() != 3)
725  throw pdal_error("Must provide 3x3 matrix to computeAspectD8");
726  if (spacing <= 0.0)
727  throw pdal_error("Must provide positive spacing to computeAspectD8");
728  Derived submatrix(3, 3);
729  submatrix.setConstant(data(1, 1));
730  submatrix -= data;
731  submatrix /= spacing;
732  submatrix(0, 1) /= std::sqrt(2.0);
733  submatrix(1, 0) /= std::sqrt(2.0);
734  submatrix(1, 2) /= std::sqrt(2.0);
735  submatrix(2, 1) /= std::sqrt(2.0);
736 
737  // find max and convert to degrees
738  double maxval = std::numeric_limits<double>::lowest();
739  int j(0);
740  for (int i = 0; i < submatrix.size(); ++i)
741  {
742  if (std::isnan(submatrix(i)))
743  continue;
744  // skip center value, which will always be 0.0
745  if (i == 5)
746  continue;
747  if (submatrix(i) > maxval)
748  {
749  maxval = submatrix(i);
750  if (i == 1) j = 6;
751  if (i == 2) j = 5;
752  if (i == 3) j = 4;
753  if (i == 4) j = 7;
754  if (i == 6) j = 3;
755  if (i == 7) j = 0;
756  if (i == 8) j = 1;
757  if (i == 9) j = 2;
758  }
759  }
760 
761  return std::pow(2.0, j);

769  \return the local slope.

772 PDAL_DLL double computeSlopeD8(const Eigen::MatrixBase<Derived>& data,
773  double spacing)
774 {
775  if (data.rows() != 3 || data.cols() != 3)
776  throw pdal_error("Must provide 3x3 matrix to computeSlopeD8");

784  submatrix(0, 2) /= std::sqrt(2.0);
785  submatrix(2, 0) /= std::sqrt(2.0);
786  submatrix(2, 2) /= std::sqrt(2.0);

793  {
794  if (std::isnan(submatrix(i)))
795  continue;
796  if (submatrix(i) > maxval)
797  maxval = submatrix(i);
798  }
799  return 100.0 * maxval;

825  Perform a morphological dilation of the input matrix.
826 
827  Performs a morphological dilation of the input matrix using a circular
828  structuring element of given radius.
829 
830  \param data the input matrix.
831  \param radius the radius of the circular structuring element.
832  \return the morphological dilation of the input matrix.
833 */
834 template <typename Derived>
835 PDAL_DLL Derived dilate(const Eigen::MatrixBase<Derived>& A, int radius)
836 {
837  Derived B = Derived::Constant(A.rows(), A.cols(), 0);
838 
839  int length = 2 * radius + 1;
840  bool match_flag;
841  for (int c = 0; c < A.cols(); ++c)
842  {
843  for (int r = 0; r < A.rows(); ++r)
844  {
845  match_flag = false;
846  for (int k = 0; k < length; ++k)
847  {
848  if (match_flag)
849  break;
850  int cdiff = k-radius;
851  int cpos = c+cdiff;
852  if (cpos < 0 || cpos >= A.cols())
853  continue;
854  for (int l = 0; l < length; ++l)
855  {
856  int rdiff = l-radius;
857  int rpos = r+rdiff;
858  if (rpos < 0 || rpos >= A.rows())
859  continue;
860  if ((cdiff*cdiff+rdiff*rdiff) > radius*radius)
861  continue;
862  if (A(rpos, cpos) == 1)
863  {
864  match_flag = true;
865  break;
866  }
867  }
868  }
869  // Assign value according to match flag
870  B(r, c) = (match_flag) ? 1 : 0;
871  }
872  }
873 
874  return B;
875 }
