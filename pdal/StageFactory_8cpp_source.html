187  { "pcd", "readers.pcd" },
188  { "ply", "readers.ply" },
189  { "pts", "readers.pts" },
190  { "qi", "readers.qfit" },
191  { "rxp", "readers.rxp" },
192  { "sbet", "readers.sbet" },
193  { "sqlite", "readers.sqlite" },
194  { "sid", "readers.mrsid" },
195  { "tindex", "readers.tindex" },
196  { "txt", "readers.text" },
197  { "h5", "readers.icebridge" }
198  };
199 
200  static const std::string ghPrefix("greyhound://");
201 
202  std::string ext;
203  // filename may actually be a greyhound uri + pipelineId
204  if (Utils::iequals(filename.substr(0, ghPrefix.size()), ghPrefix))
205  ext = ".greyhound"; // Make it look like an extension.
206  else
207  ext = FileUtils::extension(filename);
208 
209  // Strip off '.' and make lowercase.
210  if (ext.length())
211  ext = Utils::tolower(ext.substr(1, ext.length() - 1));
212 
213  return drivers[ext]; // will be "" if not found
214 }

354 Stage *StageFactory::createStage(std::string const& stage_name)
355 {
356  static_assert(0 < sizeof(Stage), "");
357  Stage *s = static_cast<Stage*>(PluginManager::createObject(stage_name));
358  if (s)
359  {
360  std::lock_guard<std::mutex> lock(m_mutex);
361  m_ownedStages.push_back(std::unique_ptr<Stage>(s));
362  }
363  return s;
364 }

367 void StageFactory::destroyStage(Stage *s)
368 {
369  std::lock_guard<std::mutex> lock(m_mutex);
370  for (auto it = m_ownedStages.begin(); it != m_ownedStages.end(); ++it)
371  {
372  if (s == it->get())
373  {
374  m_ownedStages.erase(it);
375  break;
376  }
377  }
378 }
