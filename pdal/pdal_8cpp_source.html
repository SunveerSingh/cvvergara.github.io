122 void App::outputDrivers()
123 {
124  // Force plugin loading.
125  StageFactory f(false);
126 
127  StringList stages = PluginManager::names(PF_PluginType_Filter |
128  PF_PluginType_Reader | PF_PluginType_Writer);
129 
130  if (!m_showJSON)
131  {
132  int nameColLen(28);
133  int descripColLen(Utils::screenWidth() - nameColLen - 1);
134 
135  std::string tablehead(std::string(nameColLen, '=') + ' ' +
136  std::string(descripColLen, '='));
137 
138  m_out << std::endl;
139  m_out << tablehead << std::endl;
140  m_out << std::left << std::setw(nameColLen) << "Name" <<
141  " Description" << std::endl;
142  m_out << tablehead << std::endl;
143 
144  m_out << std::left;
145 
146 
147  for (auto name : stages)
148  {
149  std::string descrip = PluginManager::description(name);
150  StringList lines = Utils::wordWrap(descrip, descripColLen - 1);
151  for (size_t i = 0; i < lines.size(); ++i)
152  {
153  m_out << std::setw(nameColLen) << name << " " <<
154  lines[i] << std::endl;
155  name.clear();
156  }
157  }
158 
159  m_out << tablehead << std::endl << std::endl;
160  }
161  else
162  {
163 
164  Json::Value array(Json::arrayValue);
165  for (auto name : stages)
166  {
167  std::string description = PluginManager::description(name);
168  std::string link = PluginManager::link(name);
169  Json::Value node(Json::objectValue);
170  node["name"] = name;
171  node["description"] = description;
172  node["link"] = link;
173  array.append(node);
174  }
175 
176  m_out << array;
177 
178  }
179 }

193 {

307 std::string App::findKernel()
308 {
309  StringList loadedKernels;
310 
311  auto kernelSurname = [](const std::string& name)
312  {
313  StringList names = Utils::split2(name, '.');
314  return names.size() == 2 ? names[1] : std::string();
315  };
316 
317  StageFactory f(true);
318  // Discover available kernels without plugins, and test to see if
319  // the positional option 'command' is a valid kernel
320  loadedKernels = PluginManager::names(PF_PluginType_Kernel);
321  for (auto& name : loadedKernels)
322  if (m_command == kernelSurname(name))
323  return name;
324 
325  // Force loading of plugins.
326  StageFactory f2(false);
327  loadedKernels = PluginManager::names(PF_PluginType_Kernel);
328  for (auto& name : loadedKernels)
329  if (m_command == kernelSurname(name))
330  return name;
331 
332  return std::string();
333 }

336 int App::execute(StringList& cmdArgs, LogPtr& log)
337 {
338  ProgramArgs args;
339 
340  addArgs(args);
341  try
342  {
343  args.parseSimple(cmdArgs);
344  }
345  catch (arg_val_error const& e)
346  {
347  Utils::printError(e.what());
348  return -1;
349  }
350 
351  log.reset(new Log("PDAL", m_log));
352  if (m_logLevel != LogLevel::None)
353  log->setLevel(m_logLevel);
354  else if (m_debug)
355  log->setLevel(LogLevel::Debug);
356  PluginManager::setLog(log);
357 #ifdef PDAL_HAVE_EXECINFO_H
358  if (m_debug)
359  {
360  signal(SIGSEGV, [](int sig)
361  {
362  logPtr->get(LogLevel::Debug) << "Segmentation fault (signal 11)\n";
363  StringList lines = Utils::backtrace();
364 
365  for (const auto& l : lines)
366  logPtr->get(LogLevel::Debug) << l << std::endl;
367  exit(1);
368  });
369  }
370 #endif
371 
372  m_command = Utils::tolower(m_command);
373  if (!m_command.empty())
374  {
375  int ret = 0;
376  std::string name(findKernel());
377  if (name.size())
378  {
379  if (m_help)
380  cmdArgs.push_back("--help");
381  void *obj = PluginManager::createObject(name);
382  Kernel *kernel(static_cast<Kernel *>(obj));
383  // This shouldn't throw. If it does, it's something awful, so
384  // not cleaning up seems inconsequential.
385  log->setLeader("pdal " + m_command);
386  ret = kernel->run(cmdArgs, log);
387  delete kernel;
388  // IMPORTANT - The kernel must be destroyed before GDAL
389  // drivers are unregistered or GDAL will attempt to destroy
390  // resources more than once, resulting in a crash.
391  gdal::unregisterDrivers();
392  }
393  else
394  log->get(LogLevel::Error) << "Command '" << m_command <<
395  "' not recognized" << std::endl << std::endl;
396  return ret;
397  }
398 
399  if (m_showVersion)
400  outputVersion();
401  else if (m_showDrivers)
402  outputDrivers();
403  else if (m_showOptions.size())
404  {
405  if (m_showOptions == "all")
406  outputOptions();
407  else
408  outputOptions(m_showOptions, m_out);
409  }
410  else
411  outputHelp(args);
412  return 0;
413 }
