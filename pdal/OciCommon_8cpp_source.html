48 
49  std::string connection(connSpec);
50 
51  if (Utils::fileExists(connection))
52  {
53  std::istream::pos_type size;
54  std::istream* input = Utils::openFile(connection, true);
55  if (!input->good())
56  {
57  Utils::closeFile(input);
58  throw connection_failed("Unable to open connection filename "
59  "for Oracle.");
60  }
61 
62  std::string output;
63  std::string line;
64  while (input->good())
65  {
66  getline(*input, line);
67  if (output.size())
68  output += "\n" + line;
69  else
70  output = line;
71  }
72  connection = output;
73 
74  Utils::closeFile(input);
75  }
76 
77  Connection con;
78 
79  auto pos = connection.find("/", 0);
80  if (pos == string::npos)
81  return con;
82 
83  string username = connection.substr(0, pos);
84  connection = connection.substr(pos + 1);
85  pos = connection.find("@", 0);
86  if (pos == string::npos)
87  return con;
88  string password = connection.substr(0, pos);
89  string instance = connection.substr(pos + 1);
90 
91  con = make_shared<pdal::OWConnection>(username.c_str(),
92  password.c_str(), instance.c_str());
93  return con;
94 }

99  // Fetch the XML that defines the schema for this point cloud
100  std::ostringstream schemaQuery;
101  OCILobLocator* metadata = NULL;
102  schemaQuery <<
103  "DECLARE" << std::endl << "PC_TABLE VARCHAR2(32) := '" <<
104  stmt->GetString(block->pc->base_table) << "';" << std::endl <<
105  "PC_ID NUMBER := " << stmt->GetInteger(&(block->pc->pc_id)) <<
106  ";" << std::endl <<
107  "PC_COLUMN VARCHAR2(32) := '" <<
108  stmt->GetString(block->pc->base_column) << "';" << std::endl <<
109  "BEGIN" << std::endl <<
110  std::endl <<
111  "EXECUTE IMMEDIATE" << std::endl <<
112  " 'SELECT T.'||PC_COLUMN||'.PC_OTHER_ATTRS.getClobVal()"
113  "FROM '||pc_table||' T WHERE T.ID='||"
114  "PC_ID INTO :metadata;" << std::endl <<
115  "END;" << std::endl;
116 
117  Statement getSchemaStmt(
118  block->m_connection->CreateStatement(schemaQuery.str().c_str()));
119  getSchemaStmt->BindName(":metadata", &metadata);
120 
121  getSchemaStmt->Execute();
122 
123  char* pc_schema = getSchemaStmt->ReadCLob(metadata);
124  std::string pc_schema_xml;
125  if (pc_schema)

127  pc_schema_xml = pc_schema;
128  CPLFree(pc_schema);
129  }
