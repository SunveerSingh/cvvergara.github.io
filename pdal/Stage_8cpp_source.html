156 PointViewSet Stage::execute(PointTableRef table)
157 {
158  startLogging();
159  table.finalize();
160 
161  PointViewSet views;
162 
163  // If the inputs are empty, we're a reader.
164  if (m_inputs.empty())
165  {
166  views.insert(PointViewPtr(new PointView(table)));
167  }
168  else
169  {
170  for (size_t i = 0; i < m_inputs.size(); ++i)
171  {
172  Stage *prev = m_inputs[i];
173  PointViewSet temp = prev->execute(table);
174  views.insert(temp.begin(), temp.end());
175  }
176  }
177 
178  PointViewSet outViews;
179  std::vector<StageRunnerPtr> runners;
180 
181  // Put the spatial references from the views onto the table.
182  // The table's spatial references are only valid as long as the stage
183  // is running.
184  // ABELL - Should we clear the references once the stage run has
185  // completed? Wondering if that would break something where a
186  // writer wants to check a table's SRS.
187  SpatialReference srs;
188  table.clearSpatialReferences();
189  // Iterating backwards will ensure that the SRS for the first view is
190  // first on the list for table.
191  for (auto it = views.rbegin(); it != views.rend(); it++)
192  table.addSpatialReference((*it)->spatialReference());
193 
194  // Count the number of views and the number of points and faces so they're
195  // available to stages.
196  m_pointCount = 0;
197  m_faceCount = 0;
198  for (auto const& it : views)
199  {
200  m_pointCount += it->size();
201  auto m = it->mesh();
202  if (m)
203  m_faceCount += m->size();
204  }
205  // Do the ready operation and then start running all the views
206  // through the stage.
207  ready(table);
208  for (auto const& it : views)
209  {
210  StageRunnerPtr runner(new StageRunner(this, it));
211  runners.push_back(runner);
212  runner->run();
213  }
214 
215  // As the stages complete (synchronously at this time), propagate the
216  // spatial reference and merge the output views.
217  srs = getSpatialReference();
218  for (auto const& it : runners)
219  {
220  StageRunnerPtr runner(it);
221  PointViewSet temp = runner->wait();
222 
223  // If our stage has a spatial reference, the view takes it on once
224  // the stage has been run.
225  if (!srs.empty())
226  for (PointViewPtr v : temp)
227  v->setSpatialReference(srs);
228  outViews.insert(temp.begin(), temp.end());
229  }
230  l_done(table);
231  stopLogging();
232  m_pointCount = 0;
233  m_faceCount = 0;
234  return outViews;
235 }

239 void Stage::execute(StreamPointTable& table)
240 {
241  struct StageList : public std::list<Stage *>
242  {
243  StageList operator - (const StageList& other) const
244  {
245  StageList resultList;
246  auto ti = rbegin();
247  auto oi = other.rbegin();
248 
249  while (oi != other.rend() && ti != rend() && *ti == *oi)
250  {
251  oi++;
252  ti++;
253  }
254  while (ti != rend())
255  resultList.push_front(*ti++);
256  return resultList;
257  };
258 
259  void ready(PointTableRef& table)
260  {
261  for (auto s : *this)
262  {
263  s->startLogging();
264  s->ready(table);
265  s->stopLogging();
266  SpatialReference srs = s->getSpatialReference();
267  if (!srs.empty())
268  table.setSpatialReference(srs);
269  }
270  }
271 
272  void done(PointTableRef& table)
273  {
274  for (auto s : *this)
275  {
276  s->startLogging();
277  s->l_done(table);
278  s->stopLogging();
279  }
280  }
281  };
282 
283  SpatialReference srs;
284  std::list<StageList> lists;
285  StageList stages;
286  StageList lastRunStages;
287 
288  table.finalize();
289 
290  // Walk from the current stage backwards. As we add each input, copy
291  // the list of stages and push it on a list. We then pull a list from the
292  // back of list and keep going. Pushing on the front and pulling from the
293  // back insures that the stages will be executed in the order that they
294  // were added. If we hit stage with no previous stages, we execute
295  // the stage list.
296  // All this often amounts to a bunch of list copying for
297  // no reason, but it's more simple than what we might otherwise do and
298  // this should be a nit in the grand scheme of execution time.
299  //
300  // As an example, if there are four paths from the end stage (writer) to
301  // reader stages, there will be four stage lists and execute(table, stages)
302  // will be called four times.
303  Stage *s = this;
304  stages.push_front(s);
305  while (true)
306  {
307  if (s->m_inputs.empty())
308  {
309  // Call done on all the stages we ran last time and aren't
310  // using this time.
311  (lastRunStages - stages).done(table);
312  // Call ready on all the stages we didn't run last time.
313  (stages - lastRunStages).ready(table);
314  execute(table, stages);
315  lastRunStages = stages;
316  }
317  else
318  {
319  for (auto s2 : s->m_inputs)
320  {
321  StageList newStages(stages);
322  newStages.push_front(s2);
323  lists.push_front(newStages);
324  }
325  }
326  if (lists.empty())
327  {
328  lastRunStages.done(table);
329  break;
330  }
331  stages = lists.back();
332  lists.pop_back();
333  s = stages.front();
334  }
335 }

338 void Stage::execute(StreamPointTable& table, std::list<Stage *>& stages)
339 {
340  std::vector<bool> skips(table.capacity());
341  std::list<Stage *> filters;
342  SpatialReference srs;
343  std::map<Stage *, SpatialReference> srsMap;
344 
345  // Separate out the first stage.
346  Stage *reader = stages.front();
347 
348  // Build a list of all stages except the first. We may have a writer in
349  // this list in addition to filters, but we treat them in the same way.
350  auto begin = stages.begin();
351  begin++;
352  std::copy(begin, stages.end(), std::back_inserter(filters));
353 
354  // Loop until we're finished. We handle the number of points up to
355  // the capacity of the StreamPointTable that we've been provided.
356 
357  bool finished = false;
358  while (!finished)
359  {
360  // Clear the spatial reference when processing starts.
361  table.clearSpatialReferences();
362  PointId idx = 0;
363  PointRef point(table, idx);
364  point_count_t pointLimit = table.capacity();
365 
366  reader->startLogging();
367  // When we get false back from a reader, we're done, so set
368  // the point limit to the number of points processed in this loop
369  // of the table.
370  if (!pointLimit)
371  finished = true;
372 
373  for (PointId idx = 0; idx < pointLimit; idx++)
374  {
375  point.setPointId(idx);
376  finished = !reader->processOne(point);
377  if (finished)
378  pointLimit = idx;
379  }
380  reader->stopLogging();
381  srs = reader->getSpatialReference();
382  if (!srs.empty())
383  table.setSpatialReference(srs);
384 
385  // When we get a false back from a filter, we're filtering out a
386  // point, so add it to the list of skips so that it doesn't get
387  // processed by subsequent filters.
388  for (Stage *s : filters)
389  {
390  if (srsMap[s] != srs)
391  {
392  s->spatialReferenceChanged(srs);
393  srsMap[s] = srs;
394  }
395  s->startLogging();
396  for (PointId idx = 0; idx < pointLimit; idx++)
397  {
398  if (skips[idx])
399  continue;
400  point.setPointId(idx);
401  if (!s->processOne(point))
402  skips[idx] = true;
403  }
404  srs = s->getSpatialReference();
405  if (!srs.empty())
406  table.setSpatialReference(srs);
407  s->stopLogging();
408  }
409 
410  // Yes, vector<bool> is terrible. Can do something better later.
411  for (size_t i = 0; i < skips.size(); ++i)
412  skips[i] = false;
413  table.reset();
414  }
415 }
