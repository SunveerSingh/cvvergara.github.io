65 class CmpZOrder
66 {
67 public:
68  bool operator()(const Coord& c1, const Coord& c2) const
69  {
70  int a[2] = {(int)(c1.first * INT_MAX), (int)(c1.second * INT_MAX)};
71  int b[2] = {(int)(c2.first * INT_MAX), (int)(c2.second * INT_MAX)};
72 
73  int j = 0;
74  int x = 0;
75 
76  for (int k = 0; k < 2; k++)
77  {
78  int y = a[k] ^ b[k];
79  if (less_msb(x, y))
80  {
81  j = k;
82  x = y;
83  }
84  }
85  return (a[j] - b[j]) < 0;
86  };
87 };
88 }

90 PointViewSet MortonOrderFilter::run(PointViewPtr inView)
91 {
92  PointViewSet viewSet;
93  if (!inView->size())
94  return viewSet;
95  CmpZOrder compare;
96  std::multimap<Coord, PointId, CmpZOrder> sorted(compare);
97 
98  BOX2D buffer_bounds;
99  inView->calculateBounds(buffer_bounds);
100  double xrange = buffer_bounds.maxx - buffer_bounds.minx;
101  double yrange = buffer_bounds.maxy - buffer_bounds.miny;
102 
103  for (PointId idx = 0; idx < inView->size(); idx++)
104  {
105  double xpos = (inView->getFieldAs<double>(Dimension::Id::X, idx) -
106  buffer_bounds.minx) / xrange;
107  double ypos = (inView->getFieldAs<double>(Dimension::Id::Y, idx) -
108  buffer_bounds.miny) / yrange;
109  Coord loc(xpos, ypos);
110  sorted.insert(std::make_pair(loc, idx));
111  }
112 
113  PointViewPtr outView = inView->makeNew();
114  std::multimap<Coord, PointId, CmpZOrder>::iterator pos;
115  for (pos = sorted.begin(); pos != sorted.end(); ++pos)
116  {
117  outView->appendPoint(*inView, pos->second);
118  }
119  viewSet.insert(outView);
120 
121  return viewSet;
122 }
