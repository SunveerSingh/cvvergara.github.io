81 GeotiffSrs::GeotiffSrs(const std::vector<uint8_t>& directoryRec,
82  const std::vector<uint8_t>& doublesRec,
83  const std::vector<uint8_t>& asciiRec)
84 {
85  GeotiffCtx ctx;
86 
87  if (directoryRec.empty())
88  return;
89 
90  // Make sure struct is 16 bytes.
91 #pragma pack(push)
92 #pragma pack(1)
93  struct ShortKeyHeader
94  {
95  uint16_t dirVersion;
96  uint16_t keyRev;
97  uint16_t minorRev;
98  uint16_t numKeys;
99  };
100 #pragma pack(pop)
101 
102  const ShortKeyHeader *header = (const ShortKeyHeader *)directoryRec.data();
103  size_t declaredSize = (header->numKeys + 1) * 4;
104  if (directoryRec.size() < declaredSize)
105  return;
106 
107  uint8_t *dirData = const_cast<uint8_t *>(directoryRec.data());
108  ST_SetKey(ctx.tiff, GEOTIFF_DIRECTORY_RECORD_ID,
109  (1 + header->numKeys) * 4, STT_SHORT, (void *)dirData);
110 
111  if (doublesRec.size())
112  {
113  uint8_t *doubleData = const_cast<uint8_t *>(doublesRec.data());
114  ST_SetKey(ctx.tiff, GEOTIFF_DOUBLES_RECORD_ID,
115  doublesRec.size() / sizeof(double), STT_DOUBLE,
116  (void *)doubleData);
117  }
118 
119  if (asciiRec.size())
120  {
121  uint8_t *asciiData = const_cast<uint8_t *>(asciiRec.data());
122  ST_SetKey(ctx.tiff, GEOTIFF_ASCII_RECORD_ID,
123  asciiRec.size(), STT_ASCII, (void *)asciiData);
124  }
125 
126  ctx.gtiff = GTIFNewSimpleTags(ctx.tiff);
127 
128  GTIFDefn sGTIFDefn;
129  if (GTIFGetDefn(ctx.gtiff, &sGTIFDefn))
130  {
131  char *wkt = GTIFGetOGISDefn(ctx.gtiff, &sGTIFDefn);
132  if (wkt)
133  m_srs.set(wkt);
134  }
135 }

138 GeotiffTags::GeotiffTags(const SpatialReference& srs)
139 {
140  if (srs.empty())
141  return;
142 
143  GeotiffCtx ctx;
144  ctx.gtiff = GTIFNewSimpleTags(ctx.tiff);
145 
146  // Set tiff tags from WKT
147  if (!GTIFSetFromOGISDefn(ctx.gtiff, srs.getWKT().c_str()))
148  throw error("Could not set m_gtiff from WKT");
149  GTIFWriteKeys(ctx.gtiff);
150 
151  auto sizeFromType = [](int type, int count) -> size_t
152  {
153  if (type == STT_ASCII)
154  return count;
155  else if (type == STT_SHORT)
156  return 2 * count;
157  else if (type == STT_DOUBLE)
158  return 8 * count;
159  return 8 * count;
160  };
161 
162  int count;
163  int st_type;
164  uint8_t *data;
165  if (ST_GetKey(ctx.tiff, GEOTIFF_DIRECTORY_RECORD_ID,
166  &count, &st_type, (void **)&data))
167  {
168  size_t size = sizeFromType(st_type, count);
169  m_directoryRec.resize(size);
170  std::copy(data, data + size, m_directoryRec.begin());
171  }
172  if (ST_GetKey(ctx.tiff, GEOTIFF_DOUBLES_RECORD_ID,
173  &count, &st_type, (void **)&data))
174  {
175  size_t size = sizeFromType(st_type, count);
176  m_doublesRec.resize(size);
177  std::copy(data, data + size, m_doublesRec.begin());
178  }
179  if (ST_GetKey(ctx.tiff, GEOTIFF_ASCII_RECORD_ID,
180  &count, &st_type, (void **)&data))
181  {
182  size_t size = sizeFromType(st_type, count);
183  m_asciiRec.resize(size);
184  std::copy(data, data + size, m_asciiRec.begin());
185  }
186 }
