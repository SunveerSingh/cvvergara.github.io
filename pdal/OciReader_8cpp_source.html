69 void OciReader::initialize()
70 {
71  m_compression = false;
72  try
73  {
74  m_connection = connect(m_connSpec);
75  }
76  catch (const connection_failed& err)
77  {
78  throwError(err.what());
79  }
80  m_block = BlockPtr(new Block(m_connection));
81 
82  gdal::registerDrivers();
83  if (m_query.empty())
84  throwError("'query' statement is empty. No data can be read.");
85 
86  m_stmt = Statement(m_connection->CreateStatement(m_query.c_str()));
87  m_stmt->Execute(0);
88 
89  validateQuery();
90 
91  // Map the query to the block in which data is to be stored.
92  defineBlock(m_stmt, m_block);
93 
94  // Fetch an initial row of data.
95  if (!m_stmt->Fetch())
96  throwError("Unable to fetch a point cloud entry entry.");
97  m_block->setFetched();
98 
99  // If the spatial reference wasn't provided as an option, fetch it from
100  // the data source.
101  if (m_spatialRef.empty())
102  m_spatialRef = fetchSpatialReference(m_stmt, m_block);
103  setSpatialReference(m_spatialRef);
104 }

107 void OciReader::defineBlock(Statement stmt, BlockPtr block) const
108 {
109  int iCol = 0;
110  char szFieldName[OWNAME];
111  int hType = 0;
112  int nSize = 0;
113  int nPrecision = 0;
114  signed short nScale = 0;
115  char szTypeName[OWNAME];
116 
117  while (stmt->GetNextField(iCol, szFieldName, &hType, &nSize,
118  &nPrecision, &nScale, szTypeName))
119  {
120  std::string typeName = Utils::toupper(szTypeName);
121  std::string fieldName = Utils::toupper(szFieldName);
122 
123  if (hType == SQLT_NTY && (typeName == "SDO_PC"))
124  stmt->Define(&(block->pc));
125  else if (fieldName == "OBJ_ID")
126  stmt->Define(&(block->obj_id));
127  else if (fieldName == "BLK_ID")
128  stmt->Define(&(block->blk_id));
129  else if (fieldName == "BLK_EXTENT")
130  stmt->Define(&(block->blk_extent));
131  else if (fieldName == "BLK_DOMAIN")
132  stmt->Define(&(block->blk_domain));
133  else if (fieldName == "PCBLK_MIN_RES")
134  stmt->Define(&(block->pcblk_min_res));
135  else if (fieldName == "PCBLK_MAX_RES")
136  stmt->Define(&(block->pcblk_max_res));
137  else if (fieldName == "NUM_POINTS")
138  stmt->Define(&(block->num_points));
139  else if (fieldName == "NUM_UNSORTED_POINTS")
140  stmt->Define(&(block->num_unsorted_points));
141  else if (fieldName == "PT_SORT_DIM")
142  stmt->Define(&(block->pt_sort_dim));
143  else if (fieldName == "POINTS")
144  stmt->Define(&(block->locator));
145  iCol++;
146  }
147 }

150 // Throws an exception if a query is invalid.
151 void OciReader::validateQuery()
152 {
153  int col = 0;
154  char fieldName[OWNAME];
155  int hType = 0;
156  int size = 0;
157  int precision = 0;
158  signed short scale = 0;
159  char typeName[OWNAME];
160  bool typeCorrect = false;
161 
162  // We must have all of these field names present to be considered block
163  // data.
164  std::set<std::string> reqFields;
165  reqFields.insert("OBJ_ID");
166  reqFields.insert("BLK_ID");
167  reqFields.insert("BLK_EXTENT");
168  reqFields.insert("BLK_DOMAIN");
169  reqFields.insert("PCBLK_MIN_RES");
170  reqFields.insert("PCBLK_MAX_RES");
171  reqFields.insert("NUM_POINTS");
172  reqFields.insert("NUM_UNSORTED_POINTS");
173  reqFields.insert("PT_SORT_DIM");
174  reqFields.insert("POINTS");
175 
176  while (m_stmt->GetNextField(col, fieldName, &hType, &size,
177  &precision, &scale, typeName))
178  {
179  reqFields.erase(fieldName);
180  if (hType == SQLT_NTY)
181  {
182  if (strcmp(typeName,"SDO_PC") == 0)
183  typeCorrect = true;
184  }
185  col++;
186  }
187 
188  if (!typeCorrect)
189  throwError("Select statement '" + m_query + "' does not fetch "
190  "a SDO_PC object.");
191 
192  // If we found all the fields, the list of required fields will be empty.
193  // If not, throw an exception.
194  if (!reqFields.empty())
195  {
196  std::ostringstream oss;
197 
198  oss << "Query returns a block but is missing reqired fields: ";
199  auto i = reqFields.begin();
200  while (i != reqFields.end())
201  {
202  oss << *i;
203  i++;
204  if (i != reqFields.end())
205  oss << ",";
206  }
207  throwError(oss.str());
208  }
209 }

247 point_count_t OciReader::read(PointViewPtr view, point_count_t count)
248 {
249  if (eof())
250  return 0;
251 
252  point_count_t totalNumRead = 0;
253  while (totalNumRead < count)
254  {
255  if (m_block->numRemaining() == 0)
256  if (!readOci(m_stmt, m_block))
257  return totalNumRead;
258  PointId bufBegin = view->size();
259 
260  point_count_t numRead = 0;
261  if (orientation() == Orientation::DimensionMajor)
262  numRead = readDimMajor(*view, m_block, count - totalNumRead);
263  else if (orientation() == Orientation::PointMajor)
264  numRead = readPointMajor(*view, m_block, count - totalNumRead);
265  PointId bufEnd = bufBegin + numRead;
266  totalNumRead += numRead;
267  }
268  return totalNumRead;
269 }

272 point_count_t OciReader::readDimMajor(PointView& view, BlockPtr block,
273  point_count_t numPts)
274 {
275  using namespace Dimension;
276 
277  point_count_t numRemaining = block->numRemaining();
278  PointId startId = view.size();
279  point_count_t blockRemaining = numRemaining;
280  point_count_t numRead = 0;
281 
282  DimTypeList dims = dbDimTypes();
283  for (auto di = dims.begin(); di != dims.end(); ++di)
284  {
285  PointId nextId = startId;
286  char *pos = seekDimMajor(*di, block);
287  blockRemaining = numRemaining;
288  numRead = 0;
289  while (numRead < numPts && blockRemaining > 0)
290  {
291  writeField(view, pos, *di, nextId);
292  pos += Dimension::size(di->m_type);
293 
294  if (di->m_id == Id::PointSourceId && m_updatePointSourceId)
295  view.setField(Id::PointSourceId, nextId, block->obj_id);
296 
297  if (m_cb && di == dims.rbegin().base() - 1)
298  m_cb(view, nextId);
299 
300  nextId++;
301  numRead++;
302  blockRemaining--;
303  }
304  }
305  block->setNumRemaining(blockRemaining);
306  return numRead;
307 }

310 point_count_t OciReader::readPointMajor(PointView& view,
311  BlockPtr block, point_count_t numPts)
312 {
313  PointId nextId = view.size();
314  point_count_t numRead = 0;
315 
316  if (m_compression)
317  {
318 #ifdef PDAL_HAVE_LAZPERF
319  auto cb = [this, &view, &nextId, &numRead](char *buf, size_t bufsize)
320  {
321  writePoint(view, nextId, buf);
322  if (m_cb)
323  m_cb(view, nextId);
324  nextId++;
325  numRead++;
326  };
327  const char *src = reinterpret_cast<const char *>(block->chunk.data());
328  size_t srcsize = block->chunk.size();
329  LazPerfDecompressor(cb, dbDimTypes(), block->numRemaining()).
330  decompress(src, srcsize);
331 #else
332  throwError("Can't decompress without LAZperf.");
333 #endif
334  }
335  else
336  {
337  char *pos = seekPointMajor(block);
338 
339  size_t cnt = block->numRemaining();
340  while (numRead < numPts && cnt--)
341  {
342  writePoint(view, nextId, pos);
343 
344  if (m_cb)
345  m_cb(view, nextId);
346 
347  pos += packedPointSize();
348  nextId++;
349  numRead++;
350  }
351  }
352  block->setNumRemaining(block->numRemaining() - numRead);
353  return numRead;
354 }

371 // Read a block (set of points) from the database.
372 bool OciReader::readOci(Statement stmt, BlockPtr block)
373 {
374  if (!block->fetched())
375  {
376  if (!stmt->Fetch())
377  {
378  m_atEnd = true;
379  return false;
380  }
381  block->setFetched();
382  }
383  // Read the points from the blob in the row.
384  readBlob(stmt, block);
385  XMLSchema *s = findSchema(stmt, block);
386  updateSchema(*s);
387  MetadataNode comp = s->getMetadata().findChild("compression");
388  m_compression = (comp.value() == "lazperf");
389 
390  block->reset();
391  block->clearFetched();
392  return true;
393 }

396 void OciReader::readBlob(Statement stmt, BlockPtr block)
397 {
398  uint32_t amountRead = 0;
399  uint32_t blobLength = stmt->GetBlobLength(block->locator);
400 
401  if (block->chunk.size() < blobLength)
402  block->chunk.resize(blobLength);
403 
404  if (!stmt->ReadBlob(block->locator, (void*)(block->chunk.data()),
405  block->chunk.size() , &amountRead))
406  throwError("Did not read all blob data.");
407 
408  block->chunk.resize(amountRead);

412 // All of the schemas should be the same with regard to actual dimension
413 // name, order, etc, but each cloud may have its own scaling for X, Y and Z.
414 // Store it away so that it can be applied later if necessary.
415 // HOBU -- nope. Each Block/cloud combo could potentially have a different
416 // schema, with same names but different composition.
417 // ABELL -- the setup doesn't allow this. Dimensions and order is all stored
418 // when the first block is read. We don't have any facility for modifying
419 // dimensions as blocks are read. If this were to actually happen, things
420 // would break. Should we do something about it?
421 XMLSchema *OciReader::findSchema(Statement stmt, BlockPtr block)
422 {
423  int32_t cloudId = stmt->GetInteger(&block->pc->pc_id);
424  auto si = m_schemas.find(cloudId);
425  if (si == m_schemas.end())
426  {
427  XMLSchema s = fetchSchema(stmt, block);
428  auto i = m_schemas.insert(std::make_pair(cloudId, s));
429  si = i.first;
430  }
431  return &(si->second);
432 }
