55  template<class T>
56  T getFieldAs(Dimension::Id dim) const
57  {
58  T val(0);
59  bool success = true;
60  Everything e;
61  Dimension::Type type = m_layout.dimDetail(dim)->type();
62 
63  m_container.getFieldInternal(dim, m_idx, &e);
64  switch (type)
65  {
66  case Dimension::Type::Unsigned8:
67  success = Utils::numericCast(e.u8, val);
68  break;
69  case Dimension::Type::Unsigned16:
70  success = Utils::numericCast(e.u16, val);
71  break;
72  case Dimension::Type::Unsigned32:
73  success = Utils::numericCast(e.u32, val);
74  break;
75  case Dimension::Type::Unsigned64:
76  success = Utils::numericCast(e.u64, val);
77  break;
78  case Dimension::Type::Signed8:
79  success = Utils::numericCast(e.s8, val);
80  break;
81  case Dimension::Type::Signed16:
82  success = Utils::numericCast(e.s16, val);
83  break;
84  case Dimension::Type::Signed32:
85  success = Utils::numericCast(e.s32, val);
86  break;
87  case Dimension::Type::Signed64:
88  success = Utils::numericCast(e.s64, val);
89  break;
90  case Dimension::Type::Float:
91  success = Utils::numericCast(e.f, val);
92  break;
93  case Dimension::Type::Double:
94  success = Utils::numericCast(e.d, val);
95  break;
96  case Dimension::Type::None:
97  val = 0;
98  break;
99  }
100  if (!success)
101  {
102  std::ostringstream oss;
103  oss << "Unable to fetch data and convert as requested: ";
104  oss << Dimension::name(dim) << ":" <<
105  Dimension::interpretationName(type) <<
106  "(" << Utils::toDouble(e, type) << ") -> " <<
107  Utils::typeidName<T>();
108  throw pdal_error(oss.str());
109  }
110  return val;
111  }

168  /// Fill a buffer with point data specified by the dimension list.
169  /// \param[in] dims List of dimensions/types to retrieve.
170  /// \param[in] idx Index of point to get.
171  /// \param[in] buf Pointer to buffer to fill.
172  void getPackedData(const DimTypeList& dims, char *buf) const
173  {
174  for (auto di = dims.begin(); di != dims.end(); ++di)
175  {
176  getField(buf, di->m_id, di->m_type);
177  buf += Dimension::size(di->m_type);
178  }
179  }
180 
181  /// Load the point buffer from memory whose arrangement is specified
182  /// by the dimension list.
183  /// \param[in] dims Dimension/types of data in packed order
184  /// \param[in] idx Index of point to write.
185  /// \param[in] buf Packed data buffer.
186  void setPackedData(const DimTypeList& dims, const char *buf)
187  {

189  {
190  setField(di->m_id, di->m_type, (const void *)buf);
