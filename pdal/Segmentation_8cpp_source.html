52 std::vector<std::vector<PointId>> extractClusters(PointView& view,
53  uint64_t min_points,
54  uint64_t max_points,
55  double tolerance)
56 {
57  // Index the incoming PointView for subsequent radius searches.
58  KD3Index kdi(view);
59  kdi.build();
60 
61  // Create variables to track PointIds that have already been added to
62  // clusters and to build the list of cluster indices.
63  std::vector<PointId> processed(view.size(), 0);
64  std::vector<std::vector<PointId>> clusters;
65 
66  for (PointId i = 0; i < view.size(); ++i)
67  {
68  // Points can only belong to a single cluster.
69  if (processed[i])
70  continue;
71 
72  // Initialize list of indices belonging to current cluster, marking the
73  // seed point as processed.
74  std::vector<PointId> seed_queue;
75  size_t sq_idx = 0;
76  seed_queue.push_back(i);
77  processed[i] = 1;
78 
79  // Check each point in the cluster for additional neighbors within the
80  // given tolerance, remembering that the list can grow if we add points
81  // to the cluster.
82  while (sq_idx < seed_queue.size())
83  {
84  // Find neighbors of the next cluster point.
85  PointId j = seed_queue[sq_idx];
86  std::vector<PointId> ids = kdi.radius(j, tolerance);
87 
88  // The case where the only neighbor is the query point.
89  if (ids.size() == 1)
90  {
91  sq_idx++;
92  continue;
93  }
94 
95  // Skip neighbors that already belong to a cluster and add the rest
96  // to this cluster.
97  for (auto const& k : ids)
98  {
99  if (processed[k])
100  continue;
101  seed_queue.push_back(k);
102  processed[k] = 1;
103  }
104 
105  sq_idx++;
106  }
107 
108  // Keep clusters that are within the min/max number of points.
109  if (seed_queue.size() >= min_points && seed_queue.size() <= max_points)
110  clusters.push_back(seed_queue);
111  }
112 
113  return clusters;
114 }

116 void ignoreDimRange(DimRange dr, PointViewPtr input, PointViewPtr keep,
117  PointViewPtr ignore)
118 {
119  PointRef point(*input, 0);
120  for (PointId i = 0; i < input->size(); ++i)
121  {
122  point.setPointId(i);
123  if (dr.valuePasses(point.getFieldAs<double>(dr.m_id)))
124  ignore->appendPoint(*input, i);
125  else
126  keep->appendPoint(*input, i);
127  }
128 }
