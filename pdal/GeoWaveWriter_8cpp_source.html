201  void GeoWaveWriter::write(const PointViewPtr view)
202  {
203 
204  using namespace Dimension;
205 
206  std::ostringstream os;
207 
208  BasicAccumuloOperations accumuloOperations;
209  try
210  {
211  accumuloOperations = java_new<BasicAccumuloOperations>(
212  java_new<String>(m_zookeeperUrl),
213  java_new<String>(m_instanceName),
214  java_new<String>(m_username),
215  java_new<String>(m_password),
216  java_new<String>(m_tableNamespace));
217  }
218  catch (AccumuloException& e)
219  {
220  log()->get(LogLevel::Error) << "There was a problem establishing a connector. " << e;
221  return;
222  }
223  catch (AccumuloSecurityException& e)
224  {
225  log()->get(LogLevel::Error) << "The credentials passed are invalid. " << e;
226  return;
227  }
228 
229  AccumuloDataStore accumuloDataStore = java_new<AccumuloDataStore>(
230  accumuloOperations);
231 
232  Index index = IndexType_JaceIndexType::createSpatialVectorIndex();
233 
234  AccumuloIndexWriter accumuloIndexWriter = java_new<AccumuloIndexWriter>(
235  index,
236  accumuloOperations,
237  accumuloDataStore);
238 
239  // treat all types as double
240  os << "location:Point:srid=4326";
241  for (auto di = m_dims.begin(); di != m_dims.end(); ++di)
242  os << "," << view->dimName(*di) << ":Double";
243 
244  SimpleFeatureType TYPE = DataUtilities::createType(
245  java_new<String>(m_featureTypeName),
246  java_new<String>(os.str()));
247 
248  String location = java_new<String>("location");
249 
250  WritableDataAdapter dataAdapter;
251  if (m_useFeatCollDataAdapter)
252  dataAdapter = java_new<FeatureCollectionDataAdapter>(
253  TYPE,
254  m_pointsPerEntry);
255  else
256  dataAdapter = java_new<FeatureDataAdapter>(TYPE);
257 
258 
259  GeometryFactory geometryFactory = JTSFactoryFinder::getGeometryFactory();
260  SimpleFeatureBuilder builder = java_new<SimpleFeatureBuilder>(TYPE);
261 
262  DefaultFeatureCollection featureCollection = java_new<DefaultFeatureCollection>(
263  UUID::randomUUID().toString(),
264  TYPE);
265 
266  for (PointId idx = 0; idx < view->size(); ++idx)
267  {
268  JDouble X = view->getFieldAs<double>(Id::X, idx);
269  JDouble Y = view->getFieldAs<double>(Id::Y, idx);
270 
271  Point point = geometryFactory.createPoint(
272  java_new<Coordinate>(
273  X,
274  Y));
275 
276  builder.set(location, point);
277 
278  for (auto di = m_dims.begin(); di != m_dims.end(); ++di)
279  if (view->hasDim(*di))
280  builder.set(java_new<String>(view->dimName(*di)), java_new<Double>(view->getFieldAs<double>(*di, idx)));
281 
282  SimpleFeature feature = builder.buildFeature(UUID::randomUUID().toString());
283 
284  if (m_useFeatCollDataAdapter)
285  featureCollection.add(feature);
286  else
287  accumuloIndexWriter.write(
288  dataAdapter,
289  feature);
290  }
291 
292  if (m_useFeatCollDataAdapter)
293  accumuloIndexWriter.write(
294  dataAdapter,
295  featureCollection);
296 
297  accumuloIndexWriter.close();
298  }
