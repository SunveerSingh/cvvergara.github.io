60 inline PGconn* pg_connect(std::string const& connection)
61 {
62  PGconn* conn;
63  if ( ! connection.size() )
64  throw pdal_error("unable to connect to database, no connection "
65  "string was given!");
66 
67  /* Validate the connection string and get verbose error (?) */
68 #ifdef PQconninfoParse
69  char *errstr;
70  PQconninfoOption *connOptions =
71  PQconninfoParse(connection.c_str(), &errstr);
72  if ( ! connOptions )
73  {
74  throw pdal_error(errstr);
75  }
76 #endif
77 
78  /* connect to database */
79  conn = PQconnectdb(connection.c_str());
80  if ( PQstatus(conn) != CONNECTION_OK )
81  {
82  throw pdal_error(PQerrorMessage(conn));
83  }
84 
85  return conn;
86 }

111 inline std::string pg_query_once(PGconn* session, std::string const& sql)
112 {
113  PGresult *result = PQexec(session, sql.c_str());
114  if ( (!result) ||
115  PQresultStatus(result) != PGRES_TUPLES_OK ||
116  PQntuples(result) == 0 )
117  {
118  PQclear(result);
119  return std::string();
120  }
121 
122  int len = PQgetlength(result, 0, 0);
123  char *value = PQgetvalue(result, 0, 0);
124  std::string out;
125  if (value)
126  out = std::string(value, len);
127  PQclear(result);
128  return out;
129 }
