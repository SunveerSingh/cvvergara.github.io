70 void FauxReader::initialize()
71 {
72  if (m_numReturns > 10)
73  throwError("Option 'number_of_returns' must be in the range [0,10].");
74  if (m_mode == Mode::Grid)
75  {
76  m_bounds.minx = ceil(m_bounds.minx);
77  m_bounds.maxx = ceil(m_bounds.maxx);
78  m_bounds.miny = ceil(m_bounds.miny);
79  m_bounds.maxy = ceil(m_bounds.maxy);
80  m_bounds.minz = ceil(m_bounds.minz);
81  m_bounds.maxz = ceil(m_bounds.maxz);
82  // Here delX/Y/Z represent the number of points in each direction.
83  m_count = 1;
84  if (m_bounds.maxx <= m_bounds.minx)
85  m_delX = 0;
86  else
87  {
88  m_delX = m_bounds.maxx - m_bounds.minx;
89  m_count *= m_delX;
90  }
91  if (m_bounds.maxy <= m_bounds.miny)
92  m_delY = 0;
93  else
94  {
95  m_delY = m_bounds.maxy - m_bounds.miny;
96  m_count *= m_delY;
97  }
98  if (m_bounds.maxz <= m_bounds.minz)
99  m_delZ = 0;
100  else
101  {
102  m_delZ = m_bounds.maxz - m_bounds.minz;
103  m_count *= m_delZ;
104  }
105  if (!m_delX && !m_delY && !m_delZ)
106  m_count = 0;
107  }
108  else
109  {
110  if (m_count > 1)
111  {
112  m_delX = (m_bounds.maxx - m_bounds.minx) / (m_count - 1);
113  m_delY = (m_bounds.maxy - m_bounds.miny) / (m_count - 1);
114  m_delZ = (m_bounds.maxz - m_bounds.minz) / (m_count - 1);
115  }
116  else
117  {
118  m_delX = 0;
119  m_delY = 0;
120  m_delZ = 0;
121  }
122  }
123 }

149 bool FauxReader::processOne(PointRef& point)
150 {
151  double x(0);
152  double y(0);
153  double z(0);
154 
155  if (m_index >= m_count)
156  return false;
157 
158  switch (m_mode)
159  {
160  case Mode::Random:
161  x = Utils::random(m_bounds.minx, m_bounds.maxx);
162  y = Utils::random(m_bounds.miny, m_bounds.maxy);
163  z = Utils::random(m_bounds.minz, m_bounds.maxz);
164  break;
165  case Mode::Constant:
166  x = m_bounds.minx;
167  y = m_bounds.miny;
168  z = m_bounds.minz;
169  break;
170  case Mode::Ramp:
171  x = m_bounds.minx + m_delX * m_index;
172  y = m_bounds.miny + m_delY * m_index;
173  z = m_bounds.minz + m_delZ * m_index;
174  break;
175  case Mode::Uniform:
176  x = Utils::uniform(m_bounds.minx, m_bounds.maxx, m_seed++);
177  y = Utils::uniform(m_bounds.miny, m_bounds.maxy, m_seed++);
178  z = Utils::uniform(m_bounds.minz, m_bounds.maxz, m_seed++);
179  break;
180  case Mode::Normal:
181  x = Utils::normal(m_mean_x, m_stdev_x, m_seed++);
182  y = Utils::normal(m_mean_y, m_stdev_y, m_seed++);
183  z = Utils::normal(m_mean_z, m_stdev_z, m_seed++);
184  break;
185  case Mode::Grid:
186  {
187  if (m_delX)
188  x = m_index % (point_count_t)m_delX;
189 
190  if (m_delY)
191  {
192  if (m_delX)
193  y = (m_index / (point_count_t)m_delX) % (point_count_t)m_delY;
194  else
195  y = m_index % (point_count_t)m_delY;
196  }
197 
198  if (m_delZ)
199  {
200  if (m_delX && m_delY)
201  z = m_index / (point_count_t)(m_delX * m_delY);
202  else if (m_delX)
203  z = m_index / (point_count_t)m_delX;
204  else if (m_delY)
205  z = m_index / (point_count_t)m_delY;
206  }
207  break;
208  }
209  }
210 
211  point.setField(Dimension::Id::X, x);
212  point.setField(Dimension::Id::Y, y);
213  point.setField(Dimension::Id::Z, z);
214  point.setField(Dimension::Id::OffsetTime, m_time++);
215  if (m_numReturns > 0)
216  {
217  point.setField(Dimension::Id::ReturnNumber, m_returnNum);
218  point.setField(Dimension::Id::NumberOfReturns, m_numReturns);
219  m_returnNum = (m_returnNum % m_numReturns) + 1;
220  }
221  m_index++;
222  return true;
223 }

226 point_count_t FauxReader::read(PointViewPtr view, point_count_t count)
227 {
228  for (PointId idx = 0; idx < count; ++idx)
229  {
230  PointRef point = view->point(idx);
231  if (!processOne(point))
232  break;
233  if (m_cb)

235  }
236  return count;
