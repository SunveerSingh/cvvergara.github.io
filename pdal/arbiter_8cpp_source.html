82  // Merge B into A, without overwriting any keys from A.
83  Json::Value merge(const Json::Value& a, const Json::Value& b)
84  {
85  Json::Value out(a);
86 
87  if (!b.isNull())
88  {
89  for (const auto& key : b.getMemberNames())
90  {
91  // If A doesn't have this key, then set it to B's value.
92  // If A has the key but it's an object, then recursively merge.
93  // Otherwise A already has a value here that we won't overwrite.
94  if (!out.isMember(key)) out[key] = b[key];
95  else if (out[key].isObject()) merge(out[key], b[key]);
96  }
97  }
98 
99  return out;
100  }

267 void Arbiter::copy(
268  const std::string src,
269  const std::string dst,
270  const bool verbose) const
271 {
272  if (src.empty()) throw ArbiterError("Cannot copy from empty source");
273  if (dst.empty()) throw ArbiterError("Cannot copy to empty destination");
274 
275  // Globify the source path if it's a directory. In this case, the source
276  // already ends with a slash.
277  const std::string srcToResolve(src + (util::isDirectory(src) ? "**" : ""));
278 
279  if (srcToResolve.back() != '*')
280  {
281  // The source is a single file.
282  copyFile(src, dst, verbose);
283  }
284  else
285  {
286  // We'll need this to mirror the directory structure in the output.
287  // All resolved paths will contain this common prefix, so we can
288  // determine any nested paths from recursive resolutions by stripping
289  // that common portion.
290  const Endpoint& srcEndpoint(getEndpoint(util::stripPostfixing(src)));
291  const std::string commonPrefix(srcEndpoint.prefixedRoot());
292 
293  const Endpoint dstEndpoint(getEndpoint(dst));
294 
295  if (srcEndpoint.prefixedRoot() == dstEndpoint.prefixedRoot())
296  {
297  throw ArbiterError("Cannot copy directory to itself");
298  }
299 
300  int i(0);
301  const auto paths(resolve(srcToResolve, verbose));
302 
303  for (const auto& path : paths)
304  {
305  const std::string subpath(path.substr(commonPrefix.size()));
306 
307  if (verbose)
308  {
309  std::cout <<
310  ++i << " / " << paths.size() << ": " <<
311  path << " -> " << dstEndpoint.fullPath(subpath) <<
312  std::endl;
313  }
314 
315  if (dstEndpoint.isLocal())
316  {
317  fs::mkdirp(util::getNonBasename(dstEndpoint.fullPath(subpath)));
318  }
319 
320  dstEndpoint.put(subpath, getBinary(path));
321  }
322  }

325 void Arbiter::copyFile(
326  const std::string file,
327  std::string dst,
328  const bool verbose) const
329 {
330  if (dst.empty()) throw ArbiterError("Cannot copy to empty destination");
331 
332  const Endpoint dstEndpoint(getEndpoint(dst));
333 
334  if (util::isDirectory(dst))
335  {
336  // If the destination is a directory, maintain the basename of the
337  // source file.
338  dst += util::getBasename(file);
339  }
340 
341  if (verbose) std::cout << file << " -> " << dst << std::endl;
342 
343  if (dstEndpoint.isLocal()) fs::mkdirp(util::getNonBasename(dst));
344 
345  if (getEndpoint(file).type() == dstEndpoint.type())
346  {
347  // If this copy is within the same driver domain, defer to the
348  // hopefully specialized copy method.
349  getDriver(file).copy(stripType(file), stripType(dst));
350  }
351  else
352  {
353  // Otherwise do a GET/PUT for the copy.
354  put(dst, getBinary(file));
355  }
