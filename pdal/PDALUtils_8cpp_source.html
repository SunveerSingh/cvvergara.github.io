57 void subnodesToJSON(const MetadataNode& parent, std::ostream& o, int level)
58 {
59  const std::string indent(level * 2, ' ');
60 
61  std::vector<std::string> names = parent.childNames();
62 
63  o << indent << "{" << endl;
64  for (auto ni = names.begin(); ni != names.end(); ++ni)
65  {
66  MetadataNodeList children = parent.children(*ni);
67 
68  MetadataNode& node = *children.begin();
69  if (node.kind() == MetadataType::Array)
70  {
71  o << indent << " \"" << node.name() << "\":" << std::endl;
72  arrayToJSON(children, o, level + 1);
73  }
74  else
75  toJSON(node, o, level + 1);
76  if (ni != names.rbegin().base() - 1)
77  o << ",";
78  o << std::endl;
79  }
80  o << indent << "}";
81 }

100 void arrayEltToJSON(const MetadataNode& m, std::ostream& o, int level)
101 {
102  std::string indent(level * 2, ' ');
103  std::string value = m.jsonValue();
104  bool children = m.hasChildren();
105 
106  // This is a case from XML. In JSON, you can't have two values.
107  if (!value.empty() && children)
108  {
109  o << value << "," << std::endl;
110  subnodesToJSON(m, o, level);
111  }
112  else if (!value.empty())
113  o << indent << value;
114  else
115  subnodesToJSON(m, o, level);
116  // There is the case where we have a name and no value to handle. What
117  // should be done?

120 void toJSON(const MetadataNode& m, std::ostream& o, int level)
121 {
122  std::string indent(level * 2, ' ');
123  std::string name = m.name();
124  std::string value = m.jsonValue();
125  bool children = m.hasChildren();
126 
127  if (name.empty())
128  name = "unnamed";
129 
130  // This is a case from XML. In JSON, you can't have two values.
131  if (!value.empty() && children)
132  {
133  o << indent << "\"" << name << "\": " << value << "," << std::endl;
134  o << indent << "\"" << name << "\": ";
135  subnodesToJSON(m, o, level);
136  }
137  else if (!value.empty())
138  o << indent << "\"" << name << "\": " << value;
139  else
140  {
141  o << indent << "\"" << name << "\":" << std::endl;
142  subnodesToJSON(m, o, level);
143  }
144  // There is the case where we have a name and no value to handle. What
145  // should be done?
146 }

360 double computeHausdorff(PointViewPtr srcView, PointViewPtr candView)
361 {
362  using namespace Dimension;
363 
364  KD3Index srcIndex(*srcView);
365  srcIndex.build();
366 
367  KD3Index candIndex(*candView);
368  candIndex.build();
369 
370  double maxDistSrcToCand = std::numeric_limits<double>::lowest();
371  double maxDistCandToSrc = std::numeric_limits<double>::lowest();
372 
373  for (PointId i = 0; i < srcView->size(); ++i)
374  {
375  std::vector<PointId> indices(1);
376  std::vector<double> sqr_dists(1);
377  PointRef srcPoint = srcView->point(i);
378  candIndex.knnSearch(srcPoint, 1, &indices, &sqr_dists);
379 
380  if (sqr_dists[0] > maxDistSrcToCand)
381  maxDistSrcToCand = sqr_dists[0];
382  }
383 
384  for (PointId i = 0; i < candView->size(); ++i)
385  {
386  std::vector<PointId> indices(1);
387  std::vector<double> sqr_dists(1);
388  PointRef candPoint = candView->point(i);
389  srcIndex.knnSearch(candPoint, 1, &indices, &sqr_dists);
390 
391  if (sqr_dists[0] > maxDistCandToSrc)
392  maxDistCandToSrc = sqr_dists[0];
393  }
394 
395  maxDistSrcToCand = std::sqrt(maxDistSrcToCand);
396  maxDistCandToSrc = std::sqrt(maxDistCandToSrc);
397 
398  return std::max(maxDistSrcToCand, maxDistCandToSrc);
399 }
