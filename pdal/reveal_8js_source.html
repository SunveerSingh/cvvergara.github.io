339  // n, page down
340  case 78: case 34: navigateNext(); triggered = true; break;
341  // h, left
342  case 72: case 37: navigateLeft(); triggered = true; break;
343  // l, right
344  case 76: case 39: navigateRight(); triggered = true; break;
345  // k, up
346  case 75: case 38: navigateUp(); triggered = true; break;

378  * enables support for swipe and pinch gestures.
379  */
380  function onDocumentTouchStart( event ) {
381  touch.startX = event.touches[0].clientX;
382  touch.startY = event.touches[0].clientY;
383  touch.startCount = event.touches.length;
384 
385  // If there's two touches we need to memorize the distance
386  // between those two points to detect pinching
387  if( event.touches.length === 2 ) {
388  touch.startSpan = distanceBetween( {
389  x: event.touches[1].clientX,
390  y: event.touches[1].clientY
391  }, {
392  x: touch.startX,
393  y: touch.startY
394  } );
395  }

398  /**
399  * Handler for the document level 'touchmove' event.
400  */
401  function onDocumentTouchMove( event ) {
402  // Each touch should only trigger one action
403  if( !touch.handled ) {
404  var currentX = event.touches[0].clientX;
405  var currentY = event.touches[0].clientY;
406 
407  // If the touch started off with two points and still has
408  // two active touches; test for the pinch gesture
409  if( event.touches.length === 2 && touch.startCount === 2 ) {
410 
411  // The current distance in pixels between the two touch points
412  var currentSpan = distanceBetween( {
413  x: event.touches[1].clientX,
414  y: event.touches[1].clientY
415  }, {
416  x: touch.startX,
417  y: touch.startY
418  } );
419 
420  // If the span is larger than the desire amount we've got
421  // ourselves a pinch
422  if( Math.abs( touch.startSpan - currentSpan ) > touch.threshold ) {
423  touch.handled = true;
424 
425  if( currentSpan < touch.startSpan ) {
426  activateOverview();
427  }
428  else {
429  deactivateOverview();
430  }
431  }
432 
433  }
434  // There was only one touch point, look for a swipe
435  else if( event.touches.length === 1 ) {
436  var deltaX = currentX - touch.startX,
437  deltaY = currentY - touch.startY;
438 
439  if( deltaX > touch.threshold && Math.abs( deltaX ) > Math.abs( deltaY ) ) {
440  touch.handled = true;
441  navigateLeft();
442  }
443  else if( deltaX < -touch.threshold && Math.abs( deltaX ) > Math.abs( deltaY ) ) {
444  touch.handled = true;
445  navigateRight();
446  }
447  else if( deltaY > touch.threshold ) {
448  touch.handled = true;
449  navigateUp();
450  }
451  else if( deltaY < -touch.threshold ) {
452  touch.handled = true;
453  navigateDown();
454  }
455  }
456 
457  event.preventDefault();
458  }
459  }

618  /**
619  * Updates one dimension of slides by showing the slide
620  * with the specified index.
621  *
622  * @param {String} selector A CSS selector that will fetch
623  * the group of slides we are working with
624  * @param {Number} index The index of the slide that should be
625  * shown
626  *
627  * @return {Number} The index of the slide that is now shown,
628  * might differ from the passed in index if it was out of
629  * bounds.
630  */
631  function updateSlides( selector, index ) {
632 
633  // Select all slides and convert the NodeList result to
634  // an array
635  var slides = Array.prototype.slice.call( document.querySelectorAll( selector ) ),
636  slidesLength = slides.length;
637 
638  if( slidesLength ) {
639 
640  // Should the index loop?
641  if( config.loop ) {
642  index %= slidesLength;
643 
644  if( index < 0 ) {
645  index = slidesLength + index;
646  }
647  }
648 
649  // Enforce max and minimum index bounds
650  index = Math.max( Math.min( index, slidesLength - 1 ), 0 );
651 
652  for( var i = 0; i < slidesLength; i++ ) {
653  var slide = slides[i];
654 
655  // Optimization; hide all slides that are three or more steps
656  // away from the present slide
657  if( overviewIsActive() === false ) {
658  // The distance loops so that it measures 1 between the first
659  // and last slides
660  var distance = Math.abs( ( index - i ) % ( slidesLength - 3 ) ) || 0;
661 
662  slide.style.display = distance > 3 ? 'none' : 'block';
663  }
664 
665  slides[i].classList.remove( 'past' );
666  slides[i].classList.remove( 'present' );
667  slides[i].classList.remove( 'future' );
668 
669  if( i < index ) {
670  // Any element previous to index is given the 'past' class
671  slides[i].classList.add( 'past' );
672  }
673  else if( i > index ) {
674  // Any element subsequent to index is given the 'future' class
675  slides[i].classList.add( 'future' );
676  }
677 
678  // If this element contains vertical slides
679  if( slide.querySelector( 'section' ) ) {
680  slides[i].classList.add( 'stack' );
681  }
682  }
683 
684  // Mark the current slide as present
685  slides[index].classList.add( 'present' );
686 
687  // If this slide has a state associated with it, add it
688  // onto the current state of the deck
689  var slideState = slides[index].getAttribute( 'data-state' );
690  if( slideState ) {
691  state = state.concat( slideState.split( ' ' ) );
692  }
693  }
694  else {
695  // Since there are no slides we can't be anywhere beyond the
696  // zeroth index
697  index = 0;
698  }
699 
700  return index;
701 
702  }

704  /**
705  * Updates the visual slides to represent the currently
706  * set indices.
707  */
708  function slide( h, v ) {
709  // Remember where we were at before
710  previousSlide = currentSlide;
711 
712  // Remember the state before this slide
713  var stateBefore = state.concat();
714 
715  // Reset the state array
716  state.length = 0;
717 
718  var indexhBefore = indexh,
719  indexvBefore = indexv;
720 
721  // Activate and transition to the new slide
722  indexh = updateSlides( HORIZONTAL_SLIDES_SELECTOR, h === undefined ? indexh : h );
723  indexv = updateSlides( VERTICAL_SLIDES_SELECTOR, v === undefined ? indexv : v );
724 
725  // Apply the new state
726  stateLoop: for( var i = 0, len = state.length; i < len; i++ ) {
727  // Check if this state existed on the previous slide. If it
728  // did, we will avoid adding it repeatedly.
729  for( var j = 0; j < stateBefore.length; j++ ) {
730  if( stateBefore[j] === state[i] ) {
731  stateBefore.splice( j, 1 );
732  continue stateLoop;
733  }
734  }
735 
736  document.documentElement.classList.add( state[i] );
737 
738  // Dispatch custom event matching the state's name
739  dispatchEvent( state[i] );
740  }
741 
742  // Clean up the remaints of the previous state
743  while( stateBefore.length ) {
744  document.documentElement.classList.remove( stateBefore.pop() );
745  }
746 
747  // Update progress if enabled
748  if( config.progress && dom.progress ) {
749  dom.progressbar.style.width = ( indexh / ( document.querySelectorAll( HORIZONTAL_SLIDES_SELECTOR ).length - 1 ) ) * window.innerWidth + 'px';
750  }
751 
752  // Close the overview if it's active
753  if( overviewIsActive() ) {
754  activateOverview();
755  }

838 
839  // Attempt to parse the hash as either an index or name
840  var bits = hash.slice( 2 ).split( '/' ),

844  // assume that this is a named link
