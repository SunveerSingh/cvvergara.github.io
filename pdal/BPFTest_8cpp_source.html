54 template<typename LeftIter, typename RightIter>
55 ::testing::AssertionResult CheckEqualCollections(
56  LeftIter left_begin, LeftIter left_end, RightIter right_begin)
57 {
58  bool equal(true);
59  std::string message;
60  size_t index(0);
61  while (left_begin != left_end)
62  {
63  if (*left_begin++ != *right_begin++)
64  {
65  equal = false;
66  message += "\n\tMismatch at index " + std::to_string(index);
67  }
68  ++index;
69  }
70  if (message.size())
71  message += "\n\t";
72  return equal ? ::testing::AssertionSuccess() :
73  ::testing::AssertionFailure() << message;
74 }

163

167  d = pts503[m_cnt - 503];
168  else
169  {
170  m_cnt++;
171  return true;
172  }
173 
174  float x = p.getFieldAs<float>(Dimension::Id::X);

180  EXPECT_FLOAT_EQ(z, d.z);
181  EXPECT_TRUE(m_cnt < 506) << "Count exceeded amount requested "
182  "in 'count' option.";
183 
184  m_cnt++;
185  return true;
186  }
187 
188  private:
189  size_t m_cnt;
190  };
