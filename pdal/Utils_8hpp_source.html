472  /**
473  Split a string into substrings based on a predicate. Characters
474  matching the predicate are discarded.
475 
476  \param s String to split.
477  \param p Unary predicate that returns true to indicate that a character
478  is a split location.
479  \return Substrings.
480  */
481  template<typename PREDICATE>
482  PDAL_DLL std::vector<std::string> split(const std::string& s, PREDICATE p)
483  {
484  std::vector<std::string> result;
485 
486  if (s.empty())
487  return result;
488 
489  auto it = s.cbegin();
490  auto const end = s.cend();
491  decltype(it) nextIt;
492  do
493  {
494  nextIt = std::find_if(it, end, p);
495  result.push_back(std::string(it, nextIt));
496 
497  // Avoid advancing the iterator past the end to avoid UB.
498  if (nextIt != end)
499  it = nextIt + 1;
500  } while (nextIt != end);
501 
502  return result;
503  }
504 
505  /**
506  Split a string into substrings. Characters matching the predicate are
507  discarded, as are empty strings otherwise produced by \ref split().
508 
509  \param s String to split.
510  \param p Predicate returns true if a char in a string is a split
511  location.
512  \return Vector of substrings.
513  */
514  template<typename PREDICATE>
515  PDAL_DLL std::vector<std::string> split2(const std::string& s, PREDICATE p)
516  {
517  std::vector<std::string> result;
518 
519  if (s.empty())
520  return result;
521 
522  auto it = s.cbegin();
523  auto const end = s.cend();
524  decltype(it) nextIt;
525  do
526  {
527  nextIt = std::find_if(it, end, p);
528  if (it != nextIt)
529  result.push_back(std::string(it, nextIt));
530 
531  // Avoid advancing the iterator past the end to avoid UB.
532  if (nextIt != end)
533  it = nextIt + 1;
534  } while (nextIt != end);
535 
536  return result;
537  }

672  /**

709  /**
710  Convert a numeric value from one type to another. Floating point
711  values are rounded to the nearest integer before a conversion is
712  attempted.
713 
714  \param in Value to convert.
715  \param out Converted value.
716  \return \c true if the conversion was successful, \c false if the
717  datatypes/input value don't allow conversion.
718  */
719  template<typename T_IN, typename T_OUT>
720  bool numericCast(T_IN in, T_OUT& out)
721  {
722  if (std::is_same<T_IN, T_OUT>::value)
723  {
724  out = static_cast<T_OUT>(in);
725  return true;
726  }
727  if (std::is_integral<T_OUT>::value)
728  in = static_cast<T_IN>(sround((double)in));
729  if ((std::is_same<T_OUT, double>::value) ||
730  (in <= static_cast<double>(std::numeric_limits<T_OUT>::max()) &&
731  in >= static_cast<double>(std::numeric_limits<T_OUT>::lowest())))
732  {
733  out = static_cast<T_OUT>(in);
734  return true;
735  }
736  return false;
