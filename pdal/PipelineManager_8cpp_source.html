66 void PipelineManager::readPipeline(const std::string& filename)
67 {
68  if (FileUtils::extension(filename) == ".json")
69  {
70  PipelineReaderJSON pipeReader(*this);
71  return pipeReader.readPipeline(filename);
72  }
73  else
74  {
75  Utils::closeFile(m_input);
76  m_input = Utils::openFile(filename);
77  if (!m_input)
78  throw pdal_error("Can't open file '" + filename + "' as pipeline "
79  "input.");
80  try
81  {
82  readPipeline(*m_input);
83  }
84  catch (const pdal_error& err)
85  {
86  throw pdal_error(filename + ": " + err.what());

140 void PipelineManager::validateStageOptions() const
141 {
142  // Make sure that the options specified are for relevant stages.
143  for (auto& si : m_stageOptions)
144  {
145  const std::string& stageName = si.first;
146  auto it = std::find_if(m_stages.begin(), m_stages.end(),
147  [stageName](Stage *s)
148  { return (s->getName() == stageName ||
149  "stage." + s->tag() == stageName); });
150 
151  // If the option stage name matches no created stage, then error.
152  if (it == m_stages.end())
153  {
154  std::ostringstream oss;
155  oss << "Argument references invalid/unused stage: '" <<
156  stageName << "'.";
157  throw pdal_error(oss.str());
158  }
159  }
160 }

184 point_count_t PipelineManager::execute()
185 {
186  prepare();
187 
188  Stage *s = getStage();
189  if (!s)
190  return 0;
191  m_viewSet = s->execute(m_table);
192  point_count_t cnt = 0;
193  for (auto pi = m_viewSet.begin(); pi != m_viewSet.end(); ++pi)
194  {
195  PointViewPtr view = *pi;
196  cnt += view->size();
197  }
198  return cnt;

365 void PipelineManager::setOptions(Stage& stage, const Options& addOps)
366 {
367  // First apply common options.
368  stage.setOptions(m_commonOptions);
369 
370  // Apply additional reader/writer options, making sure they replace any
371  // common options.
372  stage.removeOptions(addOps);
373  stage.addOptions(addOps);
374 
375  // Apply options provided on the command line, overriding others.
376  Options ops = stageOptions(stage);
377  stage.removeOptions(ops);
378  stage.addOptions(ops);
379 }

382 Options PipelineManager::stageOptions(Stage& stage)
383 {
384  Options opts;
385 
386  std::string tag = stage.tag();
387  if (tag.size())
388  {
389  tag = "stage." + tag;
390  auto oi = m_stageOptions.find(tag);
391  if (oi != m_stageOptions.end())
392  opts.add(oi->second);
393  }
394  // Tag-based options options override stagename-based options, so
395  // we call addConditional.
396  auto oi = m_stageOptions.find(stage.getName());
397  if (oi != m_stageOptions.end())
398  opts.addConditional(oi->second);
399  return opts;
400 }

424 void PipelineManager::replace(Stage *sOld, Stage *sNew)
425 {
426  Utils::remove(m_stages, sNew);
427  for (Stage * & s : m_stages)
428  {
429  if (s == sOld)
430  {
431  s = sNew;
432  // Copy inputs from the old stage to new one.
433  for (Stage *ss : sOld->getInputs())
434  sNew->setInput(*ss);
435  }
436  // Reset the inputs that refer to the replaced stage.
437  for (Stage * & ss : s->getInputs())
438  if (ss == sOld)
439  ss = sNew;
440  }
441 }
