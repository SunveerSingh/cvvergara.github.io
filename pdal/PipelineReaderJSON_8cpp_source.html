58 void PipelineReaderJSON::parsePipeline(Json::Value& tree)
59 {
60  TagMap tags;
61  std::vector<Stage*> inputs;
62 
63  Json::ArrayIndex last = tree.size() - 1;
64  for (Json::ArrayIndex i = 0; i < tree.size(); ++i)
65  {
66  Json::Value& node = tree[i];
67 
68  std::string filename;
69  std::string tag;
70  std::string type;
71  std::vector<Stage*> specifiedInputs;
72  Options options;
73 
74  // strings are assumed to be filenames
75  if (node.isString())
76  {
77  filename = node.asString();
78  }
79  else
80  {
81  type = extractType(node);
82  filename = extractFilename(node);
83  tag = extractTag(node, tags);
84  specifiedInputs = extractInputs(node, tags);
85  if (!specifiedInputs.empty())
86  inputs = specifiedInputs;
87  options = extractOptions(node);
88  }
89 
90  Stage *s = nullptr;
91 
92  // The type is inferred from a filename as a reader if it's not
93  // the last stage or if there's only one.
94  if ((type.empty() && (i == 0 || i != last)) ||
95  Utils::startsWith(type, "readers."))
96  {
97  StringList files = FileUtils::glob(filename);
98  if (files.empty())
99  files.push_back(filename);
100 
101  for (const std::string& path : files)
102  {
103  StageCreationOptions ops { path, type, nullptr, options, tag };
104  s = &m_manager.makeReader(ops);
105 
106  if (specifiedInputs.size())
107  throw pdal_error("JSON pipeline: Inputs not permitted for "
108  " reader: '" + path + "'.");
109  inputs.push_back(s);
110  }
111  }
112  else if (type.empty() || Utils::startsWith(type, "writers."))
113  {
114  StageCreationOptions ops { filename, type, nullptr, options, tag };
115  s = &m_manager.makeWriter(ops);
116  for (Stage *ts : inputs)
117  s->setInput(*ts);
118  inputs.clear();
119  }
120  else
121  {
122  if (filename.size())
123  options.add("filename", filename);
124  StageCreationOptions ops { "", type, nullptr, options, tag };
125  s = &m_manager.makeFilter(ops);
126  for (Stage *ts : inputs)
127  s->setInput(*ts);
128  inputs.clear();
129  inputs.push_back(s);
130  }
131  // 's' should be valid at this point. makeXXX will throw if the stage
132  // couldn't be constructed.
133  if (tag.size())
134  tags[tag] = s;
135  }
136 }

139 void PipelineReaderJSON::readPipeline(std::istream& input)
140 {
141  Json::Value root;
142  Json::Reader jsonReader;
143  if (!jsonReader.parse(input, root))
144  {
145  std::string err = "JSON pipeline: Unable to parse pipeline:\n";
146  err += jsonReader.getFormattedErrorMessages();
147  throw pdal_error(err);
148  }
149 
150  Json::Value& subtree = root["pipeline"];
151  if (!subtree)
152  throw pdal_error("JSON pipeline: Root element is not a Pipeline");
153  parsePipeline(subtree);
154 }

183 std::string PipelineReaderJSON::extractType(Json::Value& node)
184 {
185  std::string type;
186 
187  if (node.isMember("type"))
188  {
189  Json::Value& val = node["type"];
190  if (!val.isNull())
191  {
192  if (val.isString())
193  type = val.asString();
194  else
195  throw pdal_error("JSON pipeline: 'type' must be specified as "
196  "a string.");
197  }
198  node.removeMember("type");
199  if (node.isMember("type"))
200  throw pdal_error("JSON pipeline: found duplicate 'type' "
201  "entry in stage definition.");
202  }
203  return type;

231 std::string PipelineReaderJSON::extractTag(Json::Value& node, TagMap& tags)
232 {
233  std::string tag;
234 
235  if (node.isMember("tag"))
236  {
237  Json::Value& val = node["tag"];
238  if (!val.isNull())
239  {
240  if (val.isString())
241  {
242  tag = val.asString();
243  if (tags.find(tag) != tags.end())
244  throw pdal_error("JSON pipeline: duplicate tag '" +
245  tag + "'.");
246  }
247  else
248  throw pdal_error("JSON pipeline: tag must be "
249  "specified as a string.");
250  }
251  node.removeMember("tag");
252  if (node.isMember("tag"))
253  throw pdal_error("JSON pipeline: found duplicate 'tag' "
254  "entry in stage definition.");
255  std::string::size_type pos = 0;
256  if (!Stage::parseTagName(tag, pos) || pos != tag.size())
257  throw pdal_error("JSON pipeline: Invalid tag name '" + tag + "'. "
258  "Must start with letter. Remainder can be letters, "
259  "digits or underscores.");
260  }
261  return tag;
262 }

277 std::vector<Stage *> PipelineReaderJSON::extractInputs(Json::Value& node,
278  TagMap& tags)
279 {
280  std::vector<Stage *> inputs;
281  std::string filename;
282 
283  if (node.isMember("inputs"))
284  {
285  Json::Value& val = node["inputs"];
286  if (val.isString())
287  handleInputTag(val.asString(), tags, inputs);
288  else if (val.isArray())
289  {
290  for (const Json::Value& input : node["inputs"])
291  {
292  if (!input.isString())
293  throw pdal_error("JSON pipeline: 'inputs' tag must "
294  " be specified as a string or array of strings.");
295  handleInputTag(input.asString(), tags, inputs);
296  }
297  }
298  else
299  throw pdal_error("JSON pipeline: 'inputs' tag must "
300  " be specified as a string or array of strings.");
301  node.removeMember("inputs");
302  if (node.isMember("inputs"))
303  throw pdal_error("JSON pipeline: found duplicate 'inputs' "
304  "entry in stage definition.");
305  }
306  return inputs;
307 }

310 Options PipelineReaderJSON::extractOptions(Json::Value& node)
311 {
312  Options options;
313 
314  for (const std::string& name : node.getMemberNames())
315  {
316  if (name == "plugin")
317  {
318  PluginManager::loadPlugin(node[name].asString());
319 
320  // Don't actually put a "plugin" option on
321  // any stage
322  continue;
323  }
324 
325  if (node[name].isString())
326  options.add(name, node[name].asString());
327  else if (node[name].isInt())
328  options.add(name, node[name].asInt64());
329  else if (node[name].isUInt())
330  options.add(name, node[name].asUInt64());
331  else if (node[name].isDouble())
332  options.add(name, node[name].asDouble());
333  else if (node[name].isBool())
334  options.add(name, node[name].asBool());
335  else if (node[name].isNull())
336  options.add(name, "");
337  else if (node[name].isArray() || node[name].isObject())
338  {
339  Json::FastWriter w;
340  options.add(name, w.write(node[name]));
341  }
342  else
343  throw pdal_error("JSON pipeline: Value of stage option '" +
344  name + "' cannot be converted.");
345  }
346  node.clear();
347  return options;
348 }
