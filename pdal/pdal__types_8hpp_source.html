82  bool set(const std::string& sval)
83  {
84  if (sval == "auto")
85  m_auto = true;

87  {
88  size_t pos;
89  try
90  {
91  m_val = std::stod(sval, &pos);

93  catch (...)
94  {
95  // Set error condition.

99  {
100  m_val = 0;
101  return false;
102  }
103  }
104  return true;
105  }

173 inline std::istream& operator>>(std::istream& in, LogLevel& level)
174 {
175  std::string sval;
176  level = LogLevel::None;
177 
178  in >> sval;
179  try
180  {
181  int val = std::stoi(sval);
182  if (val >= 0 && val < (int)logNames.size())
183  level = (LogLevel)val;
184  }
185  catch (std::exception)
186  {
187  sval = Utils::tolower(sval);
188  for (size_t i = 0; i < logNames.size(); ++i)
189  if (logNames[i] == sval)
190  {
191  level = (LogLevel)i;
192  break;
193  }
194  }
195  if (level == LogLevel::None)
196  in.setstate(std::ios_base::failbit);
197  return in;
198 }

207  sval[0] = (char)toupper(sval[0]); // Make "Debug", "Error", etc.
208  }
