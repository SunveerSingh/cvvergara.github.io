81 void NitfFileWriter::write()
82 {
83  if (m_fileTitle.empty())
84  m_fileTitle = FileUtils::getFilename(m_filename);
85 
86  ::nitf::Record record(NITF_VER_21);
87  ::nitf::FileHeader header = record.getHeader();
88 
89  // This check would be better in an initialize() function, but we can't
90  // because this is a flex writer and we don't know the filename
91  // until the execute() step.
92  if (m_fileTitle.size() > header.getFileTitle().getLength())
93  throw error("Can't write file. FTITLE field (usually filename) "
94  "can't be longer than " +
95  Utils::toString(header.getFileTitle().getLength()) +
96  ". Use 'ftitle' option to set appropriately sized FTITLE.");
97 
98  header.getFileHeader().set("NITF");
99  header.getComplianceLevel().set(m_cLevel);
100  header.getSystemType().set(m_sType);
101  header.getOriginStationID().set(m_oStationId);
102  header.getFileTitle().set(m_fileTitle);
103  header.getClassification().set(m_fileClass);
104  header.getMessageCopyNum().set("00000");
105  header.getMessageNumCopies().set("00000");
106  header.getEncrypted().set("0");
107  header.getBackgroundColor().setRawData(const_cast<char*>("000"), 3);
108  header.getOriginatorName().set(m_origName);
109  header.getOriginatorPhone().set(m_origPhone);
110  header.getSecurityGroup().getClassificationSystem().set(
111  m_securityClassificationSystem);
112  header.getSecurityGroup().getControlAndHandling().set(
113  m_securityControlAndHandling);
114  header.getSecurityGroup().getClassificationText().set(m_sic);
115 
116  ::nitf::DESegment des = record.newDataExtensionSegment();
117 
118  des.getSubheader().getFilePartType().set("DE");
119  des.getSubheader().getTypeID().set("LIDARA DES");
120  des.getSubheader().getVersion().set("01");
121  // We currently use the file classification for the subheader, since the
122  // only thing we're writing is the data extension.
123  des.getSubheader().getSecurityClass().set(m_fileClass);
124  ::nitf::FileSecurity security = record.getHeader().getSecurityGroup();
125  des.getSubheader().setSecurityGroup(security.clone());
126 
127  ::nitf::TRE usrHdr("LIDARA DES", "raw_data");
128  usrHdr.setField("raw_data", "not");
129  ::nitf::Field fld = usrHdr.getField("raw_data");
130  fld.setType(::nitf::Field::BINARY);
131 
132  des.getSubheader().setSubheaderFields(usrHdr);
133 
134  ::nitf::ImageSegment image = record.newImageSegment();
135  ::nitf::ImageSubheader subheader = image.getSubheader();
136 
137  double corners[4][2];
138  corners[0][0] = m_bounds.maxy;
139  corners[0][1] = m_bounds.minx;
140  corners[1][0] = m_bounds.maxy;
141  corners[1][1] = m_bounds.maxx;
142  corners[2][0] = m_bounds.miny;
143  corners[2][1] = m_bounds.maxx;
144  corners[3][0] = m_bounds.miny;
145  corners[3][1] = m_bounds.minx;
146  subheader.setCornersFromLatLons(NRT_CORNERS_DECIMAL, corners);
147  subheader.getImageSecurityClass().set(m_imgSecurityClass);
148  subheader.setSecurityGroup(security.clone());
149  if (m_imgDate.size())
150  subheader.getImageDateAndTime().set(m_imgDate);
151 
152  ::nitf::BandInfo info;
153  ::nitf::LookupTable lt(0,0);
154  info.init(" ", /* The band representation, Nth band */
155  " ", /* The band subcategory */
156  "N", /* The band filter condition */
157  " ", /* The band standard image filter code */
158  0, /* The number of look-up tables */
159  0, /* The number of entries/LUT */
160  lt); /* The look-up tables */
161 
162  std::vector<::nitf::BandInfo> bands;
163  bands.push_back(info);
164  subheader.setPixelInformation(
165  "INT", /* Pixel value type */
166  8, /* Number of bits/pixel */
167  8, /* Actual number of bits/pixel */
168  "R", /* Pixel justification */
169  "NODISPLY", /* Image representation */
170  "VIS", /* Image category */
171  1, /* Number of bands */
172  bands);
173 
174  subheader.setBlocking(
175  8, /*!< The number of rows */
176  8, /*!< The number of columns */
177  8, /*!< The number of rows/block */
178  8, /*!< The number of columns/block */
179  "B"); /*!< Image mode */
180 
181  //Image Header fields to set
182  subheader.getImageId().set("None");
183  subheader.getImageTitle().set(m_imgIdentifier2);
184 
185  //AIMIDB
186  ::nitf::TRE aimidbTre("AIMIDB");
187 
188  //LIDAR defaults
189  if (m_imgDate.size())
190  aimidbTre.setField("ACQUISITION_DATE", m_imgDate);
191  aimidbTre.setField("MISSION_NO", "UNKN");
192  aimidbTre.setField("MISSION_IDENTIFICATION", "NOT AVAIL.");
193  aimidbTre.setField("FLIGHT_NO", "00");
194  aimidbTre.setField("CURRENT_SEGMENT", "AA");
195  aimidbTre.setField("START_TILE_COLUMN", "001");
196  aimidbTre.setField("START_TILE_ROW", "00001");
197  aimidbTre.setField("END_SEGMENT", "00");
198  aimidbTre.setField("END_TILE_COLUMN", "001");
199  aimidbTre.setField("END_TILE_ROW", "00001");
200 
201  for (auto& s : m_aimidb)
202  {
203  StringList v = Utils::split2(s, ':');
204  if (v.size() != 2)
205  throw error("Invalid name/value for AIMIDB '" + s +
206  "'. Format: <name>:<value>.");
207  Utils::trim(v[0]);
208  Utils::trim(v[1]);
209  aimidbTre.setField(v[0], v[1]);
210  }
211  subheader.getExtendedSection().appendTRE(aimidbTre);
212 
213  //if IDATIM is empty set it equal to AIMIDB.ACQUISITION_DATE
214  if (!m_imgDate.size())
215  {
216  m_imgDate = aimidbTre.getField("ACQUISITION_DATE").toString();
217  if (m_imgDate.size())
218  subheader.getImageDateAndTime().set(m_imgDate);
219  }
220 
221  //ACFTB
222  ::nitf::TRE acftbTre("ACFTB");
223 
224  //LIDAR defaults
225  acftbTre.setField("AC_MSN_ID", "NOT AVAILABLE");
226  acftbTre.setField("SCENE_SOURCE", " ");
227  if (m_imgDate.size()>7)
228  acftbTre.setField("PDATE", m_imgDate.substr(0,8));
229  acftbTre.setField("MPLAN", "999");
230  acftbTre.setField("LOC_ACCY", "000.00");
231  acftbTre.setField("ROW_SPACING", "0000000");
232  acftbTre.setField("ROW_SPACING_UNITS", "u");
233  acftbTre.setField("COL_SPACING", "0000000");
234  acftbTre.setField("COL_SPACING_UNITS", "u");
235  acftbTre.setField("FOCAL_LENGTH", "999.99");
236 
237  for (auto& s : m_acftb)
238  {
239  StringList v = Utils::split2(s, ':');
240  if (v.size() != 2)
241  throw error("Invalid name/value for ACFTB '" + s +
242  "'. Format: <name>:<value>.");
243  Utils::trim(v[0]);
244  Utils::trim(v[1]);
245  acftbTre.setField(v[0], v[1]);
246  }
247  subheader.getExtendedSection().appendTRE(acftbTre);
248 
249  ::nitf::Writer writer;
250  ::nitf::IOHandle output_io(m_filename, NITF_ACCESS_WRITEONLY,
251  NITF_CREATE);
252  writer.prepare(output_io, record);
253 
254  ::nitf::SegmentWriter sWriter = writer.newDEWriter(0);
255  sWriter.attachSource(*m_source);
256 
257  // 64 char string
258  std::string zeros(64, '0');
259 
260  ::nitf::MemorySource band(
261  const_cast<char*>(zeros.c_str()),
262  zeros.size() /* memory size */,
263  0 /* starting offset */,
264  1 /* bytes per pixel */,
265  0 /*skip*/);
266  ::nitf::ImageSource iSource;
267  iSource.addBand(band);
268 
269  ::nitf::ImageWriter iWriter = writer.newImageWriter(0);
270  iWriter.attachSource(iSource);
271 
272  writer.write();
273  output_io.close();
274 }

277 void NitfFileWriter::setBounds(const BOX3D& bounds)
278 {
279  m_bounds = bounds;
280 
281  //NITF decimal degree values for corner coordinates only has a
282  // precision of 3 after the decimal. This may cause an invalid
283  // polygon due to rounding errors with a small tile. Therefore
284  // instead of rounding min values will use the floor value and
285  // max values will use the ceiling values.
286  m_bounds.minx = (floor(m_bounds.minx * 1000)) / 1000.0;
287  m_bounds.miny = (floor(m_bounds.miny * 1000)) / 1000.0;
288  m_bounds.maxx = (ceil(m_bounds.maxx * 1000)) / 1000.0;
289  m_bounds.maxy = (ceil(m_bounds.maxy * 1000)) / 1000.0;
290 }
