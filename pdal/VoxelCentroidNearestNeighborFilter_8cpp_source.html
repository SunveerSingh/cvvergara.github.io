68 PointViewSet VoxelCentroidNearestNeighborFilter::run(PointViewPtr view)
69 {
70  BOX3D bounds;
71  view->calculateBounds(bounds);
72 
73  KD3Index kdi(*view);
74  kdi.build();
75 
76  // Make an initial pass through the input PointView to index PointIds by
77  // row, column, and depth.
78  std::map<std::tuple<size_t, size_t, size_t>, std::vector<PointId>>
79  populated_voxel_ids;
80  for (PointId id = 0; id < view->size(); ++id)
81  {
82  double y = view->getFieldAs<double>(Dimension::Id::Y, id);
83  double x = view->getFieldAs<double>(Dimension::Id::X, id);
84  double z = view->getFieldAs<double>(Dimension::Id::Z, id);
85  size_t r = static_cast<size_t>(floor(y - bounds.miny) / m_cell);
86  size_t c = static_cast<size_t>(floor(x - bounds.minx) / m_cell);
87  size_t d = static_cast<size_t>(floor(z - bounds.minz) / m_cell);
88  populated_voxel_ids[std::make_tuple(r, c, d)].push_back(id);
89  }
90 
91  // Make a second pass through the populated voxels to compute the voxel
92  // centroid and to find its nearest neighbor.
93  PointViewPtr output = view->makeNew();
94  for (auto const& t : populated_voxel_ids)
95  {
96  Eigen::Vector3f centroid = eigen::computeCentroid(*view, t.second);
97  std::vector<PointId> neighbors =
98  kdi.neighbors(centroid[0], centroid[1], centroid[2], 1);
99  output->appendPoint(*view, neighbors[0]);
100  }
101 
102  PointViewSet viewSet;
103  viewSet.insert(output);
104  return viewSet;
105 }
