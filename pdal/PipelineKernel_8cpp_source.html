95 int PipelineKernel::execute()
96 {
97  if (!Utils::fileExists(m_inputFile))
98  throw pdal_error("file not found: " + m_inputFile);
99  if (m_progressFile.size())
100  {
101  m_progressFd = Utils::openProgress(m_progressFile);
102  m_manager.setProgressFd(m_progressFd);
103  }
104 
105  m_manager.readPipeline(m_inputFile);
106 
107  if (m_validate)
108  {
109  // Validate the options of the pipeline we were
110  // given, and once we succeed, we're done
111  m_manager.prepare();
112  Utils::closeProgress(m_progressFd);
113  return 0;
114  }
115 
116  if (m_stream)
117  {
118  FixedPointTable table(10000);
119  m_manager.executeStream(table);
120  }
121  else
122  m_manager.execute();
123 
124  if (m_metadataFile.size())
125  {
126  std::ostream *out = Utils::createFile(m_metadataFile, false);
127  if (!out)
128  throw pdal_error("Can't open file '" + m_metadataFile +
129  "' for metadata output.");
130  Utils::toJSON(m_manager.getMetadata(), *out);
131  Utils::closeFile(out);
132  }
133  if (m_pipelineFile.size())
134  PipelineWriter::writePipeline(m_manager.getStage(), m_pipelineFile);
135 
136  if (m_PointCloudSchemaOutput.size() > 0)
137  {
138 #ifdef PDAL_HAVE_LIBXML2
139  XMLSchema schema(m_manager.pointTable().layout());
140 
141  std::ostream *out = Utils::createFile(m_PointCloudSchemaOutput);
142  std::string xml(schema.xml());
143  out->write(xml.c_str(), xml.size());
144  Utils::closeFile(out);
145 #else
146  std::cerr << "libxml2 support not available, no schema is produced" <<
147  std::endl;
148 #endif
149 
150  }
151  Utils::closeProgress(m_progressFd);
152  return 0;
153 }
