54 TransformationMatrix transformationMatrixFromString(const std::string& s)
55 {
56  std::istringstream iss(s);
57  TransformationMatrix matrix{{ 0 }};
58  double entry;
59  TransformationMatrix::size_type i = 0;
60  while (iss >> entry)
61  {
62  if (i + 1 > matrix.size())
63  {
64  std::stringstream msg;
65  msg << "Too many entries in transformation matrix, should be "
66  << matrix.size();
67  throw pdal_error("filters.transformation: " + msg.str());
68  }
69  matrix[i++] = entry;
70  }
71 
72  if (i != matrix.size())
73  {
74  std::stringstream msg;
75  msg << "Too few entries in transformation matrix: "
76  << i
77  << " (should be "
78  << matrix.size()
79  << ")";
80 
81  throw pdal_error("filters.transformation: " + msg.str());
82  }
83 
84  return matrix;
85 }

100 bool TransformationFilter::processOne(PointRef& point)
101 {
102  double x = point.getFieldAs<double>(Dimension::Id::X);
103  double y = point.getFieldAs<double>(Dimension::Id::Y);
104  double z = point.getFieldAs<double>(Dimension::Id::Z);
105 
106  point.setField(Dimension::Id::X,
107  x * m_matrix[0] + y * m_matrix[1] + z * m_matrix[2] + m_matrix[3]);
108 
109  point.setField(Dimension::Id::Y,
110  x * m_matrix[4] + y * m_matrix[5] + z * m_matrix[6] + m_matrix[7]);
111 
112  point.setField(Dimension::Id::Z,
113  x * m_matrix[8] + y * m_matrix[9] + z * m_matrix[10] + m_matrix[11]);
114  return true;
115 }
