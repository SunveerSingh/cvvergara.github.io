64 void SortFilter::filter(PointView& view)
65 {
66  auto cmp = [this](const PointIdxRef& p1, const PointIdxRef& p2)
67  {
68  bool result = p1.compare(m_dim, p2);
69  return (m_order == SortOrder::ASC) ? result : !result;
70  };
71 
72  std::stable_sort(view.begin(), view.end(), cmp);
73 }
