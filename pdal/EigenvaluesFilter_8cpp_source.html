75 void EigenvaluesFilter::filter(PointView& view)
76 {
77  using namespace Eigen;
78 
79  KD3Index kdi(view);
80  kdi.build();
81 
82  for (PointId i = 0; i < view.size(); ++i)
83  {
84  // find the k-nearest neighbors
85  auto ids = kdi.neighbors(i, m_knn);
86 
87  // compute covariance of the neighborhood
88  auto B = eigen::computeCovariance(view, ids);
89 
90  // perform the eigen decomposition
91  SelfAdjointEigenSolver<Matrix3f> solver(B);
92  if (solver.info() != Success)
93  throwError("Cannot perform eigen decomposition.");
94  auto ev = solver.eigenvalues();
95 
96  view.setField(m_e0, i, ev[0]);
97  view.setField(m_e1, i, ev[1]);
98  view.setField(m_e2, i, ev[2]);
99  }
100 }
