156 Options Options::fromFile(const std::string& filename, bool throwOnOpenError)
157 {

179 Options Options::fromJsonFile(const std::string& filename, const std::string& s)
180 {
181  Options options;
182 
183  Json::Reader reader;
184  Json::Value node;
185 
186  if (!reader.parse(s, node))
187  throw pdal_error("Unable to parse options file '" + filename +
188  "' as JSON: \n" + reader.getFormattedErrorMessages());
189 
190  for (const std::string& name : node.getMemberNames())
191  {
192  if (node[name].isString())
193  options.add(name, node[name].asString());
194  else if (node[name].isInt())
195  options.add(name, node[name].asInt64());
196  else if (node[name].isUInt())
197  options.add(name, node[name].asUInt64());
198  else if (node[name].isDouble())
199  options.add(name, node[name].asDouble());
200  else if (node[name].isBool())
201  options.add(name, node[name].asBool());
202  else if (node[name].isNull())
203  options.add(name, "");
204  else if (node[name].isArray() || node[name].isObject())
205  {
206  Json::FastWriter w;
207  options.add(name, w.write(node[name]));
208  }
209  else
210  throw pdal_error("Value of stage option '" +
211  name + "' in options file '" + filename +
212  "' cannot be converted.");
213  }
214  return options;
215 }

218 Options Options::fromCmdlineFile(const std::string& filename,
219  const std::string& s)
220 {
221  Options options;
222  StringList args = Utils::simpleWordexp(s);
223 
224  for (size_t i = 0; i < args.size(); ++i)
225  {
226  std::string option = args[i];
227  std::string value;
228  if (i + 1 < args.size())
229  value = args[i + 1];
230 
231  if (option.size() < 3)
232  throw pdal_error("Invalid option '" + option + "' in option "
233  "file '" + filename + "'.");
234  if (option[0] != '-' || option[1] != '-')
235  throw pdal_error("Option '" + option + "' missing leading \"--\" "
236  "in option file '" + filename + "'.");
237 
238  std::string::size_type pos = 2;
239  std::string::size_type count = Option::parse(option, pos);
240  std::string optionName = option.substr(2, count);
241  pos += count;
242  if (option[pos++] == '=')
243  value = option.substr(pos);
244  else
245  i++;
246  if (value.empty())
247  throw pdal_error("No value found for option '" + option + "' in "
248  "option file '" + filename + "'.");
249  Option o(optionName, value);
250  options.add(o);
251  }
252  return options;
253 }
