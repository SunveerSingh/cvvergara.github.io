45 TEST(SupportTest, test_paths)
46 {
47  // does the data path work?
48  string data_file = Support::datapath("las/simple.las");
49  EXPECT_TRUE(FileUtils::fileExists(data_file));
50 
51  // make sure we have read access
52  istream* istr = FileUtils::openFile(data_file);
53  string yow;
54  *istr >> yow;
55  FileUtils::closeFile(istr);
56 
57  // does the temp path work?
58  string temp_file_ok = Support::temppath("README.txt");
59  EXPECT_TRUE(FileUtils::fileExists(temp_file_ok));
60  string temp_file = Support::temppath("my_temp_file.dat");
61  EXPECT_TRUE(!FileUtils::fileExists(temp_file));
62 
63  // make sure we have write access to the temp dir
64  ostream* ostr = FileUtils::createFile(temp_file);
65  *ostr << "yow";
66  FileUtils::closeFile(ostr);
67  EXPECT_TRUE(FileUtils::fileExists(temp_file));
68  EXPECT_TRUE(FileUtils::deleteFile(temp_file));
69  EXPECT_TRUE(!FileUtils::fileExists(temp_file));
70 
71  // does binpath (and exename) work?
72  string this_bin = Support::exename("pdal");
73 #ifdef _WIN32
74  EXPECT_EQ(this_bin, "pdal.exe");
75 #else
76  EXPECT_EQ(this_bin, "pdal");
77 #endif
78  this_bin = Support::binpath(this_bin);

125 TEST(SupportTest, test_diff_file_ignorable)

128 
129  // no ignorable region
130  {
131  diffs = Support::diff_files(Support::datapath("misc/data4a.dat"), Support::datapath("misc/data4b.dat"));
132  EXPECT_TRUE(diffs == 6U);
133  EXPECT_TRUE(Support::compare_files(Support::datapath("misc/data4a.dat"), Support::datapath("misc/data4b.dat")) == false);
134  }
135 
136  // treat whole file as ignorable
137  {
138  uint32_t start[1] = {0};
139  uint32_t len[1] = {100};
140  diffs = Support::diff_files(Support::datapath("misc/data4a.dat"), Support::datapath("misc/data4b.dat"), start, len, 1);
141  EXPECT_EQ(diffs, 0U);
142  }
143 
144  // just ignore the first region
145  {
146  uint32_t start[1] = {3};
147  uint32_t len[1] = {4};
148  diffs = Support::diff_files(Support::datapath("misc/data4a.dat"), Support::datapath("misc/data4b.dat"), start, len, 1);
149  EXPECT_EQ(diffs, 2U);
150  }
151 
152  // ignore the first and second regions
153  {
154  uint32_t start[2] = {3, 23};
155  uint32_t len[2] = {4, 2};
156  diffs = Support::diff_files(Support::datapath("misc/data4a.dat"), Support::datapath("misc/data4b.dat"), start, len, 2);
157  EXPECT_EQ(diffs, 0U);
158  }
159 
160  // ignore first and part of second region
