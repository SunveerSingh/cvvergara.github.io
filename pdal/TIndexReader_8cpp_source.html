52 TIndexReader::FieldIndexes TIndexReader::getFields()
53 {
54  FieldIndexes indexes;
55 
56  void *fDefn = OGR_L_GetLayerDefn(m_layer);
57 
58  indexes.m_filename = OGR_FD_GetFieldIndex(fDefn,
59  m_tileIndexColumnName.c_str());
60  if (indexes.m_filename < 0)
61  throwError("Unable to find field '" + m_tileIndexColumnName +
62  "' in file '" + m_filename + "'.");
63  indexes.m_srs = OGR_FD_GetFieldIndex(fDefn, m_srsColumnName.c_str());
64  if (indexes.m_srs < 0)
65  throwError("Unable to find field '" + m_srsColumnName + "' in file '" +
66  m_filename + "'.");
67 
68  indexes.m_ctime = OGR_FD_GetFieldIndex(fDefn, "created");
69  indexes.m_mtime = OGR_FD_GetFieldIndex(fDefn, "modified");
70 
71  return indexes;
72 }

75 std::vector<TIndexReader::FileInfo> TIndexReader::getFiles()
76 {
77  std::vector<TIndexReader::FileInfo> output;
78 
79  OGR_L_ResetReading(m_layer);
80  FieldIndexes indexes = getFields();
81 
82  while (true)
83  {
84  OGRFeatureH feature = OGR_L_GetNextFeature(m_layer);
85  if (!feature)
86  break;
87 
88  FileInfo fileInfo;
89  fileInfo.m_filename =
90  OGR_F_GetFieldAsString(feature, indexes.m_filename);
91  fileInfo.m_srs =
92  OGR_F_GetFieldAsString(feature, indexes.m_srs);
93  output.push_back(fileInfo);
94 
95  OGR_F_Destroy(feature);
96  }
97 
98  return output;
99 }

136 void TIndexReader::initialize()
137 {
138  if (!m_bounds.empty())
139  m_wkt = m_bounds.toWKT();
140  m_out_ref.reset(new gdal::SpatialRef());
141 
142  log()->get(LogLevel::Debug) << "Opening file " << m_filename <<
143  std::endl;
144 
145  gdal::registerDrivers();
146  m_dataset = OGROpen(m_filename.c_str(), FALSE, NULL);
147  if (!m_dataset)
148  throwError("Unable to datasource '" + m_filename + "'");
149 
150  OGRGeometryH geometry(0);
151  if (m_sql.size())
152  {
153  m_layer = OGR_DS_ExecuteSQL(m_dataset, m_sql.c_str(), geometry,
154  m_dialect.c_str());
155  }
156  else
157  {
158  m_layer = OGR_DS_GetLayerByName(m_dataset, m_layerName.c_str());
159  }
160  if (!m_layer)
161  throwError("Unable to open layer '" + m_layerName +
162  "' from OGR datasource '" + m_filename + "'");
163 
164  m_out_ref->setFromLayer(m_layer);
165 
166  // Override the SRS if the user set one, otherwise, take it
167  // from the layer
168  if (m_tgtSrsString.size())
169  m_out_ref.reset(new gdal::SpatialRef(m_tgtSrsString));
170  else
171  m_out_ref.reset(new gdal::SpatialRef(m_out_ref->wkt()));
172 
173  setSpatialReference(SpatialReference(m_out_ref->wkt()));
174 
175  std::unique_ptr<gdal::Geometry> wkt_g;
176 
177  // If the user set either explicit 'polygon' or 'boundary' options
178  // we will filter by that geometry. The user can set a 'filter_srs'
179  // option to override the SRS of the input geometry and we will
180  // reproject to the output projection as needed.
181  if (m_wkt.size())
182  {
183  // Reproject the given wkt to the output SRS so
184  // filtering/cropping works
185  gdal::SpatialRef assign(m_filterSRS);
186  gdal::Geometry before(m_wkt, assign);
187  before.transform(*m_out_ref);
188 
189  wkt_g.reset (new gdal::Geometry(before.wkt(), *m_out_ref));
190 
191  geometry = wkt_g->get();
192  m_wkt = wkt_g->wkt();
193  OGR_L_SetSpatialFilter(m_layer, geometry);
194  }
195 
196  if (m_attributeFilter.size())
197  {
198  OGRErr err = OGR_L_SetAttributeFilter(m_layer,
199  m_attributeFilter.c_str());
200  if (err != OGRERR_NONE)
201  throwError("Unable to set attribute filter '" + m_attributeFilter +
202  "' for OGR datasource '" + m_filename + "'");
203  }
204 
205  Options cropOptions;
206  if (m_wkt.size())
207  cropOptions.add("polygon", m_wkt);
208 
209  for (auto f : getFiles())
210  {
211  log()->get(LogLevel::Debug) << "Adding file " << f.m_filename <<
212  " to merge filter" << std::endl;
213 
214  std::string driver = m_factory.inferReaderDriver(f.m_filename);
215  Stage *reader = m_factory.createStage(driver);
216  if (!reader)
217  throwError("Unable to create reader for file '" + f.m_filename +
218  "'.");
219  Options readerOptions;
220  readerOptions.add("filename", f.m_filename);
221  reader->setOptions(readerOptions);
222  Stage *premerge = reader;
223 
224  if (m_tgtSrsString != f.m_srs &&
225  (m_tgtSrsString.size() && f.m_srs.size()))
226  {
227  Stage *repro = m_factory.createStage("filters.reprojection");
228  repro->setInput(*reader);
229  Options reproOptions;
230  reproOptions.add("out_srs", m_tgtSrsString);
231  reproOptions.add("in_srs", f.m_srs);
232  log()->get(LogLevel::Debug2) << "Repro = "
233  << m_tgtSrsString << "/"
234  << f.m_srs << "!\n";
235  repro->setOptions(reproOptions);
236  premerge = repro;
237  }
238 
239  // WKT is set even if we're using a bounding box for filtering, so
240  // can be used as a test here.
241  if (!m_wkt.empty())
242  {
243  Stage *crop = m_factory.createStage("filters.crop");
244  crop->setOptions(cropOptions);
245  crop->setInput(*premerge);
246  log()->get(LogLevel::Debug3) << "Cropping data with wkt '"
247  << m_wkt << "'" << std::endl;
248  premerge = crop;
249  }
250 
251  m_merge.setInput(*premerge);
252  }
253 
254  if (m_sql.size())
255  {
256  // We were created with OGR_DS_ExecuteSQL which needs to have
257  // its layer explicitly released
258  OGR_DS_ReleaseResultSet(m_dataset, m_layer);
259  }
260  else
261  {
262  OGR_DS_Destroy(m_dataset);
263  }
264  m_layer = 0;
265  m_dataset = 0;
266 }
