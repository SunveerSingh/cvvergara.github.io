54 void Scaling::setAutoXForm(const PointViewPtr view)
55 {
56  double xmin = (std::numeric_limits<double>::max)();
57  double xmax = (std::numeric_limits<double>::lowest)();
58  bool xmod = m_xXform.m_offset.m_auto || m_xXform.m_scale.m_auto;
59 
60  double ymin = (std::numeric_limits<double>::max)();
61  double ymax = (std::numeric_limits<double>::lowest)();
62  bool ymod = m_yXform.m_offset.m_auto || m_yXform.m_scale.m_auto;
63 
64  double zmin = (std::numeric_limits<double>::max)();
65  double zmax = (std::numeric_limits<double>::lowest)();

67

74  {

92  }
93  }
94

108  zmax -= zmin;
109  }
