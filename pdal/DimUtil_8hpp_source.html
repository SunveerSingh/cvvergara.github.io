143 /// Get the type corresponding to a type name.

173 /// Extract a dimension name of a string. Dimension names start with an alpha
174 /// and continue with numbers or underscores.
175 /// \param s String from which to extract dimension name.
176 /// \param p Position at which to start extracting.
177 /// \return Number of characters in the extracted name.
178 inline std::size_t extractName(const std::string& s, std::string::size_type p)
179 {
180  if (!std::isalpha(s[p++]))
181  return 0;
182  auto isvalid = [](int c)
183  {
184  return std::isalpha(c) || std::isdigit(c) || c == '_';
185  };
186  return Utils::extract(s, p, isvalid) + 1;
187 }
