60 PointViewSet MovingLeastSquaresFilter::run(PointViewPtr input)
61 {
62  PointViewPtr output = input->makeNew();
63  PointViewSet viewSet;
64  viewSet.insert(output);
65 
66  bool logOutput = log()->getLevel() > LogLevel::Debug1;
67  if (logOutput)
68  log()->floatPrecision(8);
69 
70  log()->get(LogLevel::Debug2) << "Process MovingLeastSquaresFilter..." << std::endl;
71 
72  BOX3D buffer_bounds;
73  input->calculateBounds(buffer_bounds);
74 
75  // convert PointView to PointNormal
76  typedef pcl::PointCloud<pcl::PointXYZ> Cloud;
77  Cloud::Ptr cloud(new Cloud);
78  pclsupport::PDALtoPCD(input, *cloud, buffer_bounds);
79 
80  pclsupport::setLogLevel(log()->getLevel());
81 
82  // initial setup
83  pcl::MovingLeastSquares<pcl::PointXYZ, pcl::PointXYZ> mls;
84  mls.setInputCloud(cloud);
85  mls.setSearchRadius(1);
86  mls.setPolynomialFit(true);
87  mls.setPolynomialOrder(2);
88  mls.setUpsamplingMethod(pcl::MovingLeastSquares<pcl::PointXYZ, pcl::PointXYZ>::NONE);
89 
90  // create PointCloud for results
91  Cloud::Ptr cloud_f(new Cloud);
92  mls.process(*cloud_f);
93 
94  if (cloud_f->points.empty())
95  {
96  log()->get(LogLevel::Debug2) << "Filtered cloud has no points!" << std::endl;
97  return viewSet;
98  }
99 
100  pclsupport::PCDtoPDAL(*cloud_f, output, buffer_bounds);
101 
102  log()->get(LogLevel::Debug2) << cloud->points.size() << " before, " <<
103  cloud_f->points.size() << " after" << std::endl;
104  log()->get(LogLevel::Debug2) << output->size() << std::endl;
105 
106  return viewSet;
107 }
