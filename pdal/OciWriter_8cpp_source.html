87 void OciWriter::wipeBlockTable()
88 {
89  std::ostringstream oss;
90 
91  oss << "DELETE FROM " << m_blockTableName;
92  try
93  {
94  runCommand(oss);
95  }
96  catch (pdal_error const&)
97  {
98  // if we failed, let's try dropping the spatial index for the
99  // block_table_name
100  oss.str("");
101  if (isDebug())
102  std::cout << "Dropping index " << m_blockTableName
103  << "_cloud_idx for block_table_name"
104  << std::endl;
105  oss << "DROP INDEX " << m_blockTableName << "_cloud_idx" ;
106  runCommand(oss);
107  oss.str("");
108 
109  oss << "DELETE FROM " << m_blockTableName;
110  runCommand(oss);
111  }
112 
113  oss.str("");
114  oss << "declare\n"
115  "begin \n"
116  " mdsys.sdo_pc_pkg.drop_dependencies('"
117  << m_baseTableName <<
118  "', '"
119  << m_cloudColumnName <<
120  "'); end;";
121  runCommand(oss);
122 
123  oss.str("");
124  oss << "DROP TABLE " << m_blockTableName;
125  runCommand(oss);
126 
127  // Oracle upper cases the table name when inserting it in the
128  // USER_SDO_GEOM_METADATA. We'll use std::transform to do it.
129  // See http://forums.devx.com/showthread.php?t=83058 for the
130  // technique
131 
132  oss.str("");
133  oss << "DELETE FROM USER_SDO_GEOM_METADATA WHERE TABLE_NAME='" <<
134  m_blockTableName << "'" ;
135  runCommand(oss);
136 }

254 bool OciWriter::isGeographic(int32_t srid)
255 {
256  std::unique_ptr<char> kind(new char[OWNAME]);
257  std::string stmt("SELECT COORD_REF_SYS_KIND from "
258  "MDSYS.SDO_COORD_REF_SYSTEM WHERE SRID = :1");
259 
260  Statement statement(m_connection->CreateStatement(stmt.c_str()));
261  long l_srid = srid;
262  statement->Bind(&l_srid);
263  statement->Define(kind.get());
264 
265  try
266  {
267  statement->Execute();
268  }
269  catch (pdal_error const& e)
270  {
271  throwError("Failed to fetch geographicness of srid " +
272  Utils::toString(srid) + ": " + e.what());
273  }
274 
275  std::string k = Utils::toupper(kind.get());
276  return ((k == "GEOGRAPHIC2D") || (k == "GEOGRAPHIC3D"));
277 }

280 std::string OciWriter::loadSQLData(std::string const& filename)
281 {
282  if (!Utils::fileExists(filename))
283  throwError("File '" + filename + "' does not exist");
284 
285  std::istream::pos_type size;
286  std::istream* input = Utils::openFile(filename, true);
287  if (!input->good())
288  {
289  Utils::closeFile(input);
290  return std::string();
291  }
292 
293  std::string output;
294  std::string line;
295  while (input->good())
296  {
297  getline(*input, line);
298  if (output.size())
299  output += "\n" + line;
300  else
301  output = line;
302  }
303 
304  Utils::closeFile(input);
305 
306  return output;
307 }

342 void OciWriter::createPCEntry()
343 {
344  std::ostringstream oss;
345 
346  oss.setf(std::ios_base::fixed, std::ios_base::floatfield);
347  oss.precision(m_precision);
348 
349  std::ostringstream columns;
350  std::ostringstream values;
351 
352  columns << m_cloudColumnName;
353  values << "pc";
354  if (!m_baseTableAuxColumns.empty())
355  {
356  columns << "," << m_baseTableAuxColumns;
357  values << "," << m_baseTableAuxValues;
358  }
359 
360  int nPCPos = 1;
361  int nSchemaPos = 1;
362  nSchemaPos++;
363 
364  int nPos = nSchemaPos; // Bind column position
365 
366  if (!m_baseTableBoundaryColumn.empty())
367  {
368  columns << "," << m_baseTableBoundaryColumn;
369  nPos++;
370  values << ", SDO_GEOMETRY(:" << nPos;
371  nPos++;
372  values << ", :"<< nPos << ")";
373  }
374 
375  std::ostringstream s_srid;
376  std::ostringstream s_geom;
377  std::ostringstream s_schema;
378 
379  if (m_srid == 0)
380  s_srid << "NULL";
381  else
382  s_srid << m_srid;
383  s_schema << "xmltype(:" << nSchemaPos << ")";
384 
385  std::string eleminfo = createPCElemInfo();
386  BOX3D bounds = m_baseTableBounds;
387  if (bounds.empty())
388  {
389  if (isGeographic(m_srid))
390  {
391  bounds.minx = -179.99;
392  bounds.miny = -89.99;
393  bounds.minz = 0.0;
394  bounds.maxx = 179.99;
395  bounds.maxy = 89.99;
396  bounds.maxz = 20000.0;
397  }
398  else
399  {
400  bounds.minx = 0.0;
401  bounds.miny = 0.0;
402  bounds.minz = 0.0;
403  bounds.maxx = 100.0;
404  bounds.maxy = 100.0;
405  bounds.maxz = 20000.0;
406  }
407  }
408  s_geom << " mdsys.sdo_geometry(" << m_gtype << ", " <<
409  s_srid.str() << ", null,\n"
410  " mdsys.sdo_elem_info_array"<< eleminfo <<",\n"
411  " mdsys.sdo_ordinate_array(\n";
412  s_geom << bounds.minx << "," << bounds.miny << ",";
413  if (m_3d)
414  s_geom << bounds.minz << ",";
415  s_geom << bounds.maxy<< "," << bounds.maxy;
416  if (m_3d)
417  s_geom << "," << bounds.maxy;
418  s_geom << "))";
419 
420  MetadataNode node;
421  if (m_compression)
422  {
423  MetadataNode r = node.add("root");
424  r.add("compression", "lazperf");
425  r.add("version", "1.0");
426  }
427  XMLSchema schema(dbDimTypes(), node, m_orientation);
428  std::string schemaData = schema.xml();
429 
430  oss << "declare\n"
431  " pc_id NUMBER := :" << nPCPos << ";\n"
432  " pc sdo_pc;\n"
433 
434  "begin\n"
435  " -- Initialize the Point Cloud object.\n"
436  " pc := sdo_pc_pkg.init( \n"
437  " '" << m_baseTableName <<
438  "', -- Table that has the SDO_POINT_CLOUD column defined\n"
439  " '" << m_cloudColumnName <<
440  "', -- Column name of the SDO_POINT_CLOUD object\n"
441  " '" << m_blockTableName <<
442  "', -- Table to store blocks of the point cloud\n"
443  " 'blk_capacity="<< m_capacity <<"', -- max # of points "
444  "per block\n"
445  << s_geom.str() <<
446  ", -- Extent\n"
447  " 0.5, -- Tolerance for point cloud\n"
448  " " << dbDimTypes().size() <<
449  ", -- Total number of dimensions\n"
450  " NULL,"
451  " NULL,"
452  " " << s_schema.str() << ");\n"
453  " :" << nPCPos << " := pc.pc_id;\n"
454 
455  " -- Insert the Point Cloud object into the \"base\" table.\n"
456  " insert into " << m_baseTableName << " ( ID, "<< columns.str() <<
457  ") values ( pc.pc_id, " << values.str() << ");\n"
458 
459  " "
460  "end;\n";
461 
462  Statement statement(m_connection->CreateStatement(oss.str().c_str()));
463  statement->Bind(&m_pc_id);
464 
465  OCILobLocator *schema_locator;
466  OCILobLocator *boundary_locator;
467 
468  statement->WriteCLob(&schema_locator, const_cast<char *>(schemaData.c_str()));
469  statement->BindClob(&schema_locator);
470 
471  std::ostringstream wkt_s;
472  if (!m_baseTableBoundaryColumn.empty())
473  {
474  if (!Utils::fileExists(m_baseTableBoundaryWkt))
475  {
476  if (!isValidWKT(m_baseTableBoundaryWkt))
477  throwError("WKT for base_table_boundary_wkt was not valid "
478  "and '" + m_baseTableBoundaryWkt + "' doesn't exist as "
479  "a file");
480  wkt_s << m_baseTableBoundaryWkt;
481  }
482  else
483  {
484  std::string wkt = loadSQLData(m_baseTableBoundaryWkt);
485  if (!isValidWKT(wkt))
486  throwError("WKT for base_table_boundary_wkt was from file '" +
487  m_baseTableBoundaryWkt + "' is not valid");
488  wkt_s << wkt;
489  }
490  }
491 
492  if (!m_baseTableBoundaryColumn.empty())
493  {
494  statement->WriteCLob(&boundary_locator,
495  const_cast<char *>(wkt_s.str().c_str()));
496  statement->BindClob(&boundary_locator);
497  statement->Bind((int*)&m_srid);
498  }
499 
500  try
501  {
502  statement->Execute();
503  }
504  catch (std::runtime_error const& e)
505  {
506  throwError("Failed at creating Point Cloud entry into " +
507  m_baseTableName + " table: " + e.what() + ".");
508  }
509 }

733 void OciWriter::writePointMajor(PointViewPtr view, std::vector<char>& outbuf)
734 {
735  if (m_compression)
736  {
737  outbuf.resize(0);
738 #ifdef PDAL_HAVE_LAZPERF
739  XMLDimList xmlDims = dbDimTypes();
740  DimTypeList dimTypes;
741  for (XMLDim& xmlDim : xmlDims)
742  dimTypes.push_back(xmlDim.m_dimType);
743 
744  auto cb = [&outbuf](char *buf, size_t bufsize)
745  {
746  outbuf.insert(outbuf.end(), buf, buf + bufsize);
747  };
748 
749  LazPerfCompressor compressor(cb, dimTypes);
750 
751  try
752  {
753  std::vector<char> ptBuf(packedPointSize());
754  for (PointId idx = 0; idx < view->size(); ++idx)
755  {
756  size_t size = readPoint(*view, idx, ptBuf.data());
757  compressor.compress(ptBuf.data(), size);
758  }
759  }
760  catch (const pdal_error& err)
761  {
762  compressor.done();
763  throwError(err.what());
764  }
765  compressor.done();
766 #else
767  throwError("Can't compress without LAZperf.");
768 #endif
769  }
770  else
771  {
772  size_t totalSize = 0;
773  char *pos = outbuf.data();
774  for (PointId idx = 0; idx < view->size(); ++idx)
775  {
776  size_t size = readPoint(*view.get(), idx, pos);
777  totalSize += size;
778  pos += size;
779  }
780  outbuf.resize(totalSize);
781  }
782 }

806 void OciWriter::writeTile(const PointViewPtr view)
807 {
808  bool usePartition = (m_blockTablePartitionColumn.size() != 0);
809 
810  std::ostringstream oss;
811  oss << "INSERT INTO "<< m_blockTableName <<
812  "(OBJ_ID, BLK_ID, NUM_POINTS, POINTS, PCBLK_MIN_RES, BLK_EXTENT, "
813  "PCBLK_MAX_RES, NUM_UNSORTED_POINTS, PT_SORT_DIM";
814  if (usePartition)
815  oss << "," << m_blockTablePartitionColumn;
816  oss << ") "
817  "VALUES ( :1, :2, :3, :4, 1, mdsys.sdo_geometry(:5, :6, null,:7, :8)"
818  ", 1, 0, 1";
819  if (usePartition)
820  oss << ", :9";
821  oss <<")";
822 
823  Statement statement(m_connection->CreateStatement(oss.str().c_str()));
824 
825  // :1
826  statement->Bind(&m_pc_id);
827  log()->get(LogLevel::Debug4) << "Block obj_id " << m_pc_id << std::endl;
828 
829  // :2
830  statement->Bind(&m_lastBlockId);
831  m_lastBlockId++;
832  log()->get(LogLevel::Debug4) << "Last BlockId " <<
833  m_lastBlockId << std::endl;
834 
835  // :3
836  long long_num_points = static_cast<long>(view->size());
837  statement->Bind(&long_num_points);
838  log()->get(LogLevel::Debug4) << "Num points " <<
839  long_num_points << std::endl;
840 
841  // :4
842 
843  //NOTE: packed point size is guaranteed to be of sufficient size to hold
844  // a point's data, but it may be larger than the actual size of a point
845  // if location scaling is being used.
846  std::vector<char> outbuf(packedPointSize() * view->size());
847  size_t totalSize = 0;
848  if (m_orientation == Orientation::DimensionMajor)
849  writeDimMajor(view, outbuf);
850  else if (m_orientation == Orientation::PointMajor)
851  writePointMajor(view, outbuf);
852 
853 
854  log()->get(LogLevel::Debug4) << "Blob size " << outbuf.size() << std::endl;
855  OCILobLocator* locator;
856  if (m_streamChunks)
857  {
858  statement->WriteBlob(&locator, outbuf.data(), (long)outbuf.size(),
859  m_chunkCount);
860  statement->BindBlob(&locator);
861  }
862  else
863  statement->Bind(outbuf.data(), (long)outbuf.size());
864 
865  // :5
866  long long_gtype = static_cast<long>(m_gtype);
867  statement->Bind(&long_gtype);
868  log()->get(LogLevel::Debug4) << "OCI geometry type " <<
869  m_gtype << std::endl;
870 
871  // :6
872  long srid;
873 
874  if (m_srid != 0)
875  srid = m_srid;
876  statement->Bind(&srid);
877  log()->get(LogLevel::Debug4) << "OCI SRID " << srid << std::endl;
878 
879 
880  // :7
881  OCIArray* sdo_elem_info = 0;
882  m_connection->CreateType(&sdo_elem_info, m_connection->GetElemInfoType());
883  setElements(statement, sdo_elem_info);
884  statement->Bind(&sdo_elem_info, m_connection->GetElemInfoType());
885 
886  // :8
887  OCIArray* sdo_ordinates = 0;
888  m_connection->CreateType(&sdo_ordinates, m_connection->GetOrdinateType());
889 
890  BOX3D bounds;
891  view->calculateBounds(bounds);
892  // Cumulate a total bounds for the file.
893  m_pcExtent.grow(bounds);
894 
895  setOrdinates(statement, sdo_ordinates, bounds);
896  statement->Bind(&sdo_ordinates, m_connection->GetOrdinateType());
897  log()->get(LogLevel::Debug4) << "Bounds " << bounds << std::endl;
898 
899  // :9
900  if (usePartition)
901  {
902  long long_partition_id = (long)m_blockTablePartitionValue;
903  statement->Bind(&long_partition_id);
904  log()->get(LogLevel::Debug4) << "Partition ID " << long_partition_id <<
905  std::endl;
906  }
907 
908  try
909  {
910  statement->Execute();
911  }
912  catch (std::runtime_error const& e)
913  {
914  throwError("Failed to insert block # into '" + m_blockTableName +
915  "' table: " + e.what());
916  }
917 
918  if (m_streamChunks)
919  // We don't use a locator unless we're stream-writing the data.
920  OWStatement::Free(&locator, 1);
921 
922  m_connection->DestroyType(&sdo_elem_info);
923  m_connection->DestroyType(&sdo_ordinates);
924 }

927 std::string OciWriter::shutOff_SDO_PC_Trigger()
928 {
929  // Don't monkey with the trigger unless the user says to.
930  if (!m_disableCloudTrigger)
931  return std::string("");
932 
933  std::ostringstream oss;
934  char szTrigger[OWNAME] = "";
935  char szStatus[OWNAME] = "";
936 
937  oss << "select trigger_name, status from all_triggers where "
938  "table_name = '" << m_baseTableName <<
939  "' AND TRIGGER_NAME like upper('%%MDTNPC_%%') ";
940  Statement statement(m_connection->CreateStatement(oss.str().c_str()));
941 
942  statement->Define(szTrigger);
943  statement->Define(szStatus);
944  statement->Execute();
945 
946  // Yes, we're assuming there's only one trigger that met these criteria.
947 
948  if (!strlen(szStatus))
949  {
950  // No rows returned, no trigger exists
951  return std::string();
952  }
953 
954  if (Utils::iequals(szStatus, "ENABLED"))
955  {
956  oss.str("");
957  oss << "ALTER TRIGGER " << szTrigger << " DISABLE ";
958 
959  statement = Statement(m_connection->CreateStatement(oss.str().c_str()));
960  statement->Execute();
961  return std::string(szTrigger);
962  }
963  return std::string();
964 }
