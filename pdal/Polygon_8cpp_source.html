89 Polygon::Polygon(OGRGeometryH g, const SpatialReference& srs) : Geometry(g, srs)
90 {
91  OGRwkbGeometryType t = OGR_G_GetGeometryType(g);
92 
93  if (!(t == wkbPolygon ||
94  t == wkbMultiPolygon ||
95  t == wkbPolygon25D ||
96  t == wkbMultiPolygon25D))
97  {
98  std::ostringstream oss;
99  oss << "pdal::Polygon cannot construct geometry because "
100  "OGR geometry is not Polygon or MultiPolygon!";
101  throw pdal::pdal_error(oss.str());
102  }
103 
104  OGRGeometry *ogr_g = (OGRGeometry*)g;
105  //
106  // Convert the the GDAL geom to WKB in order to avoid the version
107  // context issues with exporting directoly to GEOS.
108  OGRwkbByteOrder bo =
109  GEOS_getWKBByteOrder() == GEOS_WKB_XDR ? wkbXDR : wkbNDR;
110  int wkbSize = ogr_g->WkbSize();
111  std::vector<unsigned char> wkb(wkbSize);
112 
113  ogr_g->exportToWkb(bo, wkb.data());
114  geos::GeometryDeleter geom_del(m_geoserr);
115  GEOSGeomPtr p(GEOSGeomFromWKB_buf_r(m_geoserr.ctx(), wkb.data(), wkbSize), geom_del);
116  m_geom.swap(p);
117  prepare();
118 
119 }

136 void Polygon::initializeFromBounds(const BOX3D& box)
137 {
138  GEOSCoordSequence* coords = GEOSCoordSeq_create_r(m_geoserr.ctx(), 5, 3);
139  auto set_coordinate = [coords, this](int pt_num, const double&x,
140  const double& y, const double& z)
141  {
142  if (!GEOSCoordSeq_setX_r(m_geoserr.ctx(), coords, pt_num, x))
143  throw pdal_error("unable to set x for coordinate sequence");
144  if (!GEOSCoordSeq_setY_r(m_geoserr.ctx(), coords, pt_num, y))
145  throw pdal_error("unable to set y for coordinate sequence");
146  if (!GEOSCoordSeq_setZ_r(m_geoserr.ctx(), coords, pt_num, z))
147  throw pdal_error("unable to set z for coordinate sequence");
148  };
149 
150  set_coordinate(0, box.minx, box.miny, box.minz);
151  set_coordinate(1, box.minx, box.maxy, box.minz);
152  set_coordinate(2, box.maxx, box.maxy, box.maxz);
153  set_coordinate(3, box.maxx, box.miny, box.maxz);
154  set_coordinate(4, box.minx, box.miny, box.minz);
155 
156 
157  GEOSGeometry* ring = GEOSGeom_createLinearRing_r(m_geoserr.ctx(), coords);
158  if (!ring)
159  throw pdal_error("unable to create linear ring from BOX2D");
160 
161 
162  geos::GeometryDeleter geom_del(m_geoserr);
163  GEOSGeomPtr p(GEOSGeom_createPolygon_r(m_geoserr.ctx(), ring, 0, 0), geom_del);
164  m_geom.swap(p);
165  if (!m_geom.get())
166  throw pdal_error("unable to create polygon from linear ring in "
167  "BOX2D constructor");
168  prepare();
169 }

172 Polygon Polygon::transform(const SpatialReference& ref) const
173 {
174  if (ref.empty() && m_srs.empty())
175  return *this;
176 
177  if (m_srs.empty())
178  throw pdal_error("Polygon::transform failed due to m_srs being empty");
179  if (ref.empty())
180  throw pdal_error("Polygon::transform failed due to ref being empty");
181 
182  gdal::SpatialRef fromRef(m_srs.getWKT());
183  gdal::SpatialRef toRef(ref.getWKT());
184  gdal::Geometry geom(wkt(12, true), fromRef);
185  geom.transform(toRef);
186  return Polygon(geom.wkt(), ref);

190 Polygon Polygon::simplify(double distance_tolerance,
191  double area_tolerance) const
192 {
193  GEOSGeometry *smoothed =
194  GEOSTopologyPreserveSimplify_r(m_geoserr.ctx(), m_geom.get(), distance_tolerance);
195  if (!smoothed)
196  throw pdal_error("Unable to simplify input geometry!");
197 
198  std::vector<GEOSGeometry*> geometries;
199 
200  int numGeom = GEOSGetNumGeometries_r(m_geoserr.ctx(), smoothed);
201  for (int n = 0; n < numGeom; ++n)
202  {
203  const GEOSGeometry* m = GEOSGetGeometryN_r(m_geoserr.ctx(), smoothed, n);
204  if (!m)
205  throw pdal::pdal_error("Unable to Get GeometryN");
206 
207  const GEOSGeometry* ering = GEOSGetExteriorRing_r(m_geoserr.ctx(), m);
208  if (!ering)
209  throw pdal::pdal_error("Unable to Get Exterior Ring");
210 
211  GEOSGeometry* exterior = GEOSGeom_clone_r(m_geoserr.ctx(),
212  GEOSGetExteriorRing_r(m_geoserr.ctx(), m));
213  if (!exterior)
214  throw pdal::pdal_error("Unable to clone exterior ring!");
215 
216  std::vector<GEOSGeometry*> keep_rings;
217  int numRings = GEOSGetNumInteriorRings_r(m_geoserr.ctx(), m);
218  for (int i = 0; i < numRings; ++i)
219  {
220  double area(0.0);
221 
222  const GEOSGeometry* iring =
223  GEOSGetInteriorRingN_r(m_geoserr.ctx(), m, i);
224  if (!iring)
225  throw pdal::pdal_error("Unable to Get Interior Ring");
226 
227  GEOSGeometry* cring = GEOSGeom_clone_r(m_geoserr.ctx(), iring);
228  if (!cring)
229  throw pdal::pdal_error("Unable to clone interior ring!");
230  GEOSGeometry* aring = GEOSGeom_createPolygon_r(m_geoserr.ctx(), cring,
231  NULL, 0);
232 
233  int errored = GEOSArea_r(m_geoserr.ctx(), aring, &area);
234  if (errored == 0)
235  throw pdal::pdal_error("Unable to get area of ring!");
236  if (area > area_tolerance)
237  {
238  keep_rings.push_back(cring);
239  }
240  }
241 
242  GEOSGeometry* p = GEOSGeom_createPolygon_r(m_geoserr.ctx(), exterior,
243  keep_rings.data(), keep_rings.size());
244  if (p == NULL)
245  throw pdal_error("smooth polygon could not be created!" );
246  geometries.push_back(p);
247  }
248 
249  GEOSGeometry* o = GEOSGeom_createCollection_r(m_geoserr.ctx(), GEOS_MULTIPOLYGON,
250  geometries.data(), geometries.size());
251  Polygon p(o, m_srs);
252  GEOSGeom_destroy_r(m_geoserr.ctx(), smoothed);
253  GEOSGeom_destroy_r(m_geoserr.ctx(), o);
254 
255  return p;
256 }

268 bool Polygon::covers(const PointRef& ref) const
269 {
270  GEOSCoordSequence* coords = GEOSCoordSeq_create_r(m_geoserr.ctx(), 1, 3);
271  if (!coords)
272  throw pdal_error("Unable to allocate coordinate sequence");
273 
274  const double x = ref.getFieldAs<double>(Dimension::Id::X);
275  const double y = ref.getFieldAs<double>(Dimension::Id::Y);
276  const double z = ref.getFieldAs<double>(Dimension::Id::Z);
277 
278  if (!GEOSCoordSeq_setX_r(m_geoserr.ctx(), coords, 0, x))
279  throw pdal_error("unable to set x for coordinate sequence");
280  if (!GEOSCoordSeq_setY_r(m_geoserr.ctx(), coords, 0, y))
281  throw pdal_error("unable to set y for coordinate sequence");
282  if (!GEOSCoordSeq_setZ_r(m_geoserr.ctx(), coords, 0, z))
283  throw pdal_error("unable to set z for coordinate sequence");
284  GEOSGeometry* p = GEOSGeom_createPoint_r(m_geoserr.ctx(), coords);
285  if (!p)
286  throw pdal_error("unable to allocate candidate test point");
287 
288  bool covers = (bool)(GEOSPreparedCovers_r(m_geoserr.ctx(), m_prepGeom, p));
289  GEOSGeom_destroy_r(m_geoserr.ctx(), p);
290 
291  return covers;
292 }
