57 int argCount(PyObject *function)
58 {
59  PyObject *module = PyImport_ImportModule("inspect");
60  if (!module)
61  return false;
62  PyObject *dictionary = PyModule_GetDict(module);
63  PyObject *getargFunc = PyDict_GetItemString(dictionary, "getargspec");
64  PyObject *inArgs = PyTuple_New(1);
65  PyTuple_SetItem(inArgs, 0, function);
66  PyObject *outArgs = PyObject_CallObject(getargFunc, inArgs);
67  PyObject *arglist = PyTuple_GetItem(outArgs, (Py_ssize_t)0);
68  return (int) PyList_Size(arglist);

104 void Invocation::compile()
105 {
106  m_bytecode = Py_CompileString(m_script.source(), m_script.module(),
107  Py_file_input);
108  if (!m_bytecode)
109  throw pdal::pdal_error(getTraceback());
110 
111  Py_INCREF(m_bytecode);
112 
113  m_module = PyImport_ExecCodeModule(const_cast<char*>(m_script.module()),
114  m_bytecode);
115  if (!m_module)
116  throw pdal::pdal_error(getTraceback());
117 
118  m_dictionary = PyModule_GetDict(m_module);
119  m_function = PyDict_GetItemString(m_dictionary, m_script.function());
120  if (!m_function)
121  {
122  std::ostringstream oss;
123  oss << "unable to find target function '" << m_script.function() <<
124  "' in module.";
125  throw pdal::pdal_error(oss.str());
126  }
127  if (!PyCallable_Check(m_function))
128  throw pdal::pdal_error(getTraceback());

177 void *Invocation::extractResult(std::string const& name,
178  Dimension::Type t)
179 {
180  PyObject* xarr = PyDict_GetItemString(m_varsOut, name.c_str());
181  if (!xarr)
182  throw pdal::pdal_error("plang output variable '" + name + "' not found.");
183  if (!PyArray_Check(xarr))
184  throw pdal::pdal_error("Plang output variable '" + name +
185  "' is not a numpy array");
186 
187  PyArrayObject* arr = (PyArrayObject*)xarr;
188 
189  npy_intp one = 0;
190  const int pyDataType = pdal::plang::Environment::getPythonDataType(t);
191  PyArray_Descr *dtype = PyArray_DESCR(arr);
192 
193  if (static_cast<uint32_t>(dtype->elsize) != Dimension::size(t))
194  {
195  std::ostringstream oss;
196  oss << "dtype of array has size " << dtype->elsize
197  << " but PDAL dimension '" << name << "' has byte size of "
198  << Dimension::size(t) << " bytes.";
199  throw pdal::pdal_error(oss.str());
200  }
201 
202  using namespace Dimension;
203  BaseType b = Dimension::base(t);
204  if (dtype->kind == 'i' && b != BaseType::Signed)
205  {
206  std::ostringstream oss;
207  oss << "dtype of array has a signed integer type but the " <<
208  "dimension data type of '" << name <<
209  "' is not pdal::Signed.";
210  throw pdal::pdal_error(oss.str());
211  }
212 
213  if (dtype->kind == 'u' && b != BaseType::Unsigned)
214  {
215  std::ostringstream oss;
216  oss << "dtype of array has a unsigned integer type but the " <<
217  "dimension data type of '" << name <<
218  "' is not pdal::Unsigned.";
219  throw pdal::pdal_error(oss.str());
220  }
221 
222  if (dtype->kind == 'f' && b != BaseType::Floating)
223  {
224  std::ostringstream oss;
225  oss << "dtype of array has a float type but the " <<
226  "dimension data type of '" << name << "' is not pdal::Floating.";
227  throw pdal::pdal_error(oss.str());
228  }
229  return PyArray_GetPtr(arr, &one);
230 }

260 bool Invocation::execute()
261 {
262  if (!m_bytecode)
263  throw pdal::pdal_error("No code has been compiled");
264 
265  Py_INCREF(m_varsIn);
266  Py_ssize_t numArgs = argCount(m_function);
267  m_scriptArgs = PyTuple_New(numArgs);
268 
269  if (numArgs > 2)
270  throw pdal::pdal_error("Only two arguments -- ins and outs numpy arrays -- can be passed!");
271 
272  PyTuple_SetItem(m_scriptArgs, 0, m_varsIn);
273  if (numArgs > 1)
274  {
275  Py_INCREF(m_varsOut);
276  PyTuple_SetItem(m_scriptArgs, 1, m_varsOut);
277  }
278 
279  int success(0);
280 
281  if (m_metadata_PyObject)
282  {
283  success = PyModule_AddObject(m_module, "metadata", m_metadata_PyObject);
284  if (success)
285  throw pdal::pdal_error("unable to set metadata global");
286  Py_INCREF(m_metadata_PyObject);
287  }
288 
289  if (m_schema_PyObject)
290  {
291  success = PyModule_AddObject(m_module, "schema", m_schema_PyObject);
292  if (success)
293  throw pdal::pdal_error("unable to set schema global");
294  Py_INCREF(m_srs_PyObject);
295  }
296 
297  if (m_srs_PyObject)
298  {
299  success = PyModule_AddObject(m_module, "spatialreference", m_srs_PyObject);
300  if (success)
301  throw pdal::pdal_error("unable to set spatialreference global");
302  Py_INCREF(m_schema_PyObject);
303  }
304 
305  if (m_pdalargs_PyObject)
306  {
307  success = PyModule_AddObject(m_module, "pdalargs", m_pdalargs_PyObject);
308  if (success)
309  throw pdal::pdal_error("unable to set pdalargs global");
310  Py_INCREF(m_pdalargs_PyObject);
311  }
312 
313  m_scriptResult = PyObject_CallObject(m_function, m_scriptArgs);
314  if (!m_scriptResult)
315  throw pdal::pdal_error(getTraceback());
316  if (!PyBool_Check(m_scriptResult))
317  throw pdal::pdal_error("User function return value not a boolean type.");
318 
319  PyObject* mod_vars = PyModule_GetDict(m_module);
320 
321  PyObject* b = PyUnicode_FromString("metadata");
322  if (PyDict_Contains(mod_vars, PyUnicode_FromString("metadata")) == 1)
323  m_metadata_PyObject = PyDict_GetItem(m_dictionary, b);
324 
325  return (m_scriptResult == Py_True);
326 }

365 void Invocation::begin(PointView& view, MetadataNode m)
366 {
367  PointLayoutPtr layout(view.m_pointTable.layout());
368  Dimension::IdList const& dims = layout->dims();
369 
370  for (auto di = dims.begin(); di != dims.end(); ++di)
371  {
372  Dimension::Id d = *di;
373  const Dimension::Detail *dd = layout->dimDetail(d);
374  void *data = malloc(dd->size() * view.size());
375  m_buffers.push_back(data); // Hold pointer for deallocation
376  char *p = (char *)data;
377  for (PointId idx = 0; idx < view.size(); ++idx)
378  {
379  view.getFieldInternal(d, idx, (void *)p);
380  p += dd->size();
381  }
382  std::string name = layout->dimName(*di);
383  insertArgument(name, (uint8_t *)data, dd->type(), view.size());
384  }
385 
386  // Put pipeline 'metadata' variable into module scope
387  Py_XDECREF(m_metadata_PyObject);
388  m_metadata_PyObject= plang::fromMetadata(m);
389 
390  // Put 'schema' dict into module scope
391  Py_XDECREF(m_schema_PyObject);
392  MetadataNode s = view.layout()->toMetadata();
393  std::ostringstream ostrm;
394  Utils::toJSON(s, ostrm);
395  m_schema_PyObject = getPyJSON(ostrm.str());
396  ostrm.str("");
397 
398  Py_XDECREF(m_srs_PyObject);
399  MetadataNode srs = view.spatialReference().toMetadata();
400  Utils::toJSON(srs, ostrm);
401  m_srs_PyObject = getPyJSON(ostrm.str());
402  ostrm.str("");
403 }

406 void Invocation::end(PointView& view, MetadataNode m)
407 {
408  // for each entry in the script's outs dictionary,
409  // look up that entry's name in the schema and then
410  // copy the data into the right dimension spot in the
411  // buffer
412 
413  std::vector<std::string> names;
414  getOutputNames(names);
415 
416  PointLayoutPtr layout(view.m_pointTable.layout());
417  Dimension::IdList const& dims = layout->dims();
418 
419  for (auto di = dims.begin(); di != dims.end(); ++di)
420  {
421  Dimension::Id d = *di;
422  const Dimension::Detail *dd = layout->dimDetail(d);
423  std::string name = layout->dimName(*di);
424  auto found = std::find(names.begin(), names.end(), name);
425  if (found == names.end()) continue; // didn't have this dim in the names
426 
427  assert(name == *found);
428  assert(hasOutputVariable(name));
429 
430  size_t size = dd->size();
431  void *data = extractResult(name, dd->type());
432  char *p = (char *)data;
433  for (PointId idx = 0; idx < view.size(); ++idx)
434  {
435  view.setField(d, dd->type(), idx, (void *)p);
436  p += size;
437  }
438  }
439  for (auto bi = m_buffers.begin(); bi != m_buffers.end(); ++bi)
440  free(*bi);
441  m_buffers.clear();
442  if (m_metadata_PyObject)
443  addMetadata(m_metadata_PyObject, m);
444 }
