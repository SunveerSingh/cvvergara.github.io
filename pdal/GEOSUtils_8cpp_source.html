143 void ErrorHandler::handle(const char *msg, bool notice)
144 {
145  std::ostringstream oss;
146  if (!notice)
147  {
148  oss << "GEOS failure: '" << msg << "'";
149  throw pdal_error(oss.str());
150  }
151  else if (m_debug)
152  {
153  oss << "GEOS debug: " << msg;
154  if (m_log)
155  m_log->get(LogLevel::Debug) << oss.str() << std::endl;
156  }
157 }
