55 void TextReader::initialize(PointTableRef table)
56 {
57  m_istream = Utils::openFile(m_filename);
58  if (!m_istream)
59  throwError("Unable to open text file '" + m_filename + "'.");
60 
61  std::string buf;
62  std::getline(*m_istream, buf);
63 
64  auto isspecial = [](char c)
65  { return (!std::isalnum(c) && c != ' '); };
66 
67  // If the separator wasn't provided on the command line extract it
68  // from the header line.
69  if (m_separator == ' ')
70  {
71  // Scan string for some character not a number, space or letter.
72  for (size_t i = 0; i < buf.size(); ++i)
73  if (isspecial(buf[i]))
74  {
75  m_separator = buf[i];
76  break;
77  }
78  }
79 
80  if (m_separator != ' ')
81  m_dimNames = Utils::split(buf, m_separator);
82  else
83  m_dimNames = Utils::split2(buf, m_separator);
84  Utils::closeFile(m_istream);
85 }

141 bool TextReader::processOne(PointRef& point)
142 {
143  if (!fillFields())
144  return false;
145 
146  double d;
147  for (size_t i = 0; i < m_fields.size(); ++i)
148  {
149  if (!Utils::fromString(m_fields[i], d))
150  {
151  log()->get(LogLevel::Error) << "Can't convert "
152  "field '" << m_fields[i] << "' to numeric value on line " <<
153  m_line << " in '" << m_filename << "'. Setting to 0." <<
154  std::endl;
155  d = 0;
156  }
157  point.setField(m_dims[i], d);
158  }
159  return true;
160 }

163 bool TextReader::fillFields()
164 {
165  while (true)
166  {
167  if (!m_istream->good())
168  return false;
169 
170  std::string buf;
171 
172  std::getline(*m_istream, buf);
173  m_line++;
174  if (buf.empty())
175  continue;
176  if (m_separator != ' ')
177  {
178  Utils::remove(buf, ' ');
179  m_fields = Utils::split(buf, m_separator);
180  }
181  else
182  m_fields = Utils::split2(buf, m_separator);
183  if (m_fields.size() != m_dims.size())
184  {
185  log()->get(LogLevel::Error) << "Line " << m_line <<
186  " in '" << m_filename << "' contains " << m_fields.size() <<
187  " fields when " << m_dims.size() << " were expected. "
188  "Ignoring." << std::endl;
189  continue;
190  }
191  return true;
192  }
193 }
