55  /** \brief Returns if a point X is visible from point R (or the origin)
56  * when taking into account the segment between the points S1 and S2
57  * \param X 2D coordinate of the point
58  * \param S1 2D coordinate of the segment's first point
59  * \param S2 2D coordinate of the segment's secont point
60  * \param R 2D coorddinate of the reference point (defaults to 0,0)
61  * \ingroup surface
62  */
63  inline bool
64  isVisible (const Eigen::Vector2d &X, const Eigen::Vector2d &S1,
65  const Eigen::Vector2d &S2,
66  const Eigen::Vector2d &R = Eigen::Vector2d::Zero ())
67  {
68  double a0 = S1[1] - S2[1];
69  double b0 = S2[0] - S1[0];
70  double c0 = S1[0]*S2[1] - S2[0]*S1[1];
71  double a1 = -X[1];
72  double b1 = X[0];
73  double c1 = 0;
74  if (R != Eigen::Vector2d::Zero())
75  {
76  a1 += R[1];
77  b1 -= R[0];
78  c1 = R[0]*X[1] - X[0]*R[1];
79  }
80  double div = a0*b1 - b0*a1;
81  double x = (b0*c1 - b1*c0) / div;
82  double y = (a1*c0 - a0*c1) / div;
83 
84  bool intersection_outside_XR;
85  if (R == Eigen::Vector2d::Zero())
86  {
87  if (X[0] > 0)
88  intersection_outside_XR = (x <= 0) || (x >= X[0]);
89  else if (X[0] < 0)
90  intersection_outside_XR = (x >= 0) || (x <= X[0]);
91  else if (X[1] > 0)
92  intersection_outside_XR = (y <= 0) || (y >= X[1]);
93  else if (X[1] < 0)
94  intersection_outside_XR = (y >= 0) || (y <= X[1]);
95  else
96  intersection_outside_XR = true;
97  }
98  else
99  {
100  if (X[0] > R[0])
101  intersection_outside_XR = (x <= R[0]) || (x >= X[0]);
102  else if (X[0] < R[0])
103  intersection_outside_XR = (x >= R[0]) || (x <= X[0]);
104  else if (X[1] > R[1])
105  intersection_outside_XR = (y <= R[1]) || (y >= X[1]);
106  else if (X[1] < R[1])
107  intersection_outside_XR = (y >= R[1]) || (y <= X[1]);
108  else
109  intersection_outside_XR = true;
110  }
111  if (intersection_outside_XR)
112  return true;
113  else
114  {
115  if (S1[0] > S2[0])
116  return (x <= S2[0]) || (x >= S1[0]);
117  else if (S1[0] < S2[0])
118  return (x >= S2[0]) || (x <= S1[0]);
119  else if (S1[1] > S2[1])
120  return (y <= S2[1]) || (y >= S1[1]);
121  else if (S1[1] < S2[1])
122  return (y >= S2[1]) || (y <= S1[1]);
123  else
124  return false;
125  }
126  }
