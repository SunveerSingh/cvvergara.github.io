110 void MrsidReader::addDimensions(PointLayoutPtr layout)
111 {
112  using namespace Dimension;
113 
114  if (!m_PS)
115  throwError("MrSID object not initialized.");
116  const LizardTech::PointInfo& pointinfo = m_PS->getPointInfo();
117 
118  // add a map for PDAL names that aren't the same as LT ones (GPSTime vs Time)
119  std::map<std::string, std::string> dimensionTranslations;
120  dimensionTranslations.insert(std::pair<std::string, std::string>("Time", "GPSTime"));
121  dimensionTranslations.insert(std::pair<std::string, std::string>("NumReturns", "NumberOfReturns"));
122  dimensionTranslations.insert(std::pair<std::string, std::string>("ReturnNum", "ReturnNumber"));
123  dimensionTranslations.insert(std::pair<std::string, std::string>("ScanDir", "ScanDirectionFlag"));
124  dimensionTranslations.insert(std::pair<std::string, std::string>("EdgeFlightLine", "EdgeOfFlightLine"));
125  dimensionTranslations.insert(std::pair<std::string, std::string>("ScanAngle", "ScanAngleRank"));
126  dimensionTranslations.insert(std::pair<std::string, std::string>("ClassId", "Classification"));
127  dimensionTranslations.insert(std::pair<std::string, std::string>("SourceId", "PointSourceId"));
128 
129  for (unsigned int i=0; i<pointinfo.getNumChannels(); i++)
130  {
131  const LizardTech::ChannelInfo &channel = pointinfo.getChannel(i);
132  const char* name = channel.getName();
133  auto translated = dimensionTranslations.find(name);
134  if (translated != dimensionTranslations.end())
135  name = translated->second.c_str();
136 
137  LizardTech::DataType t = channel.getDataType();
138  Dimension::Type pdal_type = getPDALType(t);
139  layout->registerOrAssignDim(name, pdal_type);
140  }
141  m_layout = layout;
142 
143 }

169 QuickInfo MrsidReader::inspect()
170 {
171  QuickInfo qi;
172  std::unique_ptr<PointLayout> layout(new PointLayout());
173 
174  MrsidReader::initialize();
175  addDimensions(layout.get());
176 
177  BOX3D b(m_PS->getBounds().x.min, m_PS->getBounds().x.max,
178  m_PS->getBounds().y.min, m_PS->getBounds().y.max,
179  m_PS->getBounds().z.min,m_PS->getBounds().z.max);
180 
181  Dimension::IdList dims = layout->dims();
182  for (auto di = dims.begin(); di != dims.end(); ++di)
183  qi.m_dimNames.push_back(layout->dimName(*di));
184  if (!Utils::numericCast(m_PS->getNumPoints(), qi.m_pointCount))
185  qi.m_pointCount = std::numeric_limits<point_count_t>::max();
186  qi.m_bounds = b;
187  qi.m_srs = pdal::SpatialReference(m_PS->getWKT());
188  qi.m_valid = true;
189 
190  PointTable table;
191  done(table);
192 
193  return qi;
194 }

196 void MrsidReader::LayoutToPointInfo(const PointLayout &layout, LizardTech::PointInfo &pointInfo) const
197 {
198  using namespace pdal::Dimension;
199  const Dimension::IdList& dims = layout.dims();
200 
201  pointInfo.init(dims.size());
202  for (unsigned int idx=0; idx<dims.size(); idx++)
203  {
204  std::string name = layout.dimName(dims[idx]);
205  Dimension::Type t = layout.dimType(dims[idx]);
206  size_t size = layout.dimSize(dims[idx]);
207 
208  if (Utils::iequals(name, "EdgeOfFlightLine")) name = CHANNEL_NAME_EdgeFlightLine;
209  if (Utils::iequals(name, "Classification")) name = CHANNEL_NAME_ClassId;
210  if (Utils::iequals(name, "ScanAngleRank")) name = CHANNEL_NAME_ScanAngle;
211  if (Utils::iequals(name, "ScanDirectionFlag")) name = CHANNEL_NAME_ScanDir;
212  if (Utils::iequals(name, "GpsTime")) name = CHANNEL_NAME_GPSTime;
213  if (Utils::iequals(name, "PointSourceId")) name = CHANNEL_NAME_SourceId;
214  if (Utils::iequals(name, "ReturnNumber")) name = CHANNEL_NAME_ReturnNum;
215  if (Utils::iequals(name, "NumberOfReturns")) name = CHANNEL_NAME_NumReturns;
216 
217  if (t == Type::Double)
218  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_FLOAT64, 64);
219  else if (t == Type::Float)
220  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_FLOAT32, 32);
221  else if (t == Type::Signed64)
222  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_SINT64, 64);
223  else if (t == Type::Signed32)
224  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_SINT32, 32);
225  else if (t == Type::Signed16)
226  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_SINT16, 16);
227  else if (t == Type::Signed8)
228  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_SINT8, 8);
229  else if (t == Type::Unsigned64)
230  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_UINT64, 64);
231  else if (t == Type::Unsigned32)
232  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_UINT32, 32);
233  else if (t == Type::Unsigned16)
234  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_UINT16, 16);
235  else if (t == Type::Unsigned8)
236  pointInfo.getChannel(idx).init(name.c_str(), LizardTech::DATATYPE_UINT8, 8);
237  }
238 }
