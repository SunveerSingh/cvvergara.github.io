159 {
160  FlexWriter::validateFilename(table);
161 
162  PointLayoutPtr layout = table.layout();
163 
164  // Make sure the dataformatID is set so that we can get the proper
165  // dimensions being written as part of the standard LAS record.
166  fillHeader();
167 
168  // If we've asked for all dimensions, add to extraDims all dimensions
169  // in the layout that aren't already destined for LAS output.
170  if (m_extraDims.size() == 1 && m_extraDims[0].m_name == "all")
171  {
172  m_extraDims.clear();
173  Dimension::IdList ids = m_lasHeader.usedDims();
174  DimTypeList dimTypes = layout->dimTypes();
175  for (auto& dt : dimTypes)
176  {
177  if (!Utils::contains(ids, dt.m_id))
178  m_extraDims.push_back(
179  ExtraDim(layout->dimName(dt.m_id), dt.m_type));
180  }
181  }
182 
183  m_extraByteLen = 0;
184  for (auto& dim : m_extraDims)
185  {
186  dim.m_dimType.m_id = table.layout()->findDim(dim.m_name);
187  if (dim.m_dimType.m_id == Dimension::Id::Unknown)
188  throwError("Dimension '" + dim.m_name + "' specified in "
189  "'extra_dim' option not found.");
190  m_extraByteLen += Dimension::size(dim.m_dimType.m_type);
191  log()->get(LogLevel::Info) << getName() << ": Writing dimension " <<
192  dim.m_name <<

194  ") " << " to LAS extra bytes." << std::endl;

233 {
234  static const StringList header = {
235  "dataformat_id", "major_version", "minor_version", "filesource_id",
236  "global_encoding", "project_id", "system_id", "software_id",
237  "creation_doy", "creation_year"
238  };
239 
240  static const StringList scale = { "scale_x", "scale_y", "scale_z" };
241 
242  static const StringList offset = { "offset_x", "offset_y", "offset_z" };
243 
244  static StringList all;
245  all.insert(all.begin(), header.begin(), header.end());
246  all.insert(all.begin(), scale.begin(), scale.end());
247  all.insert(all.begin(), offset.begin(), offset.end());
248 
249  // Build the forward list, replacing special keywords with the proper
250  // field names.
251  for (auto& name : m_forwardSpec)
252  {
253  if (name == "all")
254  {
255  m_forwards.insert(all.begin(), all.end());
256  m_forwardVlrs = true;
257  }
258  else if (name == "header")
259  m_forwards.insert(header.begin(), header.end());
260  else if (name == "scale")
261  m_forwards.insert(scale.begin(), scale.end());
262  else if (name == "offset")
263  m_forwards.insert(offset.begin(), offset.end());
264  else if (name == "format")
265  m_forwards.insert("dataformat_id");
266  else if (name == "vlr")
267  m_forwardVlrs = true;
268  else if (Utils::contains(all, name))
269  m_forwards.insert(name);
270  else
271  throwError("Error in 'forward' option. Unknown field for "
272  "forwarding: '" + name + "'.");
273  }

304 void LasWriter::prepOutput(std::ostream *outStream, const SpatialReference& srs)
305 {
306  // Use stage SRS if provided.
307  m_srs = getSpatialReference().empty() ? srs : getSpatialReference();
308 
309  handleHeaderForwards(m_forwardMetadata);
310 
311  // Filling the header here gives the VLR functions below easy access to
312  // the version information and so on.
313  fillHeader();
314 
315  // Spatial reference can potentially change for multiple output files.
316  addSpatialRefVlrs();
317 
318  m_summaryData.reset(new LasSummaryData());
319  m_ostream = outStream;
320  if (m_lasHeader.compressed())
321  readyCompression();
322 
323  // Compression should cause the last of the VLRs to get filled. We now
324  // have a valid count, so fill the header again.
325  fillHeader();
326 
327  // Write the header.
328  OLeStream out(m_ostream);
329  out << m_lasHeader;
330 
331  m_lasHeader.setVlrOffset((uint32_t)m_ostream->tellp());
332 
333  for (auto vi = m_vlrs.begin(); vi != m_vlrs.end(); ++vi)
334  {
335  LasVLR& vlr = *vi;
336  vlr.write(out, m_lasHeader.versionEquals(1, 0) ? 0xAABB : 0);
337  }
338 
339  // Write the point data start signature for version 1.0.
340  if (m_lasHeader.versionEquals(1, 0))
341  out << (uint16_t)0xCCDD;
342  m_lasHeader.setPointOffset((uint32_t)m_ostream->tellp());
343  if (m_compression == LasCompression::LasZip)
344  openCompression();
345 
346  // Set the point buffer size here in case we're using the streaming
347  // interface.
348  m_pointBuf.resize(m_lasHeader.pointLen());
349 }

352 /// Search for metadata associated with the provided recordId and userId.
353 /// \param node - Top-level node to use for metadata search.
354 /// \param recordId - Record ID to match.
355 /// \param userId - User ID to match.
356 MetadataNode LasWriter::findVlrMetadata(MetadataNode node,
357  uint16_t recordId, const std::string& userId)
358 {
359  std::string sRecordId = std::to_string(recordId);
360 
361  // Find a node whose name starts with vlr and that has child nodes
362  // with the name and recordId we're looking for.
363  auto pred = [sRecordId,userId](MetadataNode n)
364  {
365  auto recPred = [sRecordId](MetadataNode n)
366  {
367  return n.name() == "record_id" &&
368  n.value() == sRecordId;
369  };
370  auto userPred = [userId](MetadataNode n)
371  {
372  return n.name() == "user_id" &&
373  n.value() == userId;
374  };
375  return (Utils::startsWith(n.name(), "vlr") &&
376  !n.findChild(recPred).empty() &&
377  !n.findChild(userPred).empty());
378  };
379  return node.find(pred);
380 }

796 bool LasWriter::writeLasZipBuf(PointRef& point)

812  if (numberOfReturns > maxReturnCount)

814  if (m_discardHighReturnNumbers)

828  throwError("Unable to convert scaled value (" +
829  Utils::toString(d) + ") to "
830  "int32 for dimension '" + Dimension::name(dim) +
831  "' when writing LAS/LAZ file " + m_curFilename + ".");
832  return i;
833  };

836  double yOrig = point.getFieldAs<double>(Id::Y);
837  double zOrig = point.getFieldAs<double>(Id::Z);

839  double y = m_scaling.m_yXform.toScaled(yOrig);
840  double z = m_scaling.m_zXform.toScaled(zOrig);

871  p.extended_classification_flags = classFlags;

876  else

878  p.synthetic_flag = classFlags & 0x1;
879  p.keypoint_flag = (classFlags >> 1) & 0x1;
880  p.withheld_flag = (classFlags >> 2) & 0x1;
881  p.return_number = returnNumber;
882  p.number_of_returns = numberOfReturns;
883  p.scan_angle_rank = point.getFieldAs<int8_t>(Id::ScanAngleRank);
884  p.classification = classification;
