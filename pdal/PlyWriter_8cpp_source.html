71 void PlyWriter::prepared(PointTableRef table)
72 {
73  if (m_dimNames.size())
74  {
75  for (auto& name : m_dimNames)
76  {
77  auto id = table.layout()->findDim(name);
78  if (id == Dimension::Id::Unknown)
79  throwError("Unknown dimension '" + name + "' in provided "
80  "dimension list.");
81  m_dims.push_back(id);

83  }
84  else
85  {
86  m_dims = table.layout()->dims();
87  for (auto dim : m_dims)
88  m_dimNames.push_back(Utils::tolower(table.layout()->dimName(dim)));
89  }

93 std::string PlyWriter::getType(Dimension::Type type) const
94 {
95  static std::map<Dimension::Type, std::string> types =
96  {
97  { Dimension::Type::Signed8, "int8" },
98  { Dimension::Type::Unsigned8, "uint8" },
99  { Dimension::Type::Signed16, "int16" },
100  { Dimension::Type::Unsigned16, "uint16" },
101  { Dimension::Type::Signed32, "int32" },
102  { Dimension::Type::Unsigned32, "uint32" },
103  { Dimension::Type::Float, "float32" },
104  { Dimension::Type::Double, "float64" }
105  };
106 
107  try
108  {
109  return types.at(type);
110  }
111  catch (std::out_of_range)
112  {
113  throwError("Can't write dimension of type '" +
114  Dimension::interpretationName(type) + "'.");
115  }
116  return "";
117 }

161 void PlyWriter::writeValue(PointRef& point, Dimension::Id dim,
162  Dimension::Type type)
163 {
164  if (m_format == Format::Ascii)
165  {
166  double d = point.getFieldAs<double>(dim);
167  *m_stream << d;
168  }
169  else if (m_format == Format::BinaryLe)
170  {
171  OLeStream out(m_stream);
172  Everything e;
173  point.getField((char *)&e, dim, type);
174  Utils::insertDim(out, type, e);
175  }
176  else if (m_format == Format::BinaryBe)
177  {
178  OBeStream out(m_stream);
179  Everything e;
180  point.getField((char *)&e, dim, type);
181  Utils::insertDim(out, type, e);
182  }
183 }

186 void PlyWriter::writePoint(PointRef& point, PointLayoutPtr layout)
187 {
188  for (auto it = m_dims.begin(); it != m_dims.end();)
189  {
190  Dimension::Id dim = *it;
191  writeValue(point, dim, layout->dimType(dim));
192  ++it;
193  if (m_format == Format::Ascii && it != m_dims.end())
194  *m_stream << " ";
195  }
196  if (m_format == Format::Ascii)
197  *m_stream << std::endl;
198 }

201 void PlyWriter::writeTriangle(const Triangle& t, size_t offset)
202 {
203  if (m_format == Format::Ascii)
204  {
205  *m_stream << "3 " << (t.m_a + offset) << " " <<
206  (t.m_b + offset) << " " << (t.m_c + offset) << std::endl;
207  }
208  else if (m_format == Format::BinaryLe)
209  {
210  OLeStream out(m_stream);
211  unsigned char count = 3;
212  uint32_t a = (uint32_t)(t.m_a + offset);
213  uint32_t b = (uint32_t)(t.m_b + offset);
214  uint32_t c = (uint32_t)(t.m_c + offset);
215  out << count << a << b << c;
216  }
217  else if (m_format == Format::BinaryBe)
218  {
219  OBeStream out(m_stream);
220  unsigned char count = 3;
221  uint32_t a = (uint32_t)(t.m_a + offset);
222  uint32_t b = (uint32_t)(t.m_b + offset);
223  uint32_t c = (uint32_t)(t.m_c + offset);
224  out << count << a << b << c;
225  }
226 }

229 // Deferring write until this time allows both points and faces from multiple
230 // point views to be written.
231 void PlyWriter::done(PointTableRef table)
232 {
233  for (auto& v : m_views)
234  {
235  PointRef point(*v, 0);
236  for (PointId idx = 0; idx < v->size(); ++idx)
237  {
238  point.setPointId(idx);
239  writePoint(point, table.layout());
240  }
241  }
242  if (m_faces)
243  {
244  PointId offset = 0;
245  for (auto& v : m_views)
246  {
247  TriangularMesh *mesh = v->mesh();
248  if (mesh)
249  {
250  for (size_t id = 0; id < mesh->size(); ++id)
251  {
252  const Triangle& t = (*mesh)[id];
253  writeTriangle(t, offset);
254  }
255  }
256  offset += v->size();
257  }
258  }
259  Utils::closeFile(m_stream);
260  m_stream = nullptr;
261  getMetadata().addList("filename", m_filename);
262 }
