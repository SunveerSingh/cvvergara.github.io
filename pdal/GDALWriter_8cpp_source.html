84 void GDALWriter::initialize()
85 {
86  for (auto& ts : m_outputTypeString)
87  {
88  Utils::trim(ts);
89  if (ts == "all")
90  {
91  m_outputTypes = ~0;
92  break;
93  }
94  if (ts == "min")
95  m_outputTypes |= GDALGrid::statMin;
96  else if (ts == "max")
97  m_outputTypes |= GDALGrid::statMax;
98  else if (ts == "count")
99  m_outputTypes |= GDALGrid::statCount;
100  else if (ts == "mean")
101  m_outputTypes |= GDALGrid::statMean;
102  else if (ts == "idw")
103  m_outputTypes |= GDALGrid::statIdw;
104  else if (ts == "stdev")
105  m_outputTypes |= GDALGrid::statStdDev;
106  else
107  throwError("Invalid output type: '" + ts + "'.");
108  }
109 
110  gdal::registerDrivers();
111 }

142 void GDALWriter::createGrid(BOX2D bounds)
143 {
144  m_curBounds = bounds;
145  size_t width = ((m_curBounds.maxx - m_curBounds.minx) / m_edgeLength) + 1;
146  size_t height = ((m_curBounds.maxy - m_curBounds.miny) / m_edgeLength) + 1;
147  m_grid.reset(new GDALGrid(width, height, m_edgeLength, m_radius,
148  m_outputTypes, m_windowSize));

152 void GDALWriter::expandGrid(BOX2D bounds)
153 {
154  if (bounds == m_curBounds)
155  return;
156 
157  bounds.grow(m_curBounds);
158  size_t xshift = ceil((m_curBounds.minx - bounds.minx) / m_edgeLength);
159  bounds.minx = m_curBounds.minx - (xshift * m_edgeLength);
160  size_t yshift = ceil((m_curBounds.miny - bounds.miny) / m_edgeLength);
161  bounds.miny = m_curBounds.miny - (yshift * m_edgeLength);
162 
163  size_t width = ((bounds.maxx - bounds.minx) / m_edgeLength) + 1;
164  size_t height = ((bounds.maxy - bounds.miny) / m_edgeLength) + 1;
165  try
166  {
167  m_grid->expand(width, height, xshift, yshift);
168  }
169  catch (const GDALGrid::error& err)
170  {
171  throwError(err.what()); // Add the stage name onto the error text.
172  }
173  m_curBounds = bounds;
174 }

177 void GDALWriter::writeView(const PointViewPtr view)
178 {
179  BOX2D bounds;
180  if (m_bounds.to2d().valid())
181  bounds = m_bounds.to2d();
182  else
183  view->calculateBounds(bounds);
184 
185  if (!m_grid)
186  createGrid(bounds);
187  else
188  expandGrid(bounds);
189 
190  PointRef point(*view, 0);
191  for (PointId idx = 0; idx < view->size(); ++idx)
192  {
193  point.setPointId(idx);
194  processOne(point);
195  }
196 }

199 bool GDALWriter::processOne(PointRef& point)
200 {
201  double x = point.getFieldAs<double>(Dimension::Id::X) -
202  m_curBounds.minx;
203  double y = point.getFieldAs<double>(Dimension::Id::Y) -
204  m_curBounds.miny;
205  double z = point.getFieldAs<double>(m_interpDim);
206 
207  m_grid->addPoint(x, y, z);
208  return true;
209 }

212 void GDALWriter::doneFile()
213 {
214  std::array<double, 6> pixelToPos;
215 
216  pixelToPos[0] = m_curBounds.minx;
217  pixelToPos[1] = m_edgeLength;
218  pixelToPos[2] = 0;
219  pixelToPos[3] = m_curBounds.miny + (m_edgeLength * m_grid->height());
220  pixelToPos[4] = 0;
221  pixelToPos[5] = -m_edgeLength;
222  gdal::Raster raster(m_outputFilename, m_drivername, m_srs, pixelToPos);
223 
224  m_grid->finalize();
225 
226  gdal::GDALError err = raster.open(m_grid->width(), m_grid->height(),
227  m_grid->numBands(), m_dataType, m_noData, m_options);
228 
229  if (err != gdal::GDALError::None)
230  throwError(raster.errorMsg());
231  int bandNum = 1;
232 
233  // Perhaps the grid should return an iterator, which would work as well.
234  double *src;
235  src = m_grid->data("min");
236  double srcNoData = std::numeric_limits<double>::quiet_NaN();
237  if (src && err == gdal::GDALError::None)
238  err = raster.writeBand(src, srcNoData, bandNum++, "min");
239  src = m_grid->data("max");
240  if (src && err == gdal::GDALError::None)
241  err = raster.writeBand(src, srcNoData, bandNum++, "max");
242  src = m_grid->data("mean");
243  if (src && err == gdal::GDALError::None)
244  err = raster.writeBand(src, srcNoData, bandNum++, "mean");
245  src = m_grid->data("idw");
246  if (src && err == gdal::GDALError::None)
247  err = raster.writeBand(src, srcNoData, bandNum++, "idw");
248  src = m_grid->data("count");
249  if (src && err == gdal::GDALError::None)
250  err = raster.writeBand(src, srcNoData, bandNum++, "count");
251  src = m_grid->data("stdev");
252  if (src && err == gdal::GDALError::None)
253  err = raster.writeBand(src, srcNoData, bandNum++, "stdev");
254  if (err != gdal::GDALError::None)
255  throwError(raster.errorMsg());
256 
257  getMetadata().addList("filename", m_filename);
258 }
