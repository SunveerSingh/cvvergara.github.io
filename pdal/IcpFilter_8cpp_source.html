84 PointViewPtr IcpFilter::icp(PointViewPtr fixed, PointViewPtr moving) const
85 {
86  typedef pcl::PointXYZ Point;
87  typedef pcl::PointCloud<Point> Cloud;
88  Cloud::Ptr fixedCloud(new Cloud());
89  pclsupport::PDALtoPCD(fixed, *fixedCloud);
90  Cloud::Ptr movingCloud(new Cloud());
91  pclsupport::PDALtoPCD(moving, *movingCloud);
92  pcl::IterativeClosestPoint<Point, Point> icp;
93  icp.setInputCloud(movingCloud);
94  icp.setInputTarget(fixedCloud);
95  Cloud result;
96  icp.align(result);
97 
98  MetadataNode root = getMetadata();
99  // I couldn't figure out the template-fu to get
100  // `MetadataNodeImpl::setValue` to work for all Eigen matrices with one
101  // function, so I'm just brute-forcing the cast for now.
102  root.add("transform",
103  Eigen::MatrixXd(icp.getFinalTransformation().cast<double>()));
104  root.add("converged", icp.hasConverged());
105  root.add("fitness", icp.getFitnessScore());
106 
107  assert(moving->size() == result.points.size());
108  for (PointId i = 0; i < moving->size(); ++i)
109  {
110  moving->setField(Dimension::Id::X, i, result.points[i].x);
111  moving->setField(Dimension::Id::Y, i, result.points[i].y);
112  moving->setField(Dimension::Id::Z, i, result.points[i].z);
113  }
114  return moving;
115 }
