100  // This essentially moves ready() and done() into write(), which means
101  // that they get executed once for each view. The check for m_hashPos
102  // is a test to see if the filename specification is a template. If it's
103  // not a template, ready() and done() are taken care of in the ready()
104  // and done() functions in this class.
105  virtual void write(const PointViewPtr view) final
106  {
107  if (m_hashPos != std::string::npos)
