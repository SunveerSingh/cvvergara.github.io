176  std::vector<PointId> radius(double const& x, double const& y,
177  double const& r) const
178  {
179  std::vector<PointId> output;
180  std::vector<std::pair<std::size_t, double>> ret_matches;
181  nanoflann::SearchParams params;
182  params.sorted = true;
183 
184  std::vector<double> pt;
185  pt.push_back(x);
186  pt.push_back(y);
187 
188  // Our distance metric is square distance, so we use the square of
189  // the radius.
190  const std::size_t count =
191  m_index->radiusSearch(&pt[0], r * r, ret_matches, params);
192 
193  for (std::size_t i = 0; i < count; ++i)
194  output.push_back(ret_matches[i].first);
195  return output;
196  }

408 // nanoflann hands us a vector that represents the position of p1. We fetch
409 // the position of p2 and and compute the square distance.
410 template<>
411 inline double KDIndex<2>::kdtree_distance(const double *p1, const PointId idx,
412  size_t /*numDims*/) const
413 {
414  double d0 = p1[0] - m_buf.getFieldAs<double>(Dimension::Id::X, idx);
415  double d1 = p1[1] - m_buf.getFieldAs<double>(Dimension::Id::Y, idx);
416 
417  return (d0 * d0 + d1 * d1);
418 }

420 template<>
421 inline double KDIndex<3>::kdtree_distance(const double *p1, const PointId idx,
422  size_t /*numDims*/) const
423 {
424  double d0 = p1[0] - m_buf.getFieldAs<double>(Dimension::Id::X, idx);
425  double d1 = p1[1] - m_buf.getFieldAs<double>(Dimension::Id::Y, idx);
426  double d2 = p1[2] - m_buf.getFieldAs<double>(Dimension::Id::Z, idx);
427 
428  return (d0 * d0 + d1 * d1 + d2 * d2);
