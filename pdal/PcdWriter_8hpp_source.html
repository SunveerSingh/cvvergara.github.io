88 template<typename CloudT>
89 void PcdWriter::writeView(const PointViewPtr view)
90 {
91  typedef typename CloudT::PointType PointT;
92  typename CloudT::Ptr cloud(new CloudT);
93  BOX3D bounds;
94  if (m_subtract_minimum)
95  {
96  view->calculateBounds(bounds);
97  bounds.grow(bounds.minx + m_offset_x,
98  bounds.miny + m_offset_y,
99  bounds.minz + m_offset_z);
100  }
101  else
102  {
103  bounds.grow(m_offset_x, m_offset_y, m_offset_z);
104  }
105  pclsupport::PDALtoPCD(view, *cloud, bounds, m_scale_x, m_scale_y, m_scale_z);
106  pcl::PCDWriter w;
107 
108  if (m_compression_string == "binary")
109  {
110  m_compression = 1;
111  }
112  else if (m_compression_string == "compressed")
113  {
114  m_compression = 2;
115  }
116  else // including "ascii"
117  {
118  m_compression = 0;
119  }
120 
121  switch (m_compression)
122  {
123  case 0 : w.writeASCII<PointT>(m_filename, *cloud); break;
124  case 1 : w.writeBinary<PointT>(m_filename, *cloud); break;
125  case 2 : w.writeBinaryCompressed<PointT>(m_filename, *cloud); break;
126  }
127 }
