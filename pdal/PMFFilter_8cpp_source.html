102 PointViewSet PMFFilter::run(PointViewPtr input)
103 {
104  PointViewSet viewSet;
105  if (!input->size())
106  return viewSet;
107 
108  // Segment input view into ignored/kept views.
109  PointViewPtr ignoredView = input->makeNew();
110  PointViewPtr keptView = input->makeNew();
111  if (m_ignored.m_id == Dimension::Id::Unknown)
112  keptView->append(*input);
113  else
114  Segmentation::ignoreDimRange(m_ignored, input, keptView, ignoredView);
115 
116  // Classify remaining points with value of 1. processGround will mark ground
117  // returns as 2.
118  for (PointId i = 0; i < keptView->size(); ++i)
119  keptView->setField(Dimension::Id::Classification, i, 1);
120 
121  // Segment kept view into last/other-than-last return views.
122  PointViewPtr lastView = keptView->makeNew();
123  PointViewPtr nonlastView = keptView->makeNew();
124  if (m_lastOnly)
125  Segmentation::segmentLastReturns(keptView, lastView, nonlastView);
126  else
127  lastView->append(*keptView);
128 
129  // Run the actual PMF algorithm.
130  processGround(lastView);
131 
132  // Prepare the output PointView.
133  PointViewPtr outView = input->makeNew();
134  outView->append(*ignoredView);
135  outView->append(*nonlastView);
136  outView->append(*lastView);
137  viewSet.insert(outView);
138 
139  return viewSet;
140 }

142 void PMFFilter::processGround(PointViewPtr view)
143 {
144  // initialize bounds, rows, columns, and surface
145  BOX2D bounds;
146  view->calculateBounds(bounds);
147  size_t cols = ((bounds.maxx - bounds.minx) / m_cellSize) + 1;
148  size_t rows = ((bounds.maxy - bounds.miny) / m_cellSize) + 1;
149 
150  // initialize surface to NaN
151  std::vector<double> ZImin(rows * cols,
152  std::numeric_limits<double>::quiet_NaN());
153 
154  // loop through all points, identifying minimum Z value for each populated
155  // cell
156  for (PointId i = 0; i < view->size(); ++i)
157  {
158  double x = view->getFieldAs<double>(Dimension::Id::X, i);
159  double y = view->getFieldAs<double>(Dimension::Id::Y, i);
160  double z = view->getFieldAs<double>(Dimension::Id::Z, i);
161  int c = static_cast<int>(floor(x - bounds.minx) / m_cellSize);
162  int r = static_cast<int>(floor(y - bounds.miny) / m_cellSize);
163  size_t idx = c * rows + r;
164  if (z < ZImin[idx] || std::isnan(ZImin[idx]))
165  ZImin[idx] = z;
166  }
167 
168  // convert vector to PointView for indexing
169  PointViewPtr temp = view->makeNew();
170  PointId i(0);
171  for (size_t c = 0; c < cols; ++c)
172  {
173  for (size_t r = 0; r < rows; ++r)
174  {
175  size_t idx = c * rows + r;
176  if (std::isnan(ZImin[idx]))
177  continue;
178  double x = bounds.minx + (c + 0.5) * m_cellSize;
179  double y = bounds.miny + (r + 0.5) * m_cellSize;
180  temp->setField(Dimension::Id::X, i, x);
181  temp->setField(Dimension::Id::Y, i, y);
182  temp->setField(Dimension::Id::Z, i, ZImin[idx]);
183  i++;
184  }
185  }
186 
187  // build the 2D KD-tree
188  KD2Index kdi(*temp);
189  kdi.build();
190 
191  // loop through all cells, and for each NaN, replace with elevation of
192  // nearest neighbor
193  std::vector<double> out = ZImin;
194  for (size_t c = 0; c < cols; ++c)
195  {
196  for (size_t r = 0; r < rows; ++r)
197  {
198  size_t idx = c * rows + r;
199  if (!std::isnan(out[idx]))
200  continue;
201  double x = bounds.minx + (c + 0.5) * m_cellSize;
202  double y = bounds.miny + (r + 0.5) * m_cellSize;
203  int k = 1;
204  std::vector<PointId> neighbors(k);
205  std::vector<double> sqr_dists(k);
206  kdi.knnSearch(x, y, k, &neighbors, &sqr_dists);
207  out[idx] = temp->getFieldAs<double>(Dimension::Id::Z, neighbors[0]);
208  }
209  }
210 
211  ZImin.swap(out);
212 
213  // initialize ground indices
214  std::vector<PointId> groundIdx;
215  for (PointId i = 0; i < view->size(); ++i)
216  groundIdx.push_back(i);
217 
218  // Compute the series of window sizes and height thresholds
219  std::vector<float> htvec;
220  std::vector<float> wsvec;
221  int iter = 0;
222  float ws = 0.0f;
223  float ht = 0.0f;
224 
225  // pre-compute window sizes and height thresholds
226  while (ws < m_maxWindowSize)
227  {
228  // Determine the initial window size.
229  if (m_exponential)
230  ws = m_cellSize * (2.0f * std::pow(2, iter) + 1.0f);
231  else
232  ws = m_cellSize * (2.0f * (iter + 1) * 2 + 1.0f);
233 
234  // Calculate the height threshold to be used in the next iteration.
235  if (iter == 0)
236  ht = m_initialDistance;
237  else
238  ht = m_slope * (ws - wsvec[iter - 1]) * m_cellSize +
239  m_initialDistance;
240 
241  // Enforce max distance on height threshold
242  if (ht > m_maxDistance)
243  ht = m_maxDistance;
244 
245  wsvec.push_back(ws);
246  htvec.push_back(ht);
247 
248  iter++;
249  }
250 
251  // Progressively filter ground returns using morphological open
252  for (size_t j = 0; j < wsvec.size(); ++j)
253  {
254  log()->get(LogLevel::Debug)
255  << "Iteration " << j << " (height threshold = " << htvec[j]
256  << ", window size = " << wsvec[j] << ")...\n";
257 
258  int iters = 0.5 * (wsvec[j] - 1);
259  using namespace eigen;
260  std::vector<double> me = erodeDiamond(ZImin, rows, cols, iters);
261  std::vector<double> mo = dilateDiamond(me, rows, cols, iters);
262 
263  std::vector<PointId> groundNewIdx;
264  for (auto p_idx : groundIdx)
265  {
266  double x = view->getFieldAs<double>(Dimension::Id::X, p_idx);
267  double y = view->getFieldAs<double>(Dimension::Id::Y, p_idx);
268  double z = view->getFieldAs<double>(Dimension::Id::Z, p_idx);
269 
270  int c = static_cast<int>(floor((x - bounds.minx) / m_cellSize));
271  int r = static_cast<int>(floor((y - bounds.miny) / m_cellSize));
272 
273  if ((z - mo[c * rows + r]) < htvec[j])
274  groundNewIdx.push_back(p_idx);
275  }
276 
277  ZImin.swap(mo);
278  groundIdx.swap(groundNewIdx);
279 
280  log()->get(LogLevel::Debug)
281  << "Ground now has " << groundIdx.size() << " points.\n";
282  }
283 
284  log()->get(LogLevel::Debug2)
285  << "Labeled " << groundIdx.size() << " ground returns!\n";
286 
287  // set the classification label of ground returns as 2
288  // (corresponding to ASPRS LAS specification)
289  for (const auto& i : groundIdx)
290  view->setField(Dimension::Id::Classification, i, 2);
291 }
